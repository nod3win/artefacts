% THEME_TITLE: Compter les chiffres significatifs
\sisetup{per-mode=symbol}

Compter les \textbf{chiffres significatifs} (CS) d'un nombre mesuré, c'est lire la précision
que ce nombre annonce. C'est une seule mécanique, entièrement gouvernée par le sort du chiffre~$0$.

\begin{cle}[ce qu'est un chiffre significatif]
Les \textbf{chiffres significatifs} d'un nombre mesuré sont tous ses chiffres \emph{certains}, plus le
\emph{premier chiffre incertain} (le dernier écrit, celui qu'on a estimé). \textbf{Tout chiffre non nul}
($1$ à $9$) est \emph{toujours} significatif : seul le chiffre $0$ demande réflexion.
\end{cle}

\begin{methode}[compter les CS d'un nombre]
On lit le nombre \textbf{de gauche à droite}, en réglant le sort de chaque zéro par trois règles :
\begin{enumerate}[label=\arabic*.,leftmargin=1.8em,topsep=2pt,itemsep=1.5pt]
  \item \textbf{zéros de tête} (à gauche du premier chiffre non nul) : \emph{jamais} significatifs ---
        ils ne servent qu'à caler la virgule (l'ordre de grandeur) ;
  \item \textbf{zéros captifs} (entre deux chiffres non nuls) : \emph{toujours} significatifs ;
  \item \textbf{zéros de queue après la virgule} : \emph{toujours} significatifs.
\end{enumerate}
Le premier chiffre significatif est donc \emph{toujours} le premier chiffre non nul ; on compte
ensuite tous les chiffres jusqu'au dernier écrit.
\end{methode}

Lecture : $0,00$ = trois zéros de tête (\emph{non} sig.) ; $3$ et $5$ non nuls ; le $0$ entre $5$ et $7$
est \emph{captif} (sig.) ; $7$ non nul ; le $0$ final, de queue \emph{après la virgule}, est sig.
Bilan : les chiffres $3,\,5,\,0,\,7,\,0$ sont significatifs, soit $\mathbf{5}$ CS.
\end{exemplebox}

\begin{piege}[deux confusions classiques]
\begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=2pt]
  \item \textbf{CS $\neq$ décimales.} $42,5$ n'a qu'une décimale mais \emph{3} CS ; $0,007$ a trois
        décimales mais un \emph{seul} CS. On compte à partir du premier chiffre \emph{non nul}.
  \item \textbf{Zéros de queue d'un entier} (sans virgule) : $100$ est \emph{ambigu} ($1$, $2$ ou $3$
        CS). Seule la \textbf{notation scientifique} tranche : $1\times10^{2}$ (1 CS),
        $1,0\times10^{2}$ (2 CS), $1,00\times10^{2}$ (3 CS).
\end{itemize}
\end{piege}

\begin{cle}[nombres exacts : une infinité de CS]
Un nombre issu d'un \emph{comptage} ($2$ atomes de \ce{H} dans une molécule d'\ce{H2O}) ou d'une
\emph{définition} ($\qty{1}{\kilo\gram}=\qty{1000}{\gram}$ exactement) est \textbf{exact} : on lui
attribue une \emph{infinité} de CS. Il ne limite donc jamais la précision d'un résultat.
\end{cle}

\begin{remarque}
Changer d'unité ou passer en notation scientifique ne fait que \emph{déplacer la virgule} : cela ne
change \textbf{jamais} le nombre de CS. Ainsi $\qty{20,96}{\milli\litre}=\qty{0,02096}{\litre}
=\num{2,096e-2}\,\si{\litre}$ ont tous les trois $4$ CS.
\end{remarque}
