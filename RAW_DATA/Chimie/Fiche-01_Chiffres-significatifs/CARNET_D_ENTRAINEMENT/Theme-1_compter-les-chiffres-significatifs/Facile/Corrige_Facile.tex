\begin{cor}[Chiffres non nuls uniquement]
Tout chiffre non nul est toujours significatif : il suffit de les compter.
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{837}$ : $\mathbf{3}$~CS. \quad b) $\num{42,5}$ : $\mathbf{3}$~CS. \quad
        c) $\num{6,1234}$ : $\mathbf{5}$~CS.
  \item d) $\num{9}$ : $\mathbf{1}$~CS. \quad e) $\num{318,49}$ : $\mathbf{5}$~CS.
\end{itemize}
La position de la virgule n'a aucune importance : seul le nombre de chiffres compte.
\end{cor}

\begin{cor}[Zéros de tête]
Les zéros situés à gauche du premier chiffre non nul ne servent qu'à caler la virgule : ils ne
comptent \emph{jamais}. Le comptage démarre au premier chiffre non nul.
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{0,007}$ : les $0{,}00$ sont de tête $\rightarrow$ seul le $7$ compte $\Rightarrow \mathbf{1}$~CS.
  \item b) $\num{0,42}$ : $\rightarrow \mathbf{2}$~CS.\qquad
        c) $\num{0,00305}$ : $3$, $0$ (captif), $5 \Rightarrow \mathbf{3}$~CS.
  \item d) $\num{0,000010}$ : zéros de tête non sig. ; le $1$ et le zéro de queue comptent
        $\Rightarrow \mathbf{2}$~CS.\qquad e) $\num{0,68}$ : $\mathbf{2}$~CS.
\end{itemize}
\end{cor}

\begin{cor}[Zéros captifs et zéros de queue]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{2005}$ : les deux zéros sont \emph{captifs} (entre $2$ et $5$) $\Rightarrow \mathbf{4}$~CS.
  \item b) $\num{60,0}$ : zéro captif ? non — c'est $6$, puis un zéro, puis un zéro de queue après
        virgule. Les deux zéros comptent $\Rightarrow \mathbf{3}$~CS.
  \item c) $\num{1,004}$ : deux zéros captifs $\Rightarrow \mathbf{4}$~CS.
  \item d) $\num{15,200}$ : deux zéros de queue après la virgule, significatifs $\Rightarrow \mathbf{5}$~CS.
  \item e) $\num{40,080}$ : $4$, zéro captif, $8$, zéro de queue $\Rightarrow \mathbf{5}$~CS.
\end{itemize}
\end{cor}

\begin{cor}[Chiffres significatifs contre décimales]
Les décimales se comptent \emph{après la virgule} ; les CS se comptent \emph{à partir du premier
chiffre non nul}. Ce sont deux comptages indépendants.
\begin{center}\renewcommand{\arraystretch}{1.25}
\begin{tabular}{c c c}
\toprule
\textbf{Nombre} & \textbf{Décimales} & \textbf{CS}\\
\midrule
$\num{0,007}$   & $3$ & $1$\\
$\num{42,5}$    & $1$ & $3$\\
$\num{15,200}$  & $3$ & $5$\\
$\num{3,0}$     & $1$ & $2$\\
\bottomrule
\end{tabular}
\end{center}
\textbf{Conclusion :} un nombre peut avoir beaucoup de décimales et peu de CS ($\num{0,007}$), ou peu
de décimales et beaucoup de CS ($\num{42,5}$). Ne jamais confondre les deux.
\end{cor}

\begin{cor}[Le rôle de chaque zéro]
$m = \qty{0,04080}{\gram}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item Lecture chiffre par chiffre :
  \begin{center}
  \begin{tikzpicture}[font=\small]
    \node[font=\Large] at (0,0)
      {$0,\textcolor{nosigcol}{0}\,\textbf{4}\,\textcolor{sigcol}{\textbf 0}\,\textbf{8}\,\textcolor{sigcol}{\textbf 0}$};
  \end{tikzpicture}
  \end{center}
  \begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=0pt]
    \item le premier zéro ($0{,}0$) est un \textcolor{nosigcol}{zéro de tête} $\rightarrow$ \emph{non} significatif ;
    \item le $4$ est non nul $\rightarrow$ significatif ;
    \item le zéro suivant est \textcolor{sigcol}{captif} (entre $4$ et $8$) $\rightarrow$ significatif ;
    \item le $8$ est non nul $\rightarrow$ significatif ;
    \item le dernier zéro est un \textcolor{sigcol}{zéro de queue après la virgule} $\rightarrow$ significatif.
  \end{itemize}
  \item Chiffres significatifs : $4,\,0,\,8,\,0 \Rightarrow \boxed{\mathbf{4}\ \text{CS}}$.
\end{enumerate}
\end{cor}
