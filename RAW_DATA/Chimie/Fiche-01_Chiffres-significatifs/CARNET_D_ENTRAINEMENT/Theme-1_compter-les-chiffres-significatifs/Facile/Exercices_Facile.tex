\begin{exo}[chiffres non nuls]
Aucun de ces nombres ne contient de zéro. Donne, pour chacun, le nombre de chiffres significatifs.
\begin{enumerate}[label=\alph*)]
  \item $837$
  \item $42,5$
  \item $6,1234$
  \item $128,77$
\end{enumerate}
\end{exo}

\begin{exo}[zéros de tête]
Ici tous les zéros sont \emph{devant} le premier chiffre non nul. Donne le nombre de chiffres
significatifs.
\begin{enumerate}[label=\alph*)]
  \item $0,007$
  \item $0,052$
  \item $0,0035$
  \item $0,4$
\end{enumerate}
\end{exo}

\begin{exo}[zéros captifs]
Chaque zéro est ici coincé \emph{entre} deux chiffres non nuls. Combien de chiffres significatifs ?
\begin{enumerate}[label=\alph*)]
  \item $205$
  \item $4007$
  \item $3,05$
  \item $60,2$
\end{enumerate}
\end{exo}

\begin{exo}[zéros de queue après la virgule]
Attention aux zéros \emph{finaux} placés après la virgule. Donne le nombre de chiffres significatifs.
\begin{enumerate}[label=\alph*)]
  \item $15,200$
  \item $0,0100$
  \item $3,40$
  \item $8,000$
\end{enumerate}
\end{exo}

\begin{exo}[tous les cas à la fois]
Ces nombres mélangent les trois règles sur les zéros. Compte les chiffres significatifs.
\begin{enumerate}[label=\alph*)]
  \item $0,0035070$
  \item $1030,0$
  \item $0,004080$
  \item $20,08$
\end{enumerate}
\end{exo}
