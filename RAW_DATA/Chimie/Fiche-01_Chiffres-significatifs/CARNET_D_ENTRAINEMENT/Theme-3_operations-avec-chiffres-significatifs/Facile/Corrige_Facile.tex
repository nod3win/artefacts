\begin{corrige}[addition et soustraction : les décimales]
On calcule, puis on arrondit au nombre de décimales de la donnée qui en a le moins.
\begin{enumerate}[label=\alph*)]
  \item $255,507 \to \boxed{255,5}$ (le terme $46,0$ n'a qu'$1$ décimale).
  \item $0,11 \to \boxed{0,1}$ ($8,0$ n'a qu'$1$ décimale ; on tombe à $1$ CS).
  \item $13,47 \to \boxed{13,5}$ ($12,4$ a $1$ décimale).
  \item $43,43 \to \boxed{43,4}$ ($45,6$ a $1$ décimale).
\end{enumerate}
\end{corrige}

\begin{corrige}[multiplication et division : les CS]
On calcule, puis on arrondit au nombre de CS de la donnée qui en a le moins.
\begin{enumerate}[label=\alph*)]
  \item $2,5\times3,42 = 8,55 \to \boxed{8,6}$ ($2,5$ a $2$ CS).
  \item $0,0361\div12,0 = 0,00300833\ldots \to \boxed{3,01\times10^{-3}}$ ($3$ CS).
  \item $4,00\times2,1 = 8,4 \to \boxed{8,4}$ ($2,1$ a $2$ CS).
  \item $6,022\div2,0 = 3,011 \to \boxed{3,0}$ ($2,0$ a $2$ CS).
\end{enumerate}
\end{corrige}

\begin{corrige}[quelle règle, quel rang ?]
On repère l'opération, puis la donnée la plus pauvre.
\begin{enumerate}[label=\alph*)]
  \item $\times$ : on compte les CS. Minimum $= 2$ CS ($2,5$) $\Rightarrow$ résultat à $2$ CS.
  \item $+$ : on compte les décimales. Minimum $= 1$ décimale ($100,2$) $\Rightarrow$ résultat à $1$ décimale.
  \item $\div$ : on compte les CS. Minimum $= 2$ CS ($8,0$) $\Rightarrow$ résultat à $2$ CS.
  \item $-$ : on compte les décimales. Minimum $= 1$ décimale ($3,1$) $\Rightarrow$ résultat à $1$ décimale.
\end{enumerate}
\end{corrige}

\begin{corrige}[aire et volume]
Ce sont des multiplications : on garde le minimum de CS des données.
\begin{enumerate}[label=\alph*)]
  \item $2,5\times4,0 = 10,0 \to \boxed{1,0\times10^{1}\ \si{\centi\metre\squared}}$ ($2$ CS ; la notation
        scientifique évite l'ambiguïté de «~$10$~»).
  \item $3,0^{3} = 27,0 \to \boxed{\qty{27}{\centi\metre\cubed}}$ ($2$ CS).
\end{enumerate}
\end{corrige}
