\begin{center}
{\Large\bfseries\color{primarydark} Exercices — Niveau Moyen}\\[1pt]
{\color{primary} Thème 3 : les deux règles des opérations}
\end{center}
\vspace{1mm}
\textit{Pour chaque opération : calcule d'abord la valeur exacte, puis arrondis avec la bonne règle
(décimales pour $+/-$, \CS\ pour $\times/\div$).}
\vspace{2mm}

\begin{exo}[Addition : la règle des décimales]
On additionne trois volumes lus sur des instruments de précisions différentes :
\[ V = \qty{12,4}{\milli\litre} + \qty{7,25}{\milli\litre} + \qty{103,682}{\milli\litre}. \]
Donne la valeur brute de la somme, puis le résultat avec le bon nombre de décimales.
\end{exo}

\begin{exo}[Soustraction : la perte de chiffres significatifs]
Calcule chaque différence, donne le résultat avec le bon nombre de décimales, puis compte les \CS\
\emph{avant} et \emph{après} l'opération.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\qty{12,47}{\gram} - \qty{12,3}{\gram}$
  \item $\qty{5,624}{\gram} - \qty{5,60}{\gram}$
\end{enumerate}
Que constate-t-on dans les deux cas ?
\end{exo}

\begin{exo}[Multiplication et division : la règle des \CS]
Calcule et arrondis au bon nombre de \CS.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{6,3} \times \num{2,1}$ \qquad b) $\num{2,50} \times \num{3,142}$ \qquad c) $\num{45,0} \div \num{1,2}$
\end{enumerate}
\end{exo}

\begin{exo}[Quelle règle s'applique ?]
Sans faire le calcul, indique pour chaque opération quelle règle donne la précision du résultat
(« décimales » ou « \CS »), et combien de décimales ou de CS aura le résultat.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{2,5} + \num{13,42}$ \qquad b) $\num{2,5} \times \num{13,42}$
  \item[] c) $\num{100,0} - \num{2,555}$ \qquad d) $\num{8,20} \div \num{4,1}$
\end{enumerate}
\end{exo}

\begin{exo}[Concentration massique]
On dissout une masse $m = \qty{4,80}{\gram}$ de soluté dans un volume $V = \qty{15,0}{\milli\litre}$
de solution. La concentration massique est $c_m = \dfrac{m}{V}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item Calcule $c_m$ en \si{\gram\per\milli\litre} avec le bon nombre de \CS.
  \item Exprime $c_m$ en \si{\gram\per\litre}, en notation scientifique.
\end{enumerate}
\end{exo}
