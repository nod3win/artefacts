\begin{cor}[Arrondir à un nombre de \CS\ donné]
On garde les CS demandés et on regarde le premier chiffre supprimé.
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{3,148}\to$ supprimé $8\,(\ge5)\to \mathbf{\num{3,15}}$.
  \item b) $\num{3,142}\to$ supprimé $2\,(<5)\to \mathbf{\num{3,14}}$.
  \item c) $\num{0,06237}\to$ CS $=6,2$ ; supprimé $3\,(<5)\to \mathbf{\num{0,062}}$.
  \item d) $\num{124,7}\to$ supprimé $7\,(\ge5)\to \mathbf{\num{125}}$.
  \item e) $\num{7,8451}\to$ supprimé $4\,(<5)\to \mathbf{\num{7,8}}$.
\end{itemize}
\end{cor}

\begin{cor}[Le premier chiffre supprimé]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{0,4271}$ à $2$~CS : premier supprimé $=7\,(\ge5)\to$ par excès $\to \mathbf{\num{0,43}}$.
  \item b) $\num{58,96}$ à $3$~CS : premier supprimé $=6\,(\ge5)\to$ par excès $\to \mathbf{\num{59,0}}$.
  \item c) $\num{1,2350}$ à $3$~CS : premier supprimé $=5\,(\ge5)\to$ par excès $\to \mathbf{\num{1,24}}$.
\end{itemize}
\end{cor}

\begin{cor}[Attention aux retenues en cascade]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{2,996}$ à $3$~CS : supprimé $6\,(\ge5)$, $\num{2,99}+\num{0,01}=\mathbf{\num{3,00}}$
        (on garde « $00$ » pour avoir $3$~CS).
  \item b) $\num{9,98}$ à $2$~CS : supprimé $8\,(\ge5)$, $\num{9,9}+\num{0,1}=\num{10}$, soit
        $\mathbf{\num{1,0e1}}$ (pour afficher clairement $2$~CS).
  \item c) $\num{0,7996}$ à $3$~CS : CS $=7,9,9$ ; supprimé $6\,(\ge5)$,
        $\num{0,799}+\num{0,001}=\mathbf{\num{0,800}}$.
\end{itemize}
\end{cor}

\begin{cor}[Arrondir à un nombre de \emph{décimales} donné]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{255,507}$ à $1$ décimale : supprimé $0\,(<5)\to \mathbf{\num{255,5}}$.
  \item b) $\num{0,1149}$ à $2$ décimales : supprimé $4\,(<5)\to \mathbf{\num{0,11}}$.
  \item c) $\num{18,016}$ à $2$ décimales : supprimé $6\,(\ge5)\to \mathbf{\num{18,02}}$.
\end{itemize}
\end{cor}

\begin{cor}[Arrondir de grands nombres]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) $\num{45678}$ à $2$~CS : $\num{4,5}\dots$, supprimé $6\,(\ge5)\to \mathbf{\num{4,6e4}}$.
  \item b) $\num{30960}$ à $3$~CS : $\num{3,09}\dots$, supprimé $6\,(\ge5)\to \mathbf{\num{3,10e4}}$.
  \item c) $\num{1249}$ à $2$~CS : $\num{1,2}\dots$, supprimé $4\,(<5)\to \mathbf{\num{1,2e3}}$.
\end{itemize}
\end{cor}
