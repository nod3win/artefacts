\begin{cor}[Une pesée dans plusieurs unités]
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $m=\qty{4,750}{\gram}$ : le zéro de queue après la virgule est significatif $\Rightarrow \mathbf{4}$~CS.
  \item Conversions (on déplace seulement la virgule) :
        \[ m=\num{4,750e3}\,\si{\milli\gram}=\qty{0,004750}{\kilo\gram}=\num{4,750e-6}\,\si{\tonne}. \]
        En milligrammes, on écrit $\num{4,750e3}$ (et non $\num{4750}$) pour ne pas rendre le dernier
        zéro ambigu.
  \item Chaque écriture possède $4$~CS : c'est le \textbf{théorème du déplacement de la virgule} —
        changer d'unité ne modifie jamais la précision.
\end{enumerate}
\end{cor}

\begin{cor}[Un volume prélevé à la pipette]
$V=\qty{3,50}{\centi\litre}$, soit $\mathbf{3}$~CS.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\qty{1}{\centi\litre}=\qty{10}{\milli\litre}$, donc $V=\qty{35,0}{\milli\litre}
        =\qty{0,0350}{\litre}=\num{3,50e4}\,\si{\micro\litre}$.
  \item En litres : $\boxed{V=\num{3,50e-2}\,\si{\litre}}$.
  \item Toutes ces écritures ont $3$~CS : la conversion n'a rien changé à la précision.
\end{enumerate}
\end{cor}

\begin{cor}[Étiquette d'eau minérale]
$[\ce{Ca^{2+}}]=\qty{90,0}{\milli\gram\per\litre}$, soit $\mathbf{3}$~CS.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\qty{1}{\milli\gram}=\num{e-3}\,\si{\gram}$ : $[\ce{Ca^{2+}}]=\qty{0,0900}{\gram\per\litre}$.
        Et $\qty{1}{\milli\gram}=\qty{1000}{\micro\gram}$ : $[\ce{Ca^{2+}}]=\num{9,00e4}\,\si{\micro\gram\per\litre}$.
  \item Forme scientifique en \si{\gram\per\litre} : $\num{9,00e-2}\,\si{\gram\per\litre}$.
  \item Non : le nombre de CS reste $3$ partout. Seuls apparaissent/disparaissent des zéros de cadrage.
\end{enumerate}
\end{cor}

\begin{cor}[Exact ou mesuré ?]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item a) \textbf{exact} (comptage d'atomes) ; \quad b) \textbf{mesuré} ($2$~CS) ;
  \item c) \textbf{exact} (définition) ; \quad d) \textbf{exact} ($\pi$ connu avec autant de décimales
        qu'on veut) ;
  \item e) \textbf{exact} (comptage) ; \quad f) \textbf{mesuré} ($3$~CS).
\end{itemize}
\textbf{Question finale :} \emph{non}. Un nombre exact possède une infinité de CS ; il n'est donc
jamais le « maillon faible » et ne limite \emph{jamais} le nombre de CS d'un résultat.
\end{cor}

\begin{cor}[Convertir avec un facteur exact]
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\displaystyle m=\frac{\num{6,25}}{\num{1000}}\ \si{\kilo\gram}=\qty{0,00625}{\kilo\gram}
        =\num{6,25e-3}\,\si{\kilo\gram}$.
  \item Le $\num{1000}$ est exact (infinité de CS) : il n'a rien limité. $m$ conserve ses
        $\mathbf{3}$~CS. On n'a fait que déplacer la virgule de $3$ rangs.
\end{enumerate}
\end{cor}
