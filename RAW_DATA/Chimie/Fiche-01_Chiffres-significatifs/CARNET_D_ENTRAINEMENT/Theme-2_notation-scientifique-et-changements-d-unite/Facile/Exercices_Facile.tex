\vspace{1mm}
\textit{On s'entraîne au geste de conversion en notation scientifique, dans les deux sens.}
\vspace{2mm}

\begin{exo}[Grands nombres $\rightarrow$ notation scientifique]
Mets chaque nombre en notation scientifique $a\times 10^{n}$ (avec $1\le a <10$).
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{45000}$ \qquad b) $\num{128}$ \qquad c) $\num{9300}$ \qquad d) $\num{6}$ \qquad e) $\num{20960}$
\end{enumerate}
\end{exo}

\begin{exo}[Petits nombres $\rightarrow$ notation scientifique]
Même consigne pour ces nombres inférieurs à $1$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{0,02096}$ \qquad b) $\num{0,00072}$ \qquad c) $\num{0,5}$ \qquad d) $\num{0,0000130}$ \qquad e) $\num{0,048}$
\end{enumerate}
\end{exo}

\begin{exo}[Notation scientifique $\rightarrow$ écriture décimale]
Écris chaque nombre sous forme décimale usuelle (avec la virgule).
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{2,096e-2}$ \qquad b) $\num{3,00e4}$ \qquad c) $\num{6,02e2}$ \qquad d) $\num{1,0e-7}$ \qquad e) $\num{4,5e1}$
\end{enumerate}
\end{exo}

\begin{exo}[Ne garder que les chiffres significatifs dans la mantisse]
Mets en notation scientifique en veillant à ce que la mantisse contienne \emph{exactement} les \CS\
du nombre de départ (donc en supprimant les zéros de cadrage, mais pas les autres).
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{0,004500}$ \qquad b) $\num{0,0100}$ \qquad c) $\num{15,200}$
\end{enumerate}
\end{exo}

\begin{exo}[Vrai ou faux : est-ce une notation scientifique correcte ?]
Pour chaque écriture, dis si c'est une notation scientifique correcte. Si non, corrige-la.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\num{12,5e3}$ \qquad b) $\num{3,2e5}$ \qquad c) $\num{0,4e-2}$ \qquad d) $\num{8,00e0}$
\end{enumerate}
\end{exo}
