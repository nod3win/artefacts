\begin{corrige}[vrai ou faux, justifié]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux.} Une mesure fidèle est reproductible, mais toutes les valeurs peuvent être décalées
        de la même quantité (biais) : fidèle $\neq$ juste.
  \item \textbf{Vrai.} Plus de CS $\Rightarrow$ incertitude relative plus faible ($2$ CS $\sim1\%$,
        $4$ CS $\sim0,01\%$).
  \item \textbf{Faux.} L'incertitude relative est un rapport $\Delta x/x$ : elle est \emph{sans unité}
        (exprimée en \%).
  \item \textbf{Faux.} Les $4$ CS traduisent la \emph{fidélité}, pas la \emph{justesse} : une balance
        biaisée peut afficher $4$ CS et rester fausse.
\end{enumerate}
\end{corrige}

\begin{corrige}[analyse d'une série de titrages]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Fidèle} (les cinq valeurs sont resserrées, écart maximal $\qty{0,04}{\milli\litre}$) et
        \textbf{juste} (leur moyenne coïncide avec la référence).
  \item $\bar V=\dfrac{12,45+12,47+12,44+12,46+12,48}{5}=\dfrac{62,30}{5}=\qty{12,46}{\milli\litre}$.
  \item $\dfrac{\Delta V}{V}=\dfrac{0,01}{12,46}=8,0\times10^{-4}\approx\boxed{0,080\,\%}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[le piège de la pesée différentielle]
\begin{enumerate}[label=\alph*)]
  \item $m=125,78-125,46=\boxed{\qty{0,32}{\gram}}$ ($2$ décimales, imposées par les données).
  \item $\dfrac{\Delta}{m_1}=\dfrac{0,01}{125,46}\approx 0,008\,\%$, mais
        $\dfrac{\Delta}{m}=\dfrac{0,01}{0,32}\approx \boxed{3\,\%}$. La soustraction de deux nombres voisins
        \emph{amplifie} l'incertitude relative (facteur $\approx 400$ ici) : mieux vaut tarer le bécher et
        peser directement le précipité.
\end{enumerate}
\end{corrige}

\begin{corrige}[la même balance, deux échantillons]
\begin{enumerate}[label=\alph*)]
  \item $\approx\qty{2}{\gram}$ : $\dfrac{0,01}{2}=\boxed{0,5\,\%}$. \quad
        $\approx\qty{200}{\gram}$ : $\dfrac{0,01}{200}=\boxed{0,005\,\%}$.
  \item La balance suffit pour la pesée de $\qty{200}{\gram}$ ($0,005\%\le0,1\%$), mais pas pour celle de
        $\qty{2}{\gram}$ ($0,5\%>0,1\%$) : il faudrait alors une balance au milligramme. Une même
        incertitude \emph{absolue} pèse bien plus lourd en \emph{relatif} sur une petite masse.
\end{enumerate}
\end{corrige}
