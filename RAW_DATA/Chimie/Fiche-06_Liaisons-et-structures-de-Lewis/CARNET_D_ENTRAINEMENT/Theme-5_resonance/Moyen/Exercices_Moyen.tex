\begin{exo}[l'ozone \ce{O3}]
\begin{enumerate}[label=\alph*)]
  \item Écris les deux formes limites de l'ozone (connectivité \ce{O-O-O}) et indique où se
        trouvent la charge $+$ et la charge $-$ dans chaque forme.
  \item Que peut-on dire des deux liaisons \ce{O-O} dans la molécule réelle (l'hybride) ?
  \item La géométrie (coudée) change-t-elle d'une forme à l'autre ?
\end{enumerate}
\end{exo}

\begin{exo}[l'ion carbonate \ce{CO3^2-}]
\begin{enumerate}[label=\alph*)]
  \item Combien de formes limites équivalentes peut-on écrire ? Pourquoi ce nombre ?
  \item Comment la charge $-2$ est-elle répartie sur les oxygènes dans l'hybride ?
  \item Les trois liaisons \ce{C-O} sont-elles simples, doubles, ou intermédiaires ?
\end{enumerate}
\end{exo}

\begin{exo}[l'ion nitrate \ce{NO3-}]
\begin{enumerate}[label=\alph*)]
  \item Combien de formes limites équivalentes ?
  \item Répartition de la charge $-1$ sur les trois oxygènes.
  \item Longueur relative des trois liaisons \ce{N-O}.
\end{enumerate}
\end{exo}

\begin{exo}[l'ion acétate \ce{CH3COO-}]
Le groupe carboxylate \ce{-CO2^-} porte une charge $-1$ sur l'un de ses deux oxygènes.
\begin{enumerate}[label=\alph*)]
  \item Écris les deux formes limites (la double liaison \ce{C=O} passe d'un oxygène à l'autre).
  \item Sur quels atomes la charge $-1$ est-elle réellement délocalisée ?
  \item Les deux liaisons \ce{C-O} du carboxylate sont-elles identiques dans l'hybride ?
\end{enumerate}
\end{exo}

\begin{exo}[le dioxyde de soufre \ce{SO2}]
\begin{enumerate}[label=\alph*)]
  \item Combien de formes limites équivalentes possède \ce{SO2} ?
  \item Les deux liaisons \ce{S-O} sont-elles identiques ?
  \item La géométrie coudée est-elle modifiée par la résonance ?
\end{enumerate}
\end{exo}
