\begin{corrige}[l'ozone \ce{O3}]
\begin{enumerate}[label=\alph*)]
  \item \ce{O=O-O <-> O-O=O}. Dans chaque forme, l'oxygène \textbf{central} porte la charge $+1$
        et l'oxygène \textbf{terminal simplement lié} porte la charge $-1$ (l'oxygène doublement
        lié est neutre).
  \item Les deux liaisons \ce{O-O} sont \textbf{identiques}, de longueur intermédiaire entre une
        simple et une double : la double liaison est délocalisée sur les deux.
  \item \textbf{Non} : la géométrie coudée est la même dans les deux formes (seuls les électrons
        se déplacent).
\end{enumerate}
\end{corrige}

\begin{corrige}[l'ion carbonate \ce{CO3^2-}]
\begin{enumerate}[label=\alph*)]
  \item \textbf{3} formes équivalentes : la double liaison \ce{C=O} peut se placer sur \emph{chacun}
        des trois oxygènes.
  \item La charge $-2$ est répartie également sur les trois \ce{O}, soit $-\tfrac{2}{3}$ chacun.
  \item \textbf{Intermédiaires} : les trois liaisons \ce{C-O} sont identiques, entre simple et double.
\end{enumerate}
\end{corrige}

\begin{corrige}[l'ion nitrate \ce{NO3-}]
\begin{enumerate}[label=\alph*)]
  \item \textbf{3} formes équivalentes (double \ce{N=O} sur chaque oxygène).
  \item Charge $-1$ répartie sur les trois \ce{O}, soit $-\tfrac{1}{3}$ chacun.
  \item Les trois liaisons \ce{N-O} sont \textbf{identiques}, de longueur intermédiaire.
\end{enumerate}
\end{corrige}

\begin{corrige}[l'ion acétate \ce{CH3COO-}]
\begin{enumerate}[label=\alph*)]
  \item \ce{CH3-CO2^-} : \ce{CH3-C(=O)-O^- <-> CH3-C(-O^-)=O}. La double liaison \ce{C=O} passe
        d'un oxygène à l'autre.
  \item La charge $-1$ est délocalisée \textbf{sur les deux oxygènes} du carboxylate
        ($-\tfrac{1}{2}$ chacun).
  \item \textbf{Oui} : les deux liaisons \ce{C-O} deviennent identiques (longueur intermédiaire) —
        c'est la grande stabilité de l'ion carboxylate.
\end{enumerate}
\end{corrige}

\begin{corrige}[le dioxyde de soufre \ce{SO2}]
\begin{enumerate}[label=\alph*)]
  \item \textbf{2} formes équivalentes : \ce{O=S-O <-> O-S=O}.
  \item \textbf{Oui}, les deux liaisons \ce{S-O} sont identiques (délocalisation de la double).
  \item \textbf{Non} : la molécule reste coudée ; la résonance ne change pas la géométrie.
\end{enumerate}
\end{corrige}
