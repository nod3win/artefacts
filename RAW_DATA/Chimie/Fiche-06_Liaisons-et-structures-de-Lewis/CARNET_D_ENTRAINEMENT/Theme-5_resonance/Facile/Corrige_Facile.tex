{\Large\bfseries\color{primarydark}Thème 5 — Corrigé du niveau Facile}
\end{center}
\medskip

\begin{cor}[le vocabulaire de la résonance]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item \rep une \textbf{structure limite} est l'une des formules de Lewis que l'on peut écrire pour
        un même édifice, obtenue en déplaçant \emph{uniquement} des électrons (jamais des atomes).
  \item \rep l'\textbf{hybride de résonance} est la molécule \emph{réelle}, unique, qui n'est aucune
        des structures limites mais une « moyenne » de toutes.
  \item \rep le symbole $\longleftrightarrow$ relie les structures limites d'\emph{une même} espèce ;
        il indique qu'aucune formule seule ne décrit correctement l'édifice.
\end{enumerate}
\end{cor}

\begin{cor}[vrai ou faux ?]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item \rep \textbf{FAUX}. D'une forme à l'autre, \emph{seuls les électrons} changent de place ; les
        noyaux (donc les atomes) restent immobiles.
  \item \rep \textbf{FAUX}. La géométrie ne change \emph{pas} : c'est une propriété de la molécule
        réelle (l'hybride), identique dans toutes les formes limites.
  \item \rep \textbf{FAUX}. $\longleftrightarrow$ est la flèche de \emph{résonance}. L'équilibre
        chimique se note $\rightleftharpoons$ (deux espèces distinctes qui s'interconvertissent).
\end{enumerate}
\end{cor}

\begin{cor}[compter les formes limites]
Le nombre de formes limites équivalentes égale le nombre de positions équivalentes que peut occuper
la double liaison (ou la charge) :
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
\toprule
Édifice & \ce{O3} & \ce{CO3^2-} & \ce{NO3-} & \ce{ClO4-}\\
Nombre de formes & 2 & 3 & 3 & 4\\
\bottomrule
\end{tabular}
\end{center}
\rep \ce{O3} : \textbf{2} ; \ce{CO3^2-} : \textbf{3} ; \ce{NO3-} : \textbf{3} ; \ce{ClO4-} : \textbf{4}.
\end{cor}

\begin{cor}[$\longleftrightarrow$ ou $\rightleftharpoons$ ?]
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item \rep description 1 (formes de résonance d'une seule espèce) $\to$ \textbf{$\longleftrightarrow$} ;
  \item \rep description 2 (équilibre entre deux espèces) $\to$ \textbf{$\rightleftharpoons$}.
\end{itemize}
En résonance il n'existe qu'\emph{une} espèce ; dans un équilibre, il en existe \emph{deux} bien
réelles qui se transforment l'une en l'autre.
\end{cor}

\begin{cor}[qui présente de la résonance ?]
Une espèce présente de la résonance dès qu'une double liaison peut occuper \emph{plusieurs} positions
équivalentes.
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item \rep \textbf{Résonance} : \ce{CO3^2-}, \ce{NO3-}, \ce{O3}.
  \item \rep \textbf{Pas de résonance} : \ce{CH4} (que des liaisons simples \ce{C-H}) et \ce{H2O}
        (que des liaisons simples \ce{O-H}, aucune liaison multiple à délocaliser).
\end{itemize}
\end{cor}

\begin{astucebox}[le réflexe résonance]
Deux idées suffisent au niveau facile : \textbf{(1)}~seuls les \emph{électrons} bougent (jamais les
atomes ni la géométrie) ; \textbf{(2)}~$\longleftrightarrow$ (résonance, \emph{une} espèce) ne se
confond jamais avec $\rightleftharpoons$ (équilibre, \emph{deux} espèces).
\end{astucebox}
