\begin{exo}[le vocabulaire de la résonance]
Définir en \textbf{une phrase} chacune des expressions suivantes :
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=1pt]
  \item une \textbf{structure limite} (ou forme de résonance) ;
  \item l'\textbf{hybride de résonance} ;
  \item le symbole $\longleftrightarrow$ placé entre deux structures limites.
\end{enumerate}
\end{exo}

\begin{exo}[vrai ou faux ?]
Indiquer si chaque affirmation est \textbf{vraie} ou \textbf{fausse}, et corriger si elle est fausse.
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=1pt]
  \item « D'une forme de résonance à l'autre, les \emph{atomes} changent de place. »
  \item « La \emph{géométrie} de la molécule change avec la résonance. »
  \item « Le symbole $\longleftrightarrow$ désigne un équilibre chimique. »
\end{enumerate}
\end{exo}

\begin{exo}[compter les formes limites]
Pour chacun des édifices suivants, donner le \textbf{nombre de structures limites équivalentes} :
\[ \ce{O3}, \qquad \ce{CO3^2-}, \qquad \ce{NO3-}, \qquad \ce{ClO4-}. \]
\end{exo}

\begin{exo}[$\longleftrightarrow$ ou $\rightleftharpoons$ ?]
On dispose de deux symboles : la double flèche de résonance $\longleftrightarrow$ et la double flèche
d'équilibre $\rightleftharpoons$. Associer chacun à la bonne description :
\begin{enumerate}[label=\arabic*.,leftmargin=2em,itemsep=1pt]
  \item les formes de résonance d'\textbf{une seule} espèce ;
  \item un \textbf{équilibre} entre \textbf{deux} espèces qui s'interconvertissent.
\end{enumerate}
\end{exo}

\begin{exo}[qui présente de la résonance ?]
Parmi les espèces suivantes, indiquer lesquelles présentent de la \textbf{résonance} et lesquelles
\textbf{non} :
\[ \ce{CH4}, \qquad \ce{CO3^2-}, \qquad \ce{H2O}, \qquad \ce{NO3-}, \qquad \ce{O3}. \]
\end{exo}
