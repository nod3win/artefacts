\newcommand{\AX}[2]{\ensuremath{\mathrm{AX_{#1}E_{#2}}}}
\begin{corrige}[la résonance de \ce{ClO4-}]
\begin{enumerate}[label=\alph*)]
  \item \textbf{4} formes limites équivalentes : la charge $-1$ (et la simple liaison) peut se
        placer sur chacun des quatre oxygènes.
  \item La charge $-1$ est délocalisée également sur les quatre \ce{O}, soit $-\tfrac{1}{4}$ chacun.
  \item Les quatre liaisons \ce{Cl-O} réelles sont \textbf{identiques} (longueur intermédiaire).
        La géométrie \textbf{tétraédrique} (\AX{4}{0}) n'est \emph{pas} affectée par la résonance.
\end{enumerate}
\end{corrige}

\begin{corrige}[le benzène \ce{C6H6}]
\begin{enumerate}[label=\alph*)]
  \item \textbf{2} formes de Kekulé (les doubles liaisons alternées peuvent occuper deux jeux de
        positions).
  \item Les six liaisons \ce{C-C} sont \textbf{identiques}, de longueur intermédiaire entre une
        simple et une double : aucune n'est « vraiment » simple ou double.
  \item Les électrons $\pi$ sont \textbf{délocalisés} sur tout le cycle (un « nuage » $\pi$ commun
        aux six carbones), ce qui confère au benzène sa grande stabilité.
\end{enumerate}
\end{corrige}

\begin{corrige}[des formes limites qui ne se valent pas]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Non} : seules les formes \emph{équivalentes} contribuent à parts égales. Ici (I)
        et (II) ont des énergies différentes.
  \item La forme \textbf{(I)} pèse le plus : selon les critères de stabilité (charges formelles
        \emph{minimales}, octet respecté), la forme la plus stable contribue davantage à l'hybride.
        La forme (II), avec sa séparation de charges, est minoritaire.
\end{enumerate}
\end{corrige}

\begin{corrige}[corriger une idée fausse]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=2pt]
  \item \textbf{Il n'y a qu'une seule espèce.} La flèche $\leftrightarrow$ est une flèche de
        \emph{résonance}, pas d'équilibre ($\rightleftharpoons$) : l'ozone réel est un
        \textbf{hybride} unique, pas un mélange de deux molécules qui s'interconvertissent.
  \item \textbf{Les noyaux ne bougent pas.} Seuls les électrons sont délocalisés : les deux
        liaisons \ce{O-O} sont \emph{en permanence} identiques (intermédiaires), et non
        alternativement simple puis double.
\end{itemize}
\end{corrige}

\begin{corrige}[synthèse : l'ion formiate \ce{HCO2-}]
\begin{enumerate}[label=\alph*)]
  \item $n_v = \underbrace{1}_{\ce{H}} + \underbrace{4}_{\ce{C}} + \underbrace{2\times 6}_{2\,\ce{O}}
        + \underbrace{1}_{-q} = 18$. Avec $n_{\max} = 8\times 3 + 2 = 26$, on a $n_l = 26-18 = 8
        \Rightarrow 4$ liaisons : les $3$ liaisons simples \ce{C-H} et \ce{C-O} ($\times 2$) plus
        \textbf{une double} \ce{C=O}. \checkmark
  \item \ce{H-C(=O)-O^- <-> H-C(-O^-)=O} : la charge $-1$ est délocalisée sur \textbf{les deux
        oxygènes} ($-\tfrac{1}{2}$ chacun) ; les deux liaisons \ce{C-O} sont identiques.
  \item Le carbone est lié à $3$ atomes ($1$ \ce{H}, $2$ \ce{O}) et n'a aucun doublet libre :
        \AX{3}{0}, géométrie \textbf{trigonale plane} ($120^\circ$). La résonance \textbf{ne la
        modifie pas} (seuls les électrons se déplacent).
\end{enumerate}
\end{corrige}
