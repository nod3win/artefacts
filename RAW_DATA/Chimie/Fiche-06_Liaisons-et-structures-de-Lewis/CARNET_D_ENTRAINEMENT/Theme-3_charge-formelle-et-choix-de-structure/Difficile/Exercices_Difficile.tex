\begin{exo}[ion perchlorate \ce{ClO4-} : deux structures concurrentes]
On compare deux structures de Lewis de l'ion perchlorate (atome central \ce{Cl}, quatre \ce{O}
autour). Rappel des valences : \ce{Cl}$=7$, \ce{O}$=6$.

\begin{center}
\begin{tikzpicture}[scale=0.85,every node/.style={font=\small\bfseries}]
  % ---- Structure A : 4 liaisons simples ----
  \begin{scope}
    \node (ClA) at (0,0) {Cl};
    \node (OA1) at (0,1.5) {O}; \node (OA2) at (0,-1.5) {O};
    \node (OA3) at (-1.5,0) {O}; \node (OA4) at (1.5,0) {O};
    \draw[bondcol,line width=1.1pt] (ClA)--(OA1);
    \draw[bondcol,line width=1.1pt] (ClA)--(OA2);
    \draw[bondcol,line width=1.1pt] (ClA)--(OA3);
    \draw[bondcol,line width=1.1pt] (ClA)--(OA4);
    \node[font=\footnotesize\itshape] at (0,-2.2) {Structure A : 4 simples liaisons};
  \end{scope}
  % ---- Structure B : 3 doubles + 1 simple ----
  \begin{scope}[xshift=6cm]
    \node (ClB) at (0,0) {Cl};
    \node (OB1) at (0,1.5) {O}; \node (OB2) at (0,-1.5) {O};
    \node (OB3) at (-1.5,0) {O}; \node (OB4) at (1.5,0) {O};
    % double bonds vers OB1, OB3, OB4 ; simple vers OB2
    \draw[bondcol,line width=1.1pt] ($(ClB)+(-0.07,0.25)$)--($(OB1)+(-0.07,-0.25)$);
    \draw[bondcol,line width=1.1pt] ($(ClB)+(0.07,0.25)$)--($(OB1)+(0.07,-0.25)$);
    \draw[bondcol,line width=1.1pt] ($(ClB)+(-0.25,-0.07)$)--($(OB3)+(0.25,-0.07)$);
    \draw[bondcol,line width=1.1pt] ($(ClB)+(-0.25,0.07)$)--($(OB3)+(0.25,0.07)$);
    \draw[bondcol,line width=1.1pt] ($(ClB)+(0.25,-0.07)$)--($(OB4)+(-0.25,-0.07)$);
    \draw[bondcol,line width=1.1pt] ($(ClB)+(0.25,0.07)$)--($(OB4)+(-0.25,0.07)$);
    \draw[bondcol,line width=1.1pt] (ClB)--(OB2);
    \node[font=\footnotesize\itshape] at (0,-2.2) {Structure B : 3 doubles + 1 simple};
  \end{scope}
\end{tikzpicture}
\end{center}

\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item \textbf{Structure A.} Le chlore fait $4$ liaisons \ce{Cl-O} simples et ne porte aucun
        doublet libre ; chaque \ce{O} porte $3$ doublets libres. Calculer $q_{\mathrm{f}}(\ce{Cl})$
        et $q_{\mathrm{f}}$ de chaque \ce{O}, puis vérifier que la somme vaut $-1$.
  \item \textbf{Structure B.} Le chlore fait un \textbf{octet étendu} : $3$ doubles liaisons
        $+$ $1$ simple, toujours sans doublet libre. Les $3$ \ce{O} doublement liés portent $2$
        doublets libres ; l'\ce{O} simplement lié en porte $3$. Calculer $q_{\mathrm{f}}(\ce{Cl})$,
        $q_{\mathrm{f}}$ d'un \ce{O} double et de l'\ce{O} simple, puis vérifier que la somme vaut $-1$.
  \item Laquelle des deux structures est la plus représentative ? Énoncer précisément le critère.
  \item Pourquoi la structure B n'est-elle \emph{possible} que parce que le chlore est en
        3\textsuperscript{e} période ?
\end{enumerate}
\end{exo}

\begin{exo}[choisir la bonne structure : l'acide nitrique \ce{HNO3}]
On veut choisir la structure de Lewis correcte de l'acide nitrique. On rappelle les
\textbf{trois critères de choix} d'une structure de Lewis.
\begin{enumerate}[label=\arabic*.,leftmargin=2em,itemsep=1pt]
  \item l'octet doit être respecté par les atomes de 1\textsuperscript{re} et 2\textsuperscript{e} période ;
  \item les charges (formelles) doivent être réparties selon l'électronégativité ;
  \item les charges formelles doivent être minimales (proches de $0$) et, à valeur égale, éloignées les unes des autres.
\end{enumerate}

Le squelette correct est \ce{H-O-N(=O)-O} : un azote \textbf{central} lié à trois oxygènes — une
\textbf{double} liaison \ce{N=O}, et deux \textbf{simples} \ce{N-O}, dont l'une porte le \ce{H}
(groupe \ce{O-H}).
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item Combien de liaisons l'azote forme-t-il dans ce squelette ? En déduire de combien d'électrons
        il s'entoure et vérifier que son octet est \emph{exactement} respecté.
  \item Une structure dans laquelle l'azote formerait $5$ liaisons (par exemple deux doubles et une
        simple) est \textbf{à rejeter}. Pourquoi cet octet étendu est-il impossible pour l'azote ?
  \item Montrer que, dans la bonne structure, l'azote porte $q_{\mathrm{f}} = +1$ et l'oxygène
        simplement lié \emph{sans} \ce{H} porte $q_{\mathrm{f}} = -1$. Vérifier que la somme des
        charges formelles vaut $0$ (molécule neutre). \emph{On donne : l'\ce{O} du groupe \ce{O-H}
        a $q_{\mathrm{f}} = 0$, l'\ce{O} doublement lié a $q_{\mathrm{f}} = 0$, chaque \ce{H} a
        $q_{\mathrm{f}} = 0$.}
\end{enumerate}
\end{exo}

\begin{exo}[ion nitrate \ce{NO3^-} : pourquoi l'azote garde-t-il sa charge ?]
L'ion nitrate a pour structure un azote \textbf{central} lié à trois oxygènes : une \textbf{double}
liaison \ce{N=O} et deux \textbf{simples} \ce{N-O^-}. L'\ce{O} doublement lié porte $2$ doublets
libres ; chaque \ce{O} simplement lié en porte $3$.
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item Rappeler la structure et le nombre total de liaisons formées par l'azote.
  \item Calculer $q_{\mathrm{f}}(\ce{N})$, $q_{\mathrm{f}}$ de l'\ce{O} double et de chaque \ce{O}
        simple. Vérifier que la somme vaut $-1$.
  \item Expliquer pourquoi cette structure est « bonne » alors même que l'azote garde une charge
        $+1$, en comparant sa situation à celle du chlore de \ce{ClO4-}.
\end{enumerate}
\end{exo}
