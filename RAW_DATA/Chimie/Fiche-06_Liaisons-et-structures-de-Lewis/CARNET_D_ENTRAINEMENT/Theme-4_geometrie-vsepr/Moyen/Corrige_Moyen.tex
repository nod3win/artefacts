\newcommand{\AX}[2]{\ensuremath{\mathrm{AX_{#1}E_{#2}}}}
\begin{corrige}[l'eau \ce{H2O}]
\begin{enumerate}[label=\alph*)]
  \item $n=2$, $m=2 \Rightarrow$ \AX{2}{2}.
  \item $n+m = 4$ : géométrie de répulsion \textbf{tétraédrique} ; géométrie moléculaire
        \textbf{coudée} (on masque les $2$ doublets libres).
  \item $\ce{H-O-H} \approx \mathbf{104{,}5^\circ}$ (les $2$ doublets libres compriment l'angle).
\end{enumerate}
\end{corrige}

\begin{corrige}[l'ammoniac \ce{NH3}]
\begin{enumerate}[label=\alph*)]
  \item $n=3$, $m=1 \Rightarrow$ \AX{3}{1}.
  \item $n+m=4$ : répulsion \textbf{tétraédrique} ; géométrie moléculaire \textbf{pyramidale}
        (à base triangulaire).
  \item $\ce{H-N-H} \approx \mathbf{107^\circ}$ : \textbf{plus grand} que dans \ce{H2O}, car
        \ce{NH3} n'a qu'\emph{un} doublet libre (au lieu de $2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[le dioxyde de carbone \ce{CO2}]
\begin{enumerate}[label=\alph*)]
  \item Deux doubles liaisons $= 2$ sites, $m=0 \Rightarrow$ \AX{2}{0}.
  \item Répulsion \textbf{linéaire} ; géométrie moléculaire \textbf{linéaire} (identiques, car
        $m=0$).
  \item $\ce{O-C-O} = \mathbf{180^\circ}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[le dioxyde de soufre \ce{SO2}]
\begin{enumerate}[label=\alph*)]
  \item $n=2$, $m=1 \Rightarrow$ \AX{2}{1}.
  \item $n+m=3$ : répulsion \textbf{triangulaire} ; géométrie moléculaire \textbf{coudée}.
  \item Non : l'angle est \textbf{légèrement inférieur} à $120^\circ$ ($\approx 119^\circ$), car le
        doublet libre du soufre repousse plus fort que les liaisons et referme un peu l'angle.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire une géométrie]
\begin{enumerate}[label=\alph*)]
  \item Le bore ne porte \textbf{aucun} doublet libre ($m=0$), il est lié à $3$ \ce{F}
        $\Rightarrow$ \AX{3}{0}.
  \item Répulsion \textbf{triangulaire} ; géométrie moléculaire \textbf{trigonale plane}
        (angles $120^\circ$).
  \item Non : le bore ne s'entoure que de $6$~e$^-$ ($3$ liaisons) — c'est une \textbf{lacune
        électronique}, exception à l'octet.
\end{enumerate}
\end{corrige}
