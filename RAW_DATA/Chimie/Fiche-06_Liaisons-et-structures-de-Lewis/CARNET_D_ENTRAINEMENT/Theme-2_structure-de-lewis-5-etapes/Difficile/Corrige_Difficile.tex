\begin{corrige}[l'ion perchlorate \ce{ClO4-}]
Atome central \ce{Cl} ; $a=5$ (\ce{Cl} et $4$ \ce{O}), $b=0$. Charge $q=-1$.
\[ n_{\max} = 8\times 5 = 40,\qquad n_v = 7 + 4\times 6 + 1 = 32. \]
\[ n_l = 40 - 32 = 8 \Rightarrow 4\ \text{liaisons \ce{Cl-O}},\qquad
   n_{nl} = 32 - 8 = 24 \Rightarrow 12\ \text{doublets libres}. \]
Les $12$ doublets libres se répartissent \textbf{sur les quatre oxygènes} ($3$ chacun) ; le
chlore central n'en porte aucun.
\end{corrige}

\begin{corrige}[le nitrite d'hydrogène \ce{HNO2}]
Atome central \ce{N} ; $a=3$ (\ce{N} et $2$ \ce{O}), $b=1$ (le \ce{H}).
\begin{enumerate}[label=\alph*)]
  \item $n_{\max} = 8\times 3 + 2\times 1 = 26$ ; \; $n_v = 1 + 5 + 2\times 6 = 18$.
  \item $n_l = 26 - 18 = 8 \Rightarrow 4$ liaisons ; \; $n_{nl} = 18 - 8 = 10 \Rightarrow 5$
        doublets libres.
  \item Structure \ce{H-O-N=O} : les $4$ liaisons sont \ce{O-H}, \ce{O-N} (simple) et \ce{N=O}
        (double). Doublets libres : \textbf{$1$ sur l'azote}, \textbf{$2$ sur l'oxygène de
        \ce{-O-H}} et \textbf{$2$ sur l'oxygène de \ce{N=O}} ($1+2+2 = 5$). \checkmark
\end{enumerate}
\end{corrige}

\begin{corrige}[l'ion sulfate \ce{SO4^2-}]
Atome central \ce{S} ; $a=5$, $b=0$. Charge $q=-2$.
\[ n_{\max} = 8\times 5 = 40,\qquad n_v = 6 + 4\times 6 + 2 = 32. \]
\[ n_l = 40 - 32 = 8 \Rightarrow 4\ \text{liaisons \ce{S-O}},\qquad
   n_{nl} = 32 - 8 = 24 \Rightarrow 12\ \text{doublets libres} \ (3\ \text{par \ce{O}}). \]
\textbf{Remarque :} on obtient \emph{exactement} les mêmes nombres que pour \ce{ClO4-}
($n_{\max}=40$, $n_v=32$, $4$ liaisons, $12$ doublets). C'est normal : les deux ions ont le même
nombre d'atomes ($1+4$) et le même $n_v$ (l'électron de valence supplémentaire de \ce{Cl} par
rapport à \ce{S} compense une unité de charge en plus).
\end{corrige}

\begin{corrige}[une molécule radicalaire : \ce{NO}]
\begin{enumerate}[label=\alph*)]
  \item $n_{\max} = 8\times 2 = 16$ ; \; $n_v = 5 + 6 = 11$. Comme $n_v = 11$ est \textbf{impair},
        la molécule est radicalaire : on applique la correction.
  \item $n_l = n_{\max} - n_v - 1 = 16 - 11 - 1 = 4 \Rightarrow \mathbf{2}$ liaisons : une
        \textbf{double liaison} \ce{N=O}.
  \item $n_{nl} = n_v - n_l = 11 - 4 = 7$ électrons non liants $= 3$ doublets libres
        \textbf{$+$ $1$ électron célibataire}. On les place : $2$ doublets sur \ce{O}, $1$ doublet
        $+$ l'électron célibataire sur \ce{N}. C'est ce célibataire qui fait de \ce{NO} un radical.
\end{enumerate}
\end{corrige}

\begin{corrige}[quand la méthode atteint sa limite : \ce{SF6}]
Atome central \ce{S} ; $a=7$ (\ce{S} et $6$ \ce{F}), $b=0$.
\begin{enumerate}[label=\alph*)]
  \item $n_{\max} = 8\times 7 = 56$ ; \; $n_v = 6 + 6\times 7 = 48$ ; \;
        $n_l = 56 - 48 = 8 \Rightarrow$ la méthode prédit seulement $\mathbf{4}$ liaisons.
  \item La méthode simplifiée suppose que \emph{tous} les atomes se limitent à l'octet ($8a$).
        Or le soufre est en \textbf{3\textsuperscript{e} période} : il dispose d'orbitales $d$ et
        peut porter un \textbf{octet étendu} ($12$~e$^-$). Il forme donc réellement $6$ liaisons
        \ce{S-F} — chose \emph{permise} justement parce que \ce{S} peut dépasser l'octet, mais que
        la formule $n_{\max}=8a$ ne sait pas anticiper. C'est la limite annoncée de la méthode.
\end{enumerate}
\end{corrige}
