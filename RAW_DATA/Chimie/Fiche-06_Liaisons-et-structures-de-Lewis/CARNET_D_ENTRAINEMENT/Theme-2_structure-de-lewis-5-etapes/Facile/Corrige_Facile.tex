\begin{corrige}[calculer $n_{\max}$]
$n_{\max} = 8a + 2b$, avec $a$ = atomes visant l'octet et $b$ = nombre de \ce{H}.
\begin{enumerate}[label=\alph*)]
  \item \ce{CH4} : $a=1,\ b=4 \Rightarrow 8 + 8 = \mathbf{16}$.
  \item \ce{H2O} : $a=1,\ b=2 \Rightarrow 8 + 4 = \mathbf{12}$.
  \item \ce{NH3} : $a=1,\ b=3 \Rightarrow 8 + 6 = \mathbf{14}$.
  \item \ce{CO2} : $a=3,\ b=0 \Rightarrow 24 + 0 = \mathbf{24}$.
  \item \ce{HCN} : $a=2,\ b=1 \Rightarrow 16 + 2 = \mathbf{18}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer $n_{v}$ (molécules neutres)]
On somme les électrons de valence de tous les atomes.
\begin{enumerate}[label=\alph*)]
  \item \ce{CH4} : $4 + 4\times 1 = \mathbf{8}$.
  \item \ce{H2O} : $6 + 2\times 1 = \mathbf{8}$.
  \item \ce{NH3} : $5 + 3\times 1 = \mathbf{8}$.
  \item \ce{CO2} : $4 + 2\times 6 = \mathbf{16}$.
  \item \ce{HCN} : $1 + 4 + 5 = \mathbf{10}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer $n_{v}$ (ions)]
$n_{v} = \sum N^{\text{val}} - q$ : on \emph{ajoute} $|q|$ pour un anion, on \emph{retranche} $q$
pour un cation.
\begin{enumerate}[label=\alph*)]
  \item \ce{OH-} : $6 + 1 + 1 = \mathbf{8}$ (anion : $-q = +1$).
  \item \ce{NH4+} : $5 + 4 - 1 = \mathbf{8}$ (cation : $-1$).
  \item \ce{H3O+} : $6 + 3 - 1 = \mathbf{8}$.
  \item \ce{NO3-} : $5 + 3\times 6 + 1 = \mathbf{24}$.
  \item \ce{CO3^2-} : $4 + 3\times 6 + 2 = \mathbf{24}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[de $n_{\max}$ et $n_v$ au nombre de liaisons]
$n_l = n_{\max} - n_v$, puis nombre de liaisons $= n_l/2$.
\begin{enumerate}[label=\alph*)]
  \item \ce{CH4} : $n_l = 16-8 = 8 \Rightarrow \mathbf{4}$ liaisons.
  \item \ce{H2O} : $n_l = 12-8 = 4 \Rightarrow \mathbf{2}$ liaisons.
  \item \ce{NH3} : $n_l = 14-8 = 6 \Rightarrow \mathbf{3}$ liaisons.
  \item \ce{CO2} : $n_l = 24-16 = 8 \Rightarrow \mathbf{4}$ liaisons (deux doubles \ce{O=C=O}).
  \item \ce{HCN} : $n_l = 18-10 = 8 \Rightarrow \mathbf{4}$ liaisons (\ce{H-C#N}).
\end{enumerate}
\end{corrige}

\begin{corrige}[du reste aux doublets libres]
$n_{nl} = n_v - n_l$, puis nombre de doublets libres $= n_{nl}/2$.
\begin{enumerate}[label=\alph*)]
  \item \ce{CH4} : $n_{nl} = 8-8 = 0 \Rightarrow \mathbf{0}$ doublet libre.
  \item \ce{H2O} : $n_{nl} = 8-4 = 4 \Rightarrow \mathbf{2}$ doublets libres (sur \ce{O}).
  \item \ce{NH3} : $n_{nl} = 8-6 = 2 \Rightarrow \mathbf{1}$ doublet libre (sur \ce{N}).
  \item \ce{CO2} : $n_{nl} = 16-8 = 8 \Rightarrow \mathbf{4}$ doublets libres ($2$ par \ce{O}).
  \item \ce{HCN} : $n_{nl} = 10-8 = 2 \Rightarrow \mathbf{1}$ doublet libre (sur \ce{N}).
\end{enumerate}
\end{corrige}
