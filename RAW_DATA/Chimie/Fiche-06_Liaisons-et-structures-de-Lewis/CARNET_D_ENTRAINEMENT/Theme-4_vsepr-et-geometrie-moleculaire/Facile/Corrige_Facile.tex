\begin{cor}[écrire \AX{n}{m} et la géométrie de répulsion]
Le total de sites est $n+m$ ; il fixe la géométrie de répulsion.
\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{c|c c c c}
\toprule
Cas & $(n,m)$ & Type & $n+m$ & Géométrie de répulsion\\
\midrule
a) & $(2,0)$ & \AX{2}{0} & 2 & linéaire\\
b) & $(3,0)$ & \AX{3}{0} & 3 & triangulaire\\
c) & $(4,0)$ & \AX{4}{0} & 4 & tétraédrique\\
d) & $(2,2)$ & \AX{2}{2} & 4 & tétraédrique\\
\bottomrule
\end{tabular}
\end{center}
\rep a) \AX{2}{0}, 2 sites, \textbf{linéaire} ; b) \AX{3}{0}, 3 sites, \textbf{triangulaire} ;
c) \AX{4}{0}, 4 sites, \textbf{tétraédrique} ; d) \AX{2}{2}, 4 sites, \textbf{tétraédrique}
(attention : ce sont bien 4 sites, car les 2 doublets libres comptent).
\end{cor}

\begin{cor}[lire la géométrie moléculaire dans la table]
On masque les doublets libres et on lit la forme des atomes seuls :
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
  \item \AX{2}{0} $\to$ \rep\textbf{linéaire} ;
  \item \AX{3}{0} $\to$ \rep\textbf{triangulaire plane} ;
  \item \AX{4}{0} $\to$ \rep\textbf{tétraédrique} ;
  \item \AX{3}{1} $\to$ \rep\textbf{pyramide trigonale} ;
  \item \AX{2}{2} $\to$ \rep\textbf{coudée (angulaire)}.
\end{itemize}
\end{cor}

\begin{cor}[compter \texorpdfstring{$n$}{n} et \texorpdfstring{$m$}{m} autour de l'atome central]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item \ce{CH4} : \rep $n=4$, $m=0$ (type \AX{4}{0}).
  \item \ce{NH3} : \rep $n=3$, $m=1$ (type \AX{3}{1}).
  \item \ce{H2O} : \rep $n=2$, $m=2$ (type \AX{2}{2}).
  \item \ce{CO2} : \rep $n=2$, $m=0$ (type \AX{2}{0}). Chaque \emph{double} liaison \ce{C=O}
        compte pour \textbf{un seul} site : le carbone a donc 2 sites, pas 4.
\end{enumerate}
\end{cor}

\begin{cor}[associer chaque géométrie à son angle idéal]
\rep \textbf{1--B} (linéaire $\leftrightarrow 180^\circ$), \textbf{2--C} (triangulaire plane
$\leftrightarrow 120^\circ$), \textbf{3--A} (tétraédrique $\leftrightarrow 109{,}5^\circ$).
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c c c}
\toprule
Géométrie & Angle idéal & Exemple\\
\midrule
linéaire & $180^\circ$ & \ce{CO2}\\
triangulaire plane & $120^\circ$ & \ce{BH3}\\
tétraédrique & $109{,}5^\circ$ & \ce{CH4}\\
\bottomrule
\end{tabular}
\end{center}
\end{cor}

\begin{cor}[vrai ou faux ?]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=2pt]
  \item \rep \textbf{FAUX}. Une liaison multiple (double ou triple) compte pour \textbf{un seul}
        site de répulsion : c'est sa direction qui compte, pas le nombre de doublets. (Ex. \ce{CO2}
        est \AX{2}{0}, linéaire.)
  \item \rep \textbf{VRAI}. Si $m=0$, il n'y a aucun doublet libre à masquer : la géométrie
        moléculaire coïncide exactement avec la géométrie de répulsion (ex. \ce{CH4}, \ce{SF6}).
\end{enumerate}
\end{cor}

\begin{astucebox}[le réflexe \AX{n}{m}]
Toujours dans l'ordre : (1) compter $n$ et $m$, (2) faire la somme $n+m$ pour la figure de
répulsion, (3) masquer les $m$ doublets pour la forme finale. Une liaison multiple = un site.
\end{astucebox}
