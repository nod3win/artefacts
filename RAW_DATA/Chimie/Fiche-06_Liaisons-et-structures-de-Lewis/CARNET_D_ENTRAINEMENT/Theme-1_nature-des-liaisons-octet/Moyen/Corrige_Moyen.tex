\begin{corrige}[repérer la liaison dative]
Le proton \ce{H+} n'a aucun électron (case vide) : c'est toujours l'\textbf{accepteur}. L'atome
qui possède un doublet non liant est le \textbf{donneur}.
\begin{enumerate}[label=\alph*)]
  \item \ce{NH4+} : une des quatre liaisons \ce{N-H} est dative. Donneur $=$ \textbf{azote}
        (il apporte son doublet non liant) ; accepteur $=$ \textbf{\ce{H+}}. Les quatre liaisons
        deviennent ensuite identiques.
  \item \ce{H3O+} : une des trois liaisons \ce{O-H} est dative. Donneur $=$ \textbf{oxygène}
        (un de ses doublets libres) ; accepteur $=$ \textbf{\ce{H+}}.
\end{enumerate}
\end{corrige}

\begin{corrige}[nombre total d'électrons de valence]
On additionne les électrons de valence de tous les atomes, puis on retranche la charge $q$
(on ajoute $|q|$ pour un anion, on retranche $q$ pour un cation).
\begin{enumerate}[label=\alph*)]
  \item \ce{CO2} : $4 + 2\times 6 = \mathbf{16}$.
  \item \ce{NH4+} : $5 + 4\times 1 - 1 = \mathbf{8}$ (cation : $-1$).
  \item \ce{SO4^2-} : $6 + 4\times 6 + 2 = \mathbf{32}$ (anion : $+2$).
  \item \ce{CH4} : $4 + 4\times 1 = \mathbf{8}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[octet, lacune ou octet étendu ?]
On compte les électrons autour de l'atome central ($2$ par liaison).
\begin{enumerate}[label=\alph*)]
  \item \ce{C} dans \ce{CH4} : $4$ liaisons $= 8$~e$^-$ $\Rightarrow$ \textbf{octet}.
  \item \ce{B} dans \ce{BF3} : $3$ liaisons $= 6$~e$^-$ $\Rightarrow$ \textbf{lacune} (sous-octet).
  \item \ce{S} dans \ce{SF6} : $6$ liaisons $= 12$~e$^-$ $\Rightarrow$ \textbf{octet étendu}.
  \item \ce{P} dans \ce{PCl5} : $5$ liaisons $= 10$~e$^-$ $\Rightarrow$ \textbf{octet étendu}.
  \item \ce{Be} dans \ce{BeCl2} : $2$ liaisons $= 4$~e$^-$ $\Rightarrow$ \textbf{lacune}.
\end{enumerate}
\end{corrige}

\begin{corrige}[qui peut dépasser l'octet ?]
Seuls les éléments de la \textbf{3\textsuperscript{e} période et au-delà} (orbitales $d$
accessibles) peuvent dépasser l'octet.
\begin{enumerate}[label=\alph*)]
  \item \ce{N} : \textbf{non} ($2^{\text{e}}$ période, strictement $8$~e$^-$).
  \item \ce{S} : \textbf{oui} ($3^{\text{e}}$ période).
  \item \ce{O} : \textbf{non} ($2^{\text{e}}$ période).
  \item \ce{Cl} : \textbf{oui} ($3^{\text{e}}$ période).
  \item \ce{C} : \textbf{non} ($2^{\text{e}}$ période).
\end{enumerate}
\end{corrige}

\begin{corrige}[électronégativité et caractère ionique]
Le caractère ionique croît avec la \textbf{différence d'électronégativité} $\Delta\chi$.
\[ \ce{Cl2} : \Delta\chi = 0 \;<\; \ce{HCl} : |3{,}2-2{,}2| = 1{,}0 \;<\;
   \ce{NaCl} : |3{,}2-0{,}9| = 2{,}3 \;<\; \ce{NaF} : |4{,}0-0{,}9| = 3{,}1. \]
Classement final : $\ce{Cl2} < \ce{HCl} < \ce{NaCl} < \ce{NaF}$. La liaison \ce{Cl2} est purement
covalente ($\Delta\chi = 0$), \ce{NaF} est la plus ionique.
\end{corrige}
