\begin{center}
{\LARGE\bfseries\color{primarydark}Thème 3 — Proportions stœchiométriques}\\[1pt]
{\large\color{excol}\textbf{Niveau Facile}}\\[2pt]
{\itshape Objectif : appliquer la règle de trois par les coefficients (données déjà en moles, rendement \SI{100}{\percent}).}\\
{\color{primary}\rule{0.62\linewidth}{1pt}}
\end{center}

\begin{exo}[la synthèse générique 2A + 3B $\to$ C + 6D]
Pour la réaction $\ce{2 A + 3 B -> C + 6 D}$ (rendement \SI{100}{\percent}), on veut produire
$\nq{C}=\SI{0,100}{\mole}$ de \ce{C}.
Quelles quantités (en mol) de \ce{A} et de \ce{B} faut-il engager, et quelle quantité de \ce{D}
obtient-on ?
\end{exo}

\begin{exo}[synthèse de l'ammoniac]
Pour $\ce{N2 + 3 H2 -> 2 NH3}$, on fait réagir \SI{5,0}{\mole} de diazote \ce{N2} (dans les
proportions stœchiométriques).
\textbf{a)} Combien de moles de \ce{H2} sont consommées ?\quad
\textbf{b)} Combien de moles d'ammoniac \ce{NH3} forme-t-on ?
\end{exo}

\begin{exo}[formation de l'eau]
Pour $\ce{2 H2 + O2 -> 2 H2O}$, on souhaite obtenir \SI{4,0}{\mole} d'eau.
Combien de moles de \ce{H2} et de \ce{O2} faut-il faire réagir ?
\end{exo}

\begin{exo}[décomposition du calcaire]
Le calcaire se décompose selon $\ce{CaCO3 -> CaO + CO2}$. On chauffe \SI{3,0}{\mole} de \ce{CaCO3}
(réaction totale).
Combien de moles de chaux \ce{CaO} et de dioxyde de carbone \ce{CO2} obtient-on ?
\end{exo}

\begin{exo}[oxydation du fer : remonter aux réactifs]
Pour $\ce{4 Fe + 3 O2 -> 2 Fe2O3}$, on veut fabriquer \SI{6,0}{\mole} d'oxyde de fer(III) \ce{Fe2O3}.
Combien de moles de fer \ce{Fe} et de dioxygène \ce{O2} faut-il engager ?
\end{exo}
