\begin{corrige}[synthèses élémentaires]
\begin{enumerate}[label=\alph*)]
  \item \ce{2 H2 + O2 -> 2 H2O} \; (H : $4=4$ ; O : $2=2$).
  \item \ce{N2 + 3 H2 -> 2 NH3} \; (N : $2=2$ ; H : $6=6$).
  \item \ce{2 Na + Cl2 -> 2 NaCl} \; (Na : $2=2$ ; Cl : $2=2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[métaux et oxygène]
\begin{enumerate}[label=\alph*)]
  \item \ce{2 Mg + O2 -> 2 MgO} \; (Mg : $2=2$ ; O : $2=2$).
  \item \ce{2 Ca + O2 -> 2 CaO} \; (Ca : $2=2$ ; O : $2=2$).
  \item \ce{4 Al + 3 O2 -> 2 Al2O3} \; (Al : $4=4$ ; O : $6=6$). L'oxygène ($3$ atomes par motif
        \ce{Al2O3}) impose le ppcm $6$ avec les paquets de \ce{O2}.
\end{enumerate}
\end{corrige}

\begin{corrige}[décompositions]
\begin{enumerate}[label=\alph*)]
  \item \ce{2 H2O2 -> 2 H2O + O2} \; (H : $4=4$ ; O : $4=2+2$).
  \item \ce{2 KClO3 -> 2 KCl + 3 O2} \; (K : $2=2$ ; Cl : $2=2$ ; O : $6=6$).
  \item \ce{CaCO3 -> CaO + CO2} \; déjà ajustée (Ca : $1=1$ ; C : $1=1$ ; O : $3=1+2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[combustions simples]
On règle C, puis H, puis O.
\begin{enumerate}[label=\alph*)]
  \item \ce{CH4 + 2 O2 -> CO2 + 2 H2O} \; (C : $1$ ; H : $4$ ; O : $4=2+2$).
  \item \ce{C3H8 + 5 O2 -> 3 CO2 + 4 H2O} \; (C : $3$ ; H : $8$ ; O : $10=6+4$).
  \item \ce{2 C2H6 + 7 O2 -> 4 CO2 + 6 H2O} \; l'oxygène donne $\tfrac{7}{2}\ce{O2}$, on double
        tout (C : $4$ ; H : $12$ ; O : $14=8+6$).
\end{enumerate}
\end{corrige}

\begin{corrige}[somme des coefficients]
Le carbone est déjà à $1$ de chaque côté ; l'oxygène ($2$ à gauche dans \ce{O2}) impose $2\,\ce{CO}$
à droite, ce qui porte le carbone à $2$ :
\[
\ce{2 C + O2 -> 2 CO}\qquad(\text{C}:2=2,\ \text{O}:2=2).
\]
Somme des coefficients : $2+1+2=\boxed{5}$.
\end{corrige}
