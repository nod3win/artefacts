\begin{cor}[la synthèse générique]
\textbf{a)} $\dfrac{\nq{A,0}}{2}=\dfrac{0{,}100}{2}=0{,}050$ ;\quad
$\dfrac{\nq{B,0}}{3}=\dfrac{0{,}200}{3}\approx0{,}0667$.\\
\textbf{b)} Le plus petit rapport est celui de \ce{A} ($0{,}050<0{,}0667$) :
\boxed{\text{\ce{A} est limitant}}, et $\xi_{\max}=\SI{0,050}{\mole}$. \ce{B} est en excès.
\end{cor}

\begin{cor}[synthèse de l'ammoniac]
$\dfrac{\nq{N2}}{1}=\dfrac{2{,}0}{1}=2{,}0$ ;\quad $\dfrac{\nq{H2}}{3}=\dfrac{9{,}0}{3}=3{,}0$.
Le plus petit est celui de \ce{N2} : \boxed{\text{\ce{N2} est limitant}} (\ce{H2} en excès).
\end{cor}

\begin{cor}[formation de l'eau]
$\dfrac{\nq{H2}}{2}=\dfrac{5{,}0}{2}=2{,}5$ ;\quad $\dfrac{\nq{O2}}{1}=\dfrac{2{,}0}{1}=2{,}0$.
Le plus petit est celui de \ce{O2} : \boxed{\text{\ce{O2} est limitant}} (\ce{H2} en excès).
\end{cor}

\begin{cor}[le piège : « plus petite quantité » $\neq$ limitant]
\textbf{a)} On a \emph{plus} de \ce{HCl} ($0{,}80$~mol) que de \ce{Zn} ($0{,}50$~mol).\\
\textbf{b)} Critère : $\dfrac{\nq{Zn}}{1}=0{,}50$ ;\quad $\dfrac{\nq{HCl}}{2}=\dfrac{0{,}80}{2}=0{,}40$.
Le plus petit rapport est celui de \ce{HCl} : \boxed{\text{\ce{HCl} est limitant}} !
\rep{Bien qu'il y ait \emph{plus} de \ce{HCl}, son coefficient (2) le rend limitant. C'est le \emph{rapport}
$n/\nu$ qui décide, jamais la quantité brute.}
\end{cor}

\begin{cor}[vrai ou faux ? notions de limitant et d'excès]
\begin{enumerate}[label=\arabic*.,leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item \textbf{Faux.} C'est le plus petit \emph{rapport} $n/\nu$ qui compte, pas la plus petite quantité
        (cf.\ le zinc et l'acide).
  \item \textbf{Vrai.} Le réactif en excès n'est que partiellement consommé : il en reste.
  \item \textbf{Vrai.} Par définition, le limitant est entièrement consommé (c'est lui qui arrête la réaction).
  \item \textbf{Faux} en général. Cela n'est vrai que dans le cas particulier d'un mélange
        \emph{exactement stœchiométrique} (tous les rapports $n/\nu$ égaux).
\end{enumerate}
\end{cor}
