\begin{corrige}[du comprimé aux ions]
\begin{enumerate}[label=\alph*)]
  \item $\qty{304}{\milli\gram}=\qty{0.304}{\gram}$ ; $n=\dfrac{0,304}{152}=\qty{2.0e-3}{\mole}.$
  \item Chaque motif \ce{FeSO4} porte un ion \ce{Fe^2+} : $N=n\,N_A=\num{2.0e-3}\times\num{6.02e23}=\num{1.20e21}$ ions.
  \item $m(\ce{Fe})=n\times M(\ce{Fe})=\num{2.0e-3}\times56=\qty{0.112}{\gram}=\qty{112}{\milli\gram}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[d'un acide concentré à la solution de travail]
\begin{enumerate}[label=\alph*)]
  \item Sur \qty{1}{\litre} : $m_{\text{sol}}=\rho V=1,20\times1000=\qty{1200}{\gram}$ ;
        $m(\ce{HCl})=0,365\times1200=\qty{438}{\gram}$ ;
        $n=\dfrac{438}{36,5}=\qty{12.0}{\mole}$, d'où $C=\qty{12.0}{\mole\per\litre}.$
  \item $n_{\text{voulue}}=C_2V_2=0,60\times0,500=\qty{0.30}{\mole}$ ;
        $V=\dfrac{n}{C}=\dfrac{0,30}{12,0}=\qty{0.025}{\litre}=\qty{25}{\milli\litre}.$
  \item Prélever \qty{25}{\milli\litre} d'acide concentré à la pipette jaugée, les verser dans une
        fiole jaugée de \qty{500}{\milli\litre} (déjà partiellement remplie d'eau), puis compléter à
        l'eau distillée jusqu'au trait de jauge.
\end{enumerate}
\end{corrige}

\begin{corrige}[dilutions]
\begin{enumerate}[label=\alph*)]
  \item $V_1=\dfrac{C_2V_2}{C_1}=\dfrac{1,00\times50,0}{12,0}\approx\qty{4.17}{\milli\litre}.$
  \item $C_2=\dfrac{C_1V_1}{V_2}=\dfrac{2,5\times10}{250}=\qty{0.10}{\mole\per\litre}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[identifier un gaz par sa masse molaire]
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{V}{V_{\mathrm{m}}}=\dfrac{5,6}{22,4}=\qty{0.25}{\mole}.$
  \item $M=\dfrac{m}{n}=\dfrac{5,0}{0,25}=\qty{20}{\gram\per\mole}.$
  \item $M=20$ : c'est le \textbf{néon \ce{Ne}} ($M=20$) ; \ce{CH4} ($M=16$) et \ce{Ar} ($M=40$) sont exclus.
\end{enumerate}
\end{corrige}

\begin{corrige}[classer quatre échantillons]
\begin{enumerate}[label=\alph*)]
  \item $n_A=\dfrac{16}{32}=\qty{0.50}{\mole}$ ; $n_B=\dfrac{5,6}{22,4}=\qty{0.25}{\mole}$ ;
        $n_C=2,0\times0,200=\qty{0.40}{\mole}$ ;
        $n_D=\dfrac{\num{3.01e23}}{\num{6.02e23}}=\qty{0.50}{\mole}.$
  \item Ordre croissant : $B\ (0,25) < C\ (0,40) < A = D\ (0,50)$ \si{\mole}.
\end{enumerate}
\end{corrige}
