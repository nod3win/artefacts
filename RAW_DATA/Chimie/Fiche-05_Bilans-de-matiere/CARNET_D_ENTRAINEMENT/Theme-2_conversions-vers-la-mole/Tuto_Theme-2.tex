% THEME_TITLE: Convertir masses, volumes et solutions en moles

\begin{center}
{\Large\bfseries\color{primarydark} Thème 2 — Convertir en quantité de matière (moles)}
\end{center}

\medskip
Les coefficients d'une équation comptent des \textbf{moles}. Mais au laboratoire on mesure des
\textbf{masses} (balance), des \textbf{volumes de gaz} (éprouvette) et des \textbf{solutions}
(concentration). Trois relations — et trois seulement — servent de \emph{portes d'entrée} vers la
mole. Les maîtriser dans les \emph{deux sens} est le cœur de tout calcul de chimie.

\begin{cle}[les trois portes vers la mole]
\[
\underbrace{n=\dfrac{m}{M}}_{\text{à partir d'une masse}}\qquad
\underbrace{n=\dfrac{V}{V_{\mathrm{m}}}}_{\text{gaz : }V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}\text{ (TPN)}}\qquad
\underbrace{n=C\times V}_{\text{solution}}
\]
$n$ en \si{\mole} ; $m$ en \si{\gram} ; $M$ en \si{\gram\per\mole} ; $V_{\mathrm{gaz}}$ et
$V_{\mathrm{solution}}$ en \si{\litre} ; $C$ en \si{\mole\per\litre}. On lit chaque relation à
l'endroit ($\to$ moles) ou à l'envers ($m=nM$, $V=nV_{\mathrm{m}}$, $C=n/V$).
\end{cle}

\begin{methode}[convertir une donnée en moles (puis en autre chose)]
\begin{enumerate}[leftmargin=1.7em,topsep=2pt,itemsep=1.5pt]
  \item \textbf{Repérer la nature de la donnée} : une masse ? un volume de gaz ? une solution ?
  \item \textbf{Homogénéiser les unités} : masses en \si{\gram}, volumes en \si{\litre}
        ($\qty{1}{\milli\litre}=\qty{1e-3}{\litre}$), $C$ en \si{\mole\per\litre}.
  \item \textbf{Appliquer la bonne porte} pour obtenir $n$ en moles.
  \item \textbf{Reconvertir} le résultat dans l'unité demandée ($m=nM$, $V=nV_{\mathrm{m}}$, $C=n/V$,
        ou nombre d'entités $N=n\,N_A$).
\end{enumerate}
\end{methode}

\begin{piege}[ne jamais confondre les unités]
\begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=1pt]
  \item une quantité en \textbf{moles n'est ni une masse ni un volume} : pour une masse, multiplier
        par $M$ ; pour un volume de gaz, par $V_{\mathrm{m}}$.
  \item $\qty{1}{\milli\litre}\neq\qty{1}{\litre}$ : convertir \emph{avant} de calculer
        ($\qty{20}{\milli\litre}=\qty{0.020}{\litre}$).
  \item $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$ vaut \emph{seulement aux TPN}
        ($\qty{0}{\celsius}$, $\qty{1}{atm}$).
\end{itemize}
\end{piege}

\begin{remarque}[le nombre d'Avogadro : compter les entités]
Une mole contient $N_A=\qty{6.02e23}{}$ entités. Pour passer des moles au \emph{nombre} de molécules
(ou d'ions, d'atomes), on multiplie : $N=n\,N_A$. La masse d'une seule entité vaut $M/N_A$.
\end{remarque}
