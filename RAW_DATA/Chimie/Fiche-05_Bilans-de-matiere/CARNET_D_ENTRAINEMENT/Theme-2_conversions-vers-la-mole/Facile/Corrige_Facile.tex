\begin{corrige}[de la masse aux moles]
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{18}{18}=\qty{1}{\mole}.$
  \item $n=\dfrac{88}{44}=\qty{2}{\mole}.$
  \item $n=\dfrac{20}{40}=\qty{0.5}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[des moles à la masse]
\begin{enumerate}[label=\alph*)]
  \item $m=3\times18=\qty{54}{\gram}.$
  \item $m=0,5\times58,5=\qty{29.25}{\gram}.$
  \item $m=0,25\times44=\qty{11}{\gram}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[gaz aux TPN]
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{V}{V_{\mathrm{m}}}=\dfrac{44,8}{22,4}=\qty{2}{\mole}.$
  \item $V=n\,V_{\mathrm{m}}=0,5\times22,4=\qty{11.2}{\litre}.$
  \item $n=\dfrac{5,6}{22,4}=\qty{0.25}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[quantité en solution]
On convertit d'abord les volumes en litres.
\begin{enumerate}[label=\alph*)]
  \item $n=0,5\times2=\qty{1}{\mole}.$
  \item $\qty{250}{\milli\litre}=\qty{0.250}{\litre}$ : $n=0,20\times0,250=\qty{0.050}{\mole}.$
  \item $\qty{100}{\milli\litre}=\qty{0.100}{\litre}$ : $n=0,10\times0,100=\qty{0.010}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[remonter à la concentration]
\begin{enumerate}[label=\alph*)]
  \item $C=\dfrac{0,10}{0,5}=\qty{0.20}{\mole\per\litre}.$
  \item $\qty{100}{\milli\litre}=\qty{0.100}{\litre}$ : $C=\dfrac{0,020}{0,100}=\qty{0.20}{\mole\per\litre}.$
\end{enumerate}
\end{corrige}
