\begin{exo}[de la masse aux moles]
Calcule la quantité de matière $n=\dfrac{m}{M}$.
\begin{enumerate}[label=\alph*)]
  \item \qty{18}{\gram} d'eau \ce{H2O} \quad ($M=\qty{18}{\gram\per\mole}$)
  \item \qty{88}{\gram} de \ce{CO2} \quad ($M=\qty{44}{\gram\per\mole}$)
  \item \qty{20}{\gram} de soude \ce{NaOH} \quad ($M=\qty{40}{\gram\per\mole}$)
\end{enumerate}
\end{exo}

\begin{exo}[des moles à la masse]
Calcule la masse $m=n\,M$.
\begin{enumerate}[label=\alph*)]
  \item \qty{3}{\mole} d'eau \quad ($M=\qty{18}{\gram\per\mole}$)
  \item \qty{0.5}{\mole} de \ce{NaCl} \quad ($M=\qty{58.5}{\gram\per\mole}$)
  \item \qty{0.25}{\mole} de \ce{CO2} \quad ($M=\qty{44}{\gram\per\mole}$)
\end{enumerate}
\end{exo}

\begin{exo}[gaz aux TPN]
On travaille aux TPN, où $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Quantité de matière dans \qty{44.8}{\litre} de \ce{O2} ?
  \item Volume occupé par \qty{0.5}{\mole} de \ce{CO2} ?
  \item Quantité de matière dans \qty{5.6}{\litre} de \ce{H2} ?
\end{enumerate}
\end{exo}

\begin{exo}[quantité en solution]
Calcule la quantité de matière de soluté $n=C\,V$ (attention aux \si{\milli\litre}).
\begin{enumerate}[label=\alph*)]
  \item \qty{2}{\litre} de solution à \qty{0.5}{\mole\per\litre}
  \item \qty{250}{\milli\litre} de solution à \qty{0.20}{\mole\per\litre}
  \item \qty{100}{\milli\litre} de solution à \qty{0.10}{\mole\per\litre}
\end{enumerate}
\end{exo}

\begin{exo}[remonter à la concentration]
Calcule la concentration molaire $C=\dfrac{n}{V}$.
\begin{enumerate}[label=\alph*)]
  \item \qty{0.10}{\mole} de soluté dans \qty{0.5}{\litre} de solution
  \item \qty{0.020}{\mole} de soluté dans \qty{100}{\milli\litre} de solution
\end{enumerate}
\end{exo}
