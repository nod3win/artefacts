\begin{center}\small\itshape
Masses molaires atomiques (\si{\gram\per\mole}) : H\,=\,1 ; C\,=\,12 ; N\,=\,14 ; O\,=\,16 ;
Na\,=\,23 ; S\,=\,32 ; Cl\,=\,35,5 ; Ca\,=\,40. \quad $N_A=\qty{6.02e23}{}$ ; TPN : $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$.
\end{center}

\begin{exo}[calculer d'abord la masse molaire]
Détermine $M$ à partir des masses atomiques, puis la quantité de matière.
\begin{enumerate}[label=\alph*)]
  \item \qty{5.85}{\gram} de \ce{NaCl}
  \item \qty{9.8}{\gram} d'acide sulfurique \ce{H2SO4}
  \item \qty{10}{\gram} de carbonate de calcium \ce{CaCO3}
\end{enumerate}
\end{exo}

\begin{exo}[enchaîner masse et volume gazeux (TPN)]
\begin{enumerate}[label=\alph*)]
  \item Quel volume occupent \qty{8}{\gram} de \ce{O2} aux TPN ?
  \item Quelle est la masse de \qty{11.2}{\litre} de \ce{N2} pris aux TPN ?
  \item Quel volume occupent \qty{3.4}{\gram} d'ammoniac \ce{NH3} aux TPN ?
\end{enumerate}
\end{exo}

\begin{exo}[solutions dans les deux sens]
\begin{enumerate}[label=\alph*)]
  \item Quelle masse de glucose \ce{C6H12O6} ($M=\qty{180}{\gram\per\mole}$) faut-il pour préparer
        \qty{500}{\milli\litre} de solution à \qty{0.10}{\mole\per\litre} ?
  \item Quel volume d'une solution à \qty{0.25}{\mole\per\litre} contient \qty{0.05}{\mole} de soluté ?
  \item On dissout \qty{4}{\gram} de \ce{NaOH} dans de l'eau pour obtenir \qty{200}{\milli\litre} de
        solution. Quelle est sa concentration molaire ?
\end{enumerate}
\end{exo}

\begin{exo}[passer par la masse volumique]
La masse volumique relie volume et masse : $m=\rho\,V$.
\begin{enumerate}[label=\alph*)]
  \item Quelle masse représentent \qty{500}{\milli\litre} de lait de masse volumique
        $\rho=\qty{1.03}{\gram\per\milli\litre}$ ?
  \item Combien de moles d'eau contient \qty{1}{\litre} d'eau ($\rho=\qty{1.00}{\gram\per\milli\litre}$,
        $M=\qty{18}{\gram\per\mole}$) ?
  \item Quelle quantité de matière y a-t-il dans \qty{50}{\milli\litre} d'éthanol \ce{C2H6O}
        ($\rho=\qty{0.79}{\gram\per\milli\litre}$, $M=\qty{46}{\gram\per\mole}$) ?
\end{enumerate}
\end{exo}

\begin{exo}[compter les entités avec $N_A$]
\begin{enumerate}[label=\alph*)]
  \item Combien de molécules dans \qty{9}{\gram} d'eau ($M=\qty{18}{\gram\per\mole}$) ?
  \item Combien d'ions \ce{Na+} dans \qty{0.20}{\mole} de sulfate de sodium \ce{Na2SO4} ?
  \item Quelle est la masse d'\emph{une} molécule d'eau ?
\end{enumerate}
\end{exo}
