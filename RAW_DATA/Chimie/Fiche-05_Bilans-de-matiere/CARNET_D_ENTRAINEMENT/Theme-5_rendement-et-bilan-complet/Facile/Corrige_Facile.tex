\begin{corrige}[calculer un rendement]
\begin{enumerate}[label=\alph*)]
  \item $R=\dfrac{1,5}{2}\times100=\qty{75}{\percent}.$
  \item $R=\dfrac{0,3}{0,4}\times100=\qty{75}{\percent}.$
  \item $R=\dfrac{4}{5}\times100=\qty{80}{\percent}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[prévoir la quantité réelle]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{réel}}=0,90\times2=\qty{1.8}{\mole}.$
  \item $n_{\text{réel}}=0,60\times0,5=\qty{0.3}{\mole}.$
  \item $n_{\text{réel}}=0,50\times10=\qty{5}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[rendement en masse]
\begin{enumerate}[label=\alph*)]
  \item $R=\dfrac{30}{40}\times100=\qty{75}{\percent}.$
  \item $R=\dfrac{85}{100}\times100=\qty{85}{\percent}.$
  \item $m_{\text{réel}}=0,80\times50=\qty{40}{\gram}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[du théorique au rendement]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{théo}}(\ce{H2O})=n(\ce{H2})\times\dfrac{2}{2}=\qty{2}{\mole}.$
  \item $R=\dfrac{1,6}{2}\times100=\qty{80}{\percent}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[rendement et reconversion]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{théo}}(\ce{NH3})=1\times\dfrac{2}{1}=\qty{2}{\mole}$, d'où
        $m_{\text{théo}}=2\times17=\qty{34}{\gram}.$
  \item $m_{\text{réel}}=0,90\times34=\qty{30.6}{\gram}.$
\end{enumerate}
\end{corrige}
