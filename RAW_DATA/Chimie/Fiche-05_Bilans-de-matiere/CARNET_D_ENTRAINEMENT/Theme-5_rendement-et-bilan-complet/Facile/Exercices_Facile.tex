\begin{center}
{\LARGE\bfseries\color{primarydark}Thème 5 — Rendement}\\[1pt]
{\large\color{excol}\textbf{Niveau Facile}}\\[2pt]
{\itshape Objectif : manier la formule $R=n_{\text{réel}}/n_{\text{théorique}}\times100$ dans les deux sens.}\\
{\color{primary}\rule{0.64\linewidth}{1pt}}
\end{center}

\begin{exo}[rendement à partir de moles]
La quantité \textbf{théorique} de produit d'une réaction est \SI{0,100}{\mole}, mais on n'en obtient
réellement que \SI{0,080}{\mole}. Quel est le rendement $R$ ?
\end{exo}

\begin{exo}[rendement à partir de masses]
On attendait \SI{45}{\gram} de produit ; on en récupère \SI{36}{\gram}.
Quel est le rendement de la réaction ? \emph{(La masse molaire se simplifie : on peut travailler
directement sur les masses.)}
\end{exo}

\begin{exo}[quantité réelle à partir du rendement]
La quantité théorique de produit est \SI{2,0}{\mole} et le rendement vaut \SI{75}{\percent}.
Quelle quantité de produit obtient-on réellement ?
\end{exo}

\begin{exo}[masse réelle à partir du rendement]
Une synthèse a un rendement de \SI{60}{\percent}. La masse théorique de produit est \SI{50}{\gram}.
Quelle masse récupère-t-on effectivement ?
\end{exo}

\begin{exo}[vrai ou faux ? notion de rendement]
Indique si chaque affirmation est \textbf{vraie} ou \textbf{fausse}, et justifie :
\begin{enumerate}[label=\arabic*.,leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item « Le rendement d'une réaction peut dépasser \SI{100}{\percent}. »
  \item « Mettre un réactif en grand excès garantit un rendement de \SI{100}{\percent}. »
  \item « Le rendement réel se mesure expérimentalement ; il peut être bien inférieur à la valeur théorique. »
\end{enumerate}
\end{exo}
