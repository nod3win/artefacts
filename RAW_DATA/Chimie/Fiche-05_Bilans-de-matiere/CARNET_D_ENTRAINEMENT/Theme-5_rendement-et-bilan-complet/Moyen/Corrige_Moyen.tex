\begin{corrige}[rendement d'une fermentation]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{théo}}(\ce{C2H5OH})=n(\ce{glucose})\times\dfrac{2}{1}=0,5\times2=\qty{1}{\mole}.$
  \item $n_{\text{réel}}=\dfrac{11,5}{46}=\qty{0.25}{\mole}.$
  \item $R=\dfrac{0,25}{1}\times100=\qty{25}{\percent}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[calcination du calcaire]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{CaCO3})=\dfrac{100}{100}=\qty{1}{\mole}$ ; rapport $1:1$ donc
        $n_{\text{théo}}(\ce{CaO})=\qty{1}{\mole}$, soit $m_{\text{théo}}=1\times56=\qty{56}{\gram}.$
  \item $R=\dfrac{42}{56}\times100=\qty{75}{\percent}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[remonter du produit voulu au réactif]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{théo}}=\dfrac{n_{\text{réel}}}{R/100}=\dfrac{0,9}{0,75}=\qty{1.2}{\mole}.$
  \item $n(\ce{N2})=n_{\text{théo}}(\ce{NH3})\times\dfrac{1}{2}=1,2\times0,5=\qty{0.6}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[rendement à partir du limitant]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{Al})=\dfrac{2,7}{27}=\qty{0.1}{\mole}$ ; $n(\ce{ZnCl2})=0,5\times0,200=\qty{0.1}{\mole}$.
        Rapports : \ce{ZnCl2} $\frac{0,1}{3}\approx0,033$ ; \ce{Al} $\frac{0,1}{2}=0,05$.
        \textbf{\ce{ZnCl2} est limitant} ; $n_{\text{théo}}(\ce{Zn})=0,1\times\frac{3}{3}=\qty{0.1}{\mole}.$
  \item $m_{\text{théo}}=0,1\times65,4=\qty{6.54}{\gram}.$
  \item $R=\dfrac{3,27}{6,54}\times100=\qty{50}{\percent}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[synthèse du trioxyde de soufre]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{SO2})=\dfrac{128}{64}=\qty{2}{\mole}$ ; rapport $2:2$ donc
        $n_{\text{théo}}(\ce{SO3})=\qty{2}{\mole}$, soit $m_{\text{théo}}=2\times80=\qty{160}{\gram}.$
  \item $R=\dfrac{120}{160}\times100=\qty{75}{\percent}.$
\end{enumerate}
\end{corrige}
