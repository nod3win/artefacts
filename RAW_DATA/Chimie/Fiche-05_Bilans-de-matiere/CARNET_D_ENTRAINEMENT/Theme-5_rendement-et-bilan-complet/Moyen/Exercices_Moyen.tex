{\LARGE\bfseries\color{primarydark}Thème 5 — Rendement}\\[1pt]
{\large\color{moycol}\textbf{Niveau Moyen}}\\[2pt]
{\itshape Objectif : calculer la quantité théorique par la stœchiométrie, puis en déduire le rendement.}\\
{\color{primary}\rule{0.66\linewidth}{1pt}}
\end{center}

\begin{exo}[fermentation du glucose]
La fermentation alcoolique s'écrit $\ce{C6H12O6 -> 2 C2H5OH + 2 CO2}$. À partir de \SI{0,50}{\mole}
de glucose, on récupère \SI{11,5}{\gram} d'éthanol \ce{C2H5OH} ($M=\SI{46}{\gram\per\mole}$).
\textbf{a)} Quelle est la quantité \emph{théorique} d'éthanol (en mol, puis en g) ?\quad
\textbf{b)} Quel est le rendement de la fermentation ?
\end{exo}

\begin{exo}[aluminothermie du fer]
La réaction $\ce{3 FeO + 2 Al -> 3 Fe + Al2O3}$ est réalisée avec \SI{144}{\gram} de \ce{FeO}
($M=72$) et \SI{27}{\gram} d'aluminium ($M=27$). On obtient \SI{42}{\gram} de fer ($M=56$).
\textbf{a)} Identifie le réactif limitant.\quad
\textbf{b)} Calcule la masse théorique de fer.\quad
\textbf{c)} En déduire le rendement.
\end{exo}

\begin{exo}[synthèse du carbure de silicium]
Le carbure de silicium (carborundum) est produit par $\ce{SiO2 + 3 C -> SiC + 2 CO}$. On fait réagir
\SI{120}{\kilo\gram} de \ce{SiO2} ($M=60$) et \SI{36}{\kilo\gram} de carbone ($M=12$) ; on obtient
\SI{32}{\kilo\gram} de \ce{SiC} ($M=40$).
\textbf{a)} Quel est le réactif limitant ?\quad
\textbf{b)} Quelle masse théorique de \ce{SiC} attend-on ?\quad
\textbf{c)} Quel est le rendement ?
\end{exo}

\begin{exo}[préparation du carbonate de sodium]
Soit $\ce{2 NaCl + CaCO3 -> Na2CO3 + CaCl2}$. On part de \SI{29,25}{\gram} de \ce{NaCl}
($M=58{,}5$) et \SI{45,0}{\gram} de \ce{CaCO3} ($M=100$) ; on récupère \SI{13,25}{\gram} de
\ce{Na2CO3} ($M=106$).
\textbf{a)} Détermine le réactif limitant.\quad
\textbf{b)} Calcule la masse théorique de \ce{Na2CO3}, puis le rendement.
\end{exo}

\begin{exo}[combustion du sulfure d'hydrogène avec pertes]
On brûle \SI{10,2}{\gram} de \ce{H2S} ($M=34$) avec \SI{9,6}{\gram} de \ce{O2} ($M=32$) selon
$\ce{2 H2S + 3 O2 -> 2 SO2 + 2 H2O}$. Le \textbf{rendement molaire est de \SI{50}{\percent}}.
\textbf{a)} Quel est le réactif limitant ?\quad
\textbf{b)} Quelle masse de \ce{SO2} ($M=64$) obtient-on \emph{réellement} ?
\end{exo}
