\begin{center}\small\itshape
Aux TPN : $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$.
\end{center}

\begin{exo}[limitant \emph{et} rendement]
On brûle du sulfure d'hydrogène : $\ce{2 H2S + 3 O2 -> 2 SO2 + 2 H2O}$. On engage \qty{10.2}{\gram}
de \ce{H2S} ($M=\qty{34}{\gram\per\mole}$) et \qty{9.6}{\gram} de \ce{O2}
($M=\qty{32}{\gram\per\mole}$). On donne $M(\ce{SO2})=\qty{64}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule les quantités de matière et identifie le réactif limitant.
  \item Quelle masse \emph{théorique} de \ce{SO2} peut-on former ?
  \item Le rendement de la réaction est de \qty{50}{\percent}. Quelle masse de \ce{SO2} obtient-on réellement ?
\end{enumerate}
\end{exo}

\begin{exo}[fermentation : rendement et gaz dégagé]
On fait fermenter \qty{90}{\gram} de glucose ($M=\qty{180}{\gram\per\mole}$) :
$\ce{C6H12O6 -> 2 C2H5OH + 2 CO2}$. On recueille \qty{0.7}{\mole} d'éthanol.
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité \emph{théorique} d'éthanol prévoit-on ?
  \item Quel est le rendement de la fermentation ?
  \item Quel volume de \ce{CO2} (aux TPN) est \emph{réellement} dégagé (même rendement) ?
\end{enumerate}
\end{exo}

\begin{exo}[remonter au réactif, rendement compris]
Dans un haut fourneau, $\ce{Fe2O3 + 3 CO -> 2 Fe + 3 CO2}$. On veut produire \emph{réellement}
\qty{112}{\gram} de fer, avec un rendement de \qty{80}{\percent}. On donne
$M(\ce{Fe})=\qty{56}{\gram\per\mole}$ et $M(\ce{Fe2O3})=\qty{160}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité de fer \emph{réelle} vise-t-on ?
  \item Quelle quantité \emph{théorique} de fer faut-il donc prévoir ?
  \item Quelle masse d'oxyde \ce{Fe2O3} faut-il engager ?
\end{enumerate}
\end{exo}

\begin{exo}[l'airbag : décomposition, rendement et volume]
Le gonflage d'un airbag repose sur $\ce{2 NaN3 -> 2 Na + 3 N2 ^}$. On introduit \qty{65}{\gram} d'azoture
de sodium \ce{NaN3} ($M=\qty{65}{\gram\per\mole}$).
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité \emph{théorique} de \ce{N2} la décomposition peut-elle fournir ?
  \item Quel volume ce diazote occuperait-il aux TPN ?
  \item En pratique le rendement est de \qty{95}{\percent}. Quel volume de \ce{N2} gonfle réellement l'airbag ?
\end{enumerate}
\end{exo}
