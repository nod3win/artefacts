\begin{corrige}[limitant \emph{et} rendement]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{H2S})=\dfrac{10,2}{34}=\qty{0.3}{\mole}$ ; $n(\ce{O2})=\dfrac{9,6}{32}=\qty{0.3}{\mole}$.
        Rapports : \ce{H2S} $\frac{0,3}{2}=0,15$ ; \ce{O2} $\frac{0,3}{3}=0,10$.
        \textbf{\ce{O2} est limitant}.
  \item $n_{\text{théo}}(\ce{SO2})=n(\ce{O2})\times\dfrac{2}{3}=0,3\times\dfrac{2}{3}=\qty{0.2}{\mole}$,
        d'où $m_{\text{théo}}=0,2\times64=\qty{12.8}{\gram}.$
  \item $m_{\text{réel}}=\dfrac{R}{100}\,m_{\text{théo}}=0,50\times12,8=\qty{6.4}{\gram}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[fermentation : rendement et gaz dégagé]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{glucose})=\dfrac{90}{180}=\qty{0.5}{\mole}$, donc
        $n_{\text{théo}}(\ce{C2H5OH})=0,5\times2=\qty{1}{\mole}.$
  \item $R=\dfrac{0,7}{1}\times100=\qty{70}{\percent}.$
  \item La théorie prévoit $n_{\text{théo}}(\ce{CO2})=0,5\times2=\qty{1}{\mole}$ ; au rendement de
        \qty{70}{\percent} : $n_{\text{réel}}(\ce{CO2})=0,70\times1=\qty{0.7}{\mole}$, d'où
        $V=0,7\times22,4=\qty{15.68}{\litre}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[remonter au réactif, rendement compris]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{réel}}(\ce{Fe})=\dfrac{112}{56}=\qty{2}{\mole}.$
  \item $n_{\text{théo}}(\ce{Fe})=\dfrac{n_{\text{réel}}}{R/100}=\dfrac{2}{0,80}=\qty{2.5}{\mole}.$
  \item $n(\ce{Fe2O3})=n_{\text{théo}}(\ce{Fe})\times\dfrac{1}{2}=2,5\times0,5=\qty{1.25}{\mole}$,
        d'où $m=1,25\times160=\qty{200}{\gram}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[l'airbag : décomposition, rendement et volume]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{NaN3})=\dfrac{65}{65}=\qty{1}{\mole}$, donc
        $n_{\text{théo}}(\ce{N2})=1\times\dfrac{3}{2}=\qty{1.5}{\mole}.$
  \item $V_{\text{théo}}=1,5\times22,4=\qty{33.6}{\litre}.$
  \item $V_{\text{réel}}=\dfrac{R}{100}\,V_{\text{théo}}=0,95\times33,6=\qty{31.92}{\litre}.$
\end{enumerate}
\end{corrige}
