\begin{center}\small\itshape
Aux TPN : $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$. Rendement \qty{100}{\percent}.
\end{center}

\begin{exo}[données mêlées : masse et solution]
Le permanganate oxyde l'acide chlorhydrique concentré :
$\ce{2 KMnO4 + 16 HCl -> 2 KCl + 2 MnCl2 + 5 Cl2 ^ + 8 H2O}$. On introduit \qty{15.8}{\gram} de
\ce{KMnO4} ($M=\qty{158}{\gram\per\mole}$) et \qty{200}{\milli\litre} d'acide à \qty{6}{\mole\per\litre}.
\begin{enumerate}[label=\alph*)]
  \item Convertis les deux réactifs en quantités de matière.
  \item Quel est le réactif limitant ?
  \item Quel volume de dichlore \ce{Cl2} (aux TPN) se dégage-t-il ?
\end{enumerate}
\end{exo}

\begin{exo}[trois réactifs, trois formes de données]
Pour $\ce{2 K2Cr2O7 + 3 C + 8 H2SO4 -> 3 CO2 ^ + 2 Cr2(SO4)3 + 2 K2SO4 + 8 H2O}$, on mélange
\qty{58.8}{\gram} de \ce{K2Cr2O7} ($M=\qty{294}{\gram\per\mole}$), \qty{3.6}{\gram} de carbone
($M=\qty{12}{\gram\per\mole}$) et \qty{200}{\milli\litre} d'acide sulfurique à \qty{3}{\mole\per\litre}.
\begin{enumerate}[label=\alph*)]
  \item Calcule les trois quantités de matière.
  \item Détermine le réactif limitant en comparant les rapports $n/\nu$.
  \item Quel volume de \ce{CO2} (aux TPN) obtient-on ?
\end{enumerate}
\end{exo}

\begin{exo}[combustion à oxygène limité]
On brûle \qty{88}{\gram} de propane dans \qty{16}{\gram} de dioxygène :
$\ce{C3H8 + 5 O2 -> 3 CO2 + 4 H2O}$. On donne $M(\ce{C3H8})=\qty{44}{\gram\per\mole}$,
$M(\ce{O2})=\qty{32}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule les quantités de \ce{C3H8} et de \ce{O2}.
  \item Quel est le réactif limitant ?
  \item Quel volume de \ce{CO2} (aux TPN) se forme-t-il ?
\end{enumerate}
\end{exo}

\begin{exo}[bilan complet et conservation de la masse]
On fait réagir \qty{28}{\gram} de \ce{N2} avec \qty{9}{\gram} de \ce{H2} : $\ce{N2 + 3 H2 -> 2 NH3}$.
On donne $M(\ce{N2})=\qty{28}{\gram\per\mole}$, $M(\ce{H2})=\qty{2}{\gram\per\mole}$,
$M(\ce{NH3})=\qty{17}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Détermine le réactif limitant.
  \item Quelle masse d'ammoniac se forme ?
  \item Quelle masse de réactif en excès reste-t-il ? Vérifie la conservation de la masse.
\end{enumerate}
\end{exo}
