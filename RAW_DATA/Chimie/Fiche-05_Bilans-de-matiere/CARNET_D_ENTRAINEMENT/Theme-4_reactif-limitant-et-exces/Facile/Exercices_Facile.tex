\begin{exo}[identifier le réactif limitant]
Dans chaque cas, calcule le rapport $\dfrac{n}{\nu}$ de chaque réactif et désigne le limitant.
\begin{enumerate}[label=\alph*)]
  \item $\ce{C + O2 -> CO2}$ : \qty{3}{\mole} de \ce{C} et \qty{2}{\mole} de \ce{O2}.
  \item $\ce{2 H2 + O2 -> 2 H2O}$ : \qty{5}{\mole} de \ce{H2} et \qty{2}{\mole} de \ce{O2}.
  \item $\ce{N2 + 3 H2 -> 2 NH3}$ : \qty{2}{\mole} de \ce{N2} et \qty{9}{\mole} de \ce{H2}.
\end{enumerate}
\end{exo}

\begin{exo}[le réactif en excès qui reste]
Pour $\ce{2 H2 + O2 -> 2 H2O}$, on mélange \qty{5}{\mole} de \ce{H2} et \qty{2}{\mole} de \ce{O2}.
\begin{enumerate}[label=\alph*)]
  \item Quel est le réactif limitant ?
  \item Quelle quantité de \ce{H2} est consommée ?
  \item Quelle quantité de \ce{H2} reste-t-il à la fin ?
\end{enumerate}
\end{exo}

\begin{exo}[calculer le produit à partir du limitant]
Pour $\ce{2 Al + 3 Cl2 -> 2 AlCl3}$, on mélange \qty{4}{\mole} de \ce{Al} et \qty{3}{\mole} de \ce{Cl2}.
\begin{enumerate}[label=\alph*)]
  \item Quel est le réactif limitant ?
  \item Quelle quantité de \ce{AlCl3} se forme ?
  \item Quelle quantité de \ce{Al} reste-t-il ?
\end{enumerate}
\end{exo}

\begin{exo}[attention au piège !]
Pour $\ce{N2 + 3 H2 -> 2 NH3}$, on part de \qty{2}{\mole} de \ce{N2} et \qty{3}{\mole} de \ce{H2}.
\begin{enumerate}[label=\alph*)]
  \item Quel réactif est présent en plus \textbf{grande quantité de matière} ?
  \item Calcule le rapport $\dfrac{n}{\nu}$ de chacun.
  \item Quel est le réactif limitant ? Que conclus-tu sur le piège « limitant = moins de moles » ?
\end{enumerate}
\end{exo}

\begin{exo}[précipitation de l'iodure de plomb]
Pour $\ce{Pb(NO3)2 + 2 KI -> PbI2 v + 2 KNO3}$, on mélange \qty{0.1}{\mole} de \ce{Pb(NO3)2} et
\qty{0.3}{\mole} de \ce{KI}.
\begin{enumerate}[label=\alph*)]
  \item Quel est le réactif limitant ?
  \item Quelle quantité de précipité \ce{PbI2} se forme ?
  \item Quelle quantité de \ce{KI} reste-t-il ?
\end{enumerate}
\end{exo}
