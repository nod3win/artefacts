\begin{corrige}[synthèse de l'ammoniac]
On applique $n_{\ce{Y}}=n_{\ce{X}}\times\frac{\nu_{\ce{Y}}}{\nu_{\ce{X}}}$.
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{H2})=2\times\dfrac{3}{1}=\qty{6}{\mole}.$
  \item $n(\ce{NH3})=2\times\dfrac{2}{1}=\qty{4}{\mole}.$
  \item $n(\ce{N2})=5\times\dfrac{1}{2}=\qty{2.5}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[combustion du méthane]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{O2})=3\times\dfrac{2}{1}=\qty{6}{\mole}.$
  \item $n(\ce{CO2})=0,5\times\dfrac{1}{1}=\qty{0.5}{\mole}.$
  \item $n(\ce{H2O})=0,5\times\dfrac{2}{1}=\qty{1}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[chlorure d'aluminium]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{Cl2})=4\times\dfrac{3}{2}=\qty{6}{\mole}.$
  \item $n(\ce{AlCl3})=4\times\dfrac{2}{2}=\qty{4}{\mole}.$
  \item $n(\ce{Al})=1\times\dfrac{2}{2}=\qty{1}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[lire un tableau d'avancement]
Les réactifs perdent $\nu\xi$, les produits gagnent $\nu\xi$.
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{N2})=1-1\times0,5=\qty{0.5}{\mole}.$
  \item $n(\ce{H2})=3-3\times0,5=3-1,5=\qty{1.5}{\mole}.$
  \item $n(\ce{NH3})=2\times0,5=\qty{1}{\mole}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[proportion puis reconversion]
\begin{enumerate}[label=\alph*)]
  \item $n(\ce{H2O})=4\times\dfrac{2}{2}=\qty{4}{\mole}.$
  \item $m=n\,M=4\times18=\qty{72}{\gram}.$
  \item $n(\ce{O2})=4\times\dfrac{1}{2}=\qty{2}{\mole}.$
\end{enumerate}
\end{corrige}
