\begin{center}\small\itshape
Aux TPN : $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$. Rendement supposé \qty{100}{\percent}.
\end{center}

\begin{exo}[du fer à l'oxyde : masse $\to$ masse]
Pour $\ce{4 Fe + 3 O2 -> 2 Fe2O3}$, on fait réagir \qty{11.2}{\gram} de fer.
On donne $M(\ce{Fe})=\qty{56}{\gram\per\mole}$ et $M(\ce{Fe2O3})=\qty{160}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité de matière de fer engage-t-on ?
  \item Quelle masse d'oxyde \ce{Fe2O3} obtient-on ?
  \item Quel volume de \ce{O2} (aux TPN) est consommé ?
\end{enumerate}
\end{exo}

\begin{exo}[volumes de gaz : le raccourci]
À température et pression fixées, les volumes de gaz sont proportionnels aux quantités de matière.
Pour la combustion du propane $\ce{C3H8 + 5 O2 -> 3 CO2 + 4 H2O}$, on brûle \qty{2}{\litre} de propane.
\begin{enumerate}[label=\alph*)]
  \item Quel volume de \ce{O2} (mêmes conditions) faut-il ?
  \item Quel volume de \ce{CO2} obtient-on ?
\end{enumerate}
\end{exo}

\begin{exo}[précipitation en solution]
On verse du nitrate d'argent dans une solution de chlorure : $\ce{AgNO3 + NaCl -> AgCl v + NaNO3}$.
On introduit \qty{100}{\milli\litre} de nitrate d'argent à \qty{0.5}{\mole\per\litre} (en excès de chlorure).
On donne $M(\ce{AgCl})=\qty{143.5}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité de matière de \ce{AgNO3} verse-t-on ?
  \item Quelle masse de précipité \ce{AgCl} se forme ?
\end{enumerate}
\end{exo}

\begin{exo}[construire un tableau d'avancement]
On chauffe du dioxyde de soufre : $\ce{2 SO2 + O2 -> 2 SO3}$, avec $n_0(\ce{SO2})=\qty{0.4}{\mole}$ et
$n_0(\ce{O2})=\qty{0.3}{\mole}$. On observe qu'à la fin \emph{tout le \ce{SO2} a été consommé}.
\begin{enumerate}[label=\alph*)]
  \item Exprime $n(\ce{SO2})$, $n(\ce{O2})$ et $n(\ce{SO3})$ en fonction de l'avancement $\xi$.
  \item Quelle est la valeur de l'avancement en fin de réaction ?
  \item En déduire les quantités finales de \ce{O2} et de \ce{SO3}.
\end{enumerate}
\end{exo}

\begin{exo}[acétylène du chalumeau]
Le carbure de calcium réagit avec l'eau : $\ce{CaC2 + 2 H2O -> C2H2 ^ + Ca(OH)2}$. On fait réagir
\qty{320}{\gram} de \ce{CaC2}, de masse molaire $M=\qty{64}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle quantité de matière de \ce{CaC2} fait-on réagir ?
  \item Quelle quantité d'acétylène \ce{C2H2} se forme ?
  \item Quel volume ce gaz occupe-t-il aux TPN ?
\end{enumerate}
\end{exo}
