\begin{exo}[l'acidité, une échelle logarithmique]
On travaille à \qty{25}{\celsius}.
\begin{enumerate}[label=\alph*)]
  \item Le sang a $\mathrm{pH} = 7{,}4$ et le suc gastrique $\mathrm{pH} = 1{,}4$. Par quel facteur la
        concentration $[\ce{H3O+}]$ du suc gastrique dépasse-t-elle celle du sang ?
  \item De façon générale, si le pH d'une solution \textbf{diminue de 3 unités}, par quel facteur
        $[\ce{H3O+}]$ est-elle multipliée ?
  \item L'acidité d'une solution est-elle « 2 fois » ou « 1000 fois » plus élevée quand le pH passe de 6 à 3 ?
\end{enumerate}
\end{exo}

\begin{exo}[ce qu'implique « basique » (à \qty{25}{\celsius})]
Une solution aqueuse est \textbf{basique} à \qty{25}{\celsius}. Pour chaque affirmation, réponds
\textbf{vrai} ou \textbf{faux} et justifie.
\begin{enumerate}[label=\alph*)]
  \item $[\ce{H3O+}] < [\ce{OH-}]$.
  \item $[\ce{H3O+}] > \qty{e-7}{mol\per L}$.
  \item $[\ce{OH-}] < \qty{e-7}{mol\per L}$.
  \item $\mathrm{pOH} < 7$.
  \item $[\ce{H3O+}] = 0$.
\end{enumerate}
\end{exo}

\begin{exo}[lire une table de $K_a$]
À \qty{25}{\celsius}, on donne les constantes d'acidité :
\begin{center}\small
\begin{tabular}{lc}
\toprule
\textbf{Couple} & \textbf{$K_a$} \\
\midrule
\ce{H3PO4/H2PO4-}     & $\qty{7,5e-3}{}$ \\
\ce{HF/F-}            & $\qty{6,3e-4}{}$ \\
\ce{CH3COOH/CH3COO-}  & $\qty{1,8e-5}{}$ \\
\ce{HClO/ClO-}        & $\qty{3,0e-8}{}$ \\
\ce{NH4+/NH3}         & $\qty{5,6e-10}{}$ \\
\bottomrule
\end{tabular}
\end{center}
Indique la \textbf{seule affirmation incorrecte} et corrige-la.
\begin{enumerate}[label=\alph*)]
  \item \ce{HF} est un acide plus fort que \ce{CH3COOH}.
  \item \ce{ClO-} est une base plus forte que \ce{F-}.
  \item Pour chaque couple, $\mathrm{p}K_a + \mathrm{p}K_b = 14$.
  \item \ce{NH3} est une base plus faible que \ce{CH3COO-}.
\end{enumerate}
\end{exo}

\begin{exo}[une seule proposition correcte sur la force]
Parmi les propositions suivantes (à \qty{25}{\celsius}), indique la \textbf{seule correcte} et explique pourquoi
les autres sont fausses.
\begin{enumerate}[label=\alph*)]
  \item Un acide faible est d'autant plus fort que le $\mathrm{p}K_a$ de son couple est élevé.
  \item Une base faible est d'autant plus faible que sa constante $K_b$ est élevée.
  \item Pour un couple acide/base conjugués, $\mathrm{p}K_a + \mathrm{p}K_b = 7$.
  \item Plus le $\mathrm{p}K_a$ d'un acide faible est grand, plus le $K_b$ de sa base conjuguée est élevé.
\end{enumerate}
\end{exo}

\begin{exo}[le bon produit ionique, et les variations croisées]
\begin{enumerate}[label=\alph*)]
  \item Parmi les écritures suivantes, laquelle est la \textbf{bonne} expression de $K_e$ ? Justifie pourquoi les
        autres sont fausses.
        \[ K_e = [\ce{H2O}][\ce{H3O+}]\ ;\quad K_e = [\ce{OH-}][\ce{H3O+}]\ ;\quad
           K_e = \frac{[\ce{OH-}][\ce{H3O+}]}{[\ce{H2O}]^2}. \]
  \item À température constante, si l'on ajoute un acide qui fait \emph{augmenter} $[\ce{H3O+}]$, que devient
        $[\ce{OH-}]$ ? Justifie avec $K_e$.
\end{enumerate}
\end{exo}
