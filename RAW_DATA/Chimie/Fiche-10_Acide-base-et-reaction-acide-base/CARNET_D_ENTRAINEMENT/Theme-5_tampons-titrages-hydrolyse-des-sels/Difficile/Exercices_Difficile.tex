\begin{center}\small\itshape On rappelle : $\log 2 \approx 0{,}30$,\quad $\log 3 \approx 0{,}48$,\quad $\log 5 \approx 0{,}70$.\end{center}

\begin{exo}[neutraliser un triacide]
On neutralise complètement \qty{25}{mL} d'acide phosphorique \ce{H3PO4} à \qty{0,10}{mol\per L} (triacide) par
de l'hydroxyde de potassium \ce{KOH} à \qty{0,15}{mol\per L}.
\begin{enumerate}[label=\alph*)]
  \item Écris la relation à l'équivalence en tenant compte des \textbf{trois} acidités de \ce{H3PO4}.
  \item Calcule le volume de \ce{KOH} nécessaire.
\end{enumerate}
\end{exo}

\begin{exo}[un tampon face à une base forte]
On dispose de \qty{200}{mL} d'un tampon contenant \ce{NH4Cl} à \qty{0,40}{mol\per L} et \ce{NH3} à
\qty{0,40}{mol\per L} (couple \ce{NH4+/NH3}, $\mathrm{p}K_a = 9{,}2$). On y ajoute \qty{20}{mL} de \ce{NaOH} à
\qty{1,0}{mol\per L}.
\begin{enumerate}[label=\alph*)]
  \item Écris la réaction de la base forte avec le tampon et calcule les nouvelles quantités de \ce{NH4+} et
        \ce{NH3}.
  \item En déduis le pH final.
\end{enumerate}
\end{exo}

\begin{exo}[préparer un tampon par titrage partiel]
On part de \qty{150}{mL} d'acide acétique \ce{CH3COOH} à \qty{0,20}{mol\per L} ($\mathrm{p}K_a = 4{,}74$) que
l'on neutralise partiellement par \ce{NaOH} à \qty{0,20}{mol\per L}.
\begin{enumerate}[label=\alph*)]
  \item Quel volume de \ce{NaOH} donne un tampon de $\mathrm{pH} = 4{,}74$ ?
  \item Quel volume de \ce{NaOH} donne un tampon de $\mathrm{pH} = 5{,}04$ ? (Indice : $5{,}04 = 4{,}74 + \log 2$.)
\end{enumerate}
\end{exo}

\begin{exo}[lire une courbe de titrage]
On titre \qty{20}{mL} d'un acide \ce{HA} par \ce{NaOH} à \qty{0,10}{mol\per L} ; on obtient la courbe suivante.

\end{enumerate}
\end{exo}

\begin{exo}[capstone : hydrolyse, y compris le sel d'acide et de base faibles]
Classe la solution de chaque sel en \textbf{acide}, \textbf{basique} ou \textbf{neutre}. Données :
\ce{HF} ($\mathrm{p}K_a = 3{,}2$), \ce{NH4+} ($\mathrm{p}K_a = 9{,}2$) ; \ce{HNO3}, \ce{HCl} forts ;
\ce{NaOH}, \ce{KOH} fortes.
\begin{enumerate}[label=\alph*)]
  \item \ce{NaNO3}
  \item \ce{KF}
  \item \ce{NH4NO3}
  \item \ce{NH4F} (sel d'acide \emph{faible} et de base \emph{faible} : utilise
        $\mathrm{pH} \approx \tfrac12(\mathrm{p}K_{a}(\ce{NH4+}) + \mathrm{p}K_{a}(\ce{HF}))$).
\end{enumerate}
\end{exo}
