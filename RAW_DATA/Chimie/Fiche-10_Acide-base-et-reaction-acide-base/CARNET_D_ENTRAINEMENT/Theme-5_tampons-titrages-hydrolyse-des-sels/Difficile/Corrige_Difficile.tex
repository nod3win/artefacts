\begin{corrige}[neutraliser un triacide]
\begin{enumerate}[label=\alph*)]
  \item Chaque \ce{H3PO4} cède \textbf{trois} \ce{H+}, donc à l'équivalence $\;3\,C_a V_a = C_b V_b$.
  \item $V_b = \dfrac{3\,C_a V_a}{C_b} = \dfrac{3 \times 0{,}10 \times 25}{0{,}15} = \dfrac{7{,}5}{0{,}15}
        = \mathbf{50\ mL}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[un tampon face à une base forte]
\begin{enumerate}[label=\alph*)]
  \item La base forte est consommée par l'acide du tampon : $\ce{NH4+ + OH- -> NH3 + H2O}$. Quantités
        initiales : $n(\ce{NH4+}) = n(\ce{NH3}) = 0{,}200\times0{,}40 = \qty{0,080}{mol}$ ;
        $n(\ce{OH-}) = 0{,}020\times1{,}0 = \qty{0,020}{mol}$. Après réaction :
        \[ n(\ce{NH4+}) = 0{,}080 - 0{,}020 = \qty{0,060}{mol},\qquad n(\ce{NH3}) = 0{,}080 + 0{,}020 = \qty{0,100}{mol}. \]
  \item $\mathrm{pH} = 9{,}2 + \log\dfrac{0{,}100}{0{,}060} = 9{,}2 + \log\tfrac{5}{3}
        = 9{,}2 + (0{,}70 - 0{,}48) = 9{,}2 + 0{,}22 = \mathbf{9{,}42}$. Variation minime : le tampon a résisté.
\end{enumerate}
\end{corrige}

\begin{corrige}[préparer un tampon par titrage partiel]
Quantité initiale d'acide : $n_0 = 0{,}150 \times 0{,}20 = \qty{0,030}{mol}$. La soude transforme \ce{CH3COOH}
en \ce{CH3COO-} : si on verse $x$ mol de \ce{NaOH}, alors $n(\ce{A-}) = x$ et $n(\ce{HA}) = 0{,}030 - x$.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = \mathrm{p}K_a$ exige $[\ce{A-}] = [\ce{HA}]$ (demi-équivalence) : $x = 0{,}015$ mol, soit
        $V = \dfrac{0{,}015}{0{,}20} = \qty{0,075}{L} = \mathbf{75\ mL}$.
  \item $\mathrm{pH} = 4{,}74 + \log 2 \Rightarrow \dfrac{[\ce{A-}]}{[\ce{HA}]} = 2$, donc
        $\dfrac{x}{0{,}030 - x} = 2 \Rightarrow x = 0{,}020$ mol, soit $V = \dfrac{0{,}020}{0{,}20}
        = \qty{0,10}{L} = \mathbf{100\ mL}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire une courbe de titrage]
\begin{enumerate}[label=\alph*)]
  \item Le saut de pH (point d'inflexion) est centré sur $V_E = \mathbf{20\ mL}$.
  \item La demi-équivalence est à $V_E/2 = \qty{10}{mL}$ ; on y lit $\mathrm{pH} = 4{,}74$. À ce point
        $[\ce{HA}] = [\ce{A-}]$, donc $\mathrm{p}K_a = \mathbf{4{,}74}$.
  \item \ce{HA} est un acide \textbf{faible} : son pH de départ ($\approx 2{,}9$) est élevé pour une
        concentration de cet ordre (un acide fort à $0{,}1$ M partirait de $\mathrm{pH} = 1$), et il présente une
        \emph{zone tampon}. Le pH à l'équivalence $\approx 8{,}7 > 7$ est \textbf{cohérent} : la base conjuguée
        \ce{A-} formée rend le milieu basique (acide faible dosé par base forte).
\end{enumerate}
\end{corrige}

\begin{corrige}[capstone : hydrolyse, y compris le sel d'acide et de base faibles]
\begin{enumerate}[label=\alph*)]
  \item \ce{NaNO3} : base forte + acide fort $\Rightarrow$ \textbf{neutre}.
  \item \ce{KF} : base forte (\ce{KOH}) + acide faible (\ce{HF}) $\Rightarrow$ \textbf{basique}
        ($\ce{F- + H2O <=> HF + OH-}$).
  \item \ce{NH4NO3} : base faible (\ce{NH3}) + acide fort (\ce{HNO3}) $\Rightarrow$ \textbf{acide}
        ($\ce{NH4+ + H2O <=> NH3 + H3O+}$).
  \item \ce{NH4F} : $\mathrm{pH} \approx \tfrac12(9{,}2 + 3{,}2) = \tfrac{12{,}4}{2} = 6{,}2 < 7$ $\Rightarrow$
        \textbf{légèrement acide}. (L'acide \ce{NH4+} l'emporte de peu sur la base \ce{F-}, très faible.)
\end{enumerate}
\end{corrige}
