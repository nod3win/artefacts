\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 2 --- Niveau Facile}\\[2pt]
{\color{primary}\itshape Convertir pH $\leftrightarrow$ concentration, classer une solution}
\end{center}

\begin{consigne}
\textbf{Objectif du niveau.} Manipuler sans hésitation les définitions $\mathrm{pH}=-\log[\ce{H3O+}]$,
$[\ce{H3O+}]=10^{-\mathrm{pH}}$, $\mathrm{pH}+\mathrm{pOH}=14$ et $K_e=[\ce{H3O+}][\ce{OH-}]=10^{-14}$.
\emph{On se place à \SI{25}{\celsius}. On donne $\log 2 \approx 0{,}30$ ; $\log 5 \approx 0{,}70$.}
\end{consigne}

\begin{exobox}[du pH vers la concentration]
Calculer $[\ce{H3O+}]$ (en \si{mol.L^{-1}}) pour chacune des solutions suivantes :
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\mathrm{pH} = 1$
  \item $\mathrm{pH} = 4$
  \item $\mathrm{pH} = 2$ (solution d'\ce{HCl} de l'énoncé d'annale 2018)
  \item $\mathrm{pH} = 7$
\end{enumerate}
\end{exobox}

\begin{exobox}[de la concentration vers le pH]
Calculer le $\mathrm{pH}$ pour chacune des valeurs de $[\ce{H3O+}]$ suivantes :
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $[\ce{H3O+}] = 10^{-3}\ \si{mol.L^{-1}}$
  \item $[\ce{H3O+}] = 10^{-9}\ \si{mol.L^{-1}}$
  \item $[\ce{H3O+}] = 5\times10^{-4}\ \si{mol.L^{-1}}$
  \item $[\ce{H3O+}] = 2\times10^{-6}\ \si{mol.L^{-1}}$
\end{enumerate}
\end{exobox}

\begin{exobox}[acide, neutre ou basique ?]
À \SI{25}{\celsius}, classer chaque solution en \textbf{acide}, \textbf{neutre} ou \textbf{basique} :
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\mathrm{pH} = 3{,}5$
  \item $\mathrm{pH} = 7$
  \item $\mathrm{pH} = 10{,}2$
  \item $[\ce{H3O+}] = 10^{-5}\ \si{mol.L^{-1}}$
  \item $[\ce{OH-}] = 10^{-3}\ \si{mol.L^{-1}}$
\end{enumerate}
\end{exobox}

\begin{exobox}[passer par la relation $\mathrm{pH}+\mathrm{pOH}=14$]
Pour chaque solution, calculer le $\mathrm{pOH}$ puis $[\ce{OH-}]$ :
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item $\mathrm{pH} = 3$
  \item $\mathrm{pH} = 11$
  \item $\mathrm{pH} = 8{,}5$
\end{enumerate}
\end{exobox}

\begin{exobox}[le produit ionique en action]
On donne $K_e = 1{,}0\times10^{-14}$ à \SI{25}{\celsius}.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
  \item Une solution a $[\ce{H3O+}] = 10^{-2}\ \si{mol.L^{-1}}$. Calculer directement $[\ce{OH-}]$ avec
        $[\ce{OH-}] = K_e/[\ce{H3O+}]$.
  \item Une solution a $[\ce{OH-}] = 10^{-4}\ \si{mol.L^{-1}}$. Calculer $[\ce{H3O+}]$.
  \item Dans l'eau pure neutre, on a $[\ce{H3O+}]=[\ce{OH-}]$. En déduire leur valeur commune à partir de $K_e$.
\end{enumerate}
\end{exobox}
