\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 2 --- Niveau Moyen}\\[2pt]
{\color{primary}\itshape Conversions croisées, facteur 10 par unité, effet de la température}
\end{center}

\begin{consigne}
\textbf{Objectif du niveau.} Naviguer entre les quatre grandeurs, exploiter le fait qu'\emph{une unité de pH
vaut un facteur 10}, et comprendre que le pH neutre dépend de $K_e$ donc de la température. \emph{On donne
$\log 2\approx 0{,}30$.}
\end{consigne}

\begin{exobox}[compléter la ligne complète d'une solution]
Une solution a $[\ce{H3O+}] = 2{,}0\times10^{-3}\ \si{mol.L^{-1}}$ à \SI{25}{\celsius}. Déterminer, dans
l'ordre : (a) son $\mathrm{pH}$ ; (b) son $\mathrm{pOH}$ ; (c) sa concentration $[\ce{OH-}]$. Conclure sur son
caractère (acide / neutre / basique).
\end{exobox}

\begin{exobox}[le sang et le facteur 10]
Le sang a un $\mathrm{pH} = 7{,}4$. On considère une deuxième solution de $\mathrm{pH} = 8{,}4$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=2pt]
  \item Sans calculer chaque concentration, dire par quel facteur $[\ce{H3O+}]$ du sang est plus grande que
        celle de la seconde solution.
  \item Vérifier en calculant le rapport $[\ce{H3O+}]_{\text{sang}}/[\ce{H3O+}]_{\text{sol.}}$ avec
        $[\ce{H3O+}]=10^{-\mathrm{pH}}$.
\end{enumerate}
\end{exobox}

\begin{exobox}[l'estomac mille fois plus acide]
Le suc gastrique passe d'un $\mathrm{pH}$ de $3$ (à jeun) à un $\mathrm{pH}$ de $6$ après un repas copieux.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=2pt]
  \item De combien d'unités le pH a-t-il varié ?
  \item Par quel facteur la concentration en \ce{H3O+} a-t-elle été divisée ?
  \item Reformuler : le milieu est-il devenu plus acide ou moins acide, et « combien de fois » ?
\end{enumerate}
\end{exobox}

\begin{exobox}[{$[\ce{H3O+}]$ monte, $[\ce{OH-}]$ descend}]
On travaille à température constante, donc $K_e$ est fixé.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=2pt]
  \item On ajoute un acide fort à de l'eau pure : $[\ce{H3O+}]$ augmente. Que devient $[\ce{OH-}]$ ?
        Justifier avec l'expression de $K_e$.
  \item Initialement $[\ce{H3O+}]=[\ce{OH-}]=10^{-7}$. Après ajout, $[\ce{H3O+}]=10^{-3}\ \si{mol.L^{-1}}$.
        Calculer la nouvelle valeur de $[\ce{OH-}]$.
\end{enumerate}
\end{exobox}

\begin{exobox}[quand la température change le neutre]
On chauffe de l'eau pure à une température $T$ où $K_e = 4\times10^{-14}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=2pt]
  \item En milieu neutre, $[\ce{H3O+}]=[\ce{OH-}]$. En déduire $[\ce{H3O+}]$ à cette température (utiliser
        $K_e=[\ce{H3O+}]^2$).
  \item En déduire le $\mathrm{pH}$ neutre à cette température.
  \item Une solution à $\mathrm{pH}=7$ est-elle acide, neutre ou basique à cette température ? Commenter.
\end{enumerate}
\end{exobox}
