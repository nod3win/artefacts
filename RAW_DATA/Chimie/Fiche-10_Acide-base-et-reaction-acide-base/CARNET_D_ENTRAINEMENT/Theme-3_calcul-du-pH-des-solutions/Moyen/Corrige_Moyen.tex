\begin{corrige}[mélange de deux acides forts]
\begin{enumerate}[label=\alph*)]
  \item $n_1 = 0,050 \times 0,10 = \qty{5,0e-3}{mol}$ ; $n_2 = 0,050 \times 0,30 = \qty{1,5e-2}{mol}$.
        Total : $n(\ce{H3O+}) = 0,0050 + 0,015 = \qty{2,0e-2}{mol}$.
  \item $V_{\text{tot}} = \qty{0,100}{L}$, donc $[\ce{H3O+}] = \dfrac{0,020}{0,100} = \qty{0,20}{mol\per L}$.
        \[ \mathrm{pH} = -\log(2{,}0\times10^{-1}) = 1 - \log 2 = 1 - 0{,}30 = \mathbf{0{,}70}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[base forte dibasique]
\begin{enumerate}[label=\alph*)]
  \item $\ce{Ba(OH)2 -> Ba^{2+} + 2 OH-}$ : chaque formule libère \textbf{deux} \ce{OH-}, donc
        $[\ce{OH-}] = 2 \times 5{,}0\times10^{-3} = \qty{1,0e-2}{mol\per L}$.
  \item $\mathrm{pOH} = -\log(10^{-2}) = 2$, d'où $\mathrm{pH} = 14 - 2 = \mathbf{12}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[dilution d'un acide fort]
\begin{enumerate}[label=\alph*)]
  \item $[\ce{H3O+}] = 10^{-\mathrm{pH}} = \qty{e-2}{mol\per L}$.
  \item Après dilution par \num{100} : $[\ce{H3O+}] = \dfrac{10^{-2}}{100} = \qty{e-4}{mol\per L}$, donc
        $\mathrm{pH} = \mathbf{4}$.
  \item Diluer un acide fort par un facteur $10^{n}$ \textbf{augmente le pH de $n$ unités} (ici $n=2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[acide faible : calculer et vérifier]
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = \tfrac12(\mathrm{p}K_a - \log C_a) = \tfrac12\bigl(4{,}0 + 1\bigr) = \mathbf{2{,}5}$.
  \item $\alpha = \sqrt{K_a/C_a} = \sqrt{\dfrac{10^{-4}}{10^{-1}}} = \sqrt{10^{-3}} \approx \qty{3,2e-2}{}
        = \qty{3,2}{\percent}$. Comme $\alpha \ll 1$ (l'acide est peu dissocié), l'approximation est bien
        justifiée.
\end{enumerate}
\end{corrige}

\begin{corrige}[reconnaître une base faible]
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH}_{\text{théo}} = 7 + \tfrac12\bigl(9{,}3 + \log(10^{-3})\bigr) = 7 + \tfrac12(9{,}3 - 3)
        = 7 + 3{,}15 = \mathbf{10{,}15}$.
  \item La valeur théorique ($10{,}15$) coïncide exactement avec la mesure : \ce{NH3} se comporte comme une
        \textbf{base faible}. (Une base forte de même concentration donnerait $\mathrm{pH} = 14 + \log(10^{-3})
        = 11$, valeur non observée.)
\end{enumerate}
\end{corrige}
