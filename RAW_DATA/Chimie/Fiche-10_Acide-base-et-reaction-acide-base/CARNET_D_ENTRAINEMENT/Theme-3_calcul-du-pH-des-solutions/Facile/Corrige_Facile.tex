\begin{corrige}[pH d'un acide fort]
$\mathrm{pH} = -\log C_a$.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = -\log(10^{-2}) = \mathbf{2}$.
  \item $\mathrm{pH} = -\log(10^{-1}) = \mathbf{1}$.
  \item $\mathrm{pH} = -\log(10^{-3}) = \mathbf{3}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[pH d'une base forte]
$\mathrm{pH} = 14 + \log C_b$.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = 14 + \log(10^{-1}) = 14 - 1 = \mathbf{13}$.
  \item $\mathrm{pH} = 14 + \log(10^{-2}) = 14 - 2 = \mathbf{12}$.
  \item $\mathrm{pH} = 14 + \log(10^{-3}) = 14 - 3 = \mathbf{11}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[pH d'un acide faible]
$\mathrm{pH} = \tfrac12(\mathrm{p}K_a - \log C_a)$.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = \tfrac12\bigl(4{,}8 - \log(10^{-1})\bigr) = \tfrac12(4{,}8 + 1) = \tfrac{5{,}8}{2} = \mathbf{2{,}9}$.
  \item $\mathrm{pH} = \tfrac12\bigl(3{,}2 - \log(10^{-2})\bigr) = \tfrac12(3{,}2 + 2) = \tfrac{5{,}2}{2} = \mathbf{2{,}6}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[pH d'une base faible]
$\mathrm{pH} = 7 + \tfrac12(\mathrm{p}K_a + \log C_b)$.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH} = 7 + \tfrac12\bigl(9{,}2 + \log(10^{-1})\bigr) = 7 + \tfrac12(9{,}2 - 1) = 7 + 4{,}1 = \mathbf{11{,}1}$.
  \item $\mathrm{pH} = 7 + \tfrac12\bigl(10{,}6 + \log(10^{-2})\bigr) = 7 + \tfrac12(10{,}6 - 2) = 7 + 4{,}3 = \mathbf{11{,}3}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[degré d'ionisation par Ostwald]
$\alpha \approx \sqrt{K_a/C_a}$.
\begin{enumerate}[label=\alph*)]
  \item $\alpha = \sqrt{\dfrac{10^{-5}}{10^{-1}}} = \sqrt{10^{-4}} = 10^{-2} = \qty{1,0}{\percent}$.
  \item $\alpha = \sqrt{\dfrac{10^{-4}}{10^{-2}}} = \sqrt{10^{-2}} = 10^{-1} = \qty{10}{\percent}$.
\end{enumerate}
\end{corrige}
