\begin{corrbox}[extraire les deux couples d'une réaction]
On regarde qui perd un \ce{H+} (acide $\to$ sa base conjuguée) et qui en gagne un (base $\to$ son acide conjugué).
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=3pt]
  \item Couple 1 : \ce{CH3COOH}/\ce{CH3COO-} (l'acide \ce{CH3COOH} cède son \ce{H+}).\\
        Couple 2 : \ce{NH4+}/\ce{NH3} (la base \ce{NH3} capte le \ce{H+} et devient \ce{NH4+}).
  \item Couple 1 : \ce{HCO3-}/\ce{CO3^{2-}} (l'acide \ce{HCO3-} cède son \ce{H+}).\\
        Couple 2 : \ce{H2O}/\ce{OH-} (la base \ce{OH-} capte le \ce{H+} et devient \ce{H2O}).
  \item Couple 1 : \ce{H2PO4-}/\ce{HPO4^{2-}} (l'acide \ce{H2PO4-} cède son \ce{H+}).\\
        Couple 2 : \ce{NH4+}/\ce{NH3} (la base \ce{NH3} capte le \ce{H+}).
\end{enumerate}
\end{corrbox}

\begin{corrbox}[la fausse paire]
La paire qui n'est pas un couple conjugué est \textbf{(b) \ce{H3O+}/\ce{OH-}}. En effet, pour passer de
\ce{H3O+} à sa base conjuguée on enlève \emph{un seul} \ce{H+} : \ce{H3O+} $\to$ \ce{H2O} (et non \ce{OH-}).
Pour passer de \ce{H3O+} à \ce{OH-}, il faudrait retirer \emph{deux} \ce{H+} : ce sont donc deux espèces
séparées par deux protons, pas un couple. \\
Les autres sont bien des couples : (a) \ce{HClO}/\ce{ClO-}, (c) \ce{H2PO4-}/\ce{HPO4^{2-}},
(d) \ce{NH4+}/\ce{NH3} (un seul \ce{H+} d'écart à chaque fois).
\end{corrbox}

\begin{corrbox}[\ce{HCO3-} est un ampholyte]
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=3pt]
  \item \ce{HCO3-} joue l'\textcolor{acidecol}{\textbf{acide}} : $\ce{HCO3- + H2O <=> CO3^{2-} + H3O+}$,
        couple \ce{HCO3-}/\ce{CO3^{2-}}.
  \item \ce{HCO3-} joue la \textcolor{basecol}{\textbf{base}} : $\ce{HCO3- + H3O+ <=> H2CO3 + H2O}$,
        couple \ce{H2CO3}/\ce{HCO3-}.
  \item \ce{HCO3-} peut \emph{à la fois} céder un \ce{H+} (rôle d'acide) et en capter un (rôle de base) : il
        appartient à deux couples différents, c'est donc un \textbf{ampholyte}.
\end{enumerate}
\end{corrbox}

\begin{corrbox}[les bornes de l'eau]
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=3pt]
  \item L'acide conjugué de \ce{H3O+} s'obtiendrait en ajoutant un \ce{H+} : on obtiendrait \ce{H4O^{2+}},
        espèce qui \textbf{n'existe pas} en solution aqueuse. Donc \ce{H3O+} n'a pas d'acide conjugué
        observable : c'est l'acide le plus fort possible dans l'eau.
  \item L'eau peut céder un \ce{H+} : couple \ce{H2O}/\ce{OH-} (elle joue l'acide) ; et capter un \ce{H+} :
        couple \ce{H3O+}/\ce{H2O} (elle joue la base). Appartenant à ces deux couples, \ce{H2O} est un
        \textbf{ampholyte}.
\end{enumerate}
\end{corrbox}

\begin{corrbox}[ampholytes de la famille du phosphate]
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=3pt]
  \item Les ampholytes sont les espèces \emph{intermédiaires} : \textbf{\ce{H2PO4-}} et \textbf{\ce{HPO4^{2-}}}.
        Elles possèdent encore un \ce{H} à céder \emph{et} une charge négative capable de capter un \ce{H+}.
        En revanche \ce{H3PO4} (que du \ce{H} à céder) est seulement un acide, et \ce{PO4^{3-}} (plus aucun
        \ce{H}) est seulement une base.
  \item \ce{H2PO4-} appartient aux deux couples : \ce{H3PO4}/\ce{H2PO4-} (où elle est la base) et
        \ce{H2PO4-}/\ce{HPO4^{2-}} (où elle est l'acide).
\end{enumerate}
\end{corrbox}

\begin{astuce}
\textbf{Repérer un ampholyte en un coup d'œil.} Dans une chaîne de déprotonations
$\ce{H3A} \to \ce{H2A-} \to \ce{HA^{2-}} \to \ce{A^{3-}}$, les espèces \emph{aux extrémités} ne jouent qu'un
rôle (acide pur à gauche, base pure à droite) ; toutes les espèces \emph{du milieu} sont des ampholytes.
\end{astuce}
