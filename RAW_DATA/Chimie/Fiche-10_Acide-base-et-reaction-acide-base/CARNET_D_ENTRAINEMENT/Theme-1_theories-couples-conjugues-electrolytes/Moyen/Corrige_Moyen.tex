\begin{corrige}[le regard de Lewis]
\begin{enumerate}[label=\alph*)]
  \item Dans \ce{NH3 + BF3} : l'azote de \ce{NH3} possède un doublet libre qu'il \emph{donne} $\Rightarrow$
        \ce{NH3} est la \textbf{base de Lewis} ; le bore de \ce{BF3} a une \textbf{lacune} électronique et
        \emph{accepte} ce doublet $\Rightarrow$ \ce{BF3} est l'\textbf{acide de Lewis}. \\
        Dans \ce{Ag+ + 2 NH3} : \ce{NH3} \emph{donne} ses doublets $\Rightarrow$ \textbf{base de Lewis} ;
        \ce{Ag+} les \emph{accepte} $\Rightarrow$ \textbf{acide de Lewis}.
  \item Aucune de ces réactions ne met en jeu un transfert de proton \ce{H+} : \ce{BF3} et \ce{Ag+} ne cèdent ni
        ne captent d'\ce{H+}. Elles échappent donc à Brønsted-Lowry, mais entrent dans le cadre \emph{plus
        général} de Lewis (échange de doublet).
\end{enumerate}
\end{corrige}

\begin{corrige}[montrer qu'une espèce est un ampholyte]
\begin{enumerate}[label=\alph*)]
  \item En \textbf{acide} (couple \ce{HCO3-/CO3^{2-}}) : $\ce{HCO3- + H2O <=> CO3^{2-} + H3O+}$.
  \item En \textbf{base} (couple \ce{H2CO3/HCO3-}) : $\ce{HCO3- + H3O+ <=> H2CO3 + H2O}$ \\
        (ou, de façon équivalente, $\ce{HCO3- + H2O <=> H2CO3 + OH-}$).
  \item \ce{HCO3-} peut à la fois \emph{céder} un \ce{H+} (a) et en \emph{capter} un (b) : il se comporte tantôt
        en acide, tantôt en base. C'est la définition d'un \textbf{ampholyte}.
\end{enumerate}
\end{corrige}

\begin{corrige}[les couples successifs d'un polyacide]
\begin{enumerate}[label=\alph*)]
  \item On retire les \ce{H+} un par un : \[ \ce{H3PO4/H2PO4-},\qquad \ce{H2PO4-/HPO4^{2-}},\qquad \ce{HPO4^{2-}/PO4^{3-}}. \]
  \item Une espèce est ampholyte si elle est \emph{à la fois} acide d'un couple et base d'un autre. Ce sont les
        espèces \textbf{intermédiaires} : \textbf{\ce{H2PO4-}} et \textbf{\ce{HPO4^{2-}}}. \\
        (\ce{H3PO4} n'est qu'acide ; \ce{PO4^{3-}} n'est que base.)
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer un degré d'ionisation]
$\alpha = \dfrac{\text{quantité dissociée}}{\text{quantité dissoute}}$.
\begin{enumerate}[label=\alph*)]
  \item $\alpha = \dfrac{0,015}{0,50} = 0,030 = \qty{3,0}{\percent}$.
  \item La fraction dissociée vaut $\alpha C_0$ :
        \[ [\ce{A-}] = \alpha\,C_0 = 0,020 \times 0,10 = \qty{2,0e-3}{mol\per L}. \]
        Le reste est non dissocié :
        \[ [\ce{HA}] = C_0 - [\ce{A-}] = 0,10 - 0,002 = \qty{0,098}{mol\per L}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[repérer les deux couples d'une réaction]
\begin{enumerate}[label=\alph*)]
  \item \ce{CH3COOH} et \ce{CH3COO-} forment le couple \ce{CH3COOH/CH3COO-} ; \ce{NH4+} et \ce{NH3} forment le
        couple \ce{NH4+/NH3}.
  \item \ce{CH3COOH} (acide du 1\textsuperscript{er} couple) \emph{cède} son \ce{H+} ; \ce{NH3} (base du
        2\textsuperscript{e} couple) le \emph{capte}. On obtient \ce{CH3COO-} et \ce{NH4+}.
\end{enumerate}
\end{corrige}
