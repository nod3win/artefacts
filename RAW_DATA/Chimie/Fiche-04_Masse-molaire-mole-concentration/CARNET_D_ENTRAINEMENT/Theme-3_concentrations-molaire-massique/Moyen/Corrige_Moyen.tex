\begin{center}{\large\bfseries\color{primarydark} Thème 3 — Corrigé Moyen}\end{center}

\begin{corrige}[quelle masse peser~?]
On calcule $n = C\,V$ puis $m = n\,M$.
\begin{enumerate}[label=\alph*)]
  \item $n = 0{,}2\times 0{,}250 = \SI{0.05}{\mole}$, puis $m = 0{,}05\times 58{,}5 = \SI{2.93}{\gram}$.
  \item $n = 0{,}1\times 0{,}500 = \SI{0.05}{\mole}$, puis $m = 0{,}05\times 180 = \SI{9.00}{\gram}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[de la concentration massique à la molarité]
La concentration massique se lit directement ($C_m = \SI{10.29}{\gram\per\litre}$, resp. $\SI{40}{\gram\per\litre}$),
puis $C = C_m/M$.
\begin{enumerate}[label=\alph*)]
  \item $C = \dfrac{10{,}29}{102{,}9} = \SI{0.100}{\mole\per\litre}$.
  \item $C = \dfrac{40}{40} = \SI{1.00}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[la plus concentrée]
On ramène tout à $C = n/V$ (en convertissant d'abord la masse de a) en moles).
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{117}{58{,}5} = \SI{2}{\mole}$, $C = \dfrac{2}{2} = \SI{1.00}{\mole\per\litre}$.
  \item $C = \dfrac{1{,}5}{2} = \SI{0.75}{\mole\per\litre}$.
  \item $\SI{250}{\milli\litre} = \SI{0.25}{\litre}$ : $C = \dfrac{0{,}5}{0{,}25} = \SI{2.00}{\mole\per\litre}$.
  \item $C = \dfrac{3}{4} = \SI{0.75}{\mole\per\litre}$.
\end{enumerate}
La plus concentrée est \textbf{c)} à $\SI{2.00}{\mole\per\litre}$.
\end{corrige}

\begin{corrige}[une application pharmaceutique]
\begin{enumerate}[label=\alph*)]
  \item $\SI{755}{\milli\gram} = \SI{0.755}{\gram}$ : $n = \dfrac{0{,}755}{151} = \SI{5.00e-3}{\mole}$.
  \item $\SI{200}{\milli\litre} = \SI{0.200}{\litre}$ : $C = \dfrac{5{,}00\times10^{-3}}{0{,}200} = \SI{2.50e-2}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[parcours dans la solution de glucose]
\begin{enumerate}[label=\alph*)]
  \item $C = \dfrac{C_m}{M} = \dfrac{18}{180} = \SI{0.100}{\mole\per\litre}$.
  \item $n = C\,V = 0{,}100\times 0{,}250 = \SI{2.50e-2}{\mole}$.
  \item $m = n\,M = 0{,}025\times 180 = \SI{4.50}{\gram}$ \ (ou directement $m = C_m\,V = 18\times 0{,}250 = \SI{4.50}{\gram}$).
\end{enumerate}
\end{corrige}
