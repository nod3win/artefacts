\begin{center}{\large\bfseries\color{primarydark} Thème 3 — Niveau Difficile}\end{center}
\emph{Données (\si{\gram\per\mole}) :} \ce{H2SO4}$=98$, \ce{NaOH}$=40$, \ce{NaF}$=42$,
\ce{HNO3}$=63$. \quad $\rho_{\text{eau}} = \SI{1.00}{\gram\per\milli\litre}$.

\begin{exo}[molarité d'un acide concentré]
L'acide sulfurique commercial est à $98\,\%$ en masse de \ce{H2SO4} et sa masse volumique vaut
$\rho = \SI{2.00}{\gram\per\milli\litre}$. Quelle est sa concentration molaire~?
\emph{(Raisonne sur $\SI{1}{\litre}$ de solution.)}
\end{exo}

\begin{exo}[le raisonnement inverse]
Une solution de soude a une concentration molaire $C = \SI{5.00}{\mole\per\litre}$ et une masse
volumique $\rho = \SI{1.10}{\gram\per\milli\litre}$ ($M(\ce{NaOH}) = \SI{40}{\gram\per\mole}$).
Quel est son pourcentage massique en \ce{NaOH}~?
\end{exo}

\begin{exo}[du pourcentage massique à la molarité]
Une solution de fluorure de sodium est à $21\,\%$ en masse de \ce{NaF} ; sa masse volumique est
pratiquement celle de l'eau ($\rho = \SI{1.00}{\gram\per\milli\litre}$). Quelle est sa molarité~?
\end{exo}

\begin{exo}[préparer une concentration massique visée]
L'éthylène glycol pur a une masse volumique $\rho = \SI{1.10}{\gram\per\milli\litre}$. On veut
préparer $\SI{1.00}{\litre}$ d'une solution aqueuse à $C_m = \SI{550}{\gram\per\litre}$ d'éthylène
glycol.
\begin{enumerate}[label=\alph*)]
  \item Quelle masse d'éthylène glycol faut-il~?
  \item Quel volume de glycol pur cela représente-t-il~?
\end{enumerate}
\end{exo}

\begin{exo}[acide nitrique concentré]
L'acide nitrique commercial est à $63\,\%$ en masse de \ce{HNO3}, de masse volumique
$\rho = \SI{1.40}{\gram\per\milli\litre}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle est sa concentration molaire~?
  \item Quelle masse de \ce{HNO3} pur contient $\SI{100}{\milli\litre}$ de cette solution~?
  \item Quel volume de solution contient exactement $\SI{1}{\mole}$ de \ce{HNO3}~?
\end{enumerate}
\end{exo}
