\begin{center}{\large\bfseries\color{primarydark} Thème 5 — Niveau Difficile}\end{center}
\emph{Données (\si{\gram\per\mole}) :} \ce{H2O}$=18$, glycérol \ce{C3H8O3}$=92$, éthanol \ce{C2H6O}$=46$,
\ce{NaOH}$=40$, \ce{NaCl}$=58{,}5$.

\begin{exo}[de la solution commerciale à la solution diluée]
On dispose de \ce{NaOH} à $20\,\%$ en masse, de masse volumique $\rho = \SI{1.10}{\gram\per\milli\litre}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle est la concentration molaire $C_1$ de cette solution mère~?
  \item Quel volume $V_1$ prélever pour préparer $\SI{250}{\milli\litre}$ de \ce{NaOH} à $\SI{0.50}{\mole\per\litre}$~?
\end{enumerate}
\end{exo}

\begin{exo}[dilution en fonction du volume prélevé]
On prélève un volume $V_1$ d'une solution de concentration $C_0$, que l'on porte à un volume total
$V_2 = 4\,V_1$ avec de l'eau.
\begin{enumerate}[label=\alph*)]
  \item Exprime la concentration finale $C_2$ en fonction de $C_0$.
  \item Application : si $C_0 = \SI{1.2}{\mole\per\litre}$, que vaut $C_2$~?
\end{enumerate}
\end{exo}

\begin{exo}[mélange à trois constituants]
Une solution contient $\SI{36}{\gram}$ d'eau \ce{H2O}, $\SI{92}{\gram}$ de glycérol \ce{C3H8O3} et
$\SI{46}{\gram}$ d'éthanol \ce{C2H6O}.
\begin{enumerate}[label=\alph*)]
  \item Calcule la quantité de matière de chaque constituant.
  \item En déduire les trois fractions molaires et vérifie que leur somme vaut $1$.
\end{enumerate}
\end{exo}

\begin{exo}[molarité \emph{ou} molalité~?]
On dissout $\SI{0.50}{\mole}$ de \ce{NaCl} dans $\SI{1.0}{\kilo\gram}$ d'eau. La solution obtenue a
une masse volumique $\rho = \SI{1.02}{\gram\per\milli\litre}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle est la molalité de la solution~?
  \item Quelle est la masse totale de la solution~?
  \item Quel est son volume~?
  \item En déduire la molarité. Est-elle égale à la molalité~?
\end{enumerate}
\end{exo}

\begin{exo}[mélange de deux solutions du même soluté]
On mélange $\SI{200}{\milli\litre}$ de glucose à $\SI{0.30}{\mole\per\litre}$ et $\SI{300}{\milli\litre}$
de glucose à $\SI{0.80}{\mole\per\litre}$ (les volumes s'additionnent). Quelle est la concentration
molaire finale~?
\end{exo}
