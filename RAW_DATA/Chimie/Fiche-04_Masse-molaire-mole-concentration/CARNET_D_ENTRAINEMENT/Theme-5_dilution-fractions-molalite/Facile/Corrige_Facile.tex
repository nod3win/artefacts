\begin{center}{\large\bfseries\color{primarydark} Thème 5 — Corrigé Facile}\end{center}

\begin{corrige}[concentration après dilution]
$C_2 = C_1 V_1 / V_2$ (les volumes en même unité).
\begin{enumerate}[label=\alph*)]
  \item $C_2 = \dfrac{1\times 10}{100} = \SI{0.10}{\mole\per\litre}$.
  \item $C_2 = \dfrac{0{,}5\times 20}{200} = \SI{0.05}{\mole\per\litre}$.
  \item $C_2 = \dfrac{2\times 25}{250} = \SI{0.20}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[volume de mère à prélever]
$V_1 = C_2 V_2 / C_1$.
\begin{enumerate}[label=\alph*)]
  \item $V_1 = \dfrac{0{,}1\times 100}{1} = \SI{10}{\milli\litre}$.
  \item $V_1 = \dfrac{0{,}2\times 250}{2} = \SI{25}{\milli\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[facteur de dilution]
\begin{enumerate}[label=\alph*)]
  \item Diluer $3$ fois divise la concentration par $3$ : $C_2 = \dfrac{1{,}5}{3} = \SI{0.5}{\mole\per\litre}$.
  \item Diluer $4$ fois multiplie le volume par $4$ : $V_2 = 4\times 50 = \SI{200}{\milli\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[fraction molaire]
\begin{enumerate}[label=\alph*)]
  \item $n_{\text{tot}} = 5$ : $\chi_{\ce{A}} = \dfrac{2}{5} = 0{,}40$, $\chi_{\ce{B}} = \dfrac{3}{5} = 0{,}60$ (somme $=1$).
  \item $n_{\text{tot}} = 4$ : $\chi_{\ce{A}} = 0{,}25$, $\chi_{\ce{B}} = 0{,}25$, $\chi_{\ce{C}} = \dfrac{2}{4} = 0{,}50$ (somme $=1$).
\end{enumerate}
\end{corrige}

\begin{corrige}[molalité]
\begin{enumerate}[label=\alph*)]
  \item $m = \dfrac{0{,}5}{0{,}25} = \SI{2.0}{\mole\per\kilo\gram}$.
  \item $\SI{500}{\gram} = \SI{0.5}{\kilo\gram}$ : $m = \dfrac{0{,}2}{0{,}5} = \SI{0.4}{\mole\per\kilo\gram}$.
\end{enumerate}
\end{corrige}
