\begin{center}{\large\bfseries\color{primarydark} Thème 5 — Niveau Facile}\end{center}
\emph{Rappels :} dilution $C_1V_1 = C_2V_2$ ; facteur $F = V_2/V_1 = C_1/C_2$ ;
$\chi_{\ce{A}} = n_{\ce{A}}/n_{\text{tot}}$ ; molalité $m = n/m_{\text{solvant}}(\si{\kilo\gram})$.

\begin{exo}[concentration après dilution]
Calcule la concentration finale $C_2 = C_1V_1/V_2$.
\begin{enumerate}[label=\alph*)]
  \item $C_1 = \SI{1}{\mole\per\litre}$, $V_1 = \SI{10}{\milli\litre}$, $V_2 = \SI{100}{\milli\litre}$ ;
  \item $C_1 = \SI{0.5}{\mole\per\litre}$, $V_1 = \SI{20}{\milli\litre}$, $V_2 = \SI{200}{\milli\litre}$ ;
  \item $C_1 = \SI{2}{\mole\per\litre}$, $V_1 = \SI{25}{\milli\litre}$, $V_2 = \SI{250}{\milli\litre}$.
\end{enumerate}
\end{exo}

\begin{exo}[volume de mère à prélever]
Calcule le volume $V_1$ à prélever ($V_1 = C_2V_2/C_1$).
\begin{enumerate}[label=\alph*)]
  \item préparer $V_2 = \SI{100}{\milli\litre}$ à $C_2 = \SI{0.1}{\mole\per\litre}$ depuis $C_1 = \SI{1}{\mole\per\litre}$ ;
  \item préparer $V_2 = \SI{250}{\milli\litre}$ à $C_2 = \SI{0.2}{\mole\per\litre}$ depuis $C_1 = \SI{2}{\mole\per\litre}$.
\end{enumerate}
\end{exo}

\begin{exo}[facteur de dilution]
\begin{enumerate}[label=\alph*)]
  \item Une solution à $\SI{1.5}{\mole\per\litre}$ est diluée $3$ fois. Quelle est sa concentration finale~?
  \item On prélève $\SI{50}{\milli\litre}$ d'une solution que l'on dilue $4$ fois. Quel est le volume final~?
\end{enumerate}
\end{exo}

\begin{exo}[fraction molaire]
Calcule la fraction molaire de chaque constituant.
\begin{enumerate}[label=\alph*)]
  \item mélange de $\SI{2}{\mole}$ de \ce{A} et $\SI{3}{\mole}$ de \ce{B} ;
  \item mélange de $\SI{1}{\mole}$ de \ce{A}, $\SI{1}{\mole}$ de \ce{B} et $\SI{2}{\mole}$ de \ce{C}.
\end{enumerate}
\end{exo}

\begin{exo}[molalité]
Calcule la molalité $m = n/m_{\text{solvant}}$.
\begin{enumerate}[label=\alph*)]
  \item $\SI{0.5}{\mole}$ de soluté dans $\SI{0.25}{\kilo\gram}$ d'eau ;
  \item $\SI{0.2}{\mole}$ de soluté dans $\SI{500}{\gram}$ d'eau.
\end{enumerate}
\end{exo}
