\begin{center}{\large\bfseries\color{primarydark} Thème 2 — Corrigé Moyen}\end{center}

\begin{corrige}[chaîne volume $\to$ masse $\to$ moles (liquide)]
On enchaîne $V \xrightarrow{\times \rho} m \xrightarrow{\div M} n$ (avec $\rho = d\times\SI{1}{\gram\per\milli\litre}$).
\begin{enumerate}[label=\alph*)]
  \item $m = 1{,}70\times 100 = \SI{170}{\gram}$, puis $n = \dfrac{170}{85} = \SI{2.00}{\mole}$.
  \item $m = 1{,}00\times 50 = \SI{50}{\gram}$, puis $n = \dfrac{50}{18} = \SI{2.78}{\mole}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[masse d'un gaz]
On enchaîne $V \xrightarrow{\div V_m} n \xrightarrow{\times M} m$.
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{11{,}2}{22{,}4} = \SI{0.5}{\mole}$, puis $m = 0{,}5\times 32 = \SI{16.0}{\gram}$.
  \item $n = \dfrac{5{,}6}{22{,}4} = \SI{0.25}{\mole}$, puis $m = 0{,}25\times 44 = \SI{11.0}{\gram}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[nombre de molécules d'un gaz]
On enchaîne $V \xrightarrow{\div V_m} n \xrightarrow{\times N_{\mathrm{A}}} N$.
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{44{,}8}{22{,}4} = \SI{2}{\mole}$, puis $N = 2\times 6{,}02\times10^{23} = 1{,}20\times10^{24}$ molécules.
  \item $n = \dfrac{2{,}24}{22{,}4} = \SI{0.1}{\mole}$, puis $N = 0{,}1\times 6{,}02\times10^{23} = 6{,}02\times10^{22}$ atomes.
\end{enumerate}
\end{corrige}

\begin{corrige}[CNTP ou conditions standard~?]
\begin{enumerate}[label=\alph*)]
  \item $V = 0{,}5\times 22{,}4 = \SI{11.2}{\litre}$.
  \item $V = 0{,}5\times 24{,}4 = \SI{12.2}{\litre}$.
  \item Le même nombre de moles occupe un volume \emph{plus grand} à $\SI{25}{\celsius}$ (le gaz s'est
        dilaté en chauffant), d'où l'importance de vérifier les conditions avant d'utiliser
        $\SI{22.4}{\litre\per\mole}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[masse volumique d'un gaz]
Une mole occupe $\SI{22.4}{\litre}$ aux CNTP et pèse $M$ grammes, d'où $\rho = M/22{,}4$.
\begin{enumerate}[label=\alph*)]
  \item $\rho(\ce{O2}) = \dfrac{32}{22{,}4} = \SI{1.43}{\gram\per\litre}$.
  \item $\rho(\ce{CO2}) = \dfrac{44}{22{,}4} = \SI{1.96}{\gram\per\litre}$.
\end{enumerate}
Le \ce{CO2} est plus dense que le \ce{O2} : à volume égal, il est plus lourd.
\end{corrige}
