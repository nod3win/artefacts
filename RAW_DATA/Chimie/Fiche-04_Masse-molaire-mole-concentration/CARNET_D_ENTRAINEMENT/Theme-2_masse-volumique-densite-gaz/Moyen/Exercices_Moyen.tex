\begin{center}{\large\bfseries\color{primarydark} Thème 2 — Niveau Moyen}\end{center}
\emph{Données (\si{\gram\per\mole}) :} $\ce{H}=1$, $\ce{C}=12$, $\ce{O}=16$, $\ce{Cl}=35{,}5$,
$\ce{He}=4$. \quad $V_m = \SI{22.4}{\litre\per\mole}$ (CNTP), $N_{\mathrm{A}} = 6{,}02\times10^{23}\ \si{\per\mole}$.

\begin{exo}[chaîne volume $\to$ masse $\to$ moles (liquide)]
Détermine la quantité de matière contenue dans :
\begin{enumerate}[label=\alph*)]
  \item $\SI{100}{\milli\litre}$ de dichlorométhane \ce{CH2Cl2} ($d = 1{,}70$, $M = \SI{85}{\gram\per\mole}$) ;
  \item $\SI{50}{\milli\litre}$ d'eau \ce{H2O} ($d = 1{,}00$, $M = \SI{18}{\gram\per\mole}$).
\end{enumerate}
\end{exo}

\begin{exo}[masse d'un gaz]
Quelle est la masse des échantillons gazeux suivants, pris aux CNTP~?
\begin{enumerate}[label=\alph*)]
  \item $\SI{11.2}{\litre}$ de dioxygène \ce{O2} ($M = \SI{32}{\gram\per\mole}$) ;
  \item $\SI{5.6}{\litre}$ de dioxyde de carbone \ce{CO2} ($M = \SI{44}{\gram\per\mole}$).
\end{enumerate}
\end{exo}

\begin{exo}[nombre de molécules d'un gaz]
Combien de molécules contiennent, aux CNTP~:
\begin{enumerate}[label=\alph*)]
  \item $\SI{44.8}{\litre}$ de dihydrogène \ce{H2}~;
  \item $\SI{2.24}{\litre}$ d'hélium \ce{He}~?
\end{enumerate}
\end{exo}

\begin{exo}[CNTP ou conditions standard~?]
On considère $\SI{0.5}{\mole}$ d'un gaz parfait.
\begin{enumerate}[label=\alph*)]
  \item Quel volume occupe-t-il aux CNTP ($V_m = \SI{22.4}{\litre\per\mole}$)~?
  \item Quel volume occupe-t-il à $\SI{25}{\celsius}$ ($V_m = \SI{24.4}{\litre\per\mole}$)~?
  \item Commente l'écart.
\end{enumerate}
\end{exo}

\begin{exo}[masse volumique d'un gaz]
La masse volumique d'un gaz vaut $\rho = M / V_m$ (aux CNTP, $V$ en litres). Calcule $\rho$ (en \si{\gram\per\litre}) pour :
\begin{enumerate}[label=\alph*)]
  \item le dioxygène \ce{O2} ($M = \SI{32}{\gram\per\mole}$) ;
  \item le dioxyde de carbone \ce{CO2} ($M = \SI{44}{\gram\per\mole}$).
\end{enumerate}
\end{exo}
