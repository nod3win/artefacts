\begin{corrige}[calculer une masse volumique]
\begin{enumerate}[label=\alph*)]
  \item $\rho = \dfrac{24}{30} = \SI{0.80}{\gram\per\milli\litre}$.
  \item $\rho = \dfrac{79}{100} = \SI{0.79}{\gram\per\milli\litre}$.
  \item $\rho = \dfrac{13{,}55}{1} = \SI{13.55}{\gram\per\milli\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[de la masse volumique à la masse]
On convertit d'abord le volume en \si{\milli\litre}, puis $m = \rho\,V$.
\begin{enumerate}[label=\alph*)]
  \item $\SI{1}{\litre} = \SI{1000}{\milli\litre}$ : $m = 1{,}10\times 1000 = \SI{1100}{\gram}$.
  \item $m = 0{,}80\times 250 = \SI{200}{\gram}$.
  \item $\SI{2}{\litre} = \SI{2000}{\milli\litre}$ : $m = 1{,}20\times 2000 = \SI{2400}{\gram}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[densité et masse volumique]
On utilise $d = \rho / \rho_{\text{eau}}$ avec $\rho_{\text{eau}} = \SI{1}{\gram\per\milli\litre}$.
\begin{enumerate}[label=\alph*)]
  \item $\rho = d\times\rho_{\text{eau}} = 0{,}79\times 1 = \SI{0.79}{\gram\per\milli\litre}$.
  \item $d = \dfrac{13{,}6}{1} = 13{,}6$ (sans unité).
  \item $\rho = 1{,}03\times 1 = \SI{1.03}{\gram\per\milli\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[volume d'un gaz aux CNTP]
$V = n\times 22{,}4$.
\begin{enumerate}[label=\alph*)]
  \item $V = 1\times 22{,}4 = \SI{22.4}{\litre}$.
  \item $V = 0{,}5\times 22{,}4 = \SI{11.2}{\litre}$.
  \item $V = 0{,}25\times 22{,}4 = \SI{5.60}{\litre}$.
  \item $V = 2\times 22{,}4 = \SI{44.8}{\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[du volume de gaz à la quantité de matière]
$n = V / 22{,}4$.
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{22{,}4}{22{,}4} = \SI{1.00}{\mole}$.
  \item $n = \dfrac{11{,}2}{22{,}4} = \SI{0.500}{\mole}$.
  \item $n = \dfrac{44{,}8}{22{,}4} = \SI{2.00}{\mole}$.
  \item $n = \dfrac{5{,}6}{22{,}4} = \SI{0.250}{\mole}$.
\end{enumerate}
\end{corrige}
