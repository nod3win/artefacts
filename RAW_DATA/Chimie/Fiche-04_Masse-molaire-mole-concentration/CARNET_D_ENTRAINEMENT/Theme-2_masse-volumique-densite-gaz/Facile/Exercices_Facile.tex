\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 2 --- Niveau Facile}\\[2pt]
{\color{primary}\itshape Masse volumique, densité et volume molaire}
\end{center}

\begin{donnees}
$V_m=\SI{22.4}{\litre\per\mole}$ aux CNTP.\quad $\rho_{\text{eau}}=\SI{1}{\gram\per\milli\litre}$.\quad
$M(\ce{H2})=\SI{2}{\gram\per\mole}$.
\end{donnees}

\begin{exo}[masse volumique à partir d'une pesée]
Au laboratoire, on prélève $\SI{30}{\milli\litre}$ d'urine à l'aide d'une seringue pesant
initialement $\SI{50.00}{\gram}$. Après le prélèvement, la seringue pèse $\SI{80.36}{\gram}$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Quelle est la masse d'urine prélevée ?
  \item En déduire la masse volumique de l'urine, exprimée en \si{\gram\per\milli\litre}, puis en
        \si{\gram\per\litre}.
\end{enumerate}
\end{exo}

\begin{exo}[de la masse volumique à la densité]
Le mercure a une masse volumique $\rho=\SI{13.6}{\gram\per\milli\litre}$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Quelle est la densité du mercure ?
  \item Quelle est la masse d'un litre de mercure ?
\end{enumerate}
\end{exo}

\begin{exo}[du volume à la masse]
On verse $\SI{250}{\milli\litre}$ d'éthanol, de masse volumique $\rho=\SI{0.79}{\gram\per\milli\litre}$,
dans un bécher. Quelle est la masse d'éthanol versée ?
\end{exo}

\begin{exo}[volume molaire d'un gaz]
Tous les gaz sont considérés aux CNTP.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Quel volume occupent $\SI{0.250}{\mole}$ de dioxyde de carbone \ce{CO2} ?
  \item Quel volume occupent $\SI{2.0}{\gram}$ de dihydrogène \ce{H2} ?
\end{enumerate}
\end{exo}

\begin{exo}[conversions d'unités de masse volumique]
Un solide a une masse volumique $\rho=\SI{2.0}{\gram\per\cubic\centi\metre}$. Exprimer cette valeur :
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item en \si{\gram\per\milli\litre} ;
  \item en \si{\gram\per\litre} ;
  \item en \si{\kilo\gram\per\litre}.
\end{enumerate}
\end{exo}
