\begin{tcolorbox}[boxbase,colback=primary!12,colframe=primary,
   title={\Large THÈME 2 --- Masse volumique, densité et volume molaire des gaz}]
\textit{À lire avant les exercices.} On ne pèse pas toujours : souvent on \textbf{mesure un volume}.
Ce thème relie le \textbf{volume} aux autres grandeurs par \emph{deux ponts} : la \textbf{masse
volumique} $\rho$ (pour les liquides et solides) et le \textbf{volume molaire} $V_m$ (pour les gaz).
\end{tcolorbox}

\section*{1. Masse volumique : relier un volume à une masse}
\begin{formule}[masse volumique]
\[ \boxed{\,\rho=\dfrac{m}{V}\,}\qquad\Longleftrightarrow\qquad m=\rho\,V. \]
Unités usuelles : \si{\gram\per\milli\litre}, \si{\gram\per\cubic\centi\metre} (identiques, car
$\SI{1}{\milli\litre}=\SI{1}{\cubic\centi\metre}$), ou \si{\kilo\gram\per\cubic\metre}.
\end{formule}
\textbf{Traduction en phrase} (l'astuce n$^\circ$1) : «~$\rho=x\ \si{\gram\per\milli\litre}$~» veut
dire «~$\SI{1}{\milli\litre}$ pèse $x$~g~», donc «~$\SI{1}{\litre}$ pèse $1000x$~g~». On part
toujours d'un volume connu dont on déduit la masse.

\begin{attention}[conversions d'unités]
$\SI{1}{\gram\per\milli\litre}=\SI{1}{\gram\per\cubic\centi\metre}=\SI{1}{\kilo\gram\per\litre}
=\SI{1000}{\gram\per\litre}=\SI{1}{\kilo\gram\per\cubic\deci\metre}$.
Beaucoup d'énoncés « piègent » sur les unités (hg, \si{\cubic\deci\metre}, kg\ldots) : convertir
\emph{d'abord} tout en g et mL (ou en g et L).
\end{attention}

\begin{definition}[densité]
La \textbf{densité} $d$ (sans unité) est le rapport de la masse volumique du corps à celle de l'eau
($\rho_{\text{eau}}=\SI{1}{\gram\per\milli\litre}$) :
$\;d=\rho_{\text{corps}}/\rho_{\text{eau}}$. Numériquement, $d$ vaut la même valeur que $\rho$
exprimée en \si{\gram\per\milli\litre} (ex. $d=0{,}8\Leftrightarrow\rho=\SI{0.8}{\gram\per\milli\litre}$).
\end{definition}

\textbf{Mesure par déplacement d'eau :} un solide non soluble plongé dans un cylindre gradué fait
monter le niveau ; l'\emph{augmentation} de volume est égale au \textbf{volume du solide}.

\section*{2. Le cas des gaz : le volume molaire}
Pour un gaz parfait, à température et pression fixées, \textbf{une mole de n'importe quel gaz occupe
le même volume} $V_m$, indépendamment de sa nature.
\begin{formule}[volume molaire]
\[ \boxed{\,V=n\times V_m\,}\qquad\Longleftrightarrow\qquad n=\dfrac{V}{V_m}. \]
Aux \textbf{CNTP} ($\SI{0}{\celsius}$, $\SI{1}{\atm}$) : $V_m=\SI{22.4}{\litre\per\mole}$.
En conditions \textbf{standard} ($\SI{25}{\celsius}$, $\SI{1}{\bar}$) : $V_m\approx\SI{24.4}{\litre\per\mole}$.
\end{formule}
\begin{attention}[CNTP $\neq$ conditions standard]
La valeur $\SI{22.4}{\litre\per\mole}$ n'est valable qu'aux CNTP ($\SI{0}{\celsius}$). Toujours
vérifier les conditions de l'énoncé. La masse volumique d'un gaz se déduit de
$\rho_{\text{gaz}}=M/V_m$ (masse d'une mole divisée par le volume d'une mole).
\end{attention}

\begin{methode}[réflexes du thème]
\begin{enumerate}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item \textbf{Liquide/solide} : le volume passe par la \emph{masse} via $\rho$ ($m=\rho V$).
  \item \textbf{Gaz} : le volume passe directement par la \emph{quantité} via $V_m$ ($n=V/V_m$).
  \item Tout convertir en unités cohérentes \textbf{avant} de calculer (g, mL, L).
  \item Pour un solide immergé, le volume est l'\textbf{augmentation} du niveau, pas le niveau final.
\end{enumerate}
\end{methode}
