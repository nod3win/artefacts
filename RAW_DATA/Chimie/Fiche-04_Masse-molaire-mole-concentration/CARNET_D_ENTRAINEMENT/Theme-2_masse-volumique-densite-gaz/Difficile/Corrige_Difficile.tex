\begin{corr}[attribuer trois solides à leur cylindre]
\textbf{a)} Le volume de chaque solide se déduit de $V=m/\rho$ (avec $m=\SI{20.0}{\gram}$) :
\[ V(\ce{X})=\frac{20}{4{,}0}=\SI{5.0}{\milli\litre};\quad
   V(\ce{Y})=\frac{20}{10{,}0}=\SI{2.0}{\milli\litre};\quad
   V(\ce{Z})=\frac{20}{2{,}0}=\SI{10.0}{\milli\litre}. \]
\textbf{b)} Le niveau final vaut $\SI{10.0}{\milli\litre}$ (eau) $+$ volume du solide :
\[ \ce{X}:10+5=\SI{15}{\milli\litre};\quad \ce{Y}:10+2=\SI{12}{\milli\litre};\quad
   \ce{Z}:10+10=\SI{20}{\milli\litre}. \]
\textbf{c)} On rapproche des lectures : (a)$=\SI{15}{\milli\litre}$, (b)$=\SI{12}{\milli\litre}$,
(c)$=\SI{20}{\milli\litre}$ :
\[ \boxed{\ce{X}\leftrightarrow(a),\quad \ce{Y}\leftrightarrow(b),\quad \ce{Z}\leftrightarrow(c)} \]
\textit{Idée-clé :} plus un solide est \emph{dense}, plus son volume est petit à masse égale, donc
moins il fait monter l'eau (\ce{Y}, le plus dense, monte le moins).
\end{corr}

\begin{corr}[combien de moles de dioxygène dans une pièce ?]
\textbf{a)} $V_{\text{pièce}}=2\times2\times5{,}6=\SI{22.4}{\cubic\metre}$. Comme
$\SI{1}{\cubic\metre}=\SI{1000}{\litre}$ : $V_{\text{pièce}}=\SI{22400}{\litre}$.\\[2pt]
\textbf{b)} Le dioxygène occupe $\SI{20}{\percent}$ du volume :
$V(\ce{O2})=0{,}20\times22400=\SI{4480}{\litre}$.\\[2pt]
\textbf{c)} $n(\ce{O2})=\dfrac{V}{V_m}=\dfrac{4480}{22{,}4}=\SI{200}{\mole}$.
\[ \boxed{n(\ce{O2})=\SI{200}{\mole}} \]
\end{corr}

\begin{corr}[de la masse volumique d'un gaz à son identité]
\textbf{a)} Une mole occupe $V_m$ et a une masse $M$ ; sa masse volumique est donc
\[ \rho=\frac{m}{V}=\frac{M}{V_m}. \]
\textbf{b)} On isole $M$ : $M=\rho\times V_m=1{,}25\times22{,}4=\SI{28}{\gram\per\mole}$. Une masse
molaire de $\SI{28}{\gram\per\mole}$ correspond par exemple au \textbf{diazote \ce{N2}}
($2\times14=28$) ou au monoxyde de carbone \ce{CO} ($12+16=28$).\\[2pt]
\textbf{c)} $m=\rho\,V=1{,}25\times10{,}0=\SI{12.5}{\gram}$ (ou :
$n=\tfrac{10}{22{,}4}=\SI{0.446}{\mole}$, $m=nM=\SI{12.5}{\gram}$).
\[ \boxed{M=\SI{28}{\gram\per\mole}\ (\ce{N2}\ \text{ou}\ \ce{CO});\quad m(\SI{10}{\litre})=\SI{12.5}{\gram}} \]
\end{corr}

\begin{corr}[même masse, volumes différents]
\textbf{a)} $\ce{CH4}$ : $M=12+4=\SI{16}{\gram\per\mole}$, $n=\tfrac{32}{16}=\SI{2.0}{\mole}$,
$V=2{,}0\times22{,}4=\SI{44.8}{\litre}$.\\
$\ce{O2}$ : $M=\SI{32}{\gram\per\mole}$, $n=\tfrac{32}{32}=\SI{1.0}{\mole}$,
$V=1{,}0\times22{,}4=\SI{22.4}{\litre}$.\\[2pt]
\textbf{b)} Le méthane occupe le plus grand volume ($\SI{44.8}{\litre}$).
\[ \boxed{V(\ce{CH4})=\SI{44.8}{\litre}>V(\ce{O2})=\SI{22.4}{\litre}} \]
\textit{Justification :} à masse égale, la molécule la plus \emph{légère} (\ce{CH4}) fournit \emph{plus
de moles}, donc plus de volume gazeux --- le volume d'un gaz ne dépend que de la quantité de matière.
\end{corr}
