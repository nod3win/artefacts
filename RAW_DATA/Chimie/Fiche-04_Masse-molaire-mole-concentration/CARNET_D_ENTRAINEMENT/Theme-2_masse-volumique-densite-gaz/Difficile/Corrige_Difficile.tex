\begin{center}{\large\bfseries\color{primarydark} Thème 2 — Corrigé Difficile}\end{center}

\begin{corrige}[identifier un gaz par sa masse molaire]
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{V}{V_m} = \dfrac{5{,}6}{22{,}4} = \SI{0.250}{\mole}$.
  \item $M = \dfrac{m}{n} = \dfrac{8{,}5}{0{,}250} = \SI{34}{\gram\per\mole}$.
  \item $M(\ce{H2X}) = 2\times 1 + M(\ce{X}) = 34 \Rightarrow M(\ce{X}) = 32$ : c'est le \textbf{soufre}.
        Le gaz est le sulfure d'hydrogène \ce{H2S}.
\end{enumerate}
\end{corrige}

\begin{corrige}[le piège des unités]
\begin{enumerate}[label=\alph*)]
  \item L'eau ajoutée pèse $21{,}45 - 13{,}45 = \SI{8.00}{\gram}$ ; comme $\rho_{\text{eau}} = \SI{1}{\gram\per\milli\litre}$,
        le volume intérieur est $V = \SI{8.00}{\milli\litre}$.
  \item Masse du liquide : $57{,}89 - 13{,}45 = \SI{44.44}{\gram}$, donc
        $\rho = \dfrac{44{,}44}{8{,}00} = \SI{5.56}{\gram\per\milli\litre}$.
  \item $\SI{1}{\milli\litre} = \SI{1}{\cubic\centi\metre}$, donc $\rho = \SI{5.56}{\gram\per\cubic\centi\metre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[jongler avec les unités de masse]
\begin{enumerate}[label=\alph*)]
  \item $\SI{7.72}{\hecto\gram} = \SI{772}{\gram}$, donc $V = \dfrac{m}{\rho} = \dfrac{772}{19{,}3} = \SI{40.0}{\cubic\centi\metre}$.
  \item $\SI{1}{\cubic\centi\metre} = \SI{1}{\milli\litre}$, donc $V = \SI{40.0}{\milli\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[du solide aux entités]
\begin{enumerate}[label=\alph*)]
  \item Volume du cube : $V = 2^3 = \SI{8}{\cubic\centi\metre}$, donc $m = \rho\,V = 2{,}0\times 8 = \SI{16}{\gram}$.
  \item $n = \dfrac{m}{M} = \dfrac{16}{40} = \SI{0.40}{\mole}$.
  \item $N = n\,N_{\mathrm{A}} = 0{,}40\times 6{,}02\times10^{23} = 2{,}41\times10^{23}$ motifs \ce{NaOH}.
\end{enumerate}
\end{corrige}

\begin{corrige}[comparer deux liquides]
\begin{enumerate}[label=\alph*)]
  \item Masses : éthanol $m = 0{,}79\times 50 = \SI{39.5}{\gram}$ ; eau $m = 1{,}00\times 50 = \SI{50}{\gram}$.
        L'\textbf{eau} est plus lourde à volume égal.
  \item $n(\text{éthanol}) = \dfrac{39{,}5}{46} = \SI{0.859}{\mole}$ ; $n(\ce{H2O}) = \dfrac{50}{18} = \SI{2.78}{\mole}$.
  \item L'\textbf{eau} contient le plus de molécules. Bien que l'éthanol et l'eau soient comparés à
        volume égal, la molécule d'eau est bien plus légère ($M$ plus petit) : à masse voisine, l'eau
        fournit donc beaucoup plus de moles — et de molécules.
\end{enumerate}
\end{corrige}
