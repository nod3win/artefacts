\begin{donnees}
Masses molaires atomiques (\si{\gram\per\mole}) : $\ce{H}=1$ ; $\ce{C}=12$ ; $\ce{N}=14$ ;
$\ce{O}=16$.\quad $V_m=\SI{22.4}{\litre\per\mole}$ aux CNTP.
\end{donnees}

\begin{exo}[attribuer trois solides à leur cylindre]
On plonge séparément trois solides non solubles \ce{X}, \ce{Y} et \ce{Z}, tous de masse
$\SI{20.0}{\gram}$, dans trois cylindres gradués contenant chacun $\SI{10.0}{\milli\litre}$ d'eau.
Les niveaux finals lus sont représentés ci-dessous :
\begin{center}
\begin{tikzpicture}
  \cylindre{0}{15}{a}
  \cylindre{3.2}{12}{b}
  \cylindre{6.4}{20}{c}
\end{tikzpicture}
\end{center}
Les masses volumiques des solides sont $\rho(\ce{X})=\SI{4.0}{\gram\per\cubic\centi\metre}$,
$\rho(\ce{Y})=\SI{10.0}{\gram\per\cubic\centi\metre}$ et
$\rho(\ce{Z})=\SI{2.0}{\gram\per\cubic\centi\metre}$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Calculer le volume de chacun des trois solides.
  \item En déduire le niveau d'eau final attendu dans chaque cas.
  \item Attribuer à chaque solide (\ce{X}, \ce{Y}, \ce{Z}) le cylindre correspondant (a, b ou c).
\end{enumerate}
\end{exo}

\begin{exo}[combien de moles de dioxygène dans une pièce ?]
On considère une pièce de dimensions $\SI{2}{\metre}\times\SI{2}{\metre}\times\SI{5.6}{\metre}$,
remplie d'air aux CNTP. L'air est constitué de $\SI{20}{\percent}$ de dioxygène et
$\SI{80}{\percent}$ de diazote (pourcentages \emph{volumiques}).
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Calculer le volume de la pièce, puis l'exprimer en litres.
  \item En déduire le volume de dioxygène qu'elle contient.
  \item Calculer la quantité de matière (nombre de moles) de dioxygène.
\end{enumerate}
\end{exo}

\begin{exo}[de la masse volumique d'un gaz à son identité]
Aux CNTP, un gaz a une masse volumique $\rho=\SI{1.25}{\gram\per\litre}$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item En considérant une mole de ce gaz, exprimer $\rho$ en fonction de la masse molaire $M$ et du
        volume molaire $V_m$.
  \item En déduire la masse molaire du gaz, et proposer une identité possible.
  \item Quelle serait la masse de $\SI{10.0}{\litre}$ de ce gaz aux CNTP ?
\end{enumerate}
\end{exo}

\begin{exo}[même masse, volumes différents]
On dispose de deux ballons aux CNTP : le premier contient $\SI{32}{\gram}$ de méthane \ce{CH4}, le
second $\SI{32}{\gram}$ de dioxygène \ce{O2}.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Calculer le volume de gaz dans chaque ballon.
  \item Lequel occupe le plus grand volume ? Justifier en une phrase.
\end{enumerate}
\end{exo}
