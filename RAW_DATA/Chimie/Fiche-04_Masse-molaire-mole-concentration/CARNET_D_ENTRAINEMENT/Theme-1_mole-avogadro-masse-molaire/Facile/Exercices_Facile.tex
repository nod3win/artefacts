\begin{center}{\large\bfseries\color{primarydark} Thème 1 — Niveau Facile}\end{center}
\emph{Données (\si{\gram\per\mole}) :} $\ce{H}=1$, $\ce{C}=12$, $\ce{N}=14$, $\ce{O}=16$,
$\ce{Na}=23$, $\ce{S}=32$, $\ce{Cl}=35{,}5$, $\ce{Ca}=40$, $\ce{Fe}=56$. \quad
$N_{\mathrm{A}} = 6{,}02\times10^{23}\ \si{\per\mole}$.

\begin{exo}[calculer une masse molaire]
Calcule la masse molaire $M$ de chacune des espèces suivantes.
\begin{enumerate}[label=\alph*)]
  \item \ce{O2}
  \item \ce{CO2}
  \item \ce{NaCl}
  \item \ce{H2SO4}
  \item \ce{CaCO3}
\end{enumerate}
\end{exo}

\begin{exo}[de la masse à la quantité de matière]
Détermine la quantité de matière $n$ correspondant à chaque échantillon ($n = m/M$).
\begin{enumerate}[label=\alph*)]
  \item $\SI{64}{\gram}$ de \ce{O2} \quad ($M=\SI{32}{\gram\per\mole}$)
  \item $\SI{22}{\gram}$ de \ce{CO2} \quad ($M=\SI{44}{\gram\per\mole}$)
  \item $\SI{29.25}{\gram}$ de \ce{NaCl} \quad ($M=\SI{58.5}{\gram\per\mole}$)
  \item $\SI{9.8}{\gram}$ de \ce{H2SO4} \quad ($M=\SI{98}{\gram\per\mole}$)
\end{enumerate}
\end{exo}

\begin{exo}[de la quantité de matière au nombre d'entités]
Combien d'entités contient chaque échantillon ($N = n\,N_{\mathrm{A}}$)~?
\begin{enumerate}[label=\alph*)]
  \item $\SI{2}{\mole}$ de molécules \ce{H2O}
  \item $\SI{0.5}{\mole}$ d'atomes de fer
  \item $\SI{0.25}{\mole}$ d'ions \ce{Cl-}
  \item $\SI{3}{\mole}$ de molécules \ce{N2}
\end{enumerate}
\end{exo}

\begin{exo}[du nombre d'entités à la quantité de matière]
Détermine la quantité de matière $n$ correspondant à ($n = N/N_{\mathrm{A}}$) :
\begin{enumerate}[label=\alph*)]
  \item $1{,}204\times10^{24}$ molécules
  \item $3{,}01\times10^{23}$ atomes
  \item $6{,}02\times10^{22}$ ions
\end{enumerate}
\end{exo}

\begin{exo}[une mole \og de quoi \fg{}~?]
Pour chaque échantillon, donne la quantité de matière \emph{d'atomes} demandée.
\begin{enumerate}[label=\alph*)]
  \item Nombre de moles d'atomes d'oxygène dans $\SI{2}{\mole}$ de \ce{CO2}.
  \item Nombre de moles d'atomes d'hydrogène dans $\SI{0.5}{\mole}$ de \ce{H2O}.
  \item Nombre total de moles d'atomes dans $\SI{1}{\mole}$ de \ce{H3PO4}.
\end{enumerate}
\end{exo}
