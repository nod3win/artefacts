\begin{center}{\large\bfseries\color{primarydark} Thème 4 — Corrigé Moyen}\end{center}

\begin{corrige}[base forte : concentration en ions hydroxyde]
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{0{,}0855}{171} = \SI{5.00e-4}{\mole}$ ; $\SI{500}{\milli\litre} = \SI{0.5}{\litre}$, donc
        $C = \dfrac{5{,}00\times10^{-4}}{0{,}5} = \SI{1.00e-3}{\mole\per\litre}$.
  \item Chaque \ce{Ba(OH)2} libère $2$ \ce{OH-} : $[\ce{OH-}] = 2\times 1{,}00\times10^{-3}
        = \SI{2.00e-3}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[ions nitrate à partir d'une quantité]
\ce{Ba(NO3)2 -> Ba^2+ + 2 NO3-} : $n(\ce{NO3-}) = 2\times 0{,}050 = \SI{0.100}{\mole}$.
$\SI{200}{\milli\litre} = \SI{0.200}{\litre}$, d'où $[\ce{NO3-}] = \dfrac{0{,}100}{0{,}200} = \SI{0.500}{\mole\per\litre}$.
\end{corrige}

\begin{corrige}[mélange de deux sources de chlorure]
On somme les quantités de \ce{Cl-} apportées, puis on divise par le volume total ($\SI{0.200}{\litre}$).
\[ n_1(\ce{Cl-}) = 0{,}100\times 0{,}5\times 1 = \SI{0.050}{\mole};\quad
   n_2(\ce{Cl-}) = 0{,}100\times 0{,}5\times 2 = \SI{0.100}{\mole}. \]
\[ [\ce{Cl-}] = \frac{0{,}050+0{,}100}{0{,}200} = \frac{0{,}150}{0{,}200} = \SI{0.750}{\mole\per\litre}. \]
\end{corrige}

\begin{corrige}[de la concentration massique aux ions]
\begin{enumerate}[label=\alph*)]
  \item $C = \dfrac{11{,}7}{58{,}5} = \SI{0.200}{\mole\per\litre}$.
  \item \ce{NaCl -> Na+ + Cl-} : $[\ce{Na+}] = [\ce{Cl-}] = \SI{0.200}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[comparer plusieurs sels à même concentration]
$[\ce{Cl-}] = (\text{nombre de \ce{Cl}})\times 0{,}2$ :
\begin{enumerate}[label=\alph*)]
  \item \ce{NaCl} : $\SI{0.2}{\mole\per\litre}$.
  \item \ce{MgCl2} : $2\times 0{,}2 = \SI{0.4}{\mole\per\litre}$.
  \item \ce{AlCl3} : $3\times 0{,}2 = \SI{0.6}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}
