\begin{center}{\large\bfseries\color{primarydark} Thème 4 — Niveau Moyen}\end{center}
\emph{Données (\si{\gram\per\mole}) :} \ce{Ba(OH)2}$=171$, \ce{NaCl}$=58{,}5$.

\begin{exo}[base forte : concentration en ions hydroxyde]
On dissout $\SI{0.0855}{\gram}$ d'hydroxyde de baryum \ce{Ba(OH)2} dans $\SI{500}{\milli\litre}$ d'eau.
\begin{enumerate}[label=\alph*)]
  \item Quelle est la concentration molaire du soluté~?
  \item En déduire $[\ce{OH-}]$ (on rappelle \ce{Ba(OH)2 -> Ba^2+ + 2 OH-}).
\end{enumerate}
\end{exo}

\begin{exo}[ions nitrate à partir d'une quantité]
Une solution de volume $\SI{200}{\milli\litre}$ contient $\SI{0.050}{\mole}$ de nitrate de baryum
\ce{Ba(NO3)2}. Quelle est $[\ce{NO3-}]$~?
\end{exo}

\begin{exo}[mélange de deux sources de chlorure]
Dans une fiole, on mélange $\SI{100}{\milli\litre}$ de \ce{NaCl} à $\SI{0.5}{\mole\per\litre}$ et
$\SI{100}{\milli\litre}$ de \ce{CaCl2} à $\SI{0.5}{\mole\per\litre}$ ; le volume total est
$\SI{200}{\milli\litre}$. Quelle est $[\ce{Cl-}]$~?
\end{exo}

\begin{exo}[de la concentration massique aux ions]
Une solution contient $\SI{11.7}{\gram}$ de \ce{NaCl} par litre.
\begin{enumerate}[label=\alph*)]
  \item Quelle est sa concentration molaire~?
  \item En déduire $[\ce{Na+}]$ et $[\ce{Cl-}]$.
\end{enumerate}
\end{exo}

\begin{exo}[comparer plusieurs sels à même concentration]
Chaque solution est à $\SI{0.2}{\mole\per\litre}$. Donne $[\ce{Cl-}]$ pour chacune.
\begin{enumerate}[label=\alph*)]
  \item \ce{NaCl} ; \quad b) \ce{MgCl2} ; \quad c) \ce{AlCl3}.
\end{enumerate}
\end{exo}
