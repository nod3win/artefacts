% THEME_TITLE: Électrolytes forts et concentration des ions
\noindent
Beaucoup de solutés sont des \textbf{électrolytes} : dissous dans l'eau, ils se \emph{dissocient} en
ions. La concentration des \emph{ions} produits n'est pas toujours celle du soluté : tout dépend de la
\textbf{stœchiométrie} de la dissociation.

\begin{cle}[électrolyte fort]
Un \textbf{électrolyte fort} se dissocie \textbf{totalement} en ions dans l'eau (sels solubles,
acides forts \ce{HCl}/\ce{HClO4}, bases fortes \ce{NaOH}/\ce{Ba(OH)2}). Ce sont ces ions mobiles qui
rendent la solution \textbf{conductrice} : plus elle est riche en ions, mieux elle conduit (l'eau
pure, très peu dissociée, conduit très mal ; le glucose, non ionique, ne conduit pas).
\end{cle}

\begin{cle}[concentration des ions]
Pour une dissociation \ce{A_xB_y -> x A + y B} d'un électrolyte de concentration $C$ :
\[ \boxed{\,[\ce{A}] = x\,C \qquad [\ce{B}] = y\,C\,}. \]
La concentration d'un ion est celle du sel \textbf{multipliée par son coefficient stœchiométrique}.
\end{cle}

\begin{methode}[mélange : plusieurs sources d'un même ion]
Quand un ion provient de \emph{plusieurs} sels (ou qu'on mélange plusieurs solutions) :
\begin{enumerate}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item on calcule la \emph{quantité de matière} de l'ion apportée par chaque source
        ($n = \text{coeff}\times C\times V$) ;
  \item on \textbf{additionne} ces quantités ;
  \item on \textbf{divise par le volume total} de la solution finale.
\end{enumerate}
\end{methode}

\begin{piege}[trois réflexes]
\begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=1pt]
  \item Ne pas oublier le \textbf{coefficient} : $\SI{1}{\mole}$ de \ce{CaCl2} libère $\SI{2}{\mole}$ de \ce{Cl-}.
  \item Distinguer une \emph{quantité} (\si{\mole}, à diviser par $V$) d'une \emph{concentration}
        (\si{\mole\per\litre}, déjà ramenée au litre).
  \item Dans un mélange, diviser par le volume \textbf{total}, pas par un volume initial.
\end{itemize}
\end{piege}
