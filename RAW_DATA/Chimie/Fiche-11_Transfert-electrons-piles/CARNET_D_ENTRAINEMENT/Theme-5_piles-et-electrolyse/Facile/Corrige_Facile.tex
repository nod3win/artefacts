\begin{center}\color{primarydark}\large\bfseries Thème 5 — Corrigé \quad$\bullet$\quad Niveau Facile\end{center}

\begin{corrige}[anode, cathode et pôles]
\begin{enumerate}[label=\alph*)]
  \item L'anode est le couple de $E_{\mathrm{r}}^{\circ}$ le plus bas : \ce{Zn} (oxydation). La cathode
        est \ce{Cu} (réduction).
  \item Anode $=$ pôle \textbf{$-$} ; cathode $=$ pôle \textbf{$+$}.
\end{enumerate}
\end{corrige}

\begin{corrige}[sens des électrons et du courant]
\begin{enumerate}[label=\alph*)]
  \item Les électrons vont de l'\textbf{anode \ce{Zn} ($-$) vers la cathode \ce{Cu} ($+$)} dans le fil.
  \item Le courant conventionnel $i$ circule en \textbf{sens inverse} : de \ce{Cu} vers \ce{Zn} à
        l'extérieur.
\end{enumerate}
\end{corrige}

\begin{corrige}[notation conventionnelle]
\begin{enumerate}[label=\alph*)]
  \item \[\ce{(-) Zn | Zn^{2+} || Cu^{2+} | Cu (+)}\]
  \item Le trait simple $|$ est l'\textbf{interface} électrode/solution ; le double trait $\big\|$
        représente le \textbf{pont salin}.
\end{enumerate}
\end{corrige}

\begin{corrige}[charge et quantité d'électrons]
\begin{enumerate}[label=\alph*)]
  \item $t=20\times 60=\qty{1200}{s}$, donc $Q=I\,t=1{,}5\times 1200=\qty{1800}{C}$.
  \item $n_{\ce{e^-}}=\dfrac{Q}{F}=\dfrac{1800}{96485}\approx\qty{0.0187}{mol}$ d'électrons.
\end{enumerate}
\end{corrige}

\begin{corrige}[premier calcul de Faraday]
\[
m=\frac{M\,I\,t}{n\,F}=\frac{108\times 1{,}0\times 965}{1\times 96485}
=\frac{104\,220}{96\,485}\approx\qty{1.08}{g}.
\]
(On pouvait remarquer que $Q=It=\qty{965}{C}\approx 0{,}01\,F$, soit $0{,}01$~mol d'\ce{Ag}, donc
$0{,}01\times 108=\qty{1.08}{g}$.)
\end{corrige}
