\begin{corrige}[vrai ou faux : les définitions]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai} : par définition, $\mathrm{n.o.}=$ charge pour un ion monoatomique
        (ex.\ $\ce{Fe^{3+}}\to +\mathrm{III}$).
  \item \textbf{Faux} : s'oxyder, c'est \emph{perdre} des électrons, donc le n.o.\ \textbf{augmente}.
  \item \textbf{Vrai} : l'oxydant capte les électrons ; il est donc réduit.
  \item \textbf{Faux} : $\mathrm{n.o.}(\ce{H})=+\mathrm{I}$ \emph{sauf} dans les hydrures métalliques
        ($-\mathrm{I}$) et dans \ce{H2} ($0$).
  \item \textbf{Vrai} : dans un corps simple, $\mathrm{n.o.}=0$.
\end{enumerate}
\textbf{Bilan : 3 propositions vraies} (a, c, e).
\end{corrige}

\begin{corrige}[tous les éléments d'un composé, et un piège]
\begin{enumerate}[label=\alph*)]
  \item \ce{K} est un alcalin : $+\mathrm{I}$ ; \ce{O} : $-\mathrm{II}$. Pour Mn, la molécule est neutre :
        $(+1)+x+4(-2)=0\Rightarrow x=+7$, soit $\mathrm{n.o.}(\ce{Mn})=+\mathrm{VII}$.
  \item L'étudiant a \textbf{oublié le potassium} : en écrivant $x+4(-2)=0$ il trouve $+8$. Il faut
        inclure $\mathrm{n.o.}(\ce{K})=+\mathrm{I}$. De plus, le manganèse (groupe~7) ne peut pas
        dépasser $+\mathrm{VII}$ : $+\mathrm{VIII}$ est impossible. La bonne valeur est
        $+\mathrm{VII}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[reconnaître une réaction d'oxydo-réduction]
Une réaction est redox \emph{si et seulement si} au moins un atome change de n.o.
\begin{enumerate}[label=\alph*)]
  \item \ce{NaCl + AgNO3 -> AgCl + NaNO3} : \ce{Na}$(+\mathrm{I})$, \ce{Cl}$(-\mathrm{I})$,
        \ce{Ag}$(+\mathrm{I})$, \ce{N}$(+\mathrm{V})$ inchangés. \textbf{Non redox} (précipitation).
  \item \ce{Cu + 4 HNO3 -> Cu(NO3)2 + 2 NO2 + 2 H2O} : \ce{Cu}$:0\to+\mathrm{II}$ (oxydé) et une
        partie de l'azote $:+\mathrm{V}\to+\mathrm{IV}$ (réduit en \ce{NO2}). \textbf{Redox}.
  \item \ce{2 Na + 2 H2O -> 2 NaOH + H2} : \ce{Na}$:0\to+\mathrm{I}$ (oxydé), \ce{H}$:+\mathrm{I}\to 0$
        (réduit en \ce{H2}). \textbf{Redox}.
  \item \ce{CO2 + H2O -> H2CO3} : \ce{C}$(+\mathrm{IV})$, \ce{O}$(-\mathrm{II})$, \ce{H}$(+\mathrm{I})$
        inchangés. \textbf{Non redox} (hydratation acido-basique).
\end{enumerate}
\textbf{Réactions redox : b) et c).}
\end{corrige}

\begin{corrige}[un élément oxydé \emph{et} réduit à la fois]
\begin{enumerate}[label=\alph*)]
  \item \ce{MnSO4} : l'ion sulfate est \ce{SO4^{2-}}, donc $\mathrm{n.o.}(\ce{Mn})=+\mathrm{II}$.
        \ce{KMnO4} : $+\mathrm{VII}$. \ce{MnO2} : $x+2(-2)=0\Rightarrow +\mathrm{IV}$.
  \item Le manganèse de \ce{MnSO4} passe de $+\mathrm{II}$ à $+\mathrm{IV}$ : il est \textbf{oxydé}.
        Celui de \ce{KMnO4} passe de $+\mathrm{VII}$ à $+\mathrm{IV}$ : il est \textbf{réduit}. Les
        deux « se rejoignent » à $+\mathrm{IV}$ (\ce{MnO2}) : c'est une \textbf{médiamutation}.
  \item \ce{KMnO4} est \textbf{réduit} : c'est donc l'\textbf{oxydant} (et \ce{MnSO4}, oxydé, est le
        réducteur).
\end{enumerate}
\end{corrige}

\begin{corrige}[relier n.o., formule et sens de la réaction]
\begin{enumerate}[label=\alph*)]
  \item \ce{FeSO4} : avec \ce{SO4^{2-}}, on a $\mathrm{n.o.}(\ce{Fe})=+\mathrm{II}$.
  \item \ce{Fe2(SO4)3} : trois sulfates apportent $3\times(-2)=-6$, donc $2\,\mathrm{n.o.}(\ce{Fe})=+6$,
        soit $\mathrm{n.o.}(\ce{Fe})=+\mathrm{III}$.
  \item Le fer passe de $+\mathrm{II}$ à $+\mathrm{III}$ : il est \textbf{oxydé}, et chaque atome de
        fer \textbf{perd 1 électron}.
\end{enumerate}
\end{corrige}
