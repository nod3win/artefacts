\begin{corrige}[corps simples et ions monoatomiques]
Corps simple $\Rightarrow \mathrm{n.o.}=0$ ; ion monoatomique $\Rightarrow \mathrm{n.o.}=$ charge.
\begin{enumerate}[label=\alph*)]
  \item \ce{O2} est un corps simple : $\mathrm{n.o.}(\ce{O})=0$.
  \item \ce{Fe} métallique est un corps simple : $\mathrm{n.o.}(\ce{Fe})=0$.
  \item $\mathrm{n.o.}(\ce{Ca^{2+}})=+\mathrm{II}$.
  \item $\mathrm{n.o.}(\ce{Cl^-})=-\mathrm{I}$.
  \item $\mathrm{n.o.}(\ce{Al^{3+}})=+\mathrm{III}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[les exceptions de O et de H]
\begin{enumerate}[label=\alph*)]
  \item \ce{H2O} : $2(+1)+x=0\Rightarrow x=-2$, soit $\mathrm{n.o.}(\ce{O})=-\mathrm{II}$ (cas général).
  \item \ce{H2O2} : \emph{peroxyde} (liaison \ce{O-O}) ; $2(+1)+2x=0\Rightarrow x=-1$, soit
        $\mathrm{n.o.}(\ce{O})=-\mathrm{I}$.
  \item \ce{HCl} : $\mathrm{n.o.}(\ce{H})=+\mathrm{I}$ (cas général).
  \item \ce{NaH} : \emph{hydrure} d'un métal ; H y est plus électronégatif que Na, donc
        $\mathrm{n.o.}(\ce{H})=-\mathrm{I}$ (et l'on vérifie $+\mathrm{I}+(-\mathrm{I})=0$).
\end{enumerate}
\end{corrige}

\begin{corrige}[atome central d'une molécule neutre]
Molécule neutre : la somme des n.o.\ vaut $0$ ; on pose $x=\mathrm{n.o.}$ de l'atome cherché.
\begin{enumerate}[label=\alph*)]
  \item \ce{H2S} : $2(+1)+x=0\Rightarrow x=-2$, soit $\mathrm{n.o.}(\ce{S})=-\mathrm{II}$.
  \item \ce{SO2} : $x+2(-2)=0\Rightarrow x=+4$, soit $\mathrm{n.o.}(\ce{S})=+\mathrm{IV}$.
  \item \ce{NH3} : $x+3(+1)=0\Rightarrow x=-3$, soit $\mathrm{n.o.}(\ce{N})=-\mathrm{III}$.
  \item \ce{CH4} : $x+4(+1)=0\Rightarrow x=-4$, soit $\mathrm{n.o.}(\ce{C})=-\mathrm{IV}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[atome au sein d'un ion polyatomique]
Ici la somme des n.o.\ vaut la \textbf{charge de l'ion}.
\begin{enumerate}[label=\alph*)]
  \item \ce{SO4^{2-}} : $x+4(-2)=-2\Rightarrow x=+6$, soit $\mathrm{n.o.}(\ce{S})=+\mathrm{VI}$.
  \item \ce{NO3^-} : $x+3(-2)=-1\Rightarrow x=+5$, soit $\mathrm{n.o.}(\ce{N})=+\mathrm{V}$.
  \item \ce{MnO4^-} : $x+4(-2)=-1\Rightarrow x=+7$, soit $\mathrm{n.o.}(\ce{Mn})=+\mathrm{VII}$.
  \item \ce{Cr2O7^{2-}} : $2x+7(-2)=-2\Rightarrow 2x=12\Rightarrow x=+6$, soit
        $\mathrm{n.o.}(\ce{Cr})=+\mathrm{VI}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[oxydation ou réduction ?]
n.o.\ qui \textbf{augmente} $=$ oxydation (perte d'électrons) ; qui \textbf{diminue} $=$ réduction
(gain d'électrons).
\begin{enumerate}[label=\alph*)]
  \item \ce{Fe}$(0)\to$\ce{Fe^{2+}}$(+\mathrm{II})$ : $\Delta=+2$, \textbf{oxydation}, l'atome
        \textbf{perd} 2 électrons.
  \item \ce{Cl2}$(0)\to$\ce{Cl^-}$(-\mathrm{I})$ : $\Delta=-1$ par atome, \textbf{réduction}, chaque
        atome \textbf{capte} 1 électron.
  \item \ce{Cu^{2+}}$(+\mathrm{II})\to$\ce{Cu}$(0)$ : $\Delta=-2$, \textbf{réduction}, l'ion
        \textbf{capte} 2 électrons.
  \item S : de $-\mathrm{II}$ (dans \ce{H2S}) à $+\mathrm{IV}$ (dans \ce{SO2}) : $\Delta=+6$,
        \textbf{oxydation}, le soufre \textbf{perd} 6 électrons.
\end{enumerate}
\end{corrige}
