\begin{center}\color{primarydark}\large\bfseries Thème 1 — Nombre d'oxydation \quad$\bullet$\quad Niveau Facile\end{center}

\begin{exo}[corps simples et ions monoatomiques]
Donne le nombre d'oxydation de l'élément indiqué dans chaque cas.
\begin{enumerate}[label=\alph*)]
  \item l'oxygène dans le dioxygène \ce{O2}
  \item le fer métallique \ce{Fe}
  \item l'ion calcium \ce{Ca^{2+}}
  \item l'ion chlorure \ce{Cl^-}
  \item l'ion aluminium \ce{Al^{3+}}
\end{enumerate}
\end{exo}

\begin{exo}[les exceptions de O et de H]
Détermine le nombre d'oxydation de l'atome indiqué.
\begin{enumerate}[label=\alph*)]
  \item l'oxygène dans l'eau \ce{H2O}
  \item l'oxygène dans le peroxyde d'hydrogène \ce{H2O2}
  \item l'hydrogène dans le chlorure d'hydrogène \ce{HCl}
  \item l'hydrogène dans l'hydrure de sodium \ce{NaH}
\end{enumerate}
\end{exo}

\begin{exo}[atome central d'une molécule neutre]
Détermine le nombre d'oxydation de l'atome indiqué.
\begin{enumerate}[label=\alph*)]
  \item le soufre dans \ce{H2S}
  \item le soufre dans \ce{SO2}
  \item l'azote dans l'ammoniac \ce{NH3}
  \item le carbone dans le méthane \ce{CH4}
\end{enumerate}
\end{exo}

\begin{exo}[atome au sein d'un ion polyatomique]
Détermine le nombre d'oxydation de l'atome indiqué.
\begin{enumerate}[label=\alph*)]
  \item le soufre dans le sulfate \ce{SO4^{2-}}
  \item l'azote dans le nitrate \ce{NO3^-}
  \item le manganèse dans le permanganate \ce{MnO4^-}
  \item le chrome dans le dichromate \ce{Cr2O7^{2-}}
\end{enumerate}
\end{exo}

\begin{exo}[oxydation ou réduction ?]
Pour chaque transformation d'un atome, calcule la variation de nombre d'oxydation, puis indique s'il
s'agit d'une \textbf{oxydation} ou d'une \textbf{réduction}, et si l'atome \textbf{perd} ou
\textbf{capte} des électrons.
\begin{enumerate}[label=\alph*)]
  \item \ce{Fe -> Fe^{2+}}
  \item \ce{Cl2 -> 2 Cl^-}
  \item \ce{Cu^{2+} -> Cu}
  \item le soufre passant de \ce{H2S} à \ce{SO2}
\end{enumerate}
\end{exo}
