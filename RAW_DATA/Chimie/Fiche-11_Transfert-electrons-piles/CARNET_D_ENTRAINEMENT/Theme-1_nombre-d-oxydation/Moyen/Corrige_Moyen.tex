\begin{center}\color{primarydark}\large\bfseries Thème 1 — Corrigé \quad$\bullet$\quad Niveau Moyen\end{center}

\begin{corrige}[un même élément à plusieurs degrés : le manganèse]
On isole $x=\mathrm{n.o.}(\ce{Mn})$ dans chaque espèce ($\mathrm{n.o.}(\ce{O})=-\mathrm{II}$).
\begin{enumerate}[label=\alph*)]
  \item \ce{MnO4^-} : $x+4(-2)=-1\Rightarrow x=+7$, soit $+\mathrm{VII}$.
  \item \ce{MnO2} : $x+2(-2)=0\Rightarrow x=+4$, soit $+\mathrm{IV}$.
  \item \ce{Mn^{2+}} : ion monoatomique, $+\mathrm{II}$.
  \item \ce{Mn} : corps simple, $0$.
\end{enumerate}
Classement du plus oxydé au plus réduit : $\ce{MnO4^-}\,(+\mathrm{VII}) > \ce{MnO2}\,(+\mathrm{IV})
> \ce{Mn^{2+}}\,(+\mathrm{II}) > \ce{Mn}\,(0)$.
\end{corrige}

\begin{corrige}[variation du n.o. entre deux formes]
\begin{enumerate}[label=\alph*)]
  \item C : $-\mathrm{IV}$ (\ce{CH4}) $\to +\mathrm{IV}$ (\ce{CO2}), $\Delta=+8$ : \textbf{oxydation}.
  \item S : $+\mathrm{IV}$ (\ce{SO2}) $\to +\mathrm{VI}$ (\ce{SO4^{2-}}), $\Delta=+2$ : \textbf{oxydation}.
  \item N : $+\mathrm{II}$ (\ce{NO}) $\to +\mathrm{V}$ (\ce{NO3^-}), $\Delta=+3$ : \textbf{oxydation}.
  \item Cr : $+\mathrm{VI}$ (\ce{Cr2O7^{2-}}) $\to +\mathrm{III}$ (\ce{Cr^{3+}}), $\Delta=-3$ :
        \textbf{réduction}.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire une équation redox globale]
On suit l'atome dont le n.o.\ varie. \emph{Rappel : le réducteur est l'espèce oxydée, l'oxydant est
l'espèce réduite.}
\begin{enumerate}[label=\alph*)]
  \item \ce{Zn}$(0)\to$\ce{Zn^{2+}}$(+\mathrm{II})$ : \ce{Zn} est \textbf{oxydé} (réducteur) ;
        \ce{Cu^{2+}}$(+\mathrm{II})\to$\ce{Cu}$(0)$ est \textbf{réduit}, donc \ce{Cu^{2+}} est
        l'\textbf{oxydant}.
  \item \ce{Fe}$(0)\to$\ce{Fe^{2+}}$(+\mathrm{II})$ : \ce{Fe} est \textbf{oxydé} (réducteur) ;
        \ce{H+}$(+\mathrm{I})\to$\ce{H2}$(0)$ est \textbf{réduit}, donc \ce{H+} est l'\textbf{oxydant}.
  \item \ce{Al}$(0)\to +\mathrm{III}$ : \ce{Al} est \textbf{oxydé} (réducteur) ;
        \ce{Cl}$(0)\to -\mathrm{I}$ dans \ce{AlCl3} est \textbf{réduit}, donc \ce{Cl2} est
        l'\textbf{oxydant}.
\end{enumerate}
\end{corrige}

\begin{corrige}[cas piégeux : exceptions et n.o.\ moyens]
\begin{enumerate}[label=\alph*)]
  \item \ce{OF2} : le fluor est \emph{plus électronégatif} que l'oxygène, donc
        $\mathrm{n.o.}(\ce{F})=-\mathrm{I}$. Alors $x+2(-1)=0\Rightarrow x=+2$ :
        $\mathrm{n.o.}(\ce{O})=+\mathrm{II}$ (rare exception où O est positif).
  \item \ce{Fe3O4} : $3x+4(-2)=0\Rightarrow 3x=8\Rightarrow x=+\tfrac{8}{3}\approx +2{,}67$. C'est un
        \textbf{n.o.\ moyen} : la magnétite contient en réalité un mélange de \ce{Fe^{2+}} et de
        \ce{Fe^{3+}} (un \ce{Fe^{2+}} pour deux \ce{Fe^{3+}}).
  \item \ce{S2O3^{2-}} : $2x+3(-2)=-2\Rightarrow 2x=4\Rightarrow x=+2$ :
        $\mathrm{n.o.}(\ce{S})=+\mathrm{II}$ (n.o.\ moyen des deux soufres).
  \item \ce{NH4+} : $x+4(+1)=+1\Rightarrow x=-3$ : $\mathrm{n.o.}(\ce{N})=-\mathrm{III}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[combien d'électrons échange un couple ?]
\begin{enumerate}[label=\alph*)]
  \item \ce{MnO4^-}/\ce{Mn^{2+}} : Mn passe de $+\mathrm{VII}$ à $+\mathrm{II}$, $n=7-2=5$ électrons.
  \item \ce{Cr2O7^{2-}}/\ce{Cr^{3+}} : Cr passe de $+\mathrm{VI}$ à $+\mathrm{III}$, soit $3$ par atome ;
        mais il y a \textbf{2} atomes de Cr, donc $n=2\times 3=6$ électrons. (Piège classique !)
  \item \ce{NO3^-}/\ce{NO} : N passe de $+\mathrm{V}$ à $+\mathrm{II}$, $n=5-2=3$ électrons.
  \item \ce{S4O6^{2-}}/\ce{S2O3^{2-}} : le soufre passe d'un n.o.\ moyen de $+2{,}5$ (dans
        \ce{S4O6^{2-}}) à $+\mathrm{II}$ (dans \ce{S2O3^{2-}}), soit $0{,}5$ par atome ; sur les
        \textbf{4} atomes de soufre, $n=4\times 0{,}5=2$ électrons. La demi-équation est donc
        $\ce{S4O6^{2-} + 2 e^- <=> 2 S2O3^{2-}}$.
\end{enumerate}
\end{corrige}
