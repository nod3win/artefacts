{\LARGE\bfseries\color{primarydark}Thème 1 — Niveau Moyen}\\[2pt]
{\small\itshape Calcul du \NO\ dans des ions polyatomiques et des oxoanions. Pose toujours l'équation $\sum\NO=$ charge.}\\[-4pt]
{\color{primary}\rule{\textwidth}{1pt}}
\end{center}

\begin{exo}[le chrome dans deux espèces]
Calcule le nombre d'oxydation du chrome :
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item dans le dichromate de potassium \ce{K2Cr2O7} (molécule neutre) ;
  \item dans l'ion chromate \ce{CrO4^{2-}}.
\end{enumerate}
Le chrome a-t-il le même nombre d'oxydation dans ces deux espèces ?
\end{exo}

\begin{exo}[le manganèse dans toute sa gamme]
Le manganèse est un élément « caméléon » du redox. Détermine son nombre d'oxydation dans :
\[
\ce{MnO4^-},\qquad \ce{Mn^{2+}},\qquad \ce{MnO2},\qquad \ce{Mn(s)},\qquad \ce{KMnO4}.
\]
Range ensuite ces cinq espèces par \NO(\ce{Mn}) croissant.
\end{exo}

\begin{exo}[l'azote sous plusieurs formes]
Détermine le nombre d'oxydation de l'azote dans :
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item l'hydrazine \ce{N2H4} (molécule neutre) ;
  \item l'ion nitrate \ce{NO3^-} ;
  \item l'ion ammonium \ce{NH4+}.
\end{enumerate}
\end{exo}

\begin{exo}[le soufre dans les oxoanions]
Détermine le nombre d'oxydation (moyen) du soufre dans les trois anions suivants :
\[
\ce{SO4^{2-}},\qquad \ce{SO3^{2-}},\qquad \ce{S2O3^{2-}}\ (\text{thiosulfate}).
\]
\end{exo}

\begin{exo}[une somme de deux nombres d'oxydation]
On considère le carbone dans deux composés bien connus : le méthane \ce{CH4} et le dioxyde de carbone
\ce{CO2}.
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item Calcule $\NO(\ce{C})$ dans \ce{CH4}, puis dans \ce{CO2}.
  \item Quelle est la valeur de la \textbf{somme} de ces deux nombres d'oxydation ?
  \item En passant de \ce{CH4} à \ce{CO2}, le carbone est-il oxydé ou réduit ? (Justifie par la variation du \NO.)
\end{enumerate}
\end{exo}
