\begin{center}\color{primarydark}\large\bfseries Thème 1 — Nombre d'oxydation \quad$\bullet$\quad Niveau Moyen\end{center}

\begin{exo}[un même élément à plusieurs degrés : le manganèse]
Détermine le nombre d'oxydation du manganèse dans chacune de ces espèces, puis range-les de la plus
\emph{oxydée} à la plus \emph{réduite}.
\begin{enumerate}[label=\alph*)]
  \item le permanganate \ce{MnO4^-}
  \item le dioxyde de manganèse \ce{MnO2}
  \item l'ion \ce{Mn^{2+}}
  \item le manganèse métallique \ce{Mn}
\end{enumerate}
\end{exo}

\begin{exo}[variation du n.o. entre deux formes]
Pour chaque élément, calcule son n.o.\ dans les deux espèces, la variation $\Delta$, et conclus
\textbf{oxydation} ou \textbf{réduction}.
\begin{enumerate}[label=\alph*)]
  \item le carbone : \ce{CH4} $\longrightarrow$ \ce{CO2}
  \item le soufre : \ce{SO2} $\longrightarrow$ \ce{SO4^{2-}}
  \item l'azote : \ce{NO} $\longrightarrow$ \ce{NO3^-}
  \item le chrome : \ce{Cr2O7^{2-}} $\longrightarrow$ \ce{Cr^{3+}}
\end{enumerate}
\end{exo}

\begin{exo}[lire une équation redox globale]
Dans chaque équation, repère l'élément \textbf{oxydé} et l'élément \textbf{réduit}, puis nomme
l'\textbf{oxydant} et le \textbf{réducteur}.
\begin{enumerate}[label=\alph*)]
  \item \ce{Zn + Cu^{2+} -> Zn^{2+} + Cu}
  \item \ce{Fe + 2 H+ -> Fe^{2+} + H2}
  \item \ce{2 Al + 3 Cl2 -> 2 AlCl3}
\end{enumerate}
\end{exo}

\begin{exo}[cas piégeux : exceptions et n.o.\ moyens]
Détermine le nombre d'oxydation de l'atome indiqué (attention aux exceptions !).
\begin{enumerate}[label=\alph*)]
  \item l'oxygène dans le difluorure d'oxygène \ce{OF2}
  \item le fer dans la magnétite \ce{Fe3O4}
  \item le soufre dans le thiosulfate \ce{S2O3^{2-}}
  \item l'azote dans l'ion ammonium \ce{NH4+}
\end{enumerate}
\end{exo}

\begin{exo}[combien d'électrons échange un couple ?]
Le nombre d'électrons d'une demi-équation vaut $n=\mathrm{n.o.}(\text{oxydant})-\mathrm{n.o.}
(\text{réducteur})$ (compté sur \textbf{tous} les atomes concernés). Donne $n$ pour chaque couple.
\begin{enumerate}[label=\alph*)]
  \item \ce{MnO4^-}/\ce{Mn^{2+}}
  \item \ce{Cr2O7^{2-}}/\ce{Cr^{3+}}
  \item \ce{NO3^-}/\ce{NO}
  \item \ce{S4O6^{2-}}/\ce{S2O3^{2-}}
\end{enumerate}
\end{exo}
