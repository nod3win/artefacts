\begin{center}\color{primarydark}\large\bfseries Thème 2 — Demi-équations et équilibrage \quad$\bullet$\quad Niveau Difficile\end{center}

\begin{exo}[dichromate et acide nitreux]
Équilibre en milieu acide (avec des \ce{H+}) la réaction non pondérée
\[
\ce{Cr2O7^{2-} + HNO2 + H+ -> Cr^{3+} + NO3^- + H2O}
\]
(l'azote de \ce{HNO2} passe de $+\mathrm{III}$ à $+\mathrm{V}$). Combien d'électrons sont échangés au
total ?
\end{exo}

\begin{exo}[la réaction aluminothermique]
La thermite fait réagir l'aluminium sur l'oxyde de fer(III) : \ce{Al + Fe2O3 -> Al2O3 + Fe}.
\begin{enumerate}[label=\alph*)]
  \item Pondère l'équation.
  \item Combien d'électrons chaque atome d'aluminium cède-t-il ?
  \item Parmi ces affirmations, laquelle est \textbf{incorrecte} ?
    \begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=0pt]
      \item[(A)] L'aluminium est le réducteur.
      \item[(B)] Il se transfère 3 électrons par atome d'aluminium.
      \item[(C)] La somme des coefficients stœchiométriques vaut 4.
      \item[(D)] Le nombre d'oxydation du fer dans \ce{Fe2O3} vaut $+\mathrm{III}$.
    \end{itemize}
\end{enumerate}
\end{exo}

\begin{exo}[oxydation en milieu basique]
En milieu basique, l'ion permanganate oxyde l'ion sulfite. Équilibre en milieu basique (avec
\ce{OH^-} et \ce{H2O}) :
\[
\ce{MnO4^- + SO3^{2-} -> MnO2 + SO4^{2-}}
\]
(Mn : $+\mathrm{VII}\to+\mathrm{IV}$ ; S : $+\mathrm{IV}\to+\mathrm{VI}$).
\end{exo}

\begin{exo}[permanganate et iodure]
En milieu acide, le permanganate oxyde l'iodure en diiode :
\[
\ce{MnO4^- + I^- + H+ -> Mn^{2+} + I2 + H2O}.
\]
\begin{enumerate}[label=\alph*)]
  \item Écris l'équation pondérée.
  \item Combien d'électrons sont échangés au total ?
\end{enumerate}
\end{exo}

\begin{exo}[le cuivre dans l'acide nitrique : dilué ou concentré]
Selon sa concentration, l'acide nitrique oxyde le cuivre en donnant deux gaz différents. Équilibre
en milieu acide chacune des deux réactions.
\begin{enumerate}[label=\alph*)]
  \item Acide dilué (gaz \ce{NO}) : \ce{Cu + NO3^- + H+ -> Cu^{2+} + NO + H2O}.
  \item Acide concentré (gaz \ce{NO2}) : \ce{Cu + NO3^- + H+ -> Cu^{2+} + NO2 + H2O}.
\end{enumerate}
\end{exo}
