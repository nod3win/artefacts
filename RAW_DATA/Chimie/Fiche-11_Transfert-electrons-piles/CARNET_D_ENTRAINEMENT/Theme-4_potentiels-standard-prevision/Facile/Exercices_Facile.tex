\begin{exo}[classer les oxydants]
Range les formes \textbf{oxydées} \ce{F2}, \ce{Ag+}, \ce{Cu^{2+}} et \ce{Zn^{2+}} par \textbf{pouvoir
oxydant croissant}. Quel est l'oxydant le plus puissant ?
\end{exo}

\begin{exo}[classer les réducteurs]
Range les formes \textbf{réduites} \ce{F^-}, \ce{Ag}, \ce{Cu} et \ce{Zn} par \textbf{pouvoir réducteur
croissant}. Quel est le réducteur le plus puissant ?
\end{exo}

\begin{exo}[deux couples face à face]
On compare les couples \ce{Fe^{3+}}/\ce{Fe^{2+}} ($+0{,}77$\,V) et \ce{Cu^{2+}}/\ce{Cu} ($+0{,}34$\,V).
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item Lequel des deux couples possède le plus fort pouvoir oxydant ?
  \item Quelle forme (\ce{Fe^{3+}} ou \ce{Cu^{2+}}) est capable d'oxyder l'autre forme réduite ?
\end{enumerate}
\end{exo}

\begin{exo}[oui ou non ?]
En comparant les potentiels, dis si chacune de ces affirmations est \textbf{vraie ou fausse}
(justifie par $\Delta E^{\circ}$) :
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item \ce{Ag+} peut oxyder le cuivre métallique \ce{Cu}.
  \item \ce{Cu^{2+}} peut oxyder l'argent métallique \ce{Ag}.
\end{enumerate}
\end{exo}

\begin{exo}[les extrêmes du tableau]
À partir des cinq couples du tableau de données :
\begin{enumerate}[label=(\alph*),leftmargin=2em,topsep=1pt,itemsep=1pt]
  \item Quelle espèce est l'\textbf{oxydant le plus fort} de tous ?
  \item Quelle espèce est le \textbf{réducteur le plus fort} de tous ?
\end{enumerate}
\end{exo}
