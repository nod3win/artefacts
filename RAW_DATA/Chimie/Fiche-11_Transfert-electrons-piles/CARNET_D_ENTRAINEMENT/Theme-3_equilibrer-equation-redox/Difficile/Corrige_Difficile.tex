\begin{corr}[du ionique au moléculaire]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item Ionique : $\ce{MnO4^- + I^- + H+ -> Mn^{2+} + I2 + H2O}$.
        Couples : \ce{MnO4^-}/\ce{Mn^{2+}} (Mn : $+\mathrm{VII}\to+\mathrm{II}$, $n=5$) et
        \ce{I2}/\ce{I^-} (I : $-\mathrm{I}\to 0$, $2\,\ce{I^-}\to\ce{I2}+2\,\ce{e^-}$).
  \item $\ce{MnO4^- + 8 H+ + 5 e^- -> Mn^{2+} + 4 H2O}$ \quad et \quad $\ce{2 I^- -> I2 + 2 e^-}$.
  \item PPCM des électrons $=10$ : réduction $\times 2$, oxydation $\times 5$ :
        \[
        \ce{2 MnO4^- + 16 H+ + 10 I^- -> 2 Mn^{2+} + 5 I2 + 8 H2O}.
        \]
  \item On ajoute les 12 \ce{K+} ($2$ de \ce{KMnO4} $+\,10$ de \ce{KI}) et les 16 \ce{Cl^-} (de \ce{HCl}) :
        \[
        \boxed{\ce{2 KMnO4 + 10 KI + 16 HCl -> 2 MnCl2 + 5 I2 + 12 KCl + 8 H2O}}
        \]
        (Cl : $16 = 2\times2 + 12$ ; K : $12=12$ ; contrôle complet OK.)
  \item \textbf{(A)} faux (coeff. \ce{MnCl2} $=2$) ; \textbf{(B)} faux (il s'échange $2\times5=10$ électrons, pas 12) ;
        \textbf{(C)} faux ($\NO(\ce{Mn})=+7$) ; \textbf{(D)} \textbf{vrai} : il y a bien 8 \ce{H2O} à droite.
        \emph{Réponse D (annales 2021-Q25).}
\end{enumerate}
\end{corr}

\begin{corr}[dichromate et acide nitreux]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item \ce{HNO2} : $(+1)+x+2(-2)=0\Rightarrow \NO(\ce{N})=+\mathrm{III}$ ; \ce{NO3^-} : $+\mathrm{V}$.
        Couples : \ce{Cr2O7^{2-}}/\ce{Cr^{3+}} (réduction, $n=6$) et \ce{NO3^-}/\ce{HNO2}
        (N : $+\mathrm{III}\to+\mathrm{V}$, oxydation, $n=2$).
  \item $\ce{Cr2O7^{2-} + 14 H+ + 6 e^- -> 2 Cr^{3+} + 7 H2O}$ \quad et \quad
        $\ce{HNO2 + H2O -> NO3^- + 3 H+ + 2 e^-}$.
  \item Réduction $\times 1$, oxydation $\times 3$ (6 e$^-$), addition et simplification :
        \[
        \ce{Cr2O7^{2-} + 5 H+ + 3 HNO2 -> 2 Cr^{3+} + 3 NO3^- + 4 H2O}.
        \]
  \item On remplace les 5 \ce{H+} par 5 \ce{H3O+} (ajout de 5 \ce{H2O} à droite : $4+5=9$) :
        \[
        \boxed{\ce{Cr2O7^{2-} + 3 HNO2 + 5 H3O+ -> 2 Cr^{3+} + 3 NO3^- + 9 H2O}}
        \]
        Suite des coefficients : $\mathbf{1,3,5 \to 2,3,9}$ (\textbf{réponse A}, annales 2022-Q23).
        Contrôle : charges $-2+5=+3=2(+3)+3(-1)$ ; O $7+6+5=18=9+9$ ; H $3+15=18=18$. \checkmark
\end{enumerate}
\end{corr}

\begin{corr}[équilibrage complet en milieu basique]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item \ce{CrO4^{2-}} : $\NO(\ce{Cr})=+\mathrm{VI}$ ; \ce{Cr2O3} : $+\mathrm{III}$. Réduction, 2 Cr $\Rightarrow$
        $2\times3=6$ e$^-$. \ce{Fe} : $0\to+\mathrm{III}$ ; oxydation, 2 Fe $\Rightarrow 2\times3=6$ e$^-$.
  \item \textbf{Réduction} (basique) — acide d'abord :
        $\ce{2 CrO4^{2-} + 10 H+ + 6 e^- -> Cr2O3 + 5 H2O}$, puis $+10\,\ce{OH^-}$, simplification de 5 \ce{H2O} :
        \[
        \ce{2 CrO4^{2-} + 5 H2O + 6 e^- -> Cr2O3 + 10 OH^-}.
        \]
  \item \textbf{Oxydation} (basique) — acide d'abord :
        $\ce{2 Fe + 3 H2O -> Fe2O3 + 6 H+ + 6 e^-}$, puis $+6\,\ce{OH^-}$, simplification de 3 \ce{H2O} :
        \[
        \ce{2 Fe + 6 OH^- -> Fe2O3 + 3 H2O + 6 e^-}.
        \]
  \item Les deux échangent 6 e$^-$ : addition et simplification ($\ce{OH^-}$ : $10-6=4$ à droite ;
        \ce{H2O} : $5-3=2$ à gauche) :
        \[
        \boxed{\ce{2 CrO4^{2-} + 2 H2O + 2 Fe -> Cr2O3 + Fe2O3 + 4 OH^-}}
        \]
        \textbf{Somme de tous les coefficients} : $2+2+2+1+1+4 = \mathbf{12}$ (et l'on remarque que
        réactifs $=6=$ produits). Le fer passe de $0$ à $+\mathrm{III}$ : il est \textbf{oxydé}, c'est donc
        un \textbf{réducteur} (jamais un oxydant). \emph{(Propositions C et D vraies, livre QCM~14.)}
\end{enumerate}
\end{corr}
