\begin{exo}[prévoir le sens et écrire l'équation]
Données : $E_{\mathrm{r}}^{\circ}(\ce{Fe^{3+}}/\ce{Fe^{2+}})=+0{,}77$~V et
$E_{\mathrm{r}}^{\circ}(\ce{I2}/\ce{I^-})=+0{,}54$~V.
\begin{enumerate}[label=\alph*)]
  \item Les ions \ce{Fe^{3+}} oxydent-ils les ions iodure \ce{I^-} ? Justifie par $\Delta E^{\circ}$.
  \item Écris l'équation de la réaction spontanée.
\end{enumerate}
\end{exo}

\begin{exo}[un oxydant face à deux métaux]
On plonge une lame de cuivre \emph{et} une lame de zinc dans une solution de nitrate d'argent
\ce{AgNO3}. Données : $E_{\mathrm{r}}^{\circ}(\ce{Ag+}/\ce{Ag})=+0{,}80$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Cu^{2+}}/\ce{Cu})=+0{,}34$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Zn^{2+}}/\ce{Zn})=-0{,}76$~V.
\begin{enumerate}[label=\alph*)]
  \item \ce{Ag+} oxyde-t-il le cuivre ? le zinc ?
  \item Écris les équations correspondantes.
\end{enumerate}
\end{exo}

\begin{exo}[un métal en déplace-t-il un autre ?]
Un métal déplace un autre métal de son sel s'il est le \textbf{réducteur le plus fort}. Données :
$E_{\mathrm{r}}^{\circ}(\ce{Fe^{2+}}/\ce{Fe})=-0{,}44$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Cu^{2+}}/\ce{Cu})=+0{,}34$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Ag+}/\ce{Ag})=+0{,}80$~V.
\begin{enumerate}[label=\alph*)]
  \item Le fer déplace-t-il le cuivre d'une solution de \ce{CuSO4} ? Écris l'équation.
  \item L'argent déplace-t-il le cuivre d'une solution de \ce{CuSO4} ?
\end{enumerate}
\end{exo}

\begin{exo}[une cascade d'oxydations]
Données : $E_{\mathrm{r}}^{\circ}(\ce{MnO4^-}/\ce{Mn^{2+}})=+1{,}51$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Fe^{3+}}/\ce{Fe^{2+}})=+0{,}77$ ;
$E_{\mathrm{r}}^{\circ}(\ce{I2}/\ce{I^-})=+0{,}54$~V. Pour chaque cas, dis si la réaction est possible.
\begin{enumerate}[label=\alph*)]
  \item Le permanganate oxyde-t-il \ce{Fe^{2+}} ?
  \item Le permanganate oxyde-t-il \ce{I^-} ?
  \item Les ions \ce{Fe^{3+}} oxydent-ils \ce{I^-} ?
\end{enumerate}
\end{exo}

\begin{exo}[qui peut réduire l'ion argent ?]
Parmi les espèces \ce{Ca}, \ce{Cu^{2+}}, \ce{Br2} et \ce{Au}, laquelle peut \textbf{réduire}
\ce{Ag+} en \ce{Ag} ? Justifie chaque cas. Données : $E_{\mathrm{r}}^{\circ}(\ce{Ag+}/\ce{Ag})
=+0{,}80$ ; $E_{\mathrm{r}}^{\circ}(\ce{Ca^{2+}}/\ce{Ca})=-2{,}84$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Cu^{2+}}/\ce{Cu})=+0{,}34$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Br2}/\ce{Br^-})=+1{,}06$ ;
$E_{\mathrm{r}}^{\circ}(\ce{Au^{3+}}/\ce{Au})=+1{,}52$~V.
\end{exo}
