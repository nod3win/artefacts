\begin{exo}[escalier partiel : glace $\rightarrow$ eau tiède]
On chauffe $\qty{0.10}{\kilogram}$ de glace de $\qty{-20}{\celsius}$ jusqu'à de l'eau liquide à $\qty{20}{\celsius}$. Données : $c_{\text{glace}}=\qty{2.060}{\kilo\joule\per\kelvin\per\kilogram}$, $L_{\text{fusion}}=\qty{334}{\kilo\joule\per\kilogram}$, $c_{\text{eau}}=\qty{4.185}{\kilo\joule\per\kelvin\per\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Quelles sont les trois étapes du parcours ?
  \item Calcule la chaleur totale $Q$.
\end{enumerate}
\end{exo}

\begin{exo}[température finale d'un mélange]
Dans un calorimètre adiabatique, on mélange $\qty{2.0}{\kilogram}$ d'eau à $\qty{80}{\celsius}$ et $\qty{1.0}{\kilogram}$ d'eau à $\qty{20}{\celsius}$ ($c_{\text{eau}}$ identique pour les deux).
\begin{enumerate}[label=\alph*)]
  \item Écris le bilan $\sum Q_i = 0$.
  \item Détermine la température finale $T_{\text{f}}$.
\end{enumerate}
\end{exo}

\begin{exo}[capacité thermique d'un métal]
On plonge une barre métallique de $\qty{0.50}{\kilogram}$ portée à $\qty{100}{\celsius}$ dans $\qty{1.0}{\kilogram}$ d'eau à $\qty{20}{\celsius}$ (calorimètre adiabatique, capacité négligeable). À l'équilibre, $T_{\text{f}}=\qty{25}{\celsius}$. On donne $c_{\text{eau}}=\qty{4.185}{\kilo\joule\per\kelvin\per\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la chaleur reçue par l'eau.
  \item En déduis la capacité thermique massique $c$ du métal.
\end{enumerate}
\end{exo}

\begin{exo}[mesurer une chaleur de neutralisation]
On mélange $\qty{100}{\milli\litre}$ d'une solution acide et $\qty{100}{\milli\litre}$ d'une base, tous deux à $\qty{21.0}{\celsius}$ ; la température monte à $\qty{27.0}{\celsius}$. On prend une masse volumique de $\qty{1}{\gram\per\milli\litre}$, le verre ne s'échauffe pas, $c_{\text{eau}}=\qty{4.185}{\kilo\joule\per\kelvin\per\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Détermine la masse totale de solution et $\Delta T$.
  \item Calcule la chaleur dégagée par la réaction (en $\si{\kilo\joule}$ puis en $\si{\joule}$).
\end{enumerate}
\end{exo}

\begin{exo}[enthalpie molaire de dissolution]
On dissout $\qty{0.25}{\mol}$ d'un sel dans $\qty{0.20}{\kilogram}$ d'eau à $\qty{20.0}{\celsius}$ (calorimètre adiabatique, verre négligé). La température monte à $\qty{22.0}{\celsius}$. On donne $c_{\text{eau}}=\qty{4.185}{\kilo\joule\per\kelvin\per\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la chaleur captée par l'eau.
  \item La dissolution est-elle exo- ou endothermique ?
  \item Calcule l'enthalpie molaire de dissolution $\Delta H$ (en $\si{\kilo\joule\per\mol}$).
\end{enumerate}
\end{exo}
