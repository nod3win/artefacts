\begin{center}
{\Large\bfseries\color{primarydark} Thème 3 — Calorimétrie}\\[2pt]
{\color{primary}\small Mesurer et bilan-ter les quantités de chaleur}
\end{center}

\begin{cle}[les deux chaleurs et le bilan]
Dans un \textbf{calorimètre} (enceinte adiabatique), on suit la température pour mesurer des chaleurs. Deux formules suffisent.
\begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=2pt]
  \item \textbf{Chaleur sensible} (la température change) : \; $\boxed{\,Q = m\,c\,\Delta T\,}$ \;, avec $\Delta T = T_{\text{final}} - T_{\text{initial}}$.
  \item \textbf{Chaleur latente} (changement d'état, à température \emph{constante}) : \; $\boxed{\,Q = m\,L\,}$.
\end{itemize}
\textbf{Bilan calorimétrique} : dans une enceinte adiabatique, aucune chaleur ne sort, donc
\[ \boxed{\;\textstyle\sum_i Q_i = 0\;} \qquad\Longleftrightarrow\qquad \big|\textstyle\sum Q_{\text{reçues}}\big| = \big|\textstyle\sum Q_{\text{cédées}}\big|. \]
Une chaleur \emph{reçue} est $>0$, une chaleur \emph{cédée} est $<0$.
\end{cle}

\begin{methode}[grandeurs de l'eau et enchaînement des étapes]
Valeurs usuelles pour l'eau :
\[ c_{\text{eau}}=\qty{4.185}{\kilo\joule\per\kelvin\per\kilogram},\quad
   c_{\text{glace}}=\qty{2.060}{\kilo\joule\per\kelvin\per\kilogram},\quad
   c_{\text{vapeur}}=\qty{1.85}{\kilo\joule\per\kelvin\per\kilogram}, \]
\[ L_{\text{fusion}}=\qty{334}{\kilo\joule\per\kilogram},\qquad L_{\text{vaporisation}}=\qty{2257}{\kilo\joule\per\kilogram}. \]
Pour un chauffage qui traverse plusieurs états, on décompose en un \textbf{escalier} : segments \emph{horizontaux} ($Q=mc\Delta T$) et \emph{verticaux} ($Q=mL$), puis on additionne. Pour un \textbf{mélange}, on écrit $\sum Q_i=0$ et on isole $T_{\text{final}}$.
\end{methode}

\begin{exemplebox}[neutralisation dans un calorimètre]
On mélange $\qty{50}{\milli\litre}$ de $\ce{NaOH}$ et $\qty{50}{\milli\litre}$ de $\ce{HCl}$, tous deux à $\qty{20}{\celsius}$ ; la température monte à $\qty{30}{\celsius}$. Masse totale d'eau $m=\qty{0.1}{\kilogram}$, $\Delta T=\qty{10}{\celsius}$. La chaleur captée par l'eau vaut
\[ Q = m\,c_{\text{eau}}\,\Delta T = 0{,}1\times 4{,}185\times 10 = \qty{4.185}{\kilo\joule} = \qty{4185}{\joule}. \]
Le verre ne s'échauffant pas, c'est toute la chaleur dégagée par la réaction.
\end{exemplebox}

\begin{piege}[erreurs fréquentes]
\begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=2pt]
  \item \textbf{Unités} : garder $m$ en $\si{\kilogram}$ et $c$ en $\si{\kilo\joule\per\kelvin\per\kilogram}$ ; $\qty{1}{\kilo\joule}=\qty{1000}{\joule}$.
  \item une \textbf{variation} $\Delta T$ a la même valeur en $\si{\kelvin}$ et en $\si{\celsius}$ (\;$\qty{10}{\celsius}$ d'écart $=\qty{10}{\kelvin}$\;).
  \item pendant un changement d'état, $T$ ne varie pas : on utilise $Q=mL$, jamais $mc\Delta T$.
  \item ne pas oublier la chaleur captée par le \textbf{calorimètre} lui-même quand l'énoncé la donne.
\end{itemize}
\end{piege}
