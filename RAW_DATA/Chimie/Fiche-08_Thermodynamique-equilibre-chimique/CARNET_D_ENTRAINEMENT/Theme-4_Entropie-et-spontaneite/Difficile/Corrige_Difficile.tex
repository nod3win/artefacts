\begin{corrige}[format concours : variation d'entropie vraie ou fausse]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux} : $4$ mol de gaz $\to 2$ mol, le désordre diminue ($\Delta S<0$).
  \item \textbf{Vrai} : dissolution d'un solide, le désordre augmente ($\Delta S>0$).
  \item \textbf{Vrai} : $2$ mol de gaz $\to$ un solide, le désordre diminue ($\Delta S<0$).
  \item \textbf{Vrai} : création de gaz à partir d'un solide ($\Delta S>0$).
  \item \textbf{Faux} : $2$ mol de gaz $\to 2$ mol, $\Delta S\approx 0$ (pas nettement $>0$).
\end{enumerate}
\end{corrige}

\begin{corrige}[choisir la plus forte augmentation de désordre]
\begin{enumerate}[label=\alph*)]
  \item signes :
  \begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=1pt]
    \item (1) $0$ mol de gaz $\to 3$ mol : $\Delta S>0$ (fortement) ;
    \item (2) $4$ mol de gaz $\to 2$ mol : $\Delta S<0$ ;
    \item (3) $2$ mol de gaz $\to 2$ mol : $\Delta S\approx 0$ ;
    \item (4) $0$ mol de gaz $\to 1$ mol : $\Delta S>0$.
  \end{itemize}
  \item la réaction \textbf{(1)} : elle crée $3$ moles de gaz à partir de solides, soit la plus forte création de désordre (davantage que (4) qui n'en crée qu'une mole).
\end{enumerate}
\end{corrige}

\begin{corrige}[3\textsuperscript{e} principe et entropie absolue]
\begin{enumerate}[label=\alph*)]
  \item 3\textsuperscript{e} principe : à $T=\qty{0}{\kelvin}$, un cristal parfait est parfaitement ordonné, $S=\qty{0}{\joule\per\mol\per\kelvin}$.
  \item à $\qty{298}{\kelvin}$, l'agitation thermique est plus grande : le cristal est plus désordonné, donc $S(\qty{298}{\kelvin}) > S(\qty{0}{\kelvin}) = 0$.
  \item le 3\textsuperscript{e} principe fournit une \emph{origine} ($S=0$ à $\qty{0}{\kelvin}$) commune à tous les corps : on peut donc mesurer une entropie absolue en cumulant depuis $\qty{0}{\kelvin}$. L'enthalpie n'a pas d'origine absolue accessible ; on ne tabule que la variation lors de la formation, $\Delta H^\circ_{\text{f}}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[vrai ou faux — 2\textsuperscript{e} et 3\textsuperscript{e} principes]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux} : un processus spontané fait \emph{croître} l'entropie de l'univers ($\Delta S_{\text{univers}}>0$).
  \item \textbf{Vrai} : gaz $>$ solide pour le désordre, donc $S_{\text{gaz}}>S_{\text{solide}}$.
  \item \textbf{Faux} : il donne le \emph{sens} d'évolution (spontanéité), pas la vitesse, qui relève de la cinétique.
  \item \textbf{Vrai} : c'est exactement le 3\textsuperscript{e} principe.
\end{enumerate}
\end{corrige}
