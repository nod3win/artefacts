\begin{corrige}[vrai ou faux : « le désordre augmente »]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux} : $3$ mol de gaz $\to 2$ mol, le désordre diminue ($\Delta S<0$).
  \item \textbf{Vrai} : la dissolution disperse le solide en ions ($\Delta S>0$).
  \item \textbf{Vrai} : sublimation solide $\to$ gaz ($\Delta S>0$).
  \item \textbf{Faux} : $2$ mol de gaz $\to 2$ mol, $\Delta S\approx 0$ (le désordre n'augmente pas vraiment).
\end{enumerate}
\end{corrige}

\begin{corrige}[comparer des entropies]
\begin{enumerate}[label=\alph*)]
  \item solide $<$ liquide $<$ gaz : \; $\ce{H2O}(s) < \ce{H2O}(l) < \ce{H2O}(g)$.
  \item le gaz est plus désordonné : \; $\ce{Br2}(l) < \ce{Br2}(g)$.
\end{enumerate}
\end{corrige}

\begin{corrige}[prévoir le signe de $\Delta S$]
\begin{enumerate}[label=\alph*)]
  \item $3$ mol de gaz $\to 2$ mol : $\Delta S<0$.
  \item $1$ mol de gaz $\to 2$ mol : $\Delta S>0$.
  \item $1$ mol de gaz $\to 0$ mol : $\Delta S<0$.
  \item $0$ mol de gaz $\to 3$ mol : $\Delta S>0$ (forte augmentation).
\end{enumerate}
\end{corrige}

\begin{corrige}[le 3\textsuperscript{e} principe et l'origine des entropies]
\begin{enumerate}[label=\alph*)]
  \item à $T=\qty{0}{\kelvin}$, un cristal parfait est parfaitement ordonné : son entropie est nulle, $S=\qty{0}{\joule\per\mol\per\kelvin}$.
  \item l'échantillon à $\qty{500}{\kelvin}$ : plus chaud $\Rightarrow$ agitation plus grande $\Rightarrow$ désordre plus grand $\Rightarrow$ entropie plus grande.
  \item le 3\textsuperscript{e} principe fixe une \emph{origine commune} ($S=0$ à $\qty{0}{\kelvin}$) ; on peut donc cumuler les variations depuis cette origine. L'enthalpie n'a pas d'origine de ce type : seule $\Delta H$ est mesurable.
\end{enumerate}
\end{corrige}

\begin{corrige}[quand le nombre de moles de gaz ne change pas]
\begin{enumerate}[label=\alph*)]
  \item $2$ mol de gaz à gauche ($\ce{H2}+\ce{I2}$), $2$ mol de gaz à droite ($2\,\ce{HI}$).
  \item le nombre de moles de gaz est \emph{inchangé} : $\Delta S\approx 0$ (très faible, ni nettement positif ni négatif).
\end{enumerate}
\end{corrige}
