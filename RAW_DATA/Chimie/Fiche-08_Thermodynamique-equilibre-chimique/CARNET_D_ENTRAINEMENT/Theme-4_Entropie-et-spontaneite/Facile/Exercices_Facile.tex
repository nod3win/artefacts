\begin{exo}[signe de $\Delta S$ d'après les gaz]
Pour chaque réaction, compte les moles de gaz de part et d'autre et donne le signe de $\Delta S$.
\begin{enumerate}[label=\alph*)]
  \item $\ce{N2}(g) + 3\,\ce{H2}(g) \rightarrow 2\,\ce{NH3}(g)$
  \item $\ce{CaCO3}(s) \rightarrow \ce{CaO}(s) + \ce{CO2}(g)$
  \item $2\,\ce{H2}(g) + \ce{O2}(g) \rightarrow 2\,\ce{H2O}(g)$
  \item $2\,\ce{NH3}(g) \rightarrow \ce{N2}(g) + 3\,\ce{H2}(g)$
\end{enumerate}
\end{exo}

\begin{exo}[changements d'état]
Donne le signe de $\Delta S$ pour chaque transformation physique.
\begin{enumerate}[label=\alph*)]
  \item fusion de la glace : $\ce{H2O}(s) \rightarrow \ce{H2O}(l)$
  \item vaporisation : $\ce{H2O}(l) \rightarrow \ce{H2O}(g)$
  \item liquéfaction : $\ce{H2O}(g) \rightarrow \ce{H2O}(l)$
  \item sublimation : $\ce{CO2}(s) \rightarrow \ce{CO2}(g)$
\end{enumerate}
\end{exo}

\begin{exo}[dissolution et précipitation]
Donne le signe de $\Delta S$ et justifie en un mot.
\begin{enumerate}[label=\alph*)]
  \item dissolution du sucre : $\ce{C12H22O11}(s) \rightarrow \ce{C12H22O11}(aq)$
  \item précipitation : $\ce{Ag+}(aq) + \ce{Cl-}(aq) \rightarrow \ce{AgCl}(s)$
\end{enumerate}
\end{exo}

\begin{exo}[ordre des états et 3\textsuperscript{e} principe]
\begin{enumerate}[label=\alph*)]
  \item Pour une même substance, range par entropie \emph{croissante} les états : gaz, solide, liquide.
  \item Que vaut l'entropie d'un cristal parfait à $T=\qty{0}{\kelvin}$ ?
\end{enumerate}
\end{exo}

\begin{exo}[le 2\textsuperscript{e} principe]
Réponds en une phrase.
\begin{enumerate}[label=\alph*)]
  \item Selon quel critère un processus est-il spontané ?
  \item Un processus spontané fait-il \emph{croître} ou \emph{décroître} le désordre de l'univers ?
  \item « Spontané » signifie-t-il « rapide » ?
\end{enumerate}
\end{exo}
