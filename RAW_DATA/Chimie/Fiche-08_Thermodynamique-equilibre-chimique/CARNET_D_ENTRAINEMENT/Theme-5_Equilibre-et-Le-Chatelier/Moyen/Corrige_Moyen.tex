\begin{corrige}[calculer $K$ à partir des concentrations]
\begin{enumerate}[label=\alph*)]
  \item $K = \dfrac{[\ce{HI}]^{2}}{[\ce{H2}]\,[\ce{I2}]}$.
  \item $K = \dfrac{(1{,}6)^{2}}{0{,}20\times 0{,}20} = \dfrac{2{,}56}{0{,}040} = 64$.
  \item $K=64\gg 1$ : l'équilibre est très déplacé vers les \textbf{produits} ($\ce{HI}$).
\end{enumerate}
\end{corrige}

\begin{corrige}[l'effet de l'écriture sur $K$]
\begin{enumerate}[label=\alph*)]
  \item coefficients $\times 3$ : tous les exposants sont triplés, donc $K' = K^{3} = 2^{3} = 8$.
  \item réaction inverse : $K'' = \dfrac{1}{K} = \dfrac{1}{2} = 0{,}5$.
\end{enumerate}
\end{corrige}

\begin{corrige}[premiers pas avec Le Chatelier]
Le système s'oppose à la perturbation (réaction exothermique dans le sens direct).
\begin{enumerate}[label=\alph*)]
  \item vers la \textbf{droite} : l'équilibre consomme l'excès de $\ce{N2}$.
  \item vers la \textbf{gauche} : $T\nearrow$ pousse dans le sens \emph{endothermique}, ici le sens inverse.
  \item vers la \textbf{droite} : $4$ mol de gaz à gauche, $2$ à droite ; $p\nearrow$ favorise le côté à moins de gaz.
  \item \textbf{aucun déplacement} : un catalyseur n'agit que sur la vitesse.
\end{enumerate}
\end{corrige}

\begin{corrige}[équilibre hétérogène : une pression à trouver]
\begin{enumerate}[label=\alph*)]
  \item $\ce{FeO}$ et $\ce{Fe}$ sont solides : $K = \dfrac{p_{\ce{CO2}}}{p_{\ce{CO}}}$.
  \item avec $p_{\ce{CO2}}=x$ et $p_{\ce{CO}}=1-x$ : $\dfrac{x}{1-x}=0{,}5 \Rightarrow x = 0{,}5(1-x) \Rightarrow 1{,}5\,x = 0{,}5 \Rightarrow x = \dfrac{0{,}5}{1{,}5} = \dfrac{1}{3}\approx\qty{0.33}{atm}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[degré d'avancement à l'équilibre]
\begin{enumerate}[label=\alph*)]
  \item dans $\qty{1.0}{\litre}$ : $[\ce{X}]=1{,}0-x$ et $[\ce{Y}]=x$ (en $\si{\mol\per\litre}$).
  \item $K = \dfrac{[\ce{Y}]}{[\ce{X}]} = \dfrac{x}{1{,}0-x} = 1{,}5 \Rightarrow x = 1{,}5(1-x) \Rightarrow 2{,}5\,x = 1{,}5 \Rightarrow x = 0{,}60$.
  \item $x = \qty{0.60}{\mol}$ sur $\qty{1.0}{\mol}$ : \; $\qty{60}{\percent}$ de $\ce{X}$ a été transformé.
\end{enumerate}
\end{corrige}
