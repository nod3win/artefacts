\begin{corrige}[écrire la constante d'équilibre]
Produits sur réactifs, exposants $=$ coefficients.
\begin{enumerate}[label=\alph*)]
  \item $K = \dfrac{[\ce{NH3}]^{2}}{[\ce{N2}]\,[\ce{H2}]^{3}}$.
  \item $K = \dfrac{[\ce{HI}]^{2}}{[\ce{H2}]\,[\ce{I2}]}$.
  \item $K = \dfrac{[\ce{PCl3}]\,[\ce{Cl2}]}{[\ce{PCl5}]}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[quelles espèces figurent dans $K$ ?]
On retire les solides et liquides purs.
\begin{enumerate}[label=\alph*)]
  \item $\ce{CaCO3}(s)$ et $\ce{CaO}(s)$ disparaissent : $K = [\ce{CO2}]$.
  \item $\ce{B}(s)$ disparaît : $K = \dfrac{[\ce{C}]^{2}\,[\ce{D}]}{[\ce{A}]^{2}}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[interpréter la valeur de $K$]
\begin{enumerate}[label=\alph*)]
  \item $K=10^{5}\gg 1$ : équilibre très déplacé vers les \textbf{produits} (réaction quasi totale).
  \item $K=10^{-4}\ll 1$ : équilibre très déplacé vers les \textbf{réactifs} (réaction très limitée).
  \item $K\approx 1$ : réactifs et produits coexistent en proportions comparables.
\end{enumerate}
\end{corrige}

\begin{corrige}[vrai ou faux : l'état d'équilibre]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai} : c'est un état stationnaire, les concentrations sont constantes.
  \item \textbf{Faux} : l'équilibre est \emph{dynamique}, les deux sens continuent à vitesses égales.
  \item \textbf{Faux} : elles sont constantes, mais pas nécessairement égales.
  \item \textbf{Vrai} : $K$ ne dépend que de la température.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer $K$ en une ligne]
\begin{enumerate}[label=\alph*)]
  \item $K = \dfrac{[\ce{PCl3}]\,[\ce{Cl2}]}{[\ce{PCl5}]}$.
  \item $K = \dfrac{1{,}40\times 0{,}50}{0{,}40} = \dfrac{0{,}70}{0{,}40} = 1{,}75$. Comme $K>1$, l'équilibre est légèrement déplacé vers les \textbf{produits}.
\end{enumerate}
\end{corrige}
