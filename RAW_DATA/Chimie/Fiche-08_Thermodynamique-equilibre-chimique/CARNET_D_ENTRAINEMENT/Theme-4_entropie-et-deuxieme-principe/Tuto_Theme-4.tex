\section{L'entropie : une mesure du désordre}

\begin{definitionb}[entropie $S$]
L'\textbf{entropie} $S$ mesure le \textbf{désordre} d'un système (le nombre d'arrangements possibles de
ses constituants). Unité : \si{\joule\per\mol\per\kelvin}. Plus un système est désordonné, plus $S$ est grande.
\end{definitionb}

\section{Prévoir le signe de \texorpdfstring{$\dS$}{DS} d'une transformation}

\begin{methodeb}[le désordre augmente-t-il ?]
On compare le désordre des produits à celui des réactifs :
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item \textbf{apparition de gaz} ou \textbf{plus de moles de gaz} $\Rightarrow \dS>0$ ;
  \item \textbf{disparition de gaz} ou \textbf{moins de moles de gaz} $\Rightarrow \dS<0$ ;
  \item \textbf{dissolution} d'un solide $\Rightarrow \dS>0$ ; précipitation $\Rightarrow \dS<0$ ;
  \item \textbf{fusion, vaporisation, sublimation} $\Rightarrow \dS>0$ ; l'inverse $\Rightarrow \dS<0$.
\end{itemize}
\emph{Le facteur décisif est presque toujours le nombre de moles de gaz.}
\end{methodeb}

\begin{exempleb}[quatre cas typiques]
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item $\ce{N2}(g)+3\,\ce{H2}(g)\rightarrow2\,\ce{NH3}(g)$ : $4\to2$ mol de gaz, $\dS<0$.
  \item $\ce{CaCO3}(s)\rightarrow\ce{CaO}(s)+\ce{CO2}(g)$ : apparition de gaz, $\dS>0$.
  \item $\ce{C12H22O11}(s)\rightarrow\ce{C12H22O11}(aq)$ : dissolution, $\dS>0$.
  \item $\ce{CO2}(s)\rightarrow\ce{CO2}(g)$ : sublimation, $\dS>0$.
\end{itemize}
\end{exempleb}

\section{Deuxième et troisième principes}

\begin{principeb}[deuxième principe — sens d'évolution spontané]
Un processus est \textbf{spontané} si le désordre \emph{total} augmente, c'est-à-dire si l'entropie de
l'\textbf{univers} croît : $\;\dS_{\text{univers}} > 0$. Quand cette condition est réalisée, la réaction
tend à devenir \textbf{complète}.
\end{principeb}

\begin{principeb}[troisième principe — origine des entropies]
À la température du zéro absolu ($T=\SI{0}{\kelvin}$), un cristal parfait est parfaitement ordonné :
$\;S=\SI{0}{\joule\per\mol\per\kelvin}$. Ce principe fixe une \emph{origine} commune aux entropies.
\end{principeb}

\begin{attentionb}
« Spontané » se juge sur l'\textbf{univers} ($\dS_{\text{univers}}$), pas sur le seul système : une
réaction \emph{endothermique} peut être spontanée si elle augmente assez le désordre (dissolution d'un
sel qui refroidit, fusion de la glace\ldots). Le sens d'évolution n'est \emph{pas} dicté par la seule énergie.
\end{attentionb}
