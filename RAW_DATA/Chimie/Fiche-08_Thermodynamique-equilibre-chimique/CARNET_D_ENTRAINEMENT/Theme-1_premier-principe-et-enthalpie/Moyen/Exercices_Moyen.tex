\begin{exercice}[travail des forces de pression]
Un gaz évolue à \textbf{pression extérieure constante}. On rappelle $W = -\,p\,\Delta V$, avec $p$ en
\si{\pascal} et $\Delta V$ en \si{\cubic\metre} ($\SI{1}{\litre}=\SI{e-3}{\cubic\metre}$).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Le gaz se \textbf{dilate} de \SI{2.0}{\litre} à \SI{5.0}{\litre} sous $p=\SI{1.0e5}{\pascal}$.
        Calculer $W$ ; le gaz reçoit-il ou cède-t-il du travail ?
  \item Le gaz est \textbf{comprimé} de \SI{4.0}{\litre} à \SI{1.0}{\litre} sous $p=\SI{2.0e5}{\pascal}$.
        Calculer $W$ et interpréter son signe.
\end{enumerate}
\end{exercice}

\begin{exercice}[relier $\dH$, $W$ et $\dU$]
À \textbf{pression constante}, la chaleur échangée est $Q_p=\dH$, et le premier principe s'écrit
$\dU = \dH + W$.

La synthèse de l'eau \;$2\,\ce{H2}(g) + \ce{O2}(g) \rightarrow 2\,\ce{H2O}(l)$\; se déroule à pression
constante. Elle libère $\SI{572}{\kilo\joule}$ sous forme de chaleur, et comme elle \emph{consomme} du
gaz, le milieu extérieur fournit au système un travail $W = \SI{+8}{\kilo\joule}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Donner $\dH$ avec son signe. La réaction est-elle exo ou endothermique ?
  \item Calculer $\dU$.
  \item Pourquoi le travail est-il ici \emph{reçu} par le système ($W>0$) ?
\end{enumerate}
\end{exercice}

\begin{exercice}[retrouver le travail échangé]
Un gaz \textbf{reçoit} $Q=\SI{800}{\joule}$ de chaleur. On mesure que son énergie interne \textbf{augmente}
de $\SI{500}{\joule}$ ($\dU=\SI{+500}{\joule}$).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item À partir de $\dU=Q+W$, calculer le travail $W$.
  \item Le gaz \emph{reçoit}-il ou \emph{cède}-t-il ce travail ? S'est-il plutôt dilaté ou comprimé ?
\end{enumerate}
\end{exercice}

\begin{exercice}[placer les produits sur le diagramme]
On veut tracer le diagramme enthalpique d'une réaction dont les \textbf{réactifs} sont au niveau
$H_{\text{réactifs}} = \SI{60}{\kilo\joule}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item La réaction a $\dH = \SI{+150}{\kilo\joule}$. À quel niveau $H$ se placent les produits ?
        Est-elle exo ou endothermique ?
  \item Même question pour une autre réaction avec $\dH = \SI{-45}{\kilo\joule}$ (mêmes réactifs à \SI{60}{\kilo\joule}).
  \item Dans lequel des deux cas la flèche $\dH$ pointe-t-elle vers le \emph{haut} sur le diagramme ?
\end{enumerate}
\end{exercice}

\begin{exercice}[lire le « + Énergie » d'une équation]
On écrit la combustion du méthane sous la forme :
\[ \ce{CH4}(g) + 2\,\ce{O2}(g) \;\longrightarrow\; \ce{CO2}(g) + 2\,\ce{H2O}(l) + \text{Énergie}. \]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Que traduit le terme « $+\,$Énergie » écrit du côté des produits ? En déduire le signe de $\dH$.
  \item D'où provient l'énergie ainsi libérée ?
  \item Les réactifs ($\ce{CH4}$, $\ce{O2}$) ont-ils une enthalpie plus \emph{élevée} ou plus \emph{basse}
        que celle des produits ? Justifier à l'aide de $\dH = H_{\text{produits}} - H_{\text{réactifs}}$.
\end{enumerate}
\end{exercice}
