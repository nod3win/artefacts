\begin{correction}[conventions de signe]
Règle : ce que le système \emph{reçoit} est positif, ce qu'il \emph{cède} est négatif.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Chaleur \emph{reçue} : \rep{$Q>0$} (ici $Q=\SI{+300}{\joule}$).
  \item Le gaz se dilate, il \emph{fournit} (cède) du travail : \rep{$W<0$} (ici $W=\SI{-120}{\joule}$).
  \item Une réaction exothermique \emph{dégage} de la chaleur, donc $Q_p<0$ ; or $Q_p=\dH$, d'où \rep{$\dH<0$}.
  \item Système isolé : aucun échange, donc \rep{$Q=0$ et $W=0$}, et par le premier principe
        $\dU = Q+W = \rep{0}$ ($U$ reste constante).
\end{enumerate}
\end{correction}

\begin{correction}[appliquer $\dU = Q + W$]
On attribue les signes puis on additionne.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=3pt]
  \item $Q=\SI{+500}{\joule}$, $W=\SI{+150}{\joule}$ : $\;\dU = 500 + 150 = \rep{\SI{+650}{\joule}}$.
  \item $Q=\SI{+400}{\joule}$, $W=\SI{-250}{\joule}$ (travail cédé) : $\;\dU = 400 - 250 = \rep{\SI{+150}{\joule}}$.
  \item $Q=\SI{-600}{\joule}$ (chaleur cédée), $W=\SI{+600}{\joule}$ : $\;\dU = -600 + 600 = \rep{\SI{0}{\joule}}$.
\end{enumerate}
Le cas (c) illustre qu'un système peut recevoir autant de travail qu'il cède de chaleur : son énergie
interne ne varie alors pas.
\end{correction}

\begin{correction}[exo ou endo à partir du signe de $\dH$]
$\dH<0 \Rightarrow$ exothermique, produits \emph{plus bas} ; $\dH>0 \Rightarrow$ endothermique, produits \emph{plus haut}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\dH=\SI{-92}{\kilo\joule}<0$ : \rep{exothermique}, produits plus bas.
  \item $\dH=\SI{+572}{\kilo\joule}>0$ : \rep{endothermique}, produits plus haut (décomposer l'eau coûte de l'énergie).
  \item $\dH=\SI{-890}{\kilo\joule}<0$ : \rep{exothermique}, produits plus bas (une combustion l'est toujours).
  \item $\dH=\SI{+178}{\kilo\joule}>0$ : \rep{endothermique}, produits plus haut.
\end{enumerate}
\end{correction}

\begin{correction}[lire un diagramme enthalpique]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\dH = H_{\text{produits}} - H_{\text{réactifs}} = 40 - 120 = \rep{\SI{-80}{\kilo\joule}}$.
  \item $\dH<0$ : la réaction est \rep{exothermique}.
  \item La chaleur est \rep{dégagée} vers l'extérieur (les produits sont moins énergétiques ; l'excédent
        d'énergie part sous forme de chaleur).
\end{enumerate}
\end{correction}

\begin{correction}[vocabulaire de base]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Un \rep{système isolé} n'échange \emph{ni matière ni énergie} avec l'extérieur. Alors $Q=0$, $W=0$
        et $\dU=0$.
  \item \rep{Athermique} qualifie une \emph{réaction} qui n'échange pas de chaleur ($\dH\approx0$) ;
        \rep{adiabatique} qualifie une \emph{enceinte} (un mur, un calorimètre) que la chaleur ne
        traverse pas. L'un décrit un bilan de réaction, l'autre une paroi.
  \item Un calorimètre parfait ne laisse rien passer : c'est un système \rep{isolé} (adiabatique).
\end{enumerate}
\end{correction}
