\begin{corrige}[cycle de Hess à partir d'un diagramme]
\begin{enumerate}[label=\alph*)]
  \item les deux chemins mènent des éléments à $2\,\ce{NO2}$, donc ont la même enthalpie :
  \[ 2\times(+90) + \Delta H^\circ_{\text{r}} = 2\times(+33). \]
  \item $\Delta H^\circ_{\text{r}} = 2(+33) - 2(+90) = 66 - 180 = \qty{-114}{\kilo\joule}$.
  \item par mole de $\ce{NO2}$ : $\dfrac{-114}{2} = \qty{-57}{\kilo\joule\per\mol}$. La réaction est \textbf{exothermique} (bien que $\ce{NO}$ et $\ce{NO2}$ aient tous deux $\Delta H^\circ_{\text{f}}>0$).
\end{enumerate}
\end{corrige}

\begin{corrige}[déterminer une enthalpie de formation par combustion]
La combustion \emph{dégage} $\qty{1367}{\kilo\joule}$, donc $\Delta H^\circ_{\text{r}} = \qty{-1367}{\kilo\joule}$.
\begin{enumerate}[label=\alph*)]
  \item $\Delta H^\circ_{\text{r}} = \big[2\,\Delta H^\circ_{\text{f}}(\ce{CO2}) + 3\,\Delta H^\circ_{\text{f}}(\ce{H2O}(l))\big] - \Delta H^\circ_{\text{f}}(\ce{C2H5OH})$ \; (le $\ce{O2}$ apporte $0$).
  \item $-1367 = [2(-393) + 3(-286)] - \Delta H^\circ_{\text{f}}(\ce{C2H5OH}) = (-786 - 858) - \Delta H^\circ_{\text{f}}(\ce{C2H5OH})$, soit
  \[ \Delta H^\circ_{\text{f}}(\ce{C2H5OH}) = -1644 - (-1367) = \qty{-277}{\kilo\joule\per\mol}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[loi de Hess par combinaison de trois équations]
\begin{enumerate}[label=\alph*)]
  \item on veut $\ce{C} + 2\,\ce{H2} \rightarrow \ce{CH4}$. On prend $(1) + 2\times(2) - (3)$ :
  \begin{itemize}[leftmargin=1.4em,topsep=1pt,itemsep=0.5pt]
    \item $(1)$ apporte $\ce{C} + \ce{O2} \rightarrow \ce{CO2}$ ;
    \item $2\times(2)$ apporte $2\,\ce{H2} + \ce{O2} \rightarrow 2\,\ce{H2O}$ ;
    \item $-(3)$ apporte $\ce{CO2} + 2\,\ce{H2O} \rightarrow \ce{CH4} + 2\,\ce{O2}$.
  \end{itemize}
  En sommant, $\ce{CO2}$, $\ce{H2O}$ et les $\ce{O2}$ se simplifient : il reste $\ce{C}(s) + 2\,\ce{H2}(g) \rightarrow \ce{CH4}(g)$.
  \item $\Delta H^\circ_{\text{f}}(\ce{CH4}) = \Delta H_1 + 2\,\Delta H_2 - \Delta H_3 = -393 + 2(-286) - (-890) = -393 - 572 + 890 = \qty{-75}{\kilo\joule\per\mol}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[vrai ou faux — loi de Hess]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai} : $\ce{O2}$ est un corps pur simple dans son état stable, $\Delta H^\circ_{\text{f}}=0$ par convention.
  \item \textbf{Vrai} : $H$ étant une fonction d'état, $\Delta H$ ne dépend que des états initial et final, ce qui autorise l'additivité de Hess.
  \item \textbf{Faux} : inverser une réaction \emph{change} le signe de $\Delta H$.
  \item \textbf{Vrai} : le retour au point de départ implique $\sum \Delta H = 0$ (une flèche à contre-sens changeant de signe).
\end{enumerate}
\end{corrige}
