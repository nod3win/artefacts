\begin{exo}[enthalpie d'une combustion par la loi de Hess]
Combustion du méthane : $\ce{CH4}(g) + 2\,\ce{O2}(g) \rightarrow \ce{CO2}(g) + 2\,\ce{H2O}(l)$.\\
Données ($\si{\kilo\joule\per\mol}$) : $\Delta H^\circ_{\text{f}}(\ce{CH4})=-75$, \; $\Delta H^\circ_{\text{f}}(\ce{CO2})=-393$, \; $\Delta H^\circ_{\text{f}}(\ce{H2O}(l))=-286$.
\begin{enumerate}[label=\alph*)]
  \item Écris l'expression de $\Delta H^\circ_{\text{r}}$.
  \item Calcule sa valeur.
\end{enumerate}
\end{exo}

\begin{exo}[inverser et diviser une réaction]
On établit que $2\,\ce{SO2}(g) + \ce{O2}(g) \rightarrow 2\,\ce{SO3}(g)$ a pour enthalpie $\Delta H^\circ_{\text{r}} = \qty{-200}{\kilo\joule}$. Déduis-en l'enthalpie de :
\begin{enumerate}[label=\alph*)]
  \item $2\,\ce{SO3}(g) \rightarrow 2\,\ce{SO2}(g) + \ce{O2}(g)$ ;
  \item $\ce{SO2}(g) + \tfrac12\,\ce{O2}(g) \rightarrow \ce{SO3}(g)$.
\end{enumerate}
\end{exo}

\begin{exo}[remonter à une enthalpie de formation]
La formation de l'eau liquide, $2\,\ce{H2}(g) + \ce{O2}(g) \rightarrow 2\,\ce{H2O}(l)$, a pour enthalpie $\Delta H^\circ_{\text{r}} = \qty{-572}{\kilo\joule}$.
\begin{enumerate}[label=\alph*)]
  \item Exprime $\Delta H^\circ_{\text{r}}$ en fonction de $\Delta H^\circ_{\text{f}}(\ce{H2O}(l))$.
  \item Détermine $\Delta H^\circ_{\text{f}}(\ce{H2O}(l))$ par mole.
\end{enumerate}
\end{exo}

\begin{exo}[combiner deux réactions]
On donne :\\
\hspace*{1em}(1) $\ce{C}(s) + \ce{O2}(g) \rightarrow \ce{CO2}(g)$, \; $\Delta H_1 = \qty{-393}{\kilo\joule}$ ;\\
\hspace*{1em}(2) $\ce{CO}(g) + \tfrac12\,\ce{O2}(g) \rightarrow \ce{CO2}(g)$, \; $\Delta H_2 = \qty{-283}{\kilo\joule}$.
\begin{enumerate}[label=\alph*)]
  \item En combinant (1) et (2), reconstitue la réaction $\ce{C}(s) + \tfrac12\,\ce{O2}(g) \rightarrow \ce{CO}(g)$.
  \item Calcule son enthalpie $\Delta H$.
\end{enumerate}
\end{exo}

\begin{exo}[quand l'état physique change]
On donne $\Delta H^\circ_{\text{f}}(\ce{H2O}(l))=\qty{-286}{\kilo\joule\per\mol}$ et $\Delta H^\circ_{\text{f}}(\ce{H2O}(g))=\qty{-242}{\kilo\joule\per\mol}$. On s'intéresse à la vaporisation $\ce{H2O}(l) \rightarrow \ce{H2O}(g)$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'enthalpie de vaporisation $\Delta H^\circ_{\text{r}}$.
  \item Le résultat (signe) est-il cohérent avec le fait que vaporiser l'eau \emph{consomme} de l'énergie ?
\end{enumerate}
\end{exo}
