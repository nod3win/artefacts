\begin{correction}[reconnaître un $\dHf$ nul]
Sont nuls les \textbf{corps purs simples dans leur état le plus stable} :
\begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=1pt]
  \item \rep{$\ce{O2}(g)$, $\ce{Fe}(s)$, $\ce{N2}(g)$, $\ce{C}$(graphite)} : $\dHf=0$ (corps simples, forme la plus stable).
  \item $\ce{CO2}$, $\ce{H2O}$, $\ce{NH3}$ : \emph{composés}, donc $\dHf\neq0$.
  \item $\ce{O3}$ : c'est bien un corps simple, mais l'ozone \emph{n'est pas} la forme la plus stable de
        l'oxygène ($\ce{O2}$ l'est). Donc \rep{$\dHf(\ce{O3})\neq0$}.
\end{itemize}
\end{correction}

\begin{correction}[écrire la loi de Hess]
« Produits $-$ réactifs », chacun $\times$ son coefficient ; les corps simples valent $0$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=3pt]
  \item $\dHr ={}$ \rep{$2\,\dHf(\ce{B}) - \dHf(\ce{A})$}.
  \item $\dHr = 2\,\dHf(\ce{NH3}) - \big[\dHf(\ce{N2}) + 3\,\dHf(\ce{H2})\big] ={}$ \rep{$2\,\dHf(\ce{NH3})$}
        (car $\dHf(\ce{N2})=\dHf(\ce{H2})=0$).
  \item $\dHr = \dHf(\ce{C2H5OH},l) - \big[2\,\dHf(\ce{C}) + 3\,\dHf(\ce{H2}) + \tfrac12\dHf(\ce{O2})\big]
        ={}$ \rep{$\dHf(\ce{C2H5OH},l)$}. C'est justement la définition de $\dHf$ de l'éthanol.
\end{enumerate}
\end{correction}

\begin{correction}[application numérique de la loi de Hess]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\dHr = \big[\dHf(\ce{CO2}) + 2\,\dHf(\ce{H2O},l)\big] - \big[\dHf(\ce{CH4}) + 2\,\dHf(\ce{O2})\big]$.
  \item Application :
  \[ \dHr = \big[(-394) + 2(-286)\big] - \big[(-75) + 0\big] = (-966) + 75 = \rep{\SI{-891}{\kilo\joule}}. \]
  $\dHr<0$ : la combustion est \rep{exothermique}.
\end{enumerate}
\end{correction}

\begin{correction}[inverser une réaction]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Inverser une réaction change le signe de $\dH$ : $\dH_{\text{inverse}} = \rep{\SI{+200}{\kilo\joule}}$.
  \item $\dH>0$ : la décomposition de $\ce{SO3}$ est \rep{endothermique} (l'inverse d'une réaction
        exothermique est endothermique).
\end{enumerate}
\end{correction}

\begin{correction}[effet des coefficients]
$\dH$ est proportionnel aux coefficients de l'équation.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Coefficients $\times 2$ : $\dH = 2\times(-286) = \rep{\SI{-572}{\kilo\joule}}$.
  \item Coefficients $\times \tfrac12$ : $\dH = \tfrac12\times(-286) = \rep{\SI{-143}{\kilo\joule}}$.
  \item C'est l'\rep{enthalpie standard de formation de $\ce{H2O}(l)$}, $\dHf(\ce{H2O},l)=\SI{-286}{\kilo\joule\per\mol}$
        (formation d'\emph{une} mole à partir des corps simples).
\end{enumerate}
\end{correction}
