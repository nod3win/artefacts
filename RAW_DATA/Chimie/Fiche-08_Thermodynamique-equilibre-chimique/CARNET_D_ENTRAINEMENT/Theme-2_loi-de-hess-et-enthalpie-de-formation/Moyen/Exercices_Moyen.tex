\begin{exercice}[combustion de l'éthanol]
On étudie la combustion complète de l'éthanol :
\;$\ce{C2H5OH}(l) + 3\,\ce{O2}(g) \rightarrow 2\,\ce{CO2}(g) + 3\,\ce{H2O}(l)$.\;
Données (\si{\kilo\joule\per\mol}) : $\dHf(\ce{C2H5OH},l)=-278$, $\dHf(\ce{CO2})=-394$, $\dHf(\ce{H2O},l)=-286$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Écrire la loi de Hess pour cette réaction.
  \item Calculer $\dHr$ et conclure sur le caractère exo/endothermique.
\end{enumerate}
\end{exercice}

\begin{exercice}[oxydation du dioxyde de soufre — deux méthodes]
Pour \;$2\,\ce{SO2}(g) + \ce{O2}(g) \rightarrow 2\,\ce{SO3}(g)$,\; on donne
$\dHf(\ce{SO2},g)=\SI{-300}{\kilo\joule\per\mol}$ et $\dHf(\ce{SO3},g)=\SI{-400}{\kilo\joule\per\mol}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Calculer $\dHr$ par la loi de Hess.
  \item Cette valeur correspond à la formation de combien de moles de $\ce{SO3}$ ?
        En déduire l'énergie dégagée \emph{par mole} de $\ce{SO3}$ formée.
\end{enumerate}
\end{exercice}

\begin{exercice}[remonter à un $\dHf$ inconnu]
La décomposition du peroxyde d'hydrogène \;$2\,\ce{H2O2}(l) \rightarrow 2\,\ce{H2O}(l) + \ce{O2}(g)$\;
a pour enthalpie $\dHr = \SI{-196}{\kilo\joule}$. On donne $\dHf(\ce{H2O},l)=\SI{-286}{\kilo\joule\per\mol}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Écrire la loi de Hess reliant $\dHr$, $\dHf(\ce{H2O},l)$ et $\dHf(\ce{H2O2},l)$.
  \item En déduire la valeur de $\dHf(\ce{H2O2},l)$.
\end{enumerate}
\end{exercice}

\begin{exercice}[combiner deux réactions]
On dispose de deux réactions de combustion :
\begin{align*}
(1)\quad & \ce{C}(s) + \ce{O2}(g) \rightarrow \ce{CO2}(g), & \dH_1 &= \SI{-394}{\kilo\joule} ; \\
(2)\quad & \ce{CO}(g) + \tfrac12\,\ce{O2}(g) \rightarrow \ce{CO2}(g), & \dH_2 &= \SI{-283}{\kilo\joule}.
\end{align*}
On cherche l'enthalpie de la réaction cible \;$\ce{C}(s) + \tfrac12\,\ce{O2}(g) \rightarrow \ce{CO}(g)$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Quelle combinaison de $(1)$ et $(2)$ redonne exactement la réaction cible ?
        (préciser laquelle est inversée).
  \item En déduire $\dH$ de la réaction cible.
\end{enumerate}
\end{exercice}
