\section{Les deux formes de chaleur}

\begin{formuleb}[chaleur sensible — la température change]
\[ \boxed{\;Q = m\,c\,\Delta T\;} \]
\begin{ou}
  \item $Q$ : chaleur en \si{\kilo\joule} \; ; \; $m$ : masse en \si{\kilogram} ;
  \item $c$ : capacité thermique massique en \si{\kilo\joule\per\kelvin\per\kilogram} ;
  \item $\Delta T = T_{\text{f}} - T_{\text{i}}$ (même valeur en \si{\kelvin} ou en \si{\celsius}).
\end{ou}
\end{formuleb}

\begin{formuleb}[chaleur latente — la température ne change pas]
\[ \boxed{\;Q = m\,L\;} \qquad (L \text{ en } \si{\kilo\joule\per\kilogram}) \]
Pendant un \textbf{changement d'état} (fusion, vaporisation\ldots), $T$ reste constante : toute la
chaleur sert à briser ou reformer les liaisons entre molécules.
\end{formuleb}

\section{Le bilan calorimétrique}

\begin{proprieteb}[calorimètre adiabatique]
Dans un calorimètre \emph{adiabatique} (aucune chaleur ne sort), les chaleurs échangées entre les corps
présents se compensent :
\[ \boxed{\;\sum_i Q_i = 0\;} \qquad\Longleftrightarrow\qquad
   \big|\textstyle\sum Q_{\text{reçues}}\big| = \big|\textstyle\sum Q_{\text{cédées}}\big|. \]
Une chaleur \textbf{reçue} est $>0$, une chaleur \textbf{cédée} est $<0$.
\end{proprieteb}

\begin{methodeb}[« escalier » pour un chauffage en plusieurs étapes]
On décompose le trajet en \emph{segments horizontaux} (échauffements, $Q=mc\Delta T$) et \emph{segments
verticaux} (changements d'état, $Q=mL$), puis on \textbf{additionne}.
\end{methodeb}

\begin{attentionb}
\textbf{Unités} : $\SI{1}{\kilo\joule}=\SI{1000}{\joule}$. Si $c$ est en \si{\kilo\joule\per\kelvin\per\kilogram}
et $m$ en \si{\kilogram}, alors $Q$ sort en \si{\kilo\joule} (ex. $\SI{4.185}{\kilo\joule}=\SI{4185}{\joule}$,
\emph{pas} \SI{4.185}{\joule}). Et $\Delta T$ se prend \emph{toujours} $T_{\text{final}}-T_{\text{initial}}$.
\end{attentionb}
