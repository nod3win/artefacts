\textit{Données pour l'eau :} $c_{\text{glace}}=\SI{2.060}{\kilo\joule\per\kelvin\per\kilogram}$,\;
$c_{\text{eau}}=\SI{4.185}{\kilo\joule\per\kelvin\per\kilogram}$,\;
$c_{\text{vap}}=\SI{1.85}{\kilo\joule\per\kelvin\per\kilogram}$,\;
$L_{\text{fus}}=\SI{334}{\kilo\joule\per\kilogram}$,\; $L_{\text{vap}}=\SI{2257}{\kilo\joule\per\kilogram}$.

\begin{exercice}[chaleur sensible]
On chauffe $m=\SI{0.50}{\kilogram}$ d'eau liquide de \SI{20}{\celsius} à \SI{80}{\celsius}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Calculer $\Delta T$.
  \item Calculer la chaleur $Q$ reçue par l'eau (en \si{\kilo\joule}).
  \item Cette chaleur est-elle reçue ou cédée par l'eau ? Quel est le signe de $Q$ ?
\end{enumerate}
\end{exercice}

\begin{exercice}[chaleur latente]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Quelle chaleur faut-il pour \textbf{faire fondre} $m=\SI{0.20}{\kilogram}$ de glace à \SI{0}{\celsius} ?
  \item Quelle chaleur faut-il pour \textbf{vaporiser} $m=\SI{0.20}{\kilogram}$ d'eau à \SI{100}{\celsius} ?
  \item Des deux, quel changement d'état demande le plus d'énergie ? Par quel facteur environ ?
\end{enumerate}
\end{exercice}

\begin{exercice}[sensible ou latente ?]
Pour chaque étape, indiquer la formule à utiliser ($Q=mc\Delta T$ ou $Q=mL$) et préciser si la
température \emph{varie} ou \emph{reste constante}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Réchauffer de l'eau liquide de \SI{10}{\celsius} à \SI{40}{\celsius}.
  \item Faire fondre de la glace à \SI{0}{\celsius}.
  \item Réchauffer de la vapeur de \SI{100}{\celsius} à \SI{150}{\celsius}.
  \item Vaporiser de l'eau à \SI{100}{\celsius}.
\end{enumerate}
\end{exercice}

\begin{exercice}[signe de la chaleur]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Un corps se \emph{réchauffe}. Quel est le signe de $\Delta T$, puis de $Q$ ?
  \item Un corps se \emph{refroidit}. Mêmes questions.
  \item On plonge un morceau de fer chaud ($m=\SI{0.10}{\kilogram}$, $c_{\text{fer}}=\SI{0.45}{\kilo\joule\per\kelvin\per\kilogram}$)
        qui passe de \SI{90}{\celsius} à \SI{30}{\celsius}. Calculer $Q_{\text{fer}}$ ; le fer cède-t-il ou reçoit-il de la chaleur ?
\end{enumerate}
\end{exercice}

\begin{exercice}[la vigilance des unités]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Convertir \SI{3.5}{\kilo\joule} en joules.
  \item On chauffe $m=\SI{0.10}{\kilogram}$ d'eau de \SI{25}{\celsius} à \SI{35}{\celsius}. Calculer $Q$ en
        \si{\kilo\joule}, puis en \si{\joule}.
  \item Un étudiant annonce $Q=\SI{4.185}{\joule}$. Quelle erreur a-t-il commise ?
\end{enumerate}
\end{exercice}
