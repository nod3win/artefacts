\begin{exo}[calculer le travail des forces de pression]
La pression extérieure est constante, $p=\qty{1.0e5}{\pascal}$. Calcule le travail $W=-p\,\Delta V$ (en $\si{\joule}$) après avoir converti le volume en $\si{\cubic\metre}$.
\begin{enumerate}[label=\alph*)]
  \item le gaz se détend : $\Delta V = \qty{+2.0}{\litre}$ ;
  \item le gaz est comprimé : $\Delta V = \qty{-0.50}{\litre}$.
\end{enumerate}
\end{exo}

\begin{exo}[appliquer $\Delta U = Q + W$]
Calcule la variation d'énergie interne $\Delta U$ du système.
\begin{enumerate}[label=\alph*)]
  \item $Q=\qty{+500}{\joule}$ et $W=\qty{-200}{\joule}$ ;
  \item $Q=\qty{-300}{\joule}$ et $W=\qty{+120}{\joule}$ ;
  \item le système reçoit $\qty{1.5}{\kilo\joule}$ de chaleur et cède $\qty{0.4}{\kilo\joule}$ de travail.
\end{enumerate}
\end{exo}

\begin{exo}[du diagramme aux valeurs]
Pour une réaction, l'enthalpie des réactifs vaut $H_{\text{réac.}}=\qty{100}{\kilo\joule}$ et celle des produits $H_{\text{prod.}}=\qty{30}{\kilo\joule}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule $\Delta H$.
  \item La réaction est-elle exothermique ou endothermique ?
  \item Quelle quantité de chaleur est échangée, et dans quel sens (dégagée / absorbée) ?
\end{enumerate}
\end{exo}

\begin{exo}[inverser et multiplier une équation]
On donne $\ce{N2} + 3\,\ce{H2} \rightarrow 2\,\ce{NH3}$, \; $\Delta H = \qty{-92}{\kilo\joule}$. Déduis-en $\Delta H$ pour :
\begin{enumerate}[label=\alph*)]
  \item la réaction inverse $2\,\ce{NH3} \rightarrow \ce{N2} + 3\,\ce{H2}$ ;
  \item la réaction $2\,\ce{N2} + 6\,\ce{H2} \rightarrow 4\,\ce{NH3}$ (coefficients doublés) ;
  \item la formation d'une seule mole : $\tfrac12\,\ce{N2} + \tfrac32\,\ce{H2} \rightarrow \ce{NH3}$.
\end{enumerate}
\end{exo}

\begin{exo}[enchaîner travail et premier principe]
À pression constante $p=\qty{1.0e5}{\pascal}$, un système se détend de $\Delta V=\qty{+3.0}{\litre}$ tout en \emph{recevant} une chaleur $Q=\qty{+600}{\joule}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le travail $W$.
  \item En déduis la variation d'énergie interne $\Delta U$.
\end{enumerate}
\end{exo}
