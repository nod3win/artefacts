\begin{corrige}[calculer le travail des forces de pression]
On convertit : $\qty{1}{\litre}=\qty{1e-3}{\cubic\metre}$, puis $W=-p\,\Delta V$.
\begin{enumerate}[label=\alph*)]
  \item $\Delta V = \qty{+2.0e-3}{\cubic\metre}$ : $W = -\,\qty{1.0e5}{\pascal}\times \qty{2.0e-3}{\cubic\metre} = \qty{-200}{\joule}$ (travail cédé).
  \item $\Delta V = \qty{-0.50e-3}{\cubic\metre}$ : $W = -\,\qty{1.0e5}{\pascal}\times(\qty{-0.50e-3}{\cubic\metre}) = \qty{+50}{\joule}$ (travail reçu).
\end{enumerate}
\end{corrige}

\begin{corrige}[appliquer $\Delta U = Q + W$]
On additionne chaleur et travail en respectant les signes.
\begin{enumerate}[label=\alph*)]
  \item $\Delta U = 500 + (-200) = \qty{+300}{\joule}$.
  \item $\Delta U = -300 + 120 = \qty{-180}{\joule}$.
  \item chaleur reçue $\Rightarrow Q=\qty{+1500}{\joule}$ ; travail cédé $\Rightarrow W=\qty{-400}{\joule}$. Donc $\Delta U = 1500 - 400 = \qty{+1100}{\joule}=\qty{+1.1}{\kilo\joule}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[du diagramme aux valeurs]
On utilise $\Delta H = H_{\text{prod.}}-H_{\text{réac.}}$.
\begin{enumerate}[label=\alph*)]
  \item $\Delta H = 30 - 100 = \qty{-70}{\kilo\joule}$.
  \item $\Delta H<0$ : la réaction est \textbf{exothermique}.
  \item elle \textbf{dégage} $\qty{70}{\kilo\joule}$ vers le milieu extérieur.
\end{enumerate}
\end{corrige}

\begin{corrige}[inverser et multiplier une équation]
$\Delta H$ est proportionnel à l'équation : inverser change le signe, multiplier par $k$ multiplie $\Delta H$ par $k$.
\begin{enumerate}[label=\alph*)]
  \item réaction inverse : $\Delta H = \qty{+92}{\kilo\joule}$.
  \item coefficients $\times 2$ : $\Delta H = 2\times(-92) = \qty{-184}{\kilo\joule}$.
  \item coefficients $\times \tfrac12$ : $\Delta H = \tfrac12\times(-92) = \qty{-46}{\kilo\joule}$ par mole de $\ce{NH3}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[enchaîner travail et premier principe]
\begin{enumerate}[label=\alph*)]
  \item $\Delta V = \qty{+3.0e-3}{\cubic\metre}$, donc $W = -\,\qty{1.0e5}{\pascal}\times\qty{3.0e-3}{\cubic\metre} = \qty{-300}{\joule}$.
  \item $\Delta U = Q + W = 600 + (-300) = \qty{+300}{\joule}$.
\end{enumerate}
\end{corrige}
