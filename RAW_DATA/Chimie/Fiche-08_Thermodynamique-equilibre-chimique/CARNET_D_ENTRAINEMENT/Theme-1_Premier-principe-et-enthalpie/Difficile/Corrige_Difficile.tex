\begin{corrige}[énergie libérée par une combustion]
\begin{enumerate}[label=\alph*)]
  \item $n = \dfrac{m}{M} = \dfrac{8{,}0}{16} = \qty{0.50}{\mol}$.
  \item l'énergie est proportionnelle à la quantité de matière : $|E| = n\times|\Delta H| = 0{,}50\times 890 = \qty{445}{\kilo\joule}$. Comme $\Delta H<0$, cette énergie est \textbf{dégagée}.
  \item $\Delta H<0$ signifie que le système \emph{perd} de l'énergie au profit du milieu extérieur, qui la reçoit sous forme de chaleur : le milieu \textbf{s'échauffe}.
\end{enumerate}
\end{corrige}

\begin{corrige}[relier $\Delta H$ et $\Delta U$]
\begin{enumerate}[label=\alph*)]
  \item $\Delta V=\qty{-20e-3}{\cubic\metre}$, donc $W=-p\,\Delta V = -\,\qty{1.0e5}{\pascal}\times(\qty{-20e-3}{\cubic\metre}) = \qty{+2000}{\joule}=\qty{+2.0}{\kilo\joule}$ (le système est comprimé, il \emph{reçoit} du travail).
  \item $\Delta U = Q + W = \Delta H + W = -100 + 2{,}0 = \qty{-98}{\kilo\joule}$.
  \item oui : $|W|=\qty{2.0}{\kilo\joule}$ est très petit devant $|\Delta H|=\qty{100}{\kilo\joule}$ (environ $2\,\%$). En thermochimie on assimile souvent $\Delta U\approx\Delta H$ pour cette raison.
\end{enumerate}
\end{corrige}

\begin{corrige}[vrai ou faux — raisonner sur le premier principe]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux} : une réaction athermique n'échange \emph{pas} de chaleur ($\Delta H\approx 0$) ; elle ne dégage rien.
  \item \textbf{Vrai} : exothermique $\Rightarrow \Delta H = H_{\text{prod.}}-H_{\text{réac.}}<0 \Rightarrow H_{\text{réac.}}>H_{\text{prod.}}$.
  \item \textbf{Vrai} : c'est la définition d'une paroi adiabatique (ni perte ni gain de chaleur avec l'extérieur).
  \item \textbf{Faux} : « dégage » $\Rightarrow Q_p<0$, donc $\Delta H = \qty{-50}{\kilo\joule}$ (et non $+50$).
\end{enumerate}
\end{corrige}

\begin{corrige}[exploiter un diagramme énergétique chiffré]
\begin{enumerate}[label=\alph*)]
  \item $\Delta H = H_{\text{prod.}}-H_{\text{réac.}} = 280 - 100 = \qty{+180}{\kilo\joule}>0$ : réaction \textbf{endothermique} (les produits sont plus haut).
  \item l'équation forme $\qty{2}{\mol}$ de produit, donc par mole : $\dfrac{180}{2}=\qty{+90}{\kilo\joule\per\mol}$.
  \item la réaction inverse a l'enthalpie opposée : $\Delta H_{\text{inverse}} = \qty{-180}{\kilo\joule}$ (elle serait exothermique).
\end{enumerate}
\end{corrige}
