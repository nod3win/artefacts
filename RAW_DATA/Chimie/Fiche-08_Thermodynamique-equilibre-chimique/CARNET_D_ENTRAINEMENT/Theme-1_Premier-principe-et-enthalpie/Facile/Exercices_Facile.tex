\begin{exo}[exothermique ou endothermique ?]
Pour chaque réaction, on donne sa variation d'enthalpie $\Delta H$. Indique si la réaction est \textbf{exothermique} ou \textbf{endothermique}, et si elle \emph{dégage} ou \emph{absorbe} de la chaleur.
\begin{enumerate}[label=\alph*)]
  \item $\Delta H = \qty{-92}{\kilo\joule}$
  \item $\Delta H = \qty{+178}{\kilo\joule}$
  \item $\Delta H = \qty{-46}{\kilo\joule}$
  \item $\Delta H = \qty{+52}{\kilo\joule}$
\end{enumerate}
\end{exo}

\begin{exo}[conventions de signe]
On se place du point de vue du \textbf{système}. Donne le signe ($>0$ ou $<0$) de la grandeur échangée dans chaque situation.
\begin{enumerate}[label=\alph*)]
  \item le système \emph{reçoit} $\qty{50}{\joule}$ de chaleur : signe de $Q$ ?
  \item le système \emph{cède} $\qty{30}{\joule}$ de chaleur : signe de $Q$ ?
  \item le système \emph{reçoit} du travail : signe de $W$ ?
  \item le système \emph{cède} du travail : signe de $W$ ?
\end{enumerate}
\end{exo}

\begin{exo}[chaleur à pression constante]
À pression constante, la chaleur échangée est égale à la variation d'enthalpie : $Q_p=\Delta H$. Donne $\Delta H$ (valeur \emph{et} signe) dans chaque cas.
\begin{enumerate}[label=\alph*)]
  \item une réaction \emph{dégage} $\qty{150}{\kilo\joule}$ à pression constante ;
  \item une réaction \emph{absorbe} $\qty{80}{\kilo\joule}$ à pression constante.
\end{enumerate}
\end{exo}

\begin{exo}[lire un diagramme enthalpique]
On donne le diagramme enthalpique d'une réaction.
\begin{center}
\begin{tikzpicture}[scale=1.0]
  \draw[->,thick] (0,-0.3)--(0,4) node[above,font=\small]{$H$};
  \draw[->,thick] (0,0)--(6,0) node[right,font=\small]{avancement};
  \draw[primary,very thick] (0.6,3.2)--(2.4,3.2) node[midway,above,font=\small]{réactifs};
  \draw[primary,very thick] (3.6,1.0)--(5.4,1.0) node[midway,above,font=\small]{produits};
  \draw[->,thick] (3.0,3.2)--(3.0,1.0) node[midway,right,font=\small]{$\Delta H$};
  \draw[dashed,gray] (2.4,3.2)--(3.6,3.2); \draw[dashed,gray] (2.4,1.0)--(3.6,1.0);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Quel est le signe de $\Delta H$ ?
  \item La réaction est-elle exothermique ou endothermique ?
  \item Qui possède la plus grande enthalpie : les réactifs ou les produits ?
\end{enumerate}
\end{exo}

\begin{exo}[signe du travail des forces de pression]
À pression extérieure constante, $W=-p\,\Delta V$. Sans calcul, donne le signe de $W$ (ou sa valeur).
\begin{enumerate}[label=\alph*)]
  \item le système se \emph{dilate} ($\Delta V>0$) ;
  \item le système se \emph{contracte} ($\Delta V<0$) ;
  \item le volume du système ne change pas ($\Delta V=0$).
\end{enumerate}
\end{exo}
