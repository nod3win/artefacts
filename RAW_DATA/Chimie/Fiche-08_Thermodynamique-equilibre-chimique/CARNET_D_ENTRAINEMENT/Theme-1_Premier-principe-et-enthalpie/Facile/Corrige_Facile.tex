\begin{corrige}[exothermique ou endothermique ?]
Règle : $\Delta H<0 \Rightarrow$ exothermique (dégage) ; $\Delta H>0 \Rightarrow$ endothermique (absorbe).
\begin{enumerate}[label=\alph*)]
  \item $\Delta H<0$ : \textbf{exothermique}, dégage $\qty{92}{\kilo\joule}$.
  \item $\Delta H>0$ : \textbf{endothermique}, absorbe $\qty{178}{\kilo\joule}$.
  \item $\Delta H<0$ : \textbf{exothermique}, dégage $\qty{46}{\kilo\joule}$.
  \item $\Delta H>0$ : \textbf{endothermique}, absorbe $\qty{52}{\kilo\joule}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[conventions de signe]
Ce que le système \emph{reçoit} est positif, ce qu'il \emph{cède} est négatif.
\begin{enumerate}[label=\alph*)]
  \item chaleur reçue : $Q>0$ (ici $Q=\qty{+50}{\joule}$).
  \item chaleur cédée : $Q<0$ (ici $Q=\qty{-30}{\joule}$).
  \item travail reçu : $W>0$.
  \item travail cédé : $W<0$.
\end{enumerate}
\end{corrige}

\begin{corrige}[chaleur à pression constante]
On applique $Q_p=\Delta H$ avec la convention de signe.
\begin{enumerate}[label=\alph*)]
  \item la réaction \emph{dégage} de la chaleur $\Rightarrow Q_p<0$, donc $\Delta H = \qty{-150}{\kilo\joule}$.
  \item la réaction \emph{absorbe} de la chaleur $\Rightarrow Q_p>0$, donc $\Delta H = \qty{+80}{\kilo\joule}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire un diagramme enthalpique]
Les réactifs sont \emph{au-dessus} des produits.
\begin{enumerate}[label=\alph*)]
  \item les produits sont plus bas que les réactifs : $\Delta H = H_{\text{prod.}}-H_{\text{réac.}}<0$, donc $\Delta H<0$.
  \item $\Delta H<0$ : la réaction est \textbf{exothermique}.
  \item ce sont les \textbf{réactifs} qui possèdent la plus grande enthalpie (ils sont plus haut sur l'axe $H$).
\end{enumerate}
\end{corrige}

\begin{corrige}[signe du travail des forces de pression]
On lit le signe de $W=-p\,\Delta V$ (avec $p>0$).
\begin{enumerate}[label=\alph*)]
  \item $\Delta V>0 \Rightarrow W=-p\,\Delta V<0$ : le système \emph{cède} du travail.
  \item $\Delta V<0 \Rightarrow W=-p\,\Delta V>0$ : le système \emph{reçoit} du travail.
  \item $\Delta V=0 \Rightarrow W=0$ : aucun travail de pression échangé.
\end{enumerate}
\end{corrige}
