\begin{exo}[Calculer le degré d'insaturation]
Calcule l'indice de déficience en hydrogène $I$ de chaque hydrocarbure (rappel : $I=\frac{2n+2-m}{2}$).
\begin{enumerate}[label=\alph*)]
  \item $\ce{C8H18}$
  \item $\ce{C8H16}$
  \item $\ce{C8H14}$
  \item $\ce{C8H10}$
\end{enumerate}
\end{exo}

\begin{exo}[Une formule, deux structures]
La formule $\ce{C5H10}$ possède un degré d'insaturation $I=1$.
\begin{enumerate}[label=\alph*)]
  \item Propose une structure \textbf{acyclique} (un alcène) compatible avec cette formule.
  \item Propose une structure \textbf{cyclique} (un cycloalcane) compatible avec cette même formule.
  \item Explique pourquoi ces deux molécules ont le même $I$ alors qu'elles n'ont pas de double liaison en commun.
\end{enumerate}
\end{exo}

\begin{exo}[Insaturation avec des hétéroatomes]
Calcule $I$ pour chaque formule brute (rappel : l'oxygène est \emph{ignoré}, l'azote compte $+1$, un halogène compte comme un H).
\begin{enumerate}[label=\alph*)]
  \item $\ce{C3H6O}$
  \item $\ce{C2H3Cl}$
  \item $\ce{C2H7N}$
\end{enumerate}
\end{exo}

\begin{exo}[Le cas du benzène]
Le benzène a pour formule brute $\ce{C6H6}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule son degré d'insaturation $I$.
  \item Sachant que la molécule contient exactement $1$ cycle, combien de liaisons $\pi$ possède-t-elle ?
  \item Le benzène est-il saturé ?
\end{enumerate}
\end{exo}

\begin{exo}[Hydrogénation et comptage des insaturations]
Un hydrocarbure de formule $\ce{C7H10}$ est \textbf{entièrement hydrogéné} (chaque liaison $\ce{C=C}$ additionne $\ce{H2}$). Le produit saturé obtenu possède encore un cycle.
\begin{enumerate}[label=\alph*)]
  \item Calcule le degré d'insaturation $I$ de $\ce{C7H10}$.
  \item Le produit saturé est un cycloalcane $\ce{C_nH_{2n}}$ : donne sa formule brute, et le nombre d'hydrogènes additionnés.
  \item Combien de liaisons $\ce{C=C}$ l'hydrocarbure de départ contenait-il ?
\end{enumerate}
\end{exo}
