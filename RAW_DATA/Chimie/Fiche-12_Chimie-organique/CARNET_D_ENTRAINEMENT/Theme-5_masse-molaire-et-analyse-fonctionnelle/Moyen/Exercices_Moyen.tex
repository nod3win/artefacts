\begin{exo}[Masse molaire avec azote et soufre]
Calcule la masse molaire (données : $M(\ce{C})=12$, $M(\ce{H})=1$, $M(\ce{O})=16$, $M(\ce{N})=14$, $M(\ce{S})=32$ en \si{\gram\per\mole}).
\begin{enumerate}[label=\alph*)]
  \item glycine \ce{C2H5NO2}
  \item urée \ce{CH4N2O}
  \item éthanethiol \ce{C2H6S}
\end{enumerate}
\end{exo}

\begin{exo}[Nombre de moles $n=m/M$]
Un comprimé contient \qty{755}{\milli\gram} de paracétamol, de masse molaire \qty{151}{\gram\per\mole}. Quelle quantité de matière (en mol) de paracétamol contient-il ?
\end{exo}

\begin{exo}[Nombre de molécules $N=n\,N_{\mathrm{A}}$]
On dispose de \qty{0.36}{\gram} d'aspirine, de masse molaire \qty{180}{\gram\per\mole} ($N_{\mathrm{A}}=\qty{6.02e23}{\per\mole}$).
\begin{enumerate}[label=\alph*)]
  \item Combien de moles d'aspirine cela représente-t-il ?
  \item Combien de molécules d'aspirine cela fait-il ?
\end{enumerate}
\end{exo}

\begin{exo}[Pourcentage massique dans l'urée]
L'urée a pour formule brute \ce{CH4N2O} (masse molaire \qty{60}{\gram\per\mole}).
\begin{enumerate}[label=\alph*)]
  \item Quel est le pourcentage massique du carbone ?
  \item Quel est le pourcentage massique de l'azote ?
\end{enumerate}
\end{exo}

\begin{exo}[Retrouver la masse molaire à partir d'un pourcentage]
Une molécule organique contient un \emph{seul} atome de soufre, qui représente \qty{8}{\percent} de sa masse ($M(\ce{S})=\qty{32}{\gram\per\mole}$). Quelle est la masse molaire de la molécule ?
\end{exo}
