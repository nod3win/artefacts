\begin{exo}[Masse molaire de petites molécules]
Calcule la masse molaire (données : $M(\ce{C})=12$, $M(\ce{H})=1$, $M(\ce{O})=16$ en \si{\gram\per\mole}).
\begin{enumerate}[label=\alph*)]
  \item méthane \ce{CH4}
  \item méthanol \ce{CH4O}
  \item éthane \ce{C2H6}
\end{enumerate}
\end{exo}

\begin{exo}[Molécules oxygénées courantes]
Calcule la masse molaire de chaque composé.
\begin{enumerate}[label=\alph*)]
  \item éthanol \ce{C2H6O}
  \item propanone (acétone) \ce{C3H6O}
  \item acide éthanoïque \ce{C2H4O2}
\end{enumerate}
\end{exo}

\begin{exo}[Masse molaire ou masse moléculaire relative ?]
On considère le méthanal \ce{CH2O}.
\begin{enumerate}[label=\alph*)]
  \item Donne sa masse molaire $M$ (avec son unité).
  \item Donne sa masse moléculaire relative $M_r$ (grandeur sans unité).
\end{enumerate}
\end{exo}

  \item Quel est le pourcentage massique de l'hydrogène ?
\end{enumerate}
\end{exo}

\begin{exo}[De la structure à la masse molaire]
On considère la molécule \ce{CH3-CO-CH3}.
\begin{enumerate}[label=\alph*)]
  \item Donne sa formule brute.
  \item Calcule sa masse molaire.
\end{enumerate}
\end{exo}
