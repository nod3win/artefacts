% ---------------------------------------------------------------------
\begin{corr}{nomenclature complète d'une molécule trifonctionnelle}
\textbf{a)} Fonctions : un $\ce{-COOH}$ (\rep{acide carboxylique}), un $\ce{-OH}$ (\rep{alcool}), un $\ce{-NH2}$ (\rep{amine}).\\
\textbf{b)} Priorité : $\ce{-COOH} > \ce{-OH} > \ce{-NH2}$. Fonction principale $=$ \rep{acide} (suffixe \textit{-oïque}). L'alcool $\to$ préfixe \textbf{hydroxy-} ; l'amine $\to$ préfixe \textbf{amino-}.\\
\textbf{c)} Chaîne principale : \textbf{5 carbones} contenant le $\ce{-COOH}$ $\Rightarrow$ acide \emph{pentanoïque}. Le C du $\ce{-COOH}$ porte le n\textsuperscript{o}1 (imposé) :
\[ \underset{1}{\ce{COOH}}-\underset{2}{\ce{CH2}}-\underset{3}{\ce{CH}}(\ce{OH})-\underset{4}{\ce{CH}}(\ce{NH2})-\underset{5}{\ce{CH3}} \]
donc $\ce{-OH}$ en 3, $\ce{-NH2}$ en 4.\\
\textbf{d)} Préfixes par ordre alphabétique (\emph{amino} avant \emph{hydroxy}) : \rep{acide 4-amino-3-hydroxypentanoïque}.
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{du nom à la structure (sens inverse)}
\textbf{a)} « acide \dots-oïque » $\Rightarrow$ fonction principale acide carboxylique ; « propan- » $\Rightarrow$ 3 carbones ; le $\ce{-COOH}$ est le C1 ; « 2-amino » $\Rightarrow$ un $\ce{-NH2}$ sur le C2. Structure :
\begin{center}\chemfig{CH_3-[:30]CH(-[:90]NH_2)-[:-30](=[:-90]O)-[:30]OH}\qquad soit \ce{CH3-CH(NH2)-COOH}\end{center}
\textbf{b)} Deux fonctions : \rep{acide carboxylique} et \rep{amine primaire}. Priorité $\ce{-COOH} > \ce{-NH2}$ : l'acide donne le suffixe, l'amine devient le préfixe \emph{amino-} $\Rightarrow$ cohérent avec le nom. (C'est bien l'alanine.)
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{alcool + double liaison : locants et suffixe -èn-ol}
\textbf{a)} Fonctions : un $\ce{-OH}$ (\rep{alcool}) et une double liaison $\ce{C=C}$ (\rep{alcène}). Priorité : $\ce{-OH} > \text{C=C}$ $\Rightarrow$ l'alcool est prioritaire (il fixe le suffixe \textit{-ol} et le sens de numérotation).\\
\textbf{b)} En partant de la \textbf{gauche} : $\ce{-OH}$ en 4, double liaison en 1. En partant de la \textbf{droite} : $\ce{-OH}$ en 2, double liaison en 4.\\
\textbf{c)} On retient le sens donnant à l'alcool (prioritaire) le plus petit indice, soit \textbf{2}. La chaîne fait 5 C. Nom : \rep{pent-4-én-2-ol} (double liaison entre C4 et C5 : locant 4 ; alcool en 2).
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{nommer un ester}
\textbf{a)} Le carbonyle $\ce{C=O}$ est lié à un $\ce{-O-CH3}$ (un O prolongé par un carbone) $\Rightarrow$ \rep{fonction ester}.\\
\textbf{b)} La partie \textbf{acide} est $\ce{CH3CH2CH2-C(=O)-}$, dérivée de l'acide \emph{butanoïque} (4 C) $\Rightarrow$ « butanoate ». La partie \textbf{alcool} est le $\ce{-O-CH3}$, dérivée du méthanol $\Rightarrow$ « de méthyle ».\\
\textbf{c)} Nom : \rep{butanoate de méthyle}.
\end{corr}
