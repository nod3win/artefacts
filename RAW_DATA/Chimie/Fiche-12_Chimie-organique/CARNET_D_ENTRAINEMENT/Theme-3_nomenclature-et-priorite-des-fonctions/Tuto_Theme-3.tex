\section{Chaîne principale et numérotation}
\begin{regle}[Chaîne principale]
La \textbf{chaîne principale} est la chaîne carbonée \emph{la plus longue} qui contient la \textbf{fonction principale} (la plus prioritaire, cf.\ §3). Le nom de cette chaîne (prop-, but-, pent-\dots) porte le \textbf{suffixe} de la fonction principale.
\end{regle}
\begin{regle}[Numérotation]
On numérote la chaîne pour donner à la fonction principale le \textbf{plus petit indice possible}. Cas imposés : pour $\ce{-COOH}$, $\ce{-COOR}$ (ester), $\ce{-CHO}$ (aldéhyde) et l'amide, le carbone de la fonction porte \textbf{toujours le n\textsuperscript{o} 1}.
\end{regle}

% =====================================================================
\section{Fonctions secondaires : elles deviennent des préfixes}
Les fonctions \emph{moins} prioritaires que la principale ne prennent pas le suffixe : elles sont citées en \textbf{préfixe} (avec leur indice), rangées par ordre \textbf{alphabétique}.
\begin{center}\small\renewcommand{\arraystretch}{1.1}
\begin{tabular}{lll}
\toprule
Fonction en secondaire & Préfixe & Exemple \\
\midrule
alcool $\ce{-OH}$ & \textit{hydroxy-} & acide 3-\textbf{hydroxy}propanoïque \\
amine $\ce{-NH2}$ & \textit{amino-} & 2-\textbf{amino}propan-1-ol \\
carbonyle $\ce{C=O}$ (cétone/ald.) & \textit{oxo-} & acide 3-\textbf{oxo}propanoïque \\
éther $\ce{-O-R}$ & \textit{(alk)oxy-} & 3-\textbf{éthoxy}butanal \\
halogène (Cl, Br\dots) & \textit{chloro-, bromo-} & 3-\textbf{chloro}butan-2-one \\
ramification $\ce{-CH3}$ & \textit{méthyl-} & 2-\textbf{méthyl}butane \\
\bottomrule
\end{tabular}
\end{center}

\section{L'échelle de priorité des fonctions}
\begin{formule}[Ordre décroissant de priorité]
\[ -\text{COOH} > -\text{COOR} > -\text{CONH}_2 > -\text{CHO} > \text{C=O} > -\text{OH} > -\text{NH}_2 > -\text{O}- > \text{C=C} > \text{C}\equiv\text{C} \]
\end{formule}
La plus prioritaire présente devient la \textbf{fonction principale} (donne le suffixe). L'\textbf{acide carboxylique} l'emporte sur tout ; les liaisons multiples $\ce{C=C}$/$\ce{C#C}$ sont les moins prioritaires. La double/triple liaison s'indique par \textit{-èn(e)-}/\textit{-yn(e)-} avec son indice.

\section{La méthode pas à pas}
\begin{meth}[Nommer une molécule]
\textbf{1.} \emph{Repérer} toutes les fonctions. \quad \textbf{2.} \emph{Classer} par priorité $\to$ fonction principale (elle donne le suffixe). \quad \textbf{3.} \emph{Choisir} la chaîne principale (la plus longue contenant la fonction principale). \quad \textbf{4.} \emph{Numéroter} (plus petit indice à la fonction principale ; n\textsuperscript{o}1 imposé pour COOH/COOR/CHO/amide). \quad \textbf{5.} \emph{Écrire} : préfixes (ramifications + fonctions secondaires, ordre alphabétique, avec indices) $+$ nom de la chaîne $+$ suffixe (avec indice si utile).
\end{meth}

\begin{exmp}[Nomenclature détaillée]
Molécule : \raisebox{-0.8em}{\chemfig{HO-[:30]CH_2-[:-30]CH(-[:90]CH_3)-[:-30]CH_2-[:30](=[:90]O)-[:-30]OH}}
\begin{itemize}[leftmargin=1.5em,itemsep=0pt,topsep=1pt]
  \item \textbf{Fonctions} : un $\ce{-COOH}$ et un $\ce{-OH}$ ; une ramification $\ce{-CH3}$.
  \item \textbf{Priorité} : $\ce{-COOH} > \ce{-OH}$ $\Rightarrow$ fonction principale $=$ \textbf{acide} (suffixe \textit{-oïque}) ; l'alcool devient préfixe \textbf{hydroxy-}.
  \item \textbf{Chaîne} : 4 carbones contenant le $\ce{-COOH}$ $\Rightarrow$ acide \textbf{butanoïque}.
  \item \textbf{Numérotation} : le C de $\ce{-COOH}$ est le n\textsuperscript{o}1 ; le méthyle et le $\ce{-OH}$ tombent en position 3.
  \item \textbf{Nom} : \textbf{acide 3-hydroxy-3-méthylbutanoïque} (préfixes « hydroxy » et « méthyl » par ordre alphabétique).
\end{itemize}
\end{exmp}
