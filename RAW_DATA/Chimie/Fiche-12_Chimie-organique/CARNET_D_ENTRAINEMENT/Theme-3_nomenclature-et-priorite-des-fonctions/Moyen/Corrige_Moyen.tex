\begin{corr}{acide + alcool : le préfixe hydroxy-}
Deux fonctions : $\ce{-COOH}$ et $\ce{-OH}$. Priorité : $\ce{-COOH} > \ce{-OH}$ $\Rightarrow$ fonction principale $=$ \rep{acide} (suffixe \textit{-oïque}). Chaîne : 3 C $\Rightarrow$ \emph{propanoïque}. Le C du $\ce{-COOH}$ est le n\textsuperscript{o}1 ; le $\ce{-OH}$ tombe en position 3 $\Rightarrow$ préfixe \textbf{hydroxy-}. Nom : \rep{acide 3-hydroxypropanoïque}.
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{cétone + halogène}
Fonction principale : la \textbf{cétone} (suffixe \textit{-one}). Chaîne : 4 C $\Rightarrow$ \emph{butan-}. On numérote pour donner à la cétone le plus petit indice : en partant de l'extrémité la plus proche du $\ce{C=O}$, celui-ci est en position \textbf{2} (et non 3). Le chlore tombe alors en position 3 (préfixe \textbf{chloro-}). Nom : \rep{3-chlorobutan-2-one}.
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{aldéhyde + cétone : le préfixe oxo-}
\textbf{a)} Les deux carbonyles : un $\ce{-CHO}$ (aldéhyde, en bout de chaîne) et un $\ce{C=O}$ interne (cétone). Priorité $\ce{-CHO} > \text{C=O}$ $\Rightarrow$ \rep{la fonction principale est l'aldéhyde} (suffixe \textit{-al}).\\
\textbf{b)} Chaîne de 4 C $\Rightarrow$ \emph{butanal}, le $\ce{-CHO}$ portant le n\textsuperscript{o}1. La cétone est en position 3 : elle se cite en préfixe \textbf{3-oxo-}. Nom : \rep{3-oxobutanal}.
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{choisir le bon sens de numérotation}
\textbf{a)} En partant de la \textbf{gauche} : $\ce{-OH}$ en position \textbf{2}. En partant de la \textbf{droite} : $\ce{-OH}$ en position \textbf{4}.\\
\textbf{b)} On retient le sens qui donne à la fonction principale (l'alcool) le \rep{plus petit indice}, soit \textbf{2}. Nom : \rep{pentan-2-ol}.
\end{corr}

% ---------------------------------------------------------------------
\begin{corr}{aldéhyde + éther : le préfixe (alk)oxy-}
Fonction principale : l'\textbf{aldéhyde} ($\ce{-CHO} > \ce{-O-}$), suffixe \textit{-al}. Chaîne de 3 C contenant le $\ce{-CHO}$ $\Rightarrow$ \emph{propanal}, le $\ce{-CHO}$ portant le n\textsuperscript{o}1. Le groupe $\ce{-O-CH3}$ (éther), en position 3, se cite en préfixe \textbf{méthoxy-}. Nom : \rep{3-méthoxypropanal}.
\end{corr}
