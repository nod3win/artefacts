\begin{corrige}[Suffixe $\rightarrow$ fonction]
\begin{enumerate}[label=\alph*)]
  \item \textit{-ol} : \textbf{alcool}.
  \item \textit{-al} : \textbf{aldéhyde}.
  \item \textit{-one} : \textbf{cétone}.
  \item \textit{acide \dots-oïque} : \textbf{acide carboxylique}.
  \item \textit{-amine} : \textbf{amine}.
\end{enumerate}
\end{corrige}

\begin{corrige}[Groupement $\rightarrow$ nom de la fonction]
\begin{enumerate}[label=\alph*)]
  \item $-\ce{OH}$ : \textbf{alcool}.
  \item $-\ce{COOH}$ : \textbf{acide carboxylique}.
  \item $-\ce{O}-$ (pont) : \textbf{éther}.
  \item $-\ce{C#N}$ : \textbf{nitrile}.
\end{enumerate}
\end{corrige}

\begin{corrige}[Aldéhyde ou cétone ?]
\begin{enumerate}[label=\alph*)]
  \item \ce{CH3-CHO} : le carbonyle porte un \textbf{H} $\Rightarrow$ \textbf{aldéhyde} (éthanal).
  \item \ce{CH3-CO-CH3} : le carbonyle est encadré par \textbf{deux carbones} $\Rightarrow$ \textbf{cétone} (propanone).
\end{enumerate}
La seule différence est le voisinage du $\ce{C=O}$.
\end{corrige}

\begin{corrige}[Identifier la fonction d'une petite molécule]
\begin{enumerate}[label=\alph*)]
  \item \ce{CH3-CH2-OH} : $-\ce{OH}$ terminal $\Rightarrow$ \textbf{alcool} (éthanol).
  \item \ce{CH3-O-CH3} : oxygène entre deux carbones $\Rightarrow$ \textbf{éther} (diméthyléther).
  \item \ce{CH3-NH2} : azote sans carbonyle $\Rightarrow$ \textbf{amine} (méthylamine).
  \item \ce{CH3-CH2-SH} : groupement $-\ce{SH}$ $\Rightarrow$ \textbf{thiol} (éthanethiol).
\end{enumerate}
\end{corrige}

\begin{corrige}[Classer un alcool]
On regarde le nombre de carbones liés au carbone porteur du $-\ce{OH}$.
\begin{enumerate}[label=\alph*)]
  \item propan-1-ol : le $\ce{C}$--OH est lié à $1$ carbone $\Rightarrow$ alcool \textbf{primaire}.
  \item propan-2-ol : le $\ce{C}$--OH est lié à $2$ carbones $\Rightarrow$ alcool \textbf{secondaire}.
  \item 2-méthylpropan-2-ol : le $\ce{C}$--OH est lié à $3$ carbones $\Rightarrow$ alcool \textbf{tertiaire}.
\end{enumerate}
\end{corrige}
