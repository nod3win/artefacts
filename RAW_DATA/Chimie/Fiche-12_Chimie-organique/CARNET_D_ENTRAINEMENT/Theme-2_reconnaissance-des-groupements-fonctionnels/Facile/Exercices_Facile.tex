\begin{consigne}
Exercices \textbf{ouverts} : chaque question isole \emph{un} réflexe d'identification. Rédige et \textbf{justifie} (pour un carbonyle : que trouve-t-on de l'autre côté ?).
\end{consigne}

% ---------------------------------------------------------------------
\begin{exo}{nommer la fonction de chaque molécule}
Pour chacune des cinq molécules ci-dessous, indique la \textbf{fonction chimique} présente.
\begin{center}\renewcommand{\arraystretch}{1.2}
\begin{tabular}{ccccc}
\chemfig{CH_3-[:30]OH} & \chemfig{CH_3-[:30](=[:90]O)-[:-30]CH_3} & \chemfig{CH_3-[:30](=[:90]O)-[:-30]H} & \chemfig{CH_3-[:30]NH_2} & \chemfig{CH_3-[:30](=[:90]O)-[:-30]OH} \\
\textbf{(A)} & \textbf{(B)} & \textbf{(C)} & \textbf{(D)} & \textbf{(E)} \\
\end{tabular}
\end{center}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{aldéhyde ou cétone ?}
Les deux molécules ci-dessous portent un carbonyle $\ce{C=O}$. Indique pour chacune s'il s'agit d'un \textbf{aldéhyde} ou d'une \textbf{cétone}, et \emph{justifie} par le voisinage du carbonyle.
\begin{center}
\begin{tabular}{ccc}
\chemfig{CH_3-[:30]CH_2-[:-30](=[:-90]O)-[:30]H} & \qquad &
\chemfig{CH_3-[:30](=[:90]O)-[:-30]CH_3} \\
\textbf{(A)} & & \textbf{(B)} \\
\end{tabular}
\end{center}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{alcool ou éther ?}
Les molécules (A) et (B) ont la \textbf{même formule brute} $\ce{C2H6O}$. Identifie la fonction de chacune et explique pourquoi ce sont des \textbf{isomères de fonction}.
\begin{center}
\begin{tabular}{ccc}
\chemfig{CH_3-[:30]CH_2-[:-30]OH} & \qquad &
\chemfig{CH_3-[:30]O-[:-30]CH_3} \\
\textbf{(A)} & & \textbf{(B)} \\
\end{tabular}
\end{center}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{amine ou amide ?}
Les deux molécules contiennent un atome d'azote. Indique pour chacune s'il s'agit d'une \textbf{amine} ou d'une \textbf{amide}, et \emph{justifie}.
\begin{center}
\begin{tabular}{ccc}
\chemfig{CH_3-[:30]NH_2} & \qquad &
\chemfig{CH_3-[:30](=[:90]O)-[:-30]NH_2} \\
\textbf{(A)} & & \textbf{(B)} \\
\end{tabular}
\end{center}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{classer un alcool (I / II / III)}
On donne le \textbf{butan-2-ol} :
\begin{center}\chemfig{CH_3-[:30]CH(-[:90]OH)-[:-30]CH_2-[:30]CH_3}\end{center}
Le carbone qui porte le groupe $\ce{-OH}$ est lié à combien d'autres carbones ? Déduis-en la \textbf{classe} de cet alcool (primaire, secondaire ou tertiaire).
\end{exo}
