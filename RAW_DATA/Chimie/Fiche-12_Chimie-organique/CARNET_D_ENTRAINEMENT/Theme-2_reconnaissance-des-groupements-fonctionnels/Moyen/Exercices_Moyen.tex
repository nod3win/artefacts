\banner{Thème 2 — Reconnaissance des groupements fonctionnels}{Niveau \textbf{Moyen} — Énoncés \quad$\vert$\quad Fiche 12, Chimie organique}

\begin{consigne}
Exercices \textbf{ouverts} sur des molécules polyfonctionnelles. Consigne clé du concours : \textbf{compter \emph{et} classer}. Parcours chaque structure de façon systématique (repère les $\ce{C=O}$, puis les hétéroatomes isolés).
\end{consigne}

% ---------------------------------------------------------------------
\begin{exo}{recenser toutes les fonctions}
Liste \textbf{toutes} les fonctions chimiques présentes dans la molécule ci-dessous (parcours-la de gauche à droite) :
\begin{center}\chemfig{HO-[:30]CH_2-[:-30](=[:-90]O)-[:30]CH_2-[:-30](=[:90]O)-[:30]OH}\end{center}
Précise, pour chaque carbonyle, à quelle fonction il appartient et pourquoi.
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{compter et classer les alcools}
Dans la molécule suivante :
\begin{center}\chemfig{HO-[:30]CH_2-[:-30]C(-[:90]OH)(-[:-90]CH_3)-[:30]CH(-[:90]OH)-[:-30]CH_3}\end{center}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=1pt]
  \item Combien de fonctions \textbf{alcool} la molécule comporte-t-elle ?
  \item Classe \emph{chacune} d'elles (primaire / secondaire / tertiaire) en examinant le carbone porteur du $\ce{-OH}$.
\end{enumerate}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{discriminer les carbonyles}
On considère la molécule :
\begin{center}\chemfig{CH_3-[:30](=[:90]O)-[:-30]CH_2-[:30]CH=[:-30]CH-[:30](=[:90]O)-[:-30]H}\end{center}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=1pt]
  \item Combien de groupes \textbf{carbonyle} ($\ce{C=O}$) contient-elle ?
  \item À quelle(s) fonction(s) ces carbonyles appartiennent-ils ? (justifie par le voisinage)
  \item Recense \emph{toutes} les fonctions de la molécule. Y a-t-il une fonction acide carboxylique ?
\end{enumerate}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{azote : amine ou amide, et de quelle classe ?}
La molécule ci-dessous contient trois motifs azotés :
\begin{center}\chemfig{H_2N-[:30]CH_2-[:-30]CH_2-[:30](=[:90]O)-[:-30]NH-[:30]CH_2-[:-30]CH_2-[:30]N(-[:90]CH_3)-[:-30]CH_3}\end{center}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=1pt]
  \item Identifie la nature de chaque motif azoté (\textbf{amine} ou \textbf{amide}) en justifiant.
  \item Pour les amines, précise leur \textbf{classe} (primaire / secondaire / tertiaire).
\end{enumerate}
\end{exo}

% ---------------------------------------------------------------------
\begin{exo}{effet d'une réduction sur le décompte des fonctions}
Soit la molécule :
\begin{center}\chemfig{CH_3-[:30]C(-[:90]OH)(-[:-90]CH_3)-[:30]CH_2-[:-30](=[:-90]O)-[:30]CH_2-[:-30]CH(-[:90]OH)-[:30]CH_3}\end{center}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=1pt]
  \item Recense et classe les fonctions oxygénées (alcools et cétone).
  \item On \textbf{réduit} le carbonyle de la cétone en $\ce{-CH(OH)-}$ (ajout de H). Combien la molécule compte-t-elle alors de fonctions alcool, et de quelles classes ?
\end{enumerate}
\end{exo}
