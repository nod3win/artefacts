\begin{exo}[n.o.\ d'un ion monoatomique]
Pour un ion monoatomique, le nombre d'oxydation est simplement égal à sa charge. Donne le n.o.\ (en
chiffres romains) de l'élément dans chaque ion.
\begin{enumerate}[label=\alph*)]
  \item \ce{Na+}
  \item \ce{Cl-}
  \item \ce{Ca^2+}
  \item \ce{Al^3+}
  \item \ce{S^2-}
\end{enumerate}
\end{exo}

\begin{exo}[le n.o.\ d'un corps simple]
Un \emph{corps simple} est un élément non combiné à un autre élément. Quel est le nombre d'oxydation de
l'élément souligné dans chacune de ces espèces ?
\begin{enumerate}[label=\alph*)]
  \item \ce{Fe} (métal)
  \item \ce{O2} (dioxygène)
  \item \ce{Cl2} (dichlore)
  \item \ce{N2} (diazote)
\end{enumerate}
\end{exo}

\begin{exo}[appliquer « oxygène $=-\mathrm{II}$ » dans un oxyde neutre]
Dans ces oxydes, l'oxygène vaut $-\mathrm{II}$ et la molécule est neutre. Détermine le n.o.\ du métal.
\begin{enumerate}[label=\alph*)]
  \item \ce{CaO}
  \item \ce{MgO}
  \item \ce{K2O}
  \item \ce{Al2O3}
\end{enumerate}
\end{exo}

\begin{exo}[appliquer « hydrogène $=+\mathrm{I}$ »]
Dans ces composés, l'hydrogène vaut $+\mathrm{I}$ et la molécule est neutre. Détermine le n.o.\ de l'autre
élément (souligné).
\begin{enumerate}[label=\alph*)]
  \item \ce{HCl} (le chlore)
  \item \ce{H2S} (le soufre)
  \item \ce{HBr} (le brome)
  \item \ce{NH3} (l'azote)
\end{enumerate}
\end{exo}

\begin{exo}[n.o.\ d'un non-métal dans un oxyde]
L'oxygène vaut $-\mathrm{II}$ et la molécule est neutre. Détermine le n.o.\ de l'élément associé.
\begin{enumerate}[label=\alph*)]
  \item \ce{CO2} (le carbone)
  \item \ce{SO3} (le soufre)
  \item \ce{CO} (le carbone)
\end{enumerate}
\end{exo}
