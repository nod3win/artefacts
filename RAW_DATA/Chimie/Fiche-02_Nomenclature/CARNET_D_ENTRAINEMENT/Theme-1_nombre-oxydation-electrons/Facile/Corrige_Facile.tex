\begin{corrige}[n.o.\ d'un ion monoatomique]
Le n.o.\ d'un ion monoatomique est \emph{égal à sa charge}.
\begin{enumerate}[label=\alph*)]
  \item \ce{Na+} : $+\mathrm{I}$.
  \item \ce{Cl-} : $-\mathrm{I}$.
  \item \ce{Ca^2+} : $+\mathrm{II}$.
  \item \ce{Al^3+} : $+\mathrm{III}$.
  \item \ce{S^2-} : $-\mathrm{II}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[le n.o.\ d'un corps simple]
Un corps simple (élément non combiné) a toujours un n.o.\ \textbf{nul}, quelle que soit son atomicité.
\begin{enumerate}[label=\alph*)]
  \item \ce{Fe} : $0$.
  \item \ce{O2} : $0$.
  \item \ce{Cl2} : $0$.
  \item \ce{N2} : $0$.
\end{enumerate}
\end{corrige}

\begin{corrige}[appliquer « oxygène $=-\mathrm{II}$ » dans un oxyde neutre]
On pose $x$ le n.o.\ du métal, $\ce{O}=-\mathrm{II}$, et somme des n.o.\ $=0$.
\begin{enumerate}[label=\alph*)]
  \item \ce{CaO} : $x+(-2)=0 \Rightarrow x=+\mathrm{II}$.
  \item \ce{MgO} : $x+(-2)=0 \Rightarrow x=+\mathrm{II}$.
  \item \ce{K2O} : $2x+(-2)=0 \Rightarrow 2x=+2 \Rightarrow x=+\mathrm{I}$.
  \item \ce{Al2O3} : $2x+3(-2)=0 \Rightarrow 2x=+6 \Rightarrow x=+\mathrm{III}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[appliquer « hydrogène $=+\mathrm{I}$ »]
On pose $x$ le n.o.\ de l'élément souligné, $\ce{H}=+\mathrm{I}$, somme $=0$.
\begin{enumerate}[label=\alph*)]
  \item \ce{HCl} : $(+1)+x=0 \Rightarrow x=-\mathrm{I}$ (le chlore).
  \item \ce{H2S} : $2(+1)+x=0 \Rightarrow x=-\mathrm{II}$ (le soufre).
  \item \ce{HBr} : $(+1)+x=0 \Rightarrow x=-\mathrm{I}$ (le brome).
  \item \ce{NH3} : $x+3(+1)=0 \Rightarrow x=-\mathrm{III}$ (l'azote).
\end{enumerate}
\end{corrige}

\begin{corrige}[n.o.\ d'un non-métal dans un oxyde]
On pose $x$ le n.o.\ du non-métal, $\ce{O}=-\mathrm{II}$, somme $=0$.
\begin{enumerate}[label=\alph*)]
  \item \ce{CO2} : $x+2(-2)=0 \Rightarrow x=+\mathrm{IV}$.
  \item \ce{SO3} : $x+3(-2)=0 \Rightarrow x=+\mathrm{VI}$.
  \item \ce{CO} : $x+(-2)=0 \Rightarrow x=+\mathrm{II}$.
\end{enumerate}
\end{corrige}
