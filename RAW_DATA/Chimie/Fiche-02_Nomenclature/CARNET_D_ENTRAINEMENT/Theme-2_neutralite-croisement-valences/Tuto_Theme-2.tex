% THEME_TITLE: Neutralité électrique et croisement des valences

\begin{cle}[la règle d'or : toute substance pure est neutre]
Un composé pur est \textbf{électriquement neutre} : la somme des charges positives (portées par le cation)
égale la somme des charges négatives (portées par l'anion). C'est cette contrainte qui fixe les
\textbf{indices} — les petits chiffres — d'une formule brute. La \textbf{valence} d'un ion est la valeur
absolue de sa charge (\ce{Na+} : 1 ; \ce{Ca^2+} : 2 ; \ce{PO4^3-} : 3).
\end{cle}

\begin{methode}[le chassé-croisé des valences en 4 gestes]
Pour assembler un cation \ce{C^{a+}} (valence $a$) et un anion \ce{A^{b-}} (valence $b$) :
\begin{enumerate}[label=\arabic*.,leftmargin=2.0em,topsep=2pt,itemsep=1pt]
  \item \textbf{identifier} le cation et l'anion, et \emph{leur valence} ;
  \item \textbf{croiser} : la valence de l'un devient l'indice de l'autre $\Rightarrow \ce{C_bA_a}$ ;
  \item \textbf{simplifier} les indices par leur diviseur commun (ex.\ $\ce{Mg2O2}\to\ce{MgO}$) ;
  \item \textbf{parenthéser} tout ion \emph{polyatomique} affecté d'un indice $\geq 2$, puis \emph{vérifier}
        la neutralité.
\end{enumerate}
\end{methode}

\begin{exemplebox}[trois croisements types]
\begin{itemize}[leftmargin=1.5em,topsep=2pt,itemsep=2pt]
  \item \ce{Ca^2+} + \ce{Cl-} : on croise $2$ et $1 \Rightarrow \ce{CaCl2}$. Vérif.\ : $+2$ et $2(-1)=-2$.
  \item \ce{Mg^2+} + \ce{O^2-} : on croise $\Rightarrow \ce{Mg2O2}$, que l'on \emph{simplifie} en \ce{MgO}.
  \item \ce{NH4+} + \ce{SO4^2-} : cation polyatomique de valence 1, anion de valence 2 $\Rightarrow$
        deux \ce{NH4+} $\Rightarrow \ce{(NH4)2SO4}$ (parenthèses car \ce{NH4} est présent 2 fois).
\end{itemize}
\end{exemplebox}

\begin{piege}[les fautes qui coûtent des points]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item \textbf{oublier de simplifier} : \ce{Ca^2+}+\ce{O^2-} donne \ce{CaO}, pas \og\ce{Ca2O2}\fg{}.
  \item \textbf{parenthèses} : seulement pour un ion \emph{polyatomique} d'indice $\geq 2$
        (\ce{Al2(SO4)3}, \ce{Ca(OH)2}) ; jamais autour d'un atome seul (\ce{FeCl3}, pas \og Fe(Cl)3\fg{}).
  \item l'\textbf{ammonium} \ce{NH4+} est un \emph{cation} de valence 1 ; l'ammoniac \ce{NH3} est neutre.
  \item bien lire la \textbf{valence variable} : nitrate d'étain(IV) utilise \ce{Sn^4+} (et non \ce{Sn^2+}).
\end{itemize}
\end{piege}

\begin{remarque}[valences à connaître par cœur]
\textbf{Valence 1 :} \ce{Na+}, \ce{K+}, \ce{Ag+}, \ce{NH4+}, \ce{Cl-}, \ce{Br-}, \ce{NO3-}, \ce{OH-},
\ce{ClO4-}, \ce{MnO4-}, \ce{CH3COO-}. \quad
\textbf{Valence 2 :} \ce{Ca^2+}, \ce{Mg^2+}, \ce{Zn^2+}, \ce{O^2-}, \ce{S^2-}, \ce{SO4^2-}, \ce{SO3^2-},
\ce{CO3^2-}, \ce{S2O3^2-}, \ce{CrO4^2-}. \quad
\textbf{Valence 3 :} \ce{Al^3+}, \ce{PO4^3-}. \quad
Valence \emph{variable} : \ce{Fe} (II/III), \ce{Cu} (I/II), \ce{Sn} (II/IV), \ce{Pb} (II/IV).
\end{remarque}
