\bandeau{Thème 1 — Moyen}{Ions, valences et nombre d'oxydation --- Exercices}

\noindent\textit{On passe au calcul du nombre d'oxydation. Pose à chaque fois l'équation « somme des n.o.\
$=$ charge globale ».}
\vspace{0.3cm}

\begin{exercice}[n.o.\ dans un oxyde neutre]
Détermine le nombre d'oxydation de l'élément \emph{autre que l'oxygène} dans chacune de ces molécules
neutres (rappel : \ce{O} vaut $-\mathrm{II}$) :
\[ \text{a) } \ce{SO3} \qquad \text{b) } \ce{N2O5} \qquad \text{c) } \ce{Cl2O5} \qquad \text{d) } \ce{CO2}. \]
\end{exercice}

\begin{exercice}[n.o.\ dans un composé avec hydrogène]
Détermine le nombre d'oxydation de l'élément central (rappel : \ce{H} vaut $+\mathrm{I}$) :
\[ \text{a) } \ce{NH3} \qquad \text{b) } \ce{H2S} \qquad \text{c) } \ce{CH4} \qquad \text{d) } \ce{H2SO4}. \]
Pour \ce{H2SO4}, pense à traiter \emph{à la fois} les 2 \ce{H} ($+\mathrm{I}$) et les 4 \ce{O} ($-\mathrm{II}$).
\end{exercice}

\begin{exercice}[n.o.\ dans un ion polyatomique]
Ici la somme des n.o.\ n'est plus nulle : elle égale la \textbf{charge de l'ion}. Détermine le n.o.\ de
l'élément central :
\[ \text{a) } \ce{MnO4-} \qquad \text{b) } \ce{Cr2O7^2-} \qquad \text{c) } \ce{SO4^2-} \qquad \text{d) } \ce{NO3-}. \]
\end{exercice}

\begin{exercice}[du n.o.\ au nombre d'électrons]
Un métal au degré d'oxydation $+n$ a cédé $n$ électrons. Complète : pour chaque cation, donne le nombre
d'électrons restants (on donne $Z$).
\begin{center}
\begin{tabular}{lcccc}
\toprule
Cation & \ce{Fe^3+} & \ce{Cu+} & \ce{Cu^2+} & \ce{Sn^4+} \\
$Z$ & 26 & 29 & 29 & 50 \\
\bottomrule
\end{tabular}
\end{center}
\end{exercice}

\begin{exercice}[un métal, deux degrés]
Le fer possède deux degrés d'oxydation usuels. En utilisant \ce{O}$=-\mathrm{II}$ et la neutralité,
détermine le n.o.\ du fer dans :
\[ \text{a) } \ce{FeO} \qquad \text{b) } \ce{Fe2O3}. \]
Nomme ensuite chaque fer (« fer(\ldots) ») et explique pourquoi il est \emph{obligatoire} d'indiquer la
valence pour le fer, mais pas pour le sodium dans \ce{Na2O}.
\end{exercice}
