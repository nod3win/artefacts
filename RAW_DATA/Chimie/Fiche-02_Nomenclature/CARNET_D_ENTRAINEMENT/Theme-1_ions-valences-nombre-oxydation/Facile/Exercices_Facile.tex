\bandeau{Thème 1 — Facile}{Ions, valences et nombre d'oxydation --- Exercices}

\noindent\textit{Chaque exercice isole une seule mécanique. Travaille-les dans l'ordre, sans regarder le
corrigé (fichier séparé).}
\vspace{0.3cm}

\begin{exercice}[métal ou non-métal ?]
Pour chacun des éléments suivants, indique s'il s'agit d'un \textbf{métal} ou d'un \textbf{non-métal}, puis
précise s'il formera plutôt un \textbf{cation} ou un \textbf{anion} :
\[ \ce{Na}, \quad \ce{Cl}, \quad \ce{Ca}, \quad \ce{O}, \quad \ce{Al}, \quad \ce{S}, \quad \ce{Fe}, \quad \ce{F}. \]
\end{exercice}

\begin{exercice}[lire une valence]
Donne la \textbf{valence} (valeur absolue de la charge) de chacun de ces ions :
\[ \ce{K+}, \quad \ce{Mg^2+}, \quad \ce{Al^3+}, \quad \ce{Br-}, \quad \ce{O^2-}, \quad \ce{PO4^3-}, \quad \ce{NH4+}, \quad \ce{SO4^2-}. \]
\end{exercice}

\begin{exercice}[électrons gagnés ou perdus]
Le tableau donne le numéro atomique $Z$. Indique, pour chaque ion, s'il a \emph{gagné} ou \emph{perdu} des
électrons, combien, et le nombre total d'électrons qu'il possède.
\begin{center}
\begin{tabular}{lccccc}
\toprule
Ion & \ce{Na+} & \ce{Ca^2+} & \ce{Al^3+} & \ce{Cl-} & \ce{S^2-} \\
$Z$ & 11 & 20 & 13 & 17 & 16 \\
\bottomrule
\end{tabular}
\end{center}
\end{exercice}

\begin{exercice}[n.o.\ des cas immédiats]
Donne, \emph{sans calcul}, le nombre d'oxydation de l'élément souligné, en citant la convention utilisée :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item le fer dans \ce{Fe} (métal solide) ; \qquad b) le chlore dans \ce{Cl2} (dichlore) ;
  \item[c)] l'ion \ce{Ca^2+} ; \qquad d) l'ion \ce{Cl-} ; \qquad e) l'oxygène dans \ce{Na2O}.
\end{enumerate}
\end{exercice}

\begin{exercice}[repérer le cation]
Parmi les ions polyatomiques suivants, \textbf{un seul} est un cation. Lequel ? Justifie en lisant le signe
de la charge, puis classe les autres comme anions.
\[ \ce{NO3-}, \quad \ce{SO4^2-}, \quad \ce{NH4+}, \quad \ce{MnO4-}, \quad \ce{CO3^2-}, \quad \ce{PO4^3-}. \]
\end{exercice}
