\begin{exo}[nommer un hydracide]
Un hydracide se nomme « [racine du non-métal]-ure d'hydrogène ». Nomme chaque composé.
\begin{enumerate}[label=\alph*)]
  \item \ce{HCl}
  \item \ce{HBr}
  \item \ce{H2S}
  \item \ce{HF}
\end{enumerate}
\end{exo}

\begin{exo}[l'anion d'un hydracide]
En perdant ses \ce{H+}, l'hydracide donne un anion en \textbf{-ure}. Donne le nom \emph{et} la formule de
cet anion.
\begin{enumerate}[label=\alph*)]
  \item \ce{HCl}
  \item \ce{H2S}
  \item \ce{HI}
\end{enumerate}
\end{exo}

\begin{exo}[la règle des suffixes acide $\to$ anion]
Complète la correspondance de suffixes (règle générale).
\begin{enumerate}[label=\alph*)]
  \item un acide en \textbf{-eux} donne un anion en \ldots
  \item un acide en \textbf{-ique} donne un anion en \ldots
  \item les préfixes \textbf{hypo-} et \textbf{per-} sont-ils conservés en passant à l'anion ?
\end{enumerate}
\end{exo}

\begin{exo}[de l'acide à l'anion (le nom)]
Applique la règle des suffixes pour donner le nom de l'anion.
\begin{enumerate}[label=\alph*)]
  \item acide nitreux
  \item acide nitrique
  \item acide sulfureux
  \item acide sulfurique
\end{enumerate}
\end{exo}

\begin{exo}[nommer un oxacide « à la fiche »]
La fiche lit un oxacide « [nom de l'anion] + d'hydrogène ». Nomme chaque acide de cette façon.
\begin{enumerate}[label=\alph*)]
  \item \ce{H2SO4} (anion sulfate)
  \item \ce{HNO3} (anion nitrate)
  \item \ce{H3PO4} (anion phosphate)
\end{enumerate}
\end{exo}
