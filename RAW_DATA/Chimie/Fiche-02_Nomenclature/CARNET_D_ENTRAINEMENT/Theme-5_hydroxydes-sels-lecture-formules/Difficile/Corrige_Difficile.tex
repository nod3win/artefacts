\begin{corrige}[coefficient contre indice]
Le coefficient (2) multiplie toute la formule ; les indices décrivent \emph{un} motif \ce{Na2CO3}
(2 \ce{Na}, 1 \ce{C}, 3 \ce{O}, soit 6 atomes).
\begin{enumerate}[label=\alph*)]
  \item \textbf{2} molécules (motifs) de carbonate de sodium.
  \item $2 \times 2 = \boxed{4}$ atomes de sodium.
  \item $2 \times 6 = \boxed{12}$ atomes au total (et non « 4 molécules » ni « 8 atomes »).
\end{enumerate}
\end{corrige}

\begin{corrige}[lire un hydrate]
\begin{enumerate}[label=\alph*)]
  \item « \ce{. 2 H2O} » signifie que \emph{deux} molécules d'eau de cristallisation sont associées à chaque
        motif \ce{CaCO3} : c'est un composé \textbf{dihydraté}.
  \item \textbf{2} molécules d'eau par motif.
  \item le composé anhydre est \ce{CaCO3} (carbonate de calcium).
\end{enumerate}
\end{corrige}

\begin{corrige}[masse molaire d'un sel hydraté]
\begin{enumerate}[label=\alph*)]
  \item \ce{Cu^2+} (valence 2) + \ce{SO4^2-} (valence 2) $\Rightarrow$ \ce{Cu2(SO4)2}\,$\to$\,\ce{CuSO4}.
  \item $M(\ce{CuSO4}) = 63{,}5 + 32 + 4\times16 = \qty{159,5}{\gram\per\mol}$.
  \item $M(\ce{CuSO4 . 5 H2O}) = 159{,}5 + 5\times(2\times1+16) = 159{,}5 + 5\times18 = \qty{249,5}{\gram\per\mol}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[identifier un métal par un sulfate hydraté]
\begin{enumerate}[label=\alph*)]
  \item $M = \dfrac{m}{n} = \dfrac{55{,}6}{0{,}20} = \qty{278}{\gram\per\mol}$.
  \item $M(\ce{SO4})=\qty{96}{\gram\per\mol}$ et $7\,M(\ce{H2O})=7\times18=\qty{126}{\gram\per\mol}$, donc
        $M(\ce{M}) = 278 - 96 - 126 = \qty{56}{\gram\per\mol}$.
  \item \qty{56}{\gram\per\mol} correspond au \textbf{fer} \ce{Fe} : le sel est \ce{FeSO4 . 7 H2O}.
\end{enumerate}
\end{corrige}

\begin{corrige}[mise en situation : un soluté physiologique]
\begin{enumerate}[label=\alph*)]
  \item chlorure de sodium \ce{NaCl} ; chlorure de potassium \ce{KCl}.
  \item chlorure de calcium \ce{CaCl2} : le calcium étant divalent, chaque motif libère \textbf{2} ions \ce{Cl-}.
  \item parce que dans chaque sel la somme des charges positives égale la somme des charges négatives :
        l'édifice est globalement neutre bien qu'il soit constitué d'ions.
\end{enumerate}
\end{corrige}
