\bandeau{Thème 2 — Moyen}{Du nom à la formule : croisement des valences --- Exercices}

\noindent\textit{On part maintenant du \emph{nom complet}. Pense à l'ordre de lecture : « [anion] de [cation] ».}
\vspace{0.3cm}

\begin{exercice}[écrire un sel neutre à partir de son nom]
Donne la formule brute (avec vérification de la neutralité) de :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item sulfure de sodium \qquad b) nitrate de calcium \qquad c) phosphate d'aluminium
  \item[d)] sulfate de potassium \qquad e) carbonate de sodium
\end{enumerate}
\end{exercice}

\begin{exercice}[quand la valence du métal est précisée]
La mention « (II), (III), (IV)\ldots » donne le degré du métal, donc sa valence. Écris la formule de :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item nitrate de fer(III) \qquad b) sulfate de cuivre(II) \qquad c) chlorure d'étain(IV)
  \item[d)] carbonate de fer(II) \qquad e) phosphate de fer(III)
\end{enumerate}
\end{exercice}

\begin{exercice}[le piège des suffixes -ite / -ate]
Une seule lettre change l'anion. Écris la formule des deux composés de chaque paire, et souligne la
différence :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item nitrite de sodium \emph{et} nitrate de sodium ;
  \item sulfite de potassium \emph{et} sulfate de potassium ;
  \item hypochlorite de sodium \emph{et} perchlorate de sodium.
\end{enumerate}
\end{exercice}

\begin{exercice}[cation \emph{et} anion polyatomiques : le thiosulfate d'ammonium]
Écris la formule du \textbf{thiosulfate d'ammonium}. Détaille : le cation, l'anion (\ce{S2O3^2-}), le
croisement, les parenthèses, la vérification. Distingue bien ta réponse de celle du \emph{sulfite}
d'ammonium.
\end{exercice}

\begin{exercice}[traduire un énoncé d'annales en formules]
Un énoncé décrit : « le mélange de \textbf{phosphate de sodium} et de \textbf{bromure de strontium} conduit
à la formation de \textbf{phosphate de strontium} et de \textbf{bromure de sodium} » (le strontium \ce{Sr}
est un alcalino-terreux, valence 2). Écris les \emph{quatre} formules brutes correctement (sans équilibrer
l'équation).
\end{exercice}
