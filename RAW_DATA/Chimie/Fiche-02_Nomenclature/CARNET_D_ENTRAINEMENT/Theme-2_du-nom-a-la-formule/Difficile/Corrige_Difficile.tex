\bandeau{Thème 2 — Difficile}{Du nom à la formule : croisement des valences --- Corrigé}

\begin{corrige}[nitrate d'étain(IV) : la bonne formule et les pièges]
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt,itemsep=1.5pt]
  \item Étain $=$ \ce{Sn}. « Étain(IV) » $\Rightarrow$ degré $+\mathrm{IV}$ $\Rightarrow$ cation \ce{Sn^4+} (valence 4).
  \item Nitrate $=$ \ce{NO3-} (valence 1). Le nitr\emph{ite} est \ce{NO2-} : un oxygène de moins.
  \item Croisement : $\ce{Sn^4+}+\ce{NO3-} \to \ce{Sn(NO3)4}$. Neutralité : $(+4)+4(-1)=0$. \checkmark
        $\boxed{\ce{Sn(NO3)4}}$
  \item \ce{Sr(NO3)2} et \ce{Sr(NO3)4} utilisent le \emph{strontium} \ce{Sr}, pas l'étain \ce{Sn} (mauvais
        élément). \ce{Sn(NO3)2} correspondrait à l'étain(\textbf{II}), pas à l'étain(IV).
\end{enumerate}
\end{corrige}

\begin{corrige}[sulfite d'aluminium : ne pas confondre l'anion]
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt,itemsep=1.5pt]
  \item sulf\emph{ite} \ce{SO3^2-} (valence 2) ; sulf\emph{ate} \ce{SO4^2-} (valence 2). Même valence, mais
        le sulfate a un oxygène de plus.
  \item \ce{Al^3+}$+$\ce{SO3^2-} : croisement 3 et 2 $\to$ \ce{Al2(SO3)3}. Neutralité : $2(+3)+3(-2)=0$.
        \checkmark $\boxed{\ce{Al2(SO3)3}}$
  \item \ce{Al2(SO4)3} est le piège majeur : c'est le \emph{sulfate} d'aluminium (mauvais anion, \ce{SO4}
        au lieu de \ce{SO3}). \ce{Al(SO3)3} viole la neutralité ($+3$ contre $-6$) ; \ce{AlSO4} aussi
        ($+3$ contre $-2$) et emploie de surcroît le sulfate.
\end{enumerate}
\end{corrige}

\begin{corrige}[la grille des croisements]
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{c|ccc}
\toprule
 & \ce{Cl-} & \ce{SO4^2-} & \ce{PO4^3-} \\
\midrule
\ce{Na+}   & \ce{NaCl}  & \ce{Na2SO4} & \ce{Na3PO4} \\
\ce{Ca^2+} & \ce{CaCl2} & \ce{CaSO4}  & \ce{Ca3(PO4)2} \\
\ce{Al^3+} & \ce{AlCl3} & \ce{Al2(SO4)3} & \ce{AlPO4} \\
\bottomrule
\end{tabular}
\end{center}
Sur la « diagonale » où cation et anion ont la même valence (\ce{CaSO4}, \ce{AlPO4}), les indices se
simplifient à 1 : la formule est la plus simple possible.
\end{corrige}

\begin{corrige}[lecture inverse : de la formule aux ions]
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt,itemsep=1.5pt]
  \item \ce{Mg(ClO4)2} : \ce{Mg^2+} $+$ 2 \ce{ClO4-}. $(+2)+2(-1)=0$. \checkmark $\Rightarrow$
        \textbf{perchlorate de magnésium}.
  \item \ce{(NH4)2CO3} : 2 \ce{NH4+} $+$ \ce{CO3^2-}. $2(+1)+(-2)=0$. \checkmark $\Rightarrow$
        \textbf{carbonate d'ammonium}.
  \item \ce{Fe2(SO4)3} : 2 \ce{Fe^3+} $+$ 3 \ce{SO4^2-}. $2(+3)+3(-2)=0$. \checkmark $\Rightarrow$
        \textbf{sulfate de fer(III)}.
  \item \ce{Ca3(PO4)2} : 3 \ce{Ca^2+} $+$ 2 \ce{PO4^3-}. $3(+2)+2(-3)=0$. \checkmark $\Rightarrow$
        \textbf{phosphate de calcium}.
\end{enumerate}
\end{corrige}

\begin{corrige}[déduire la valence d'un métal à partir de la formule]
On écrit la neutralité et on isole la charge du cation.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt,itemsep=1.5pt]
  \item \ce{Cu3(PO4)2} : $3q + 2(-3)=0 \Rightarrow 3q = 6 \Rightarrow q=+2$. Cuivre(II) (valence 2).
  \item \ce{Sn(SO4)2} : $q + 2(-2)=0 \Rightarrow q=+4$. Étain(IV) (valence 4).
  \item \ce{Pb(NO3)2} : $q+2(-1)=0 \Rightarrow q=+2 \Rightarrow$ plomb(II). \quad
        \ce{Pb(NO3)4} : $q+4(-1)=0 \Rightarrow q=+4 \Rightarrow$ plomb(IV). Le plomb prend bien
        \emph{deux} valences : sa valence n'est donc \emph{pas} « toujours 4 ».
\end{enumerate}
\end{corrige}
