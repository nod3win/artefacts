\bandeau{Thème 2 — Facile}{Du nom à la formule : croisement des valences --- Exercices}

\noindent\textit{On entraîne le geste du croisement, brique par brique.}
\vspace{0.3cm}

\begin{exercice}[croiser deux ions monoatomiques]
Écris la formule brute neutre obtenue en associant les ions suivants (croise les valences, puis simplifie
si besoin) :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item \ce{Na+} et \ce{Cl-} \qquad b) \ce{Ca^2+} et \ce{O^2-} \qquad c) \ce{Al^3+} et \ce{O^2-}
  \item[d)] \ce{Mg^2+} et \ce{Cl-} \qquad e) \ce{Na+} et \ce{S^2-}
\end{enumerate}
\end{exercice}

\begin{exercice}[où mettre les parenthèses ?]
Chaque composé associe un cation à un \emph{ion polyatomique}. Écris la formule et mets les parenthèses au
bon endroit (seulement si l'indice de l'ion polyatomique est $\geq 2$) :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item \ce{Ca^2+} et \ce{OH-} \qquad b) \ce{Na+} et \ce{NO3-} \qquad c) \ce{Al^3+} et \ce{NO3-}
  \item[d)] \ce{NH4+} et \ce{SO4^2-} \qquad e) \ce{Ca^2+} et \ce{PO4^3-}
\end{enumerate}
\end{exercice}

\begin{exercice}[simplifier les indices]
Le croisement « brut » donne parfois des indices à simplifier. Donne la formule croisée \emph{puis} sa forme
simplifiée :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item \ce{Ca^2+} et \ce{O^2-} \qquad b) \ce{Sn^4+} et \ce{O^2-} \qquad c) \ce{Mg^2+} et \ce{S^2-}
\end{enumerate}
\end{exercice}

\begin{exercice}[du nom à l'ion et sa valence]
Pour chaque nom, donne la \textbf{formule de l'ion} et sa \textbf{valence} (c'est l'étape 1--2 de la méthode) :
\[ \text{nitrate}, \quad \text{sulfate}, \quad \text{phosphate}, \quad \text{hydroxyde}, \quad \text{carbonate}, \quad \text{ammonium}. \]
\end{exercice}

\begin{exercice}[l'ammonium joue le rôle d'un métal]
L'ion ammonium \ce{NH4+} (valence 1) se comporte comme \ce{Na+}. Écris la formule de :
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=1pt,itemsep=1pt]
  \item ammonium $+$ chlorure \ce{Cl-} \qquad b) ammonium $+$ nitrate \ce{NO3-}
  \item[c)] ammonium $+$ carbonate \ce{CO3^2-} \qquad d) ammonium $+$ phosphate \ce{PO4^3-}
\end{enumerate}
\end{exercice}
