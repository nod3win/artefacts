\begin{corrige}[nommer un oxyde de métal à valence variable]
On trouve la valence par $\ce{O}=-\mathrm{II}$ et neutralité.
\begin{enumerate}[label=\alph*)]
  \item \ce{FeO} : fer $+\mathrm{II}$ $\Rightarrow$ oxyde de fer(II).
  \item \ce{Fe2O3} : fer $+\mathrm{III}$ $\Rightarrow$ oxyde de fer(III).
  \item \ce{Cu2O} : cuivre $+\mathrm{I}$ $\Rightarrow$ oxyde de cuivre(I).
  \item \ce{CuO} : cuivre $+\mathrm{II}$ $\Rightarrow$ oxyde de cuivre(II).
\end{enumerate}
\end{corrige}

\begin{corrige}[écrire un oxyde de métal à valence variable]
On croise la valence indiquée avec $2$, puis on simplifie.
\begin{enumerate}[label=\alph*)]
  \item oxyde de fer(II) : \ce{Fe^2+} $\Rightarrow$ \ce{Fe2O2}\,$\to$\,\ce{FeO}.
  \item oxyde de cuivre(II) : \ce{Cu^2+} $\Rightarrow$ \ce{CuO}.
  \item oxyde de mercure(II) : \ce{Hg^2+} $\Rightarrow$ \ce{HgO}.
  \item oxyde de plomb(IV) : \ce{Pb^4+} $\Rightarrow$ \ce{Pb2O4}\,$\to$\,\ce{PbO2}.
\end{enumerate}
\end{corrige}

\begin{corrige}[nommer un oxyde non métallique \ce{X2O_n}]
Rapport O/X $=n/2$.
\begin{enumerate}[label=\alph*)]
  \item \ce{N2O3} : O/N $=3/2 \Rightarrow$ \emph{sesqui} $\Rightarrow$ sesquioxyde d'azote.
  \item \ce{N2O5} : O/N $=5/2 \Rightarrow$ \emph{hémipent} $\Rightarrow$ hémipentoxyde d'azote.
  \item \ce{Cl2O7} : O/Cl $=7/2 \Rightarrow$ \emph{hémihept} $\Rightarrow$ hémiheptoxyde de chlore.
\end{enumerate}
\end{corrige}

\begin{corrige}[écrire un oxyde non métallique fractionnaire]
Deux atomes de $X$ ; le rapport donne le nombre d'oxygènes ($n = 2 \times$ rapport).
\begin{enumerate}[label=\alph*)]
  \item sesquioxyde de phosphore : $3/2 \Rightarrow$ \ce{P2O3}.
  \item hémipentoxyde de phosphore : $5/2 \Rightarrow$ \ce{P2O5}.
  \item hémioxyde de chlore : $1/2 \Rightarrow$ \ce{Cl2O}.
\end{enumerate}
\end{corrige}

\begin{corrige}[une même paire d'éléments, plusieurs rapports]
Seul le rapport O/Cl change le préfixe.
\begin{enumerate}[label=\alph*)]
  \item \ce{Cl2O} : $1/2 \Rightarrow$ hémioxyde de chlore.
  \item \ce{Cl2O3} : $3/2 \Rightarrow$ sesquioxyde de chlore.
  \item \ce{Cl2O5} : $5/2 \Rightarrow$ hémipentoxyde de chlore.
\end{enumerate}
\end{corrige}
