\begin{exo}[écrire l'expression de $K_{\mathrm{ps}}$]
À partir de l'équation de dissolution, écris l'expression littérale du produit de solubilité de chaque
sel.
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl}
  \item \ce{PbBr2}
  \item \ce{Fe(OH)3}
  \item \ce{CaCO3}
  \item \ce{Ag2CrO4}
\end{enumerate}
\end{exo}

\begin{exo}[coefficient ou exposant ?]
On considère le phosphate d'argent \ce{Ag3PO4}.
\begin{enumerate}[label=\alph*)]
  \item Écris son équation de dissolution.
  \item Écris l'expression correcte de $K_{\mathrm{ps}}$.
  \item Explique pourquoi l'écriture $K_{\mathrm{ps}} = 3\,[\ce{Ag+}]\,[\ce{PO4^{3-}}]$ est
        \textbf{fausse}.
\end{enumerate}
\end{exo}

\begin{exo}[vrai ou faux : la nature de $K_{\mathrm{ps}}$]
Indique si chaque affirmation est vraie ou fausse (et corrige si nécessaire).
\begin{enumerate}[label=\alph*)]
  \item Le produit de solubilité s'exprime en \si{\mole\per\litre}.
  \item $K_{\mathrm{ps}}$ est une constante d'équilibre.
  \item La solubilité $s$ s'exprime en \si{\mole\per\litre}.
  \item $K_{\mathrm{ps}}$ dépend de la température.
\end{enumerate}
\end{exo}

\begin{exo}[calculer $K_{\mathrm{ps}}$ à partir des concentrations]
Dans chaque solution saturée, on a mesuré les concentrations des ions. Calcule $K_{\mathrm{ps}}$
(sel de type $1{:}1$).
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl} : $[\ce{Ag+}]=[\ce{Cl-}]=\qty{1.3e-5}{\mole\per\litre}$.
  \item \ce{BaSO4} : $[\ce{Ba^{2+}}]=[\ce{SO4^{2-}}]=\qty{1.0e-5}{\mole\per\litre}$.
  \item \ce{SnS} : $[\ce{Sn^{2+}}]=[\ce{S^{2-}}]=\qty{1.0e-13}{\mole\per\litre}$.
\end{enumerate}
\end{exo}

\begin{exo}[cas $MX_n$ : l'exposant est $n$]

\end{exo}
