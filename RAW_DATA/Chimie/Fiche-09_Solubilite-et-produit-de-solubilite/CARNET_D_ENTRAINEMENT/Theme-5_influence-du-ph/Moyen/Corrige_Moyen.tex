{Réponses détaillées et justifications}

\begin{correction}[pH de précipitation de \ce{Al(OH)3}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $[\ce{OH-}]=\sqrt[3]{\dfrac{\Kps}{[\ce{Al^3+}]}}=\sqrt[3]{\dfrac{\num{2{,}0e-32}}{\num{1{,}0e-3}}}
        =\sqrt[3]{\num{2{,}0e-29}}\approx\qty{2{,}7e-10}{\molar}$.
  \item $\mathrm{pH}=14+\log(\num{2{,}7e-10})=14-9{,}57\approx\boxed{4{,}4}$.
  \item C'est \textbf{acide} (pH $<7$) : \ce{Al(OH)3} précipite dès que l'eau cesse d'être franchement
        acide — d'où la floculation de l'aluminium en traitement des eaux potables.
\end{enumerate}
\end{correction}

\begin{correction}[pH de précipitation de \ce{Mg(OH)2}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $[\ce{OH-}]=\sqrt{\dfrac{\Kps}{[\ce{Mg^2+}]}}=\sqrt{\dfrac{\num{9{,}0e-12}}{0{,}06}}
        =\sqrt{\num{1{,}5e-10}}\approx\qty{1{,}2e-5}{\molar}$.
  \item $\mathrm{pH}=14+\log(\num{1{,}2e-5})=14-4{,}91\approx\boxed{9{,}1}$.
        Il faut donc rendre l'eau nettement \textbf{basique} pour précipiter le magnésium (cf. l'ajout
        de chaux au Thème 3).
\end{enumerate}
\end{correction}

\begin{correction}[pH de précipitation de \ce{Fe(OH)3}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $[\ce{OH-}]=\sqrt[3]{\dfrac{\num{2{,}0e-39}}{\num{1{,}0e-2}}}=\sqrt[3]{\num{2{,}0e-37}}
        \approx\qty{5{,}9e-13}{\molar}$, d'où
        $\mathrm{pH}=14+\log(\num{5{,}9e-13})\approx\boxed{1{,}8}$.
  \item Oui : \ce{Fe(OH)3} précipite dès $\mathrm{pH}\approx 1{,}8$, c'est-à-dire en milieu
        \textbf{franchement acide}. Il est donc \textbf{très difficile} de conserver du fer(III) en
        solution : il faut un pH extrêmement bas (son $\Kps$ minuscule le rend quasi insoluble).
\end{enumerate}
\end{correction}

\begin{correction}[le calcaire et l'acide]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\ce{CaCO3(s) + 2 H3O+ -> Ca^{2+}(aq) + CO2(g) + 3 H2O}$.
  \item L'anion \ce{CO3^2-} (base faible) est \textbf{consommé} par \ce{H3O+}. En retirant un produit de
        la dissolution $\ce{CaCO3 <=> Ca^2+ + CO3^2-}$, on déplace l'équilibre vers la \textbf{droite}
        (Le Chatelier) : plus de \ce{CaCO3} se dissout, la solubilité \textbf{augmente}.
  \item L'effervescence vient du dégagement de \ce{CO2} gazeux. L'eau pure (neutre) ne fournit pas
        assez de \ce{H3O+} pour attaquer le carbonate : \textbf{pas d'effervescence}. Ce mécanisme
        explique aussi l'érosion des roches calcaires par les pluies acides.
\end{enumerate}
\end{correction}

\begin{correction}[classer trois hydroxydes]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Par pH de précipitation croissant : \boxed{\ce{Fe(OH)3}\,(1{,}8) < \ce{Al(OH)3}\,(4{,}4) < \ce{Mg(OH)2}\,(9{,}1)}.
        C'est l'ordre dans lequel ils précipitent quand on monte le pH.
  \item Le fer précipite dès $\mathrm{pH}\approx 1{,}8$, le magnésium seulement vers $\mathrm{pH}\approx 9{,}1$.
        En maintenant le pH dans un intervalle intermédiaire, par exemple \textbf{$4 \lesssim \mathrm{pH}
        \lesssim 8$}, on précipite \textbf{tout le fer} sous forme de \ce{Fe(OH)3} tout en gardant le
        \ce{Mg^{2+}} en solution : c'est une \textbf{séparation sélective} par le pH.
\end{enumerate}
\end{correction}
