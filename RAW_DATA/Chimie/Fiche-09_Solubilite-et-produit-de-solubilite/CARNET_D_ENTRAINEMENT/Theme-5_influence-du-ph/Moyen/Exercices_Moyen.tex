\begin{exo}[pH de début de précipitation de \ce{Al(OH)3}]
Une solution contient $[\ce{Al^{3+}}]=\qty{1.0e-3}{\mole\per\litre}$ ; on donne
$K_{\mathrm{ps}}(\ce{Al(OH)3})=\num{2.0e-32}$. À partir de quel pH l'hydroxyde d'aluminium commence-t-il
à précipiter ? (Calcule d'abord $[\ce{OH-}]$, puis le pH.)
\end{exo}

\begin{exo}[pH de début de précipitation de \ce{Mg(OH)2}]
Une solution contient $[\ce{Mg^{2+}}]=\qty{0.010}{\mole\per\litre}$ ; on donne
$K_{\mathrm{ps}}(\ce{Mg(OH)2})=\num{9.0e-12}$. Détermine le pH de début de précipitation de
\ce{Mg(OH)2}.
\end{exo}

\begin{exo}[dissolution des sels d'acides faibles en milieu acide]
Pour chaque sel, écris la réaction de dissolution par un acide fort (\ce{H3O+}) et explique en une
phrase pourquoi la solubilité augmente.
\begin{enumerate}[label=\alph*)]
  \item \ce{CaCO3} (l'acide faible est \ce{H2CO3}, qui donne \ce{CO2} + \ce{H2O}).
  \item \ce{FeS} (l'acide faible est \ce{H2S}, gazeux).
\end{enumerate}
\end{exo}

\begin{exo}[\ce{Fe(OH)3} précipite-t-il à pH $=3$ ?]
Une eau industrielle contient $[\ce{Fe^{3+}}]=\qty{1.0e-2}{\mole\per\litre}$ à
$\mathrm{pH}=3$. On donne $K_{\mathrm{ps}}(\ce{Fe(OH)3})=\num{2.0e-39}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule $[\ce{OH-}]$ à ce pH.
  \item Calcule $Q$ et conclus : \ce{Fe(OH)3} précipite-t-il déjà à $\mathrm{pH}=3$ ?
\end{enumerate}
\end{exo}

\begin{exo}[solubilité d'un hydroxyde à pH imposé]
La solubilité d'un hydroxyde est $s=[\ce{M^{n+}}]=K_{\mathrm{ps}}/[\ce{OH-}]^{n}$ lorsque le pH est
fixé. Pour \ce{Mg(OH)2} ($K_{\mathrm{ps}}=\num{9.0e-12}$), calcule $s$ à :
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH}=9$ ;
  \item $\mathrm{pH}=11$.
\end{enumerate}
Que constate-t-on ?
\end{exo}
