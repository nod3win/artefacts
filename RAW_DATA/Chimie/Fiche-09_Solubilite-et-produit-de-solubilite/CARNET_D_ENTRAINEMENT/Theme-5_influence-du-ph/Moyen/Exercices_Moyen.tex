\bandeau{Thème 5 — Influence du pH}%
{Exercices — Niveau Moyen}%
{Calculs de pH de précipitation et dissolution en milieu acide}

\noindent\textit{Consigne : mène les calculs de pH jusqu'à une valeur numérique. Rappel :
$\mathrm{pH}=14+\log[\ce{OH-}]$.}
\medskip

\begin{exercice}[pH de précipitation de \ce{Al(OH)3}]
Solution avec $[\ce{Al^3+}]=\qty{1{,}0e-3}{\molar}$ ; $\Kps(\ce{Al(OH)3})=\num{2{,}0e-32}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Calcule $[\ce{OH-}]$ de début de précipitation.
  \item En déduis le pH de début de précipitation (valeur numérique).
  \item Commente : cette valeur est-elle acide ou basique ?
\end{enumerate}
\end{exercice}

\begin{exercice}[pH de précipitation de \ce{Mg(OH)2}]
Dans l'eau de mer, $[\ce{Mg^2+}]=\qty{0{,}06}{\molar}$ ; $\Kps(\ce{Mg(OH)2})=\num{9{,}0e-12}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Calcule $[\ce{OH-}]$ nécessaire pour amorcer la précipitation de \ce{Mg(OH)2}.
  \item Calcule le pH correspondant.
\end{enumerate}
\end{exercice}

\begin{exercice}[pH de précipitation de \ce{Fe(OH)3}]
Solution avec $[\ce{Fe^3+}]=\qty{1{,}0e-2}{\molar}$ ; $\Kps(\ce{Fe(OH)3})=\num{2{,}0e-39}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Calcule $[\ce{OH-}]$ de début de précipitation, puis le pH.
  \item \ce{Fe(OH)3} précipite-t-il déjà en milieu acide ? Que peux-tu en conclure sur la difficulté de
        garder du fer(III) en solution ?
\end{enumerate}
\end{exercice}

\begin{exercice}[le calcaire et l'acide]
On verse de l'acide (par exemple du vinaigre) sur du carbonate de calcium \ce{CaCO3} (craie).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris l'équation de la réaction de \ce{CaCO3} solide avec les ions \ce{H3O+} (il se dégage du
        \ce{CO2}).
  \item Explique, avec l'argument de Le Chatelier, pourquoi la solubilité de \ce{CaCO3}
        \textbf{augmente} en milieu acide.
  \item Pourquoi observe-t-on une \textbf{effervescence} ? L'eau pure produit-elle le même effet ?
\end{enumerate}
\end{exercice}

\begin{exercice}[classer trois hydroxydes]
On reprend les trois pH de début de précipitation calculés ci-dessus :
\ce{Al(OH)3}\,($\approx 4{,}4$), \ce{Mg(OH)2}\,($\approx 9{,}1$), \ce{Fe(OH)3}\,($\approx 1{,}8$).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Range ces hydroxydes dans l'ordre où ils précipitent quand on \textbf{augmente
        progressivement} le pH.
  \item Une solution contient à la fois \ce{Fe^{3+}} et \ce{Mg^{2+}}. À quel pH pourrait-on précipiter
        \emph{sélectivement} le fer sans faire précipiter le magnésium ? Donne un intervalle plausible.
\end{enumerate}
\end{exercice}
