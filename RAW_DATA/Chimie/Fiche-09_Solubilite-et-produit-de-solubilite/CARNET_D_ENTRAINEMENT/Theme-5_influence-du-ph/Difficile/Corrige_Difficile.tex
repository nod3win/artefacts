\begin{corrige}[la bonne expression du pH de précipitation]
\begin{enumerate}[label=\alph*)]
  \item À $Q=K_{\mathrm{ps}}$ : $K_{\mathrm{ps}}=[\ce{Al^{3+}}][\ce{OH-}]^{3}$, donc
        \[ [\ce{OH-}] = \sqrt[3]{\dfrac{K_{\mathrm{ps}}}{[\ce{Al^{3+}}]}}
                     = \sqrt[3]{\dfrac{\num{2.0e-32}}{\num{1.0e-3}}} = \sqrt[3]{\num{2.0e-29}}. \]
  \item $\mathrm{pH}=14+\log[\ce{OH-}]=14+\log\sqrt[3]{\num{2.0e-29}}$ : c'est la réponse \textbf{D}
        ($\mathrm{pH}\approx 4{,}4$).
  \item Les erreurs :
    \begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=1.5pt]
      \item \textbf{A} : le $\tfrac{1}{2}\log$ (racine carrée) correspondrait à un hydroxyde
            \ce{M(OH)2} ($n=2$) ; ici $n=3$, il faut une racine \emph{cubique}.
      \item \textbf{B} : aucune racine — cela reviendrait à $n=1$.
      \item \textbf{C} : $-\log$ donne le pOH, et le terme $+14$ a disparu.
    \end{itemize}
\end{enumerate}
\end{corrige}

\begin{corrige}[séparer deux cations par le pH]
\begin{enumerate}[label=\alph*)]
  \item \ce{Fe(OH)3} ($n=3$) : $[\ce{OH-}]=\sqrt[3]{\dfrac{\num{2.0e-39}}{0{,}010}}=\sqrt[3]{\num{2.0e-37}}
        \approx\qty{5.9e-13}{\mole\per\litre}$, d'où $\mathrm{pH}=14+\log(\num{5.9e-13})\approx 1{,}8$.
  \item \ce{Mg(OH)2} ($n=2$) : $[\ce{OH-}]=\sqrt{\dfrac{\num{9.0e-12}}{0{,}010}}=\sqrt{\num{9.0e-10}}
        =\qty{3.0e-5}{\mole\per\litre}$, d'où $\mathrm{pH}=14+\log(\num{3.0e-5})\approx 9{,}5$.
  \item \textbf{Oui.} \ce{Fe(OH)3} précipite dès $\mathrm{pH}\approx 1{,}8$, alors que \ce{Mg(OH)2}
        attend $\mathrm{pH}\approx 9{,}5$. En maintenant le pH entre $\sim 3$ et $\sim 9$, on précipite
        \emph{tout} le fer(III) tandis que le magnésium reste dissous : la séparation est possible.
\end{enumerate}
\end{corrige}

\begin{corrige}[traitement des eaux : éliminer l'aluminium]
\begin{enumerate}[label=\alph*)]
  \item $[\ce{OH-}]=\sqrt[3]{\dfrac{\num{2.0e-32}}{\num{1.0e-4}}}=\sqrt[3]{\num{2.0e-28}}
        \approx\qty{5.9e-10}{\mole\per\litre}$, d'où $\mathrm{pH}=14+\log(\num{5.9e-10})\approx 4{,}8$.
  \item Pour $[\ce{Al^{3+}}]=\qty{1.0e-7}{\mole\per\litre}$ :
        $[\ce{OH-}]=\sqrt[3]{\dfrac{\num{2.0e-32}}{\num{1.0e-7}}}=\sqrt[3]{\num{2.0e-25}}
        \approx\qty{5.9e-9}{\mole\per\litre}$, d'où $\mathrm{pH}\approx 5{,}8$.
  \item Augmenter le pH d'une seule unité (de $4{,}8$ à $5{,}8$) divise la concentration résiduelle en
        \ce{Al^{3+}} par \num{1000} : un petit ajustement de pH suffit à rendre la précipitation quasi
        complète.
\end{enumerate}
\end{corrige}

\begin{corrige}[pourquoi la pluie acide ronge le marbre]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Cas 2} : sel d'acide faible — l'anion \ce{CO3^{2-}} est la base conjuguée de l'acide
        faible \ce{H2CO3}.
  \item $\ce{CaCO3(s) + 2 H3O+ -> Ca^{2+}(aq) + CO2(g) + 3 H2O}$. La réaction est quasi totale car le
        \ce{CO2} \emph{gazeux} s'échappe : le produit étant retiré en permanence, l'équilibre est
        déplacé complètement vers la dissolution.
  \item Dans l'eau pure, seule joue la faible solubilité ($s=\sqrt{K_{\mathrm{ps}}}=\qty{5.7e-5}{\mole\per\litre}$,
        infime). En milieu acide, \ce{H3O+} consomme sans cesse les ions \ce{CO3^{2-}}, ce qui « tire »
        l'équilibre vers la dissolution : le marbre se dissout beaucoup plus, d'où l'érosion des
        monuments par les pluies acides.
\end{enumerate}
\end{corrige}
