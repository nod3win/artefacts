\bandeau{Thème 5 — Influence du pH sur la solubilité}%
{Tuto — trois cas, trois effets}%
{Hydroxydes, sels d'acides faibles, et sels « indifférents » au pH}

\noindent Le pH mesure $[\ce{H3O+}]$ (donc aussi $[\ce{OH-}]$). Comme de nombreux sels peu solubles
mettent en jeu des ions sensibles au pH, celui-ci peut \textbf{fortement} modifier la solubilité. Le
livre distingue \textbf{trois cas}.

\section*{1. La relation pH $\leftrightarrow [\ce{OH-}]$}
\begin{theoreme}[produit ionique de l'eau, à \qty{25}{\degreeCelsius}]
\[ K_e=[\ce{H3O+}][\ce{OH-}]=\num{1{,}0e-14}
   \quad\Longrightarrow\quad
   \boxed{\ \mathrm{pH}=14+\log[\ce{OH-}]\,,\qquad [\ce{OH-}]=10^{\,\mathrm{pH}-14}\ } \]
\end{theoreme}
Plus le pH est \textbf{élevé}, plus $[\ce{OH-}]$ est grand.

\section*{2. Les trois cas}
\begin{center}\renewcommand{\arraystretch}{1.35}\small
\begin{tabular}{p{3.4cm} p{4.6cm} p{5.6cm}}
\toprule
\rowcolor{primary!12}
\textbf{Type de sel} & \textbf{Effet du pH} & \textbf{Exemples \& mécanisme}\\
\midrule
\textbf{Hydroxydes} \ce{M(OH)_n} & pH $\uparrow\Rightarrow s\downarrow$ (précipitent en milieu basique) &
\ce{Fe(OH)3}, \ce{Al(OH)3}, \ce{Mg(OH)2} — l'anion \emph{est} \ce{OH-}\\
\textbf{Sels d'acides faibles} & pH $\downarrow\Rightarrow s\uparrow$ (se dissolvent en milieu acide) &
\ce{CaCO3}, \ce{CaC2O4}, \ce{FeS} — \ce{H3O+} consomme l'anion\\
\textbf{Autres sels} (acide fort) & \textbf{pas d'effet} du pH &
\ce{AgCl}, \ce{BaSO4}, \ce{PbBr2} — l'anion ne réagit pas avec \ce{H3O+}\\
\bottomrule
\end{tabular}
\end{center}

\section*{3. Cas n°1 : pH de début de précipitation d'un hydroxyde}
Pour $\ce{M(OH)_n}$ à la concentration $[\ce{M^{n+}}]$, la précipitation débute quand $Q=\Kps$ :
\begin{formule}[pH de précipitation]
\[ [\ce{OH-}]=\sqrt[\scriptstyle n]{\dfrac{\Kps}{[\ce{M^{n+}}]}}
   \quad\Longrightarrow\quad
   \boxed{\ \mathrm{pH}_{\text{précip.}}=14+\log\!\left(\sqrt[\scriptstyle n]{\dfrac{\Kps}{[\ce{M^{n+}}]}}\right)\ } \]
\end{formule}

\begin{exemple}[\ce{Al(OH)3} : à partir de quel pH précipite-t-il ?]
$[\ce{Al^3+}]=\qty{1{,}0e-3}{\molar}$, $\Kps=\num{2{,}0e-32}$, $n=3$ :
\[ [\ce{OH-}]=\sqrt[3]{\dfrac{\num{2{,}0e-32}}{\num{1{,}0e-3}}}=\sqrt[3]{\num{2{,}0e-29}}\approx\qty{2{,}7e-10}{\molar},
   \quad \mathrm{pH}=14+\log(\num{2{,}7e-10})\approx 4{,}4. \]
\ce{Al(OH)3} précipite dès que l'eau cesse d'être franchement acide — d'où la floculation de
l'aluminium en traitement des eaux.
\end{exemple}

\section*{4. Cas n°2 : sels d'acides faibles}
\begin{theoreme}[dissolution en milieu acide]
Si l'anion est la base conjuguée d'un \textbf{acide faible}, \ce{H3O+} le consomme :
$\ce{X^{p-} + H3O+ -> HX^{(p-1)-} + H2O}$, ce qui \textbf{déplace la dissolution vers la droite}
(Le Chatelier) : la solubilité \textbf{augmente} quand le pH baisse.
\end{theoreme}
\begin{exemple}[le calcaire attaqué par l'acide]
$\ce{CaCO3(s) + 2 H3O+ -> Ca^{2+} + CO2(g) + 3 H2O}$ : l'anion \ce{CO3^2-} est transformé en \ce{CO2}
qui s'échappe. Une goutte de vinaigre sur de la craie fait \textbf{effervescence} — l'eau pure, non.
\end{exemple}

\begin{attention}[retenir la \emph{direction}]
Hydroxydes : basique $\Rightarrow$ précipite. Sels d'acides faibles : acide $\Rightarrow$ se dissout.
Autres (\ce{AgCl}, \ce{BaSO4}\dots) : indifférents. Ne pas confondre pH et pOH ($\mathrm{pH}+\mathrm{pOH}=14$).
\end{attention}
