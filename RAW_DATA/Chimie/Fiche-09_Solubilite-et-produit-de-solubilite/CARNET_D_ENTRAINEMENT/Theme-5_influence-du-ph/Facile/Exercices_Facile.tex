\bandeau{Thème 5 — Influence du pH}%
{Exercices — Niveau Facile}%
{Relation pH$/[\ce{OH-}]$, sens de l'effet, et mise en place du calcul}

\noindent\textit{Consigne : une mécanique par question. On travaille à \qty{25}{\degreeCelsius}
($K_e=\num{1{,}0e-14}$).}
\medskip

\begin{exercice}[relation pH $\leftrightarrow$ concentration en \ce{OH-}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Une solution a $[\ce{OH-}]=\qty{1{,}0e-3}{\molar}$. Calcule son pH.
  \item Une solution a un $\mathrm{pH}=5$. Calcule $[\ce{OH-}]$.
\end{enumerate}
\end{exercice}

\begin{exercice}[sens de l'effet du pH]
Pour chacun des sels suivants, indique si sa solubilité \textbf{augmente}, \textbf{diminue} ou
\textbf{ne change pas} lorsque le pH \emph{augmente} :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item l'hydroxyde de fer(III) \ce{Fe(OH)3} ;
  \item le carbonate de calcium \ce{CaCO3} (sel d'acide faible) ;
  \item le chlorure d'argent \ce{AgCl}.
\end{enumerate}
\end{exercice}

\begin{exercice}[qui se dissout en milieu acide ?]
Parmi ces sels, lesquels voient leur solubilité \textbf{augmenter} en milieu acide, et lesquels sont
\textbf{indifférents} au pH ? Justifie en une ligne (nature de l'anion) :
\[ \ce{CaCO3},\quad \ce{BaSO4},\quad \ce{CaC2O4},\quad \ce{AgCl},\quad \ce{FeS}. \]
\end{exercice}

\begin{exercice}[mise en équation — \ce{Al(OH)3}]
Une solution contient $[\ce{Al^3+}]=\qty{1{,}0e-3}{\molar}$ ; $\Kps(\ce{Al(OH)3})=\num{2{,}0e-32}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris la condition de début de précipitation ($Q=\Kps$) pour \ce{Al(OH)3}.
  \item Isole $[\ce{OH-}]$ (laisse sous forme de racine cubique).
  \item Écris l'expression du pH de début de précipitation (forme littérale, sans calcul final).
\end{enumerate}
\end{exercice}

\begin{exercice}[seuil en \ce{OH-} d'un hydroxyde]
Pour l'hydroxyde de magnésium \ce{Mg(OH)2} ($\Kps=\num{9{,}0e-12}$) dans une solution où
$[\ce{Mg^2+}]=\qty{9{,}0e-2}{\molar}$ :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris $[\ce{OH-}]$ de début de précipitation en fonction de $\Kps$ et $[\ce{Mg^2+}]$.
  \item Calcule la valeur numérique de $[\ce{OH-}]$.
\end{enumerate}
\end{exercice}
