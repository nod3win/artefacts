\begin{exo}[{convertir entre $[\ce{OH-}]$ et pH}]
À \qty{25}{\degreeCelsius}, utilise $\mathrm{pH}=14+\log[\ce{OH-}]$ et $[\ce{OH-}]=10^{\mathrm{pH}-14}$.
\begin{enumerate}[label=\alph*)]
  \item $[\ce{OH-}]=\qty{1.0e-3}{\mole\per\litre}$ : quel pH ?
  \item $[\ce{OH-}]=\qty{1.0e-6}{\mole\per\litre}$ : quel pH ?
  \item $\mathrm{pH}=10$ : quelle $[\ce{OH-}]$ ?
  \item $\mathrm{pH}=12$ : quelle $[\ce{OH-}]$ ?
\end{enumerate}
\end{exo}

\begin{exo}[classer les sels dans les trois cas]
Pour chaque sel, indique le cas (1 : hydroxyde, 2 : sel d'acide faible, 3 : autre) et l'effet du pH.
\begin{enumerate}[label=\alph*)]
  \item \ce{Mg(OH)2}
  \item \ce{CaCO3}
  \item \ce{AgCl}
  \item \ce{FeS}
  \item \ce{BaSO4}
\end{enumerate}
\end{exo}

\begin{exo}[vrai ou faux : sens de l'effet du pH]
Indique si chaque affirmation est vraie ou fausse.
\begin{enumerate}[label=\alph*)]
  \item Un hydroxyde métallique précipite quand le pH augmente.
  \item Le carbonate de calcium est plus soluble en milieu basique qu'en milieu acide.
  \item La solubilité de \ce{AgCl} dépend fortement du pH.
  \item En milieu très acide, un hydroxyde métallique est plus soluble.
\end{enumerate}
\end{exo}

\begin{exo}[le pH décide-t-il la précipitation ?]
Une solution contient $[\ce{Mg^{2+}}]=\qty{0.010}{\mole\per\litre}$. On donne
$K_{\mathrm{ps}}(\ce{Mg(OH)2})=\num{9.0e-12}$. Pour chaque pH, calcule $[\ce{OH-}]$ puis
$Q=[\ce{Mg^{2+}}][\ce{OH-}]^{2}$ et conclus.
\begin{enumerate}[label=\alph*)]
  \item $\mathrm{pH}=10$.
  \item $\mathrm{pH}=8$.
\end{enumerate}
\end{exo}

\end{exo}
