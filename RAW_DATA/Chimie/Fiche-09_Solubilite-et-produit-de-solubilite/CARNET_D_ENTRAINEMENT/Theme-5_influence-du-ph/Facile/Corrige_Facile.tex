{Réponses détaillées et justifications}

\begin{correction}[relation pH $\leftrightarrow$ concentration en \ce{OH-}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\mathrm{pH}=14+\log[\ce{OH-}]=14+\log(\num{1{,}0e-3})=14-3=\boxed{11}$ (solution basique).
  \item $[\ce{OH-}]=10^{\mathrm{pH}-14}=10^{5-14}=10^{-9}=\boxed{\qty{1{,}0e-9}{\molar}}$ (solution acide).
\end{enumerate}
\end{correction}

\begin{correction}[sens de l'effet]
Effet quand le pH \emph{augmente} :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item \ce{Fe(OH)3} (hydroxyde) : $s$ \textbf{diminue} (plus de \ce{OH-} $\Rightarrow$ précipitation).
  \item \ce{CaCO3} (sel d'acide faible) : $s$ \textbf{diminue} (il est très soluble en milieu acide ;
        en montant le pH, on n'attaque plus l'anion \ce{CO3^2-}, la solubilité retombe à sa faible valeur).
  \item \ce{AgCl} : \textbf{ne change pas} (l'anion \ce{Cl-} ne réagit pas avec \ce{H3O+}).
\end{enumerate}
\end{correction}

\begin{correction}[qui se dissout en milieu acide ?]
On regarde si l'anion est la base conjuguée d'un acide \textbf{faible} (alors \ce{H3O+} le consomme) :
\begin{itemize}[leftmargin=1.5em,topsep=2pt,itemsep=1pt]
  \item \textbf{Solubilité $\uparrow$ en acide} : \ce{CaCO3} (\ce{CO3^2-}), \ce{CaC2O4} (oxalate
        \ce{C2O4^2-}), \ce{FeS} (\ce{S^2-}) — anions d'acides faibles.
  \item \textbf{Indifférents} : \ce{BaSO4} et \ce{AgCl} — anions \ce{SO4^2-} et \ce{Cl-} (bases
        conjuguées d'acides forts, ne réagissent pas avec \ce{H3O+}).
\end{itemize}
\end{correction}

\begin{correction}[mise en équation — \ce{Al(OH)3}]
$\ce{Al(OH)3(s) <=> Al^{3+} + 3 OH-}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Début de précipitation : $Q=\Kps$, soit $[\ce{Al^3+}][\ce{OH-}]^{3}=\Kps$.
  \item $[\ce{OH-}]=\sqrt[3]{\dfrac{\Kps}{[\ce{Al^3+}]}}=\sqrt[3]{\dfrac{\num{2{,}0e-32}}{\num{1{,}0e-3}}}
        =\sqrt[3]{\num{2{,}0e-29}}$.
  \item $\boxed{\ \mathrm{pH}=14+\log\!\left(\sqrt[3]{\num{2{,}0e-29}}\right)\ }$.
        \emph{(C'est la réponse D du \textsc{qcm}\,n\textsuperscript{o}\,11.)}
\end{enumerate}
\end{correction}

\begin{correction}[seuil en \ce{OH-} — \ce{Mg(OH)2}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $\Kps=[\ce{Mg^2+}][\ce{OH-}]^{2}\Rightarrow [\ce{OH-}]=\sqrt{\dfrac{\Kps}{[\ce{Mg^2+}]}}$.
  \item $[\ce{OH-}]=\sqrt{\dfrac{\num{9{,}0e-12}}{\num{9{,}0e-2}}}=\sqrt{\num{1{,}0e-10}}
        =\boxed{\qty{1{,}0e-5}{\molar}}$.
\end{enumerate}
\end{correction}
