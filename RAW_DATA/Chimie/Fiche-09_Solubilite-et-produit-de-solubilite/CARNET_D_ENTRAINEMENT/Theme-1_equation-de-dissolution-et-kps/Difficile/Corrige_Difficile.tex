{Corrigé — Niveau Difficile}%
{Réponses détaillées et justifications}

\begin{correction}[anatomie complète d'un sel $3{:}2$]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item $\ce{Ca3(PO4)2(s) <=> 3 Ca^{2+}(aq) + 2 PO4^{3-}(aq)}$.
  \item $\Kps=\Ce{Ca^2+}^{3}\,\Ce{PO4^3-}^{2}$.
  \item $p+q = 3+2 = \mathbf{5}$. On s'attend donc à un $\Kps$ proportionnel à $s^{5}$
        (précisément $\Kps=108\,s^{5}$, établi au Thème 2).
  \item \textbf{Commun} : les deux sont des « types $3$ » où un ion porte le coefficient 3, ce qui
        donne des expressions à exposants élevés. \textbf{Différence} : \ce{Ag3PO4} est de type $3{:}1$
        ($\Kps=\Ce{Ag+}^{3}\Ce{PO4^3-}$, donc $p+q=4$ et $\Kps\propto s^{4}$), tandis que
        \ce{Ca3(PO4)2} est de type $3{:}2$ ($p+q=5$, $\Kps\propto s^{5}$). Le second libère \emph{deux}
        anions phosphate par unité, d'où l'exposant 2 supplémentaire.
\end{enumerate}
\end{correction}

\begin{correction}[lecture d'un schéma de dissolution]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item Dans le cristal, on compte \textbf{1 cation} $(2{+})$ pour \textbf{2 anions} $(-)$. La
        neutralité impose la formule $\mathrm{M}\mathrm{X}_2$, soit $n=2$ (type $1{:}2$).
  \item Avec \ce{Pb^{2+}} et \ce{Br-} : formule $\ce{PbBr2}$, équation
        $\ce{PbBr2(s) <=> Pb^{2+}(aq) + 2 Br-(aq)}$.
  \item $\Kps=\Ce{Pb^2+}\,\Ce{Br-}^{2}$.
  \item Une unité dissoute libère 1 \ce{Pb^{2+}} et 2 \ce{Br-} : donc $\Ce{Pb^2+}=s$ et
        $\Ce{Br-}=2s$. (C'est exactement le rapport $1{:}2$ visible sur la figure.)
\end{enumerate}
\end{correction}

\begin{correction}[unités et statut du $\Kps$]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item $\ce{Pb3(PO4)2(s) <=> 3 Pb^{2+}(aq) + 2 PO4^{3-}(aq)}$, donc
        $\Kps=\Ce{Pb^2+}^{3}\,\Ce{PO4^3-}^{2}$.
  \item Dimensionnellement, on aurait $(\si{\mole\per\litre})^{3}\times(\si{\mole\per\litre})^{2}
        =(\si{\mole\per\litre})^{5}=\mathrm{mol^{5}\,L^{-5}}$ : une « unité apparente » à la puissance 5.
  \item En toute rigueur, chaque concentration est divisée par la \textbf{concentration standard}
        $c^{\circ}=\qty{1}{\mole\per\litre}$ (c'est l'\emph{activité}). Les unités se simplifient
        alors et $\Kps$ devient un \textbf{nombre pur, sans dimension}. On écrit donc
        $\Kps=\num{1{,}0e-54}$, sans unité.
  \item \textbf{Faux.} La solubilité $s$ est une \textbf{concentration molaire} : son unité est
        \si{\mole\per\litre} (soit \si{\molar}), \emph{jamais} $\mathrm{M}^{5}$. La puissance 5
        n'apparaît que dans le \emph{$\Kps$}, pas dans $s$. \emph{(Item A faux du \textsc{qcm}\,12.)}
  \item \textbf{Oui}, $\Kps$ dépend de $T$ (constante d'équilibre). La dissolution étant le plus
        souvent \textbf{endothermique}, $\Kps$ \textbf{augmente} avec $T$ : le sel devient plus soluble
        à chaud. \emph{(Item D vrai du \textsc{qcm}\,12.)}
\end{enumerate}
\end{correction}
