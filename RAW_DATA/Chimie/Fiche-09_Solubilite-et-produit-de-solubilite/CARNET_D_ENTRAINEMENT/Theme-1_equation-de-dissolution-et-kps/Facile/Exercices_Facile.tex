\bandeau{Thème 1 — Équation de dissolution \& $\Kps$}%
{Exercices — Niveau Facile}%
{Un geste par question : équation, coefficients, ou expression du $\Kps$}

\noindent\textit{Consigne : réponds directement, sans calcul numérique. À faire avant de regarder le
corrigé (fichier séparé).}
\medskip

\begin{exercice}[écrire des équations de dissolution simples]
Écris l'équation de dissolution dans l'eau (double flèche, états physiques $(s)$ et $(aq)$) des trois
sels suivants :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item le chlorure d'argent \ce{AgCl} ;
  \item le carbonate de calcium \ce{CaCO3} ;
  \item l'iodure de potassium \ce{KI}.
\end{enumerate}
\end{exercice}

\begin{exercice}[repérer le type $p{:}q$]
Pour chacun des sels ci-dessous, écris l'équation de dissolution puis donne son \textbf{type} $p{:}q$
(nombre de cations : nombre d'anions libérés par une unité de sel) :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item le bromure de plomb(II) \ce{PbBr2} ;
  \item le fluorure de calcium \ce{CaF2} ;
  \item l'hydroxyde de fer(III) \ce{Fe(OH)3}.
\end{enumerate}
\end{exercice}

\begin{exercice}[appliquer la règle d'or]
On donne l'équation de dissolution du phosphate d'argent :
\[ \ce{Ag3PO4(s) <=> 3 Ag+(aq) + PO4^{3-}(aq)}. \]
Écris l'expression littérale de son produit de solubilité $\Kps$. Indique clairement quel est
l'exposant de chaque ion et d'où il vient.
\end{exercice}

\begin{exercice}[du $\Kps$ à partir d'équations données]
On donne les équations de dissolution suivantes. Écris l'expression de $\Kps$ pour chacune :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item $\ce{Mn(OH)2(s) <=> Mn^{2+}(aq) + 2 OH-(aq)}$ ;
  \item $\ce{Ag2CrO4(s) <=> 2 Ag+(aq) + CrO4^{2-}(aq)}$ ;
  \item $\ce{Al(OH)3(s) <=> Al^{3+}(aq) + 3 OH-(aq)}$.
\end{enumerate}
\end{exercice}

\begin{exercice}[vrai ou faux : la nature du $\Kps$]
Réponds par \textbf{vrai} ou \textbf{faux} et justifie en une phrase :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Le solide \ce{AgCl(s)} apparaît dans l'expression du $\Kps$.
  \item Le produit de solubilité s'exprime en \si{\mole\per\litre}.
  \item Le $\Kps$ est une constante d'équilibre.
  \item La valeur du $\Kps$ change si l'on modifie la température.
\end{enumerate}
\end{exercice}
