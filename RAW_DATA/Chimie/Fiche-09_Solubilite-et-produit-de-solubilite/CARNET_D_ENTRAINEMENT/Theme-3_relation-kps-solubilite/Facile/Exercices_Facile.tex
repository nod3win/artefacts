\begin{exo}[$K_{\mathrm{ps}}$ en fonction de $s$]
Pour chaque sel, donne l'expression de $K_{\mathrm{ps}}$ en fonction de la solubilité $s$ (utilise son
type).
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl} (type $1{:}1$)
  \item \ce{PbBr2} (type $1{:}2$)
  \item \ce{Ag2CrO4} (type $2{:}1$)
  \item \ce{Fe(OH)3} (type $1{:}3$)
  \item \ce{Ca3(PO4)2} (type $3{:}2$)
\end{enumerate}
\end{exo}

\begin{exo}[calculer $K_{\mathrm{ps}}$ à partir de $s$]
Connaissant la solubilité $s$ et le type du sel, calcule $K_{\mathrm{ps}}$.
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl} ($1{:}1$) : $s=\qty{1.3e-5}{\mole\per\litre}$.
  \item \ce{PbBr2} ($1{:}2$) : $s=\qty{1.3e-2}{\mole\per\litre}$.
  \item \ce{Fe(OH)3} ($1{:}3$) : $s=\qty{1.0e-10}{\mole\per\litre}$.
\end{enumerate}
\end{exo}

\begin{exo}[calculer $s$ à partir de $K_{\mathrm{ps}}$ (type $1{:}1$)]
Pour ces sels $1{:}1$, on a $s=\sqrt{K_{\mathrm{ps}}}$. Calcule $s$.
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl} : $K_{\mathrm{ps}}=\num{1.8e-10}$.
  \item \ce{BaSO4} : $K_{\mathrm{ps}}=\num{1.1e-10}$.
  \item \ce{CaCO3} : $K_{\mathrm{ps}}=\num{3.3e-9}$.
\end{enumerate}
\end{exo}

\begin{exo}[calculer $s$ à partir de $K_{\mathrm{ps}}$ (type $1{:}2$ ou $2{:}1$)]
Pour ces sels, $K_{\mathrm{ps}}=4s^{3}$, donc $s=\sqrt[3]{K_{\mathrm{ps}}/4}$. Calcule $s$.
\begin{enumerate}[label=\alph*)]
  \item \ce{PbBr2} : $K_{\mathrm{ps}}=\num{8.0e-6}$.
  \item \ce{Ag2CrO4} : $K_{\mathrm{ps}}=\num{1.1e-12}$.
\end{enumerate}
\end{exo}

\begin{exo}[comparer deux sels de même type]
Dans chaque paire, les deux sels sont de type $1{:}1$. Sans calcul, indique le \textbf{moins soluble}
et justifie.
\begin{enumerate}[label=\alph*)]
  \item \ce{AgCl} ($K_{\mathrm{ps}}=\num{1.8e-10}$) ou \ce{AgBr} ($K_{\mathrm{ps}}=\num{5.4e-13}$) ?
  \item \ce{BaSO4} ($K_{\mathrm{ps}}=\num{1.1e-10}$) ou \ce{SrSO4} ($K_{\mathrm{ps}}=\num{3.4e-7}$) ?
\end{enumerate}
\end{exo}
