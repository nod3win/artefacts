\begin{exo}[établir la relation par le tableau d'avancement]
Retrouve l'expression de $K_{\mathrm{ps}}$ en fonction de $s$ en posant $[\ce{M^{q+}}]=p\,s$ et
$[\ce{X^{p-}}]=q\,s$, puis en développant.
\begin{enumerate}[label=\alph*)]
  \item \ce{Ag3PO4} (type $3{:}1$)
  \item \ce{Ca3(PO4)2} (type $3{:}2$)
\end{enumerate}
\end{exo}

\begin{exo}[calculer $s$ à partir de $K_{\mathrm{ps}}$ (types variés)]
Calcule la solubilité $s$ de chaque sel à partir de son $K_{\mathrm{ps}}$.
\begin{enumerate}[label=\alph*)]
  \item \ce{CaCO3} (type $1{:}1$) : $K_{\mathrm{ps}}=\num{3.3e-9}$.
  \item \ce{PbBr2} (type $1{:}2$) : $K_{\mathrm{ps}}=\num{6.6e-6}$.
  \item \ce{Fe(OH)3} (type $1{:}3$) : $K_{\mathrm{ps}}=\num{2.0e-39}$.
\end{enumerate}
\end{exo}

\begin{exo}[$K_{\mathrm{ps}}$ à partir d'une variable imposée]
Dans une solution saturée d'hydroxyde de manganèse \ce{Mn(OH)2}, la concentration en ions
\ce{Mn^{2+}} est supposée égale à $\dfrac{x}{2}$ \si{\mole\per\litre}. Exprime le produit de solubilité
$K_{\mathrm{ps}}$ en fonction de $x$.
\end{exo}

\begin{exo}[classer par solubilité (même type)]
On donne trois sels tous de type $1{:}1$. Classe-les par \textbf{solubilité croissante}.
\begin{enumerate}[label=\alph*)]
  \item \ce{BaSO4} : $K_{\mathrm{ps}}=\num{1.1e-10}$.
  \item \ce{CaC2O4} : $K_{\mathrm{ps}}=\num{2.3e-9}$.
  \item \ce{CaCO3} : $K_{\mathrm{ps}}=\num{3.3e-9}$.
\end{enumerate}
\end{exo}

\begin{exo}[de la solubilité massique au $K_{\mathrm{ps}}$]
Le bromure de plomb \ce{PbBr2} a une solubilité massique $s_m=\qty{4.84}{\gram\per\litre}$ dans l'eau
pure à \qty{25}{\degreeCelsius}. On donne $M(\ce{PbBr2})=\qty{367.0}{\gram\per\mole}$. Calcule son
produit de solubilité $K_{\mathrm{ps}}$.
\end{exo}
