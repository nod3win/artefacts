\begin{corrige}[repérer les équations correctes]
On vérifie deux choses : la \emph{charge} de chaque ion et la \emph{conservation} (atomes + charge
globale nulle, puisque le solide est neutre).
\begin{enumerate}[label=\alph*)]
  \item \textbf{Fausse.} Le fer y est ici \ce{Fe^{2+}}, or dans \ce{Fe2(SO4)3} il est \ce{Fe^{3+}}
        (et la charge ne serait pas neutre : $2(+2)+3(-2)=-2$). Correct :
        $\ce{Fe2(SO4)3(s) -> 2 Fe^{3+}(aq) + 3 SO4^{2-}(aq)}$.
  \item \textbf{Correcte.} $3(+2)+2(-3)=0$ : atomes et charges équilibrés.
  \item \textbf{Fausse.} L'argent est \ce{Ag+}, pas \ce{Ag^{2+}}. Correct :
        $\ce{AgCl(s) -> Ag^{+}(aq) + Cl^{-}(aq)}$.
  \item \textbf{Correcte.} $2(+1)+(-2)=0$.
  \item \textbf{Correcte.} $(+3)+3(-1)=0$.
\end{enumerate}
Les équations \textbf{b, d, e} sont correctes.
\end{corrige}

\begin{corrige}[dissolution de l'iodate de strontium]
\begin{enumerate}[label=\alph*)]
  \item $\ce{Sr(IO3)2(s) -> Sr^{2+}(aq) + 2 IO3^{-}(aq)}$ (type $1{:}2$).
  \item Une unité libère 1 \ce{Sr^{2+}} et 2 \ce{IO3-} :
        \[ n(\ce{Sr^{2+}}) = \qty{0.010}{\mole}, \qquad n(\ce{IO3-}) = 2\times\qty{0.010}{\mole}=\qty{0.020}{\mole}. \]
        \textbf{Non} : il y a \emph{deux fois plus} d'ions iodate que d'ions strontium.
  \item Charges positives : $2\times n(\ce{Sr^{2+}}) = 2\times 0{,}010 = \qty{0.020}{\mole}$ de charges~$+$.
        Charges négatives : $1\times n(\ce{IO3-}) = \qty{0.020}{\mole}$ de charges~$-$. Elles se
        compensent : la solution est \textbf{neutre}.
  \item \textbf{Oui} : \ce{Sr(IO3)2} est un composé ionique, donc un \textbf{électrolyte} ; la solution
        contient des ions mobiles et \textbf{conduit} le courant.
\end{enumerate}
\end{corrige}

\begin{corrige}[mélange de deux sels : bilan des ions]
\begin{enumerate}[label=\alph*)]
  \item $\ce{NaCl(s) -> Na^{+}(aq) + Cl^{-}(aq)}$ \quad et \quad
        $\ce{CaCl2(s) -> Ca^{2+}(aq) + 2 Cl^{-}(aq)}$.
  \item Quantités : $n(\ce{NaCl})=\dfrac{5{,}85}{58{,}5}=\qty{0.100}{\mole}$ et
        $n(\ce{CaCl2})=\dfrac{1{,}11}{111{,}0}=\qty{0.0100}{\mole}$, dans $V=\qty{0.500}{\litre}$.
        \begin{itemize}[leftmargin=1.6em,topsep=1pt,itemsep=1pt]
          \item $[\ce{Na+}] = \dfrac{0{,}100}{0{,}500} = \qty{0.200}{\mole\per\litre}$ ;
          \item $[\ce{Ca^{2+}}] = \dfrac{0{,}0100}{0{,}500} = \qty{0.0200}{\mole\per\litre}$ ;
          \item $[\ce{Cl-}] = \dfrac{0{,}100 + 2\times 0{,}0100}{0{,}500}
                 = \dfrac{0{,}120}{0{,}500} = \qty{0.240}{\mole\per\litre}$.
        \end{itemize}
  \item Charges~$+$ : $[\ce{Na+}] + 2[\ce{Ca^{2+}}] = 0{,}200 + 2\times 0{,}0200 = \qty{0.240}{\mole\per\litre}$,
        égales aux charges~$-$ : $[\ce{Cl-}]=\qty{0.240}{\mole\per\litre}$. La solution est neutre.
\end{enumerate}
\end{corrige}

\begin{corrige}[identifier un sel par sa solubilité]
\begin{enumerate}[label=\alph*)]
  \item La masse molaire est le rapport des deux solubilités :
        \[ M = \dfrac{s_m}{s} = \dfrac{\qty{1.64e-2}{\gram\per\litre}}{\qty{2.1e-4}{\mole\per\litre}}
             = \qty{78.1}{\gram\per\mole}. \]
  \item Comme $M(\ce{MF2}) = M(\ce{M}) + 2\,M(\ce{F})$ :
        \[ M(\ce{M}) = 78{,}1 - 2\times 19{,}0 = \qty{40.1}{\gram\per\mole}. \]
        C'est le \textbf{calcium} : le sel est \ce{CaF2} (fluorine).
  \item Type $1{:}2$ : $[\ce{Ca^{2+}}]=s=\qty{2.1e-4}{\mole\per\litre}$ et
        $[\ce{F-}]=2s=\qty{4.2e-4}{\mole\per\litre}$.
\end{enumerate}
\end{corrige}
