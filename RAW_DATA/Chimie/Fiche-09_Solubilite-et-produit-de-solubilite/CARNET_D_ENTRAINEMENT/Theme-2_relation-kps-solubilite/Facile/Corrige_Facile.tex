\bandeau{Thème 2 — Relation $\Kps \leftrightarrow s$}%
{Corrigé — Niveau Facile}%
{Réponses détaillées et justifications}

\begin{correction}[type $1{:}1$ — \ce{CaCO3}]
$\ce{CaCO3(s) <=> Ca^{2+}(aq) + CO3^{2-}(aq)}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Une unité libère 1 \ce{Ca^{2+}} et 1 \ce{CO3^{2-}} : $[\ce{Ca^2+}]=s$ et $[\ce{CO3^2-}]=s$.
  \item $\Kps=[\ce{Ca^2+}][\ce{CO3^2-}]=s\times s=\boxed{s^{2}}$.
  \item $s=\sqrt{\Kps}$.
\end{enumerate}
\emph{(Réponse du \textsc{qcm}\,n\textsuperscript{o}\,3 du livre : $\Kps=s^{2}$.)}
\end{correction}

\begin{correction}[type $1{:}2$ — \ce{CaF2}]
$\ce{CaF2(s) <=> Ca^{2+}(aq) + 2 F-(aq)}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $[\ce{Ca^2+}]=s$ ; $[\ce{F-}]=2s$ (le coefficient 2 !).
  \item $\Kps=[\ce{Ca^2+}][\ce{F-}]^{2}=(s)(2s)^{2}=s\times 4s^{2}=\boxed{4\,s^{3}}$.
\end{enumerate}
\textbf{Piège évité} : $(2s)^2 = 4s^2$, pas $2s^2$.
\end{correction}

\begin{correction}[type $1{:}3$ — \ce{Fe(OH)3}]
$\ce{Fe(OH)3(s) <=> Fe^{3+}(aq) + 3 OH-(aq)}$, donc $[\ce{Fe^3+}]=s$ et $[\ce{OH-}]=3s$ :
\[ \Kps=[\ce{Fe^3+}][\ce{OH-}]^{3}=(s)(3s)^{3}=s\times 27\,s^{3}=\boxed{27\,s^{4}}. \]
Le coefficient est $k=\mathbf{27}$. \emph{(C'est pourquoi le \textsc{qcm}\,n\textsuperscript{o}\,2 du livre,
qui ne propose que $1s^4,3s^4,6s^4,9s^4$, a pour réponse « Autre » : la bonne valeur est $27\,s^4$.)}
\end{correction}

\begin{correction}[conversion massique $\to$ molaire — \ce{PbBr2}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $s_m = s\times M$, donc $s = \dfrac{s_m}{M}$.
  \item $s = \dfrac{\qty{4{,}84}{\gram\per\litre}}{\qty{367{,}0}{\gram\per\mole}}
        = \boxed{\qty{1{,}32e-2}{\mole\per\litre}}$.
\end{enumerate}
Un litre d'eau saturée contient donc \qty{1,32e-2}{\mole} de \ce{PbBr2}, soit bien \qty{4,84}{\gram}.
\end{correction}
