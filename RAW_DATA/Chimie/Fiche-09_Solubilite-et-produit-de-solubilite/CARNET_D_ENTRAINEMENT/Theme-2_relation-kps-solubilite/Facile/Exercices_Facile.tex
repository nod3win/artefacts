\bandeau{Thème 2 — Relation $\Kps \leftrightarrow s$}%
{Exercices — Niveau Facile}%
{Établir la relation $\Kps=f(s)$ type par type}

\noindent\textit{Consigne : pour établir chaque relation, écris l'équation, pose le tableau
d'avancement ($[\text{cation}]=p\,s$, $[\text{anion}]=q\,s$), puis reporte dans le $\Kps$. Réponses
littérales, sauf mention.}
\medskip

\begin{exercice}[type $1{:}1$]
Soit $s$ la solubilité de \ce{CaCO3} dans l'eau pure à \qty{25}{\degreeCelsius}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Exprime $[\ce{Ca^2+}]$ et $[\ce{CO3^2-}]$ en fonction de $s$.
  \item En déduis l'expression de $\Kps$ en fonction de $s$.
  \item Exprime réciproquement $s$ en fonction de $\Kps$.
\end{enumerate}
\end{exercice}

\begin{exercice}[type $1{:}2$]
Soit $s$ la solubilité du fluorure de calcium \ce{CaF2}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris $[\ce{Ca^2+}]$ et $[\ce{F-}]$ en fonction de $s$.
  \item Montre que $\Kps=4\,s^{3}$.
\end{enumerate}
\end{exercice}

\begin{exercice}[type $1{:}3$]
Soit $s$ la solubilité de \ce{Fe(OH)3} dans l'eau pure à \qty{25}{\degreeCelsius}. Établis l'expression
de son produit de solubilité en fonction de $s$. \emph{(On attend une expression de la forme
$k\,s^{4}$ : donne la valeur du coefficient $k$.)}
\end{exercice}

\begin{exercice}[conversion massique $\to$ molaire]
La solubilité massique du bromure de plomb \ce{PbBr2} vaut $s_m=\qty{4{,}84}{\gram\per\litre}$ à
\qty{25}{\degreeCelsius}. On donne $M(\ce{PbBr2})=\qty{367{,}0}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Quelle relation lie $s_m$, $s$ et $M$ ?
  \item Calcule la solubilité molaire $s$ (en \si{\mole\per\litre}).
\end{enumerate}
\end{exercice}
