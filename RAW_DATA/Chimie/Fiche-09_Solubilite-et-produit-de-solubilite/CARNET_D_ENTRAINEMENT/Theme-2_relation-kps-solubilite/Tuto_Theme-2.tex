\bandeau{Thème 2 — Relation entre $\Kps$ et solubilité $s$}%
{Tuto — le tableau d'avancement}%
{Passer de $\Kps$ à $s$ (et inversement), et gérer les conversions massiques}

\noindent $\Kps$ et $s$ décrivent le \textbf{même} état de saturation : $s$ est un \emph{nombre de
moles dissoutes par litre}, $\Kps$ un \emph{produit de concentrations}. Les relier se fait \emph{toujours}
par un petit \textbf{tableau d'avancement} de la dissolution.

\section*{1. Le tableau d'avancement}
On note $s$ la quantité de sel qui se dissout par litre. Chaque unité libère $p$ cations et $q$ anions,
donc à l'équilibre $[\mathrm{M}^{q+}]=p\,s$ et $[\mathrm{X}^{p-}]=q\,s$.

\begin{center}
\begin{tikzpicture}[scale=1.0]
  \node[anchor=north west,font=\footnotesize] at (0,0){%
    \begin{tabular}{|l|c|c|c|}
      \hline
      \rowcolor{primary!15}
       & $\ce{M_{p}X_{q}(s)}$ & $\ce{M^{q+}}$ & $\ce{X^{p-}}$\\
      \hline
      État initial  & excès & $0$ & $0$\\
      \hline
      À l'équilibre & excès & $\boldsymbol{p\,s}$ & $\boldsymbol{q\,s}$\\
      \hline
    \end{tabular}};
\end{tikzpicture}
\end{center}

\noindent On reporte dans $\Kps=[\mathrm{M}^{q+}]^{p}[\mathrm{X}^{p-}]^{q}$ :
\[ \boxed{\ \Kps = (p\,s)^{p}(q\,s)^{q} = p^{\,p}\,q^{\,q}\;s^{\,p+q}\ } \]
Formule \textbf{générale}, valable pour tout sel $\mathrm{M}_p\mathrm{X}_q$.

\section*{2. Le tableau à connaître par cœur}
\begin{center}\renewcommand{\arraystretch}{1.45}\small
\begin{tabular}{c c c c}
\toprule
\rowcolor{primary!12}
\textbf{Type} & \textbf{Exemples} & \textbf{$\Kps=f(s)$} & \textbf{$s=f(\Kps)$}\\
\midrule
$1{:}1$ & \ce{AgCl}, \ce{CaCO3}, \ce{BaSO4} & $\Kps=s^{2}$ & $s=\sqrt{\Kps}$\\
$1{:}2$ ou $2{:}1$ & \ce{PbBr2}, \ce{CaF2}, \ce{Ag2CrO4} & $\Kps=4\,s^{3}$ & $s=\sqrt[3]{\Kps/4}$\\
$1{:}3$ ou $3{:}1$ & \ce{Fe(OH)3}, \ce{Ag3PO4} & $\Kps=27\,s^{4}$ & $s=\sqrt[4]{\Kps/27}$\\
$3{:}2$ & \ce{Ca3(PO4)2}, \ce{Ba3(PO4)2} & $\Kps=108\,s^{5}$ & $s=\sqrt[5]{\Kps/108}$\\
\bottomrule
\end{tabular}
\end{center}

\begin{exemple}[deux dérivations à refaire soi-même]
\ce{Fe(OH)3} ($1{:}3$) : $[\ce{Fe^3+}]=s,\ [\ce{OH-}]=3s \Rightarrow \Kps=(s)(3s)^{3}=27\,s^{4}$.\\
\ce{Ag3PO4} ($3{:}1$) : $[\ce{Ag+}]=3s,\ [\ce{PO4^3-}]=s \Rightarrow \Kps=(3s)^{3}(s)=27\,s^{4}$.\\
Même expression $27\,s^{4}$ : les types $1{:}3$ et $3{:}1$ sont symétriques.
\end{exemple}

\section*{3. Solubilité massique $\leftrightarrow$ molaire}
\begin{formule}[conversion par la masse molaire]
\[ \boxed{\ s_m = s\times M\ }\qquad\Longleftrightarrow\qquad \boxed{\ s = \dfrac{s_m}{M}\ } \]
$s_m$ en \si{\gram\per\litre}, $s$ en \si{\mole\per\litre}, $M$ en \si{\gram\per\mole}.
\end{formule}
\noindent\textbf{Réflexe} : si l'énoncé donne une solubilité en \si{\gram\per\litre}, on la convertit en
\si{\mole\per\litre} \emph{avant} tout calcul de $\Kps$.

\begin{astuce}[comparer deux sels — le piège classique]
\textbf{Même type} : le plus petit $\Kps$ = le moins soluble (comparaison directe permise).\\
\textbf{Types différents} : il faut \textbf{repasser par $s$}. Ex. $\Kps(\ce{AgCl})=\num{1{,}8e-10}$
donne $s=\qty{1{,}3e-5}{\molar}$, mais $\Kps(\ce{Ag2CrO4})=\num{1{,}1e-12}$ (plus petit !) donne
$s=\qty{6{,}5e-5}{\molar}$ : \ce{Ag2CrO4} est en fait \emph{plus} soluble.
\end{astuce}

\begin{astuce}[$s$ est une concentration : la règle de trois]
Puisque $s$ est un « nombre de moles par litre à saturation », la quantité maximale dissoute dans un
volume $V$ est $n=s\times V$. On en tire par exemple le volume d'eau nécessaire : $V=n/s$.
\end{astuce}
