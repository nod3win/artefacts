\bandeau{Thème 3 — Quotient $Q$ \& précipitation}%
{Corrigé — Niveau Difficile}%
{Réponses détaillées et justifications}

\begin{correction}[précipitation sélective — Mohr]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item \ce{AgCl} : $Q=\Kps \Rightarrow [\ce{Ag+}][\ce{Cl-}]=\Kps$, donc
        $[\ce{Ag+}]=\dfrac{\Kps}{[\ce{Cl-}]}=\dfrac{\num{1{,}8e-10}}{\num{1{,}0e-2}}=\boxed{\qty{1{,}8e-8}{\molar}}$.
  \item \ce{Ag2CrO4} (type $2{:}1$) : $\Kps=[\ce{Ag+}]^{2}[\ce{CrO4^2-}]$, donc
        $[\ce{Ag+}]=\sqrt{\dfrac{\Kps}{[\ce{CrO4^2-}]}}=\sqrt{\dfrac{\num{1{,}1e-12}}{\num{1{,}0e-2}}}
        =\sqrt{\num{1{,}1e-10}}\approx\boxed{\qty{1{,}0e-5}{\molar}}$.
  \item Il faut bien \textbf{moins} d'\ce{Ag+} pour \ce{AgCl} ($\num{1{,}8e-8}$) que pour \ce{Ag2CrO4}
        ($\num{1{,}0e-5}$). C'est donc \textbf{\ce{AgCl} qui précipite en premier} : les chlorures sont
        « chassés » avant les chromates.
  \item Tant qu'il reste des \ce{Cl-}, $[\ce{Ag+}]$ reste faible et le chromate ne précipite pas.
        Lorsque presque tous les \ce{Cl-} ont précipité, $[\ce{Ag+}]$ grimpe jusqu'à \qty{1{,}0e-5}{\molar}
        et \ce{Ag2CrO4} \textbf{rouge brique} apparaît : ce virage de couleur signale la fin de la
        précipitation des chlorures (principe du dosage de Mohr).
\end{enumerate}
\end{correction}

\begin{correction}[mélange à volumes inégaux — \ce{PbI2}]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item $[\ce{Pb^2+}]=\dfrac{\num{1{,}5e-2}\times 20}{60}=\qty{5{,}0e-3}{\molar}$ ;
        $[\ce{I-}]=\dfrac{\num{3{,}0e-2}\times 40}{60}=\qty{2{,}0e-2}{\molar}$.
  \item $Q=[\ce{Pb^2+}][\ce{I-}]^{2}=(\num{5{,}0e-3})(\num{2{,}0e-2})^{2}
        =\num{5{,}0e-3}\times\num{4{,}0e-4}=\boxed{\num{2{,}0e-6}}$.
  \item $Q=\num{2{,}0e-6}>\Kps=\num{7{,}1e-9}$ (environ $280\times$) : $Q>\Kps$, un précipité de
        \textbf{\ce{PbI2}} (jaune) se forme.
  \item Sur l'axe, $Q\approx\num{2e-6}$ se situe \textbf{à droite} du repère $\Kps=\num{7{,}1e-9}$ :
        dans la \textbf{zone de précipitation} (sursaturée).
\end{enumerate}
\end{correction}

\begin{correction}[précipitation par mélange concentré — \ce{Cu3(PO4)2}]
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item $\ce{3 Cu^{2+}(aq) + 2 PO4^{3-}(aq) -> Cu3(PO4)2(s)}$.
  \item $[\ce{Cu^2+}]=\dfrac{\num{1{,}0}\times 20}{40}=\qty{0{,}50}{\molar}$ ;
        $[\ce{PO4^3-}]=\dfrac{\num{0{,}20}\times 20}{40}=\qty{0{,}10}{\molar}$.
  \item $Q=[\ce{Cu^2+}]^{3}[\ce{PO4^3-}]^{2}=(0{,}50)^{3}(0{,}10)^{2}=0{,}125\times 0{,}01
        =\boxed{\num{1{,}25e-3}}$.
  \item $Q=\num{1{,}25e-3}\ggg\Kps=\num{1{,}4e-37}$ (34 ordres de grandeur !) : la solution est
        immensément sursaturée. La précipitation de \ce{Cu3(PO4)2} a lieu et est \textbf{quasi-totale}
        (les ions s'associent en solide jusqu'à ce que $Q$ retombe à la valeur $\Kps$, infime). C'est
        pourquoi, dans l'exercice d'annales correspondant, on considère que \ce{Cu^{2+}} et \ce{PO4^{3-}}
        sont presque entièrement consommés.
\end{enumerate}
\end{correction}
