\bandeau{Thème 3 — Quotient $Q$ \& précipitation}%
{Corrigé — Niveau Facile}%
{Réponses détaillées et justifications}

\begin{correction}[expression de $Q$]
$\ce{CaSO4(s) <=> Ca^{2+}(aq) + SO4^{2-}(aq)}$ (type $1{:}1$). Le quotient a la même forme que le $\Kps$ :
\[ \boxed{\ Q=[\ce{Ca^2+}]\,[\ce{SO4^2-}]\ } \]
avec les concentrations \emph{effectives} du mélange (pas nécessairement à l'équilibre).
\end{correction}

\begin{correction}[lire le critère]
On compare $Q$ à $\Kps=\num{1{,}0e-10}$ :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $Q=\num{1{,}0e-13}<\Kps$ : \textbf{non saturée}, pas de précipité.
  \item $Q=\num{1{,}0e-10}=\Kps$ : \textbf{saturée} (équilibre exact).
  \item $Q=\num{1{,}0e-6}>\Kps$ : \textbf{sursaturée}, il y a \textbf{précipitation} jusqu'à ce que $Q$
        redescende à $\Kps$.
\end{enumerate}
\end{correction}

\begin{correction}[concentration maximale — \ce{AgCl}]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item Saturation : $Q=\Kps$, soit $[\ce{Ag+}][\ce{Cl-}]=\Kps$. Comme $[\ce{Ag+}]=[\ce{Cl-}]=c$ :
        $c^{2}=\Kps$.
  \item $c=\sqrt{\Kps}=\sqrt{\num{1{,}8e-10}}=\boxed{\qty{1{,}3e-5}{\molar}}$. Au-delà de cette valeur
        commune, $Q>\Kps$ et \ce{AgCl} précipite.
\end{enumerate}
\end{correction}

\begin{correction}[\ce{SnS} saturée ?]
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=2pt]
  \item $Q=[\ce{Sn^2+}][\ce{S^2-}]=(\num{1{,}0e-19})^{2}=\num{1{,}0e-38}$.
  \item $Q=\num{1{,}0e-38}\ll\Kps=\num{1{,}0e-26}$ : $Q<\Kps$, la solution est \textbf{non saturée}.
        \emph{(Réponse B du \textsc{qcm}\,n\textsuperscript{o}\,6.)} Elle pourrait dissoudre encore beaucoup
        de \ce{SnS} : à saturation on aurait $[\ce{Sn^2+}]=\sqrt{\Kps}=\qty{1{,}0e-13}{\molar}$, bien
        au-dessus de \num{1{,}0e-19}.
\end{enumerate}
\end{correction}

\begin{correction}[\ce{BaSO4} précipite-t-il ?]
$Q=[\ce{Ba^2+}][\ce{SO4^2-}]=(\num{1{,}0e-4})(\num{1{,}0e-4})=\num{1{,}0e-8}$.
On compare : $Q=\num{1{,}0e-8}>\Kps=\num{1{,}1e-10}$ (car $\num{1{,}0e-8}=100\times10^{-10}$). Donc
$Q>\Kps$ : la solution est sursaturée et \ce{BaSO4} \textbf{précipite}.
\end{correction}
