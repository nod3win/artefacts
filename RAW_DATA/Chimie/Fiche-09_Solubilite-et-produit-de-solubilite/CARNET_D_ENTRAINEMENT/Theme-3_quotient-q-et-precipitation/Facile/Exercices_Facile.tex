\bandeau{Thème 3 — Quotient $Q$ \& précipitation}%
{Exercices — Niveau Facile}%
{Écrire $Q$, comparer à $\Kps$, conclure sur l'état de la solution}

\noindent\textit{Consigne : une mécanique par question. Compare toujours dans le même sens : $Q$ face à
$\Kps$.}
\medskip

\begin{exercice}[écrire l'expression de $Q$]
Une solution contient des ions calcium et sulfate, susceptibles de former le sulfate de calcium
\ce{CaSO4}. Écris l'expression du quotient réactionnel $Q$ associé, en fonction des concentrations
effectives $[\ce{Ca^2+}]$ et $[\ce{SO4^2-}]$.
\end{exercice}

\begin{exercice}[lire le critère $Q$ vs $\Kps$]
Pour un même sel de $\Kps=\num{1{,}0e-10}$, indique dans chaque cas si la solution est \emph{non
saturée}, \emph{saturée} ou \emph{sursaturée}, et s'il y a précipitation :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item $Q=\num{1{,}0e-13}$ ;
  \item $Q=\num{1{,}0e-10}$ ;
  \item $Q=\num{1{,}0e-6}$.
\end{enumerate}
\end{exercice}

\begin{exercice}[concentration maximale avant précipitation]
On dispose d'une solution où $[\ce{Ag+}]=[\ce{Cl-}]$ (ils proviennent de la même source). On donne
$\Kps(\ce{AgCl})=\num{1{,}8e-10}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris la condition exacte de saturation ($Q=\Kps$).
  \item Quelle est la concentration commune maximale $[\ce{Ag+}]=[\ce{Cl-}]$ que la solution peut
        contenir sans qu'il y ait précipitation ?
\end{enumerate}
\end{exercice}

\begin{exercice}[la solution de \ce{SnS} est-elle saturée ?]
On considère une solution aqueuse de sulfure d'étain \ce{SnS} dont la concentration en ions vérifie
$[\ce{Sn^2+}]=[\ce{S^2-}]=\qty{1{,}0e-19}{\molar}$. On donne $\Kps(\ce{SnS})=\num{1{,}0e-26}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Calcule le quotient $Q=[\ce{Sn^2+}][\ce{S^2-}]$.
  \item Compare à $\Kps$ et conclus : la solution est-elle saturée, non saturée ou sursaturée ?
\end{enumerate}
\end{exercice}

\begin{exercice}[y a-t-il précipitation ?]
Dans un bécher, on mesure $[\ce{Ba^2+}]=\qty{1{,}0e-4}{\molar}$ et $[\ce{SO4^2-}]=\qty{1{,}0e-4}{\molar}$.
On donne $\Kps(\ce{BaSO4})=\num{1{,}1e-10}$. Le sulfate de baryum \ce{BaSO4} va-t-il précipiter ?
Justifie par le calcul de $Q$.
\end{exercice}
