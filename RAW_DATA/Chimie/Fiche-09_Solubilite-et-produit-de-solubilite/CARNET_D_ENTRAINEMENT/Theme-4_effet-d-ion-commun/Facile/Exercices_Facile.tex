\bandeau{Thème 4 — Effet d'ion commun}%
{Exercices — Niveau Facile}%
{Sens de l'effet, et application directe de $s'=\Kps/C^{\,n}$}

\noindent\textit{Consigne : une idée par question. Rappelle-toi que l'ion commun \emph{diminue} la
solubilité.}
\medskip

\begin{exercice}[sens de l'effet]
On a une solution saturée de \ce{AgCl} dans l'eau pure. On y ajoute du sel très soluble \ce{NaCl}.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Quel est l'ion \textbf{commun} apporté par \ce{NaCl} ?
  \item La solubilité de \ce{AgCl} va-t-elle augmenter ou diminuer ? Dans quel sens l'équilibre de
        dissolution se déplace-t-il (Le Chatelier) ?
\end{enumerate}
\end{exercice}

\begin{exercice}[repérer l'ion commun]
Pour chaque situation, dis si le sel ajouté apporte un \textbf{ion commun} au \ce{AgCl}
($\ce{AgCl <=> Ag+ + Cl-}$), et lequel :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item ajout de \ce{NaCl} ;
  \item ajout de \ce{KNO3} ;
  \item ajout de \ce{AgNO3}.
\end{enumerate}
\end{exercice}

\begin{exercice}[application directe — \ce{AgCl}]
On a une solution saturée en \ce{AgCl} ($\Kps=\num{1{,}8e-10}$) à laquelle on ajoute des ions chlorure
jusqu'à $[\ce{Cl-}]=\qty{0{,}10}{\molar}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris la nouvelle solubilité $s'=[\ce{Ag+}]$ en fonction de $\Kps$ et de $[\ce{Cl-}]$.
  \item Calcule sa valeur numérique.
\end{enumerate}
\end{exercice}

\begin{exercice}[comparer eau pure / ion commun]
Toujours pour \ce{AgCl} ($\Kps=\num{1{,}8e-10}$) :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Rappelle la solubilité $s_0$ dans l'eau pure.
  \item Sachant que $s'=\qty{1{,}8e-9}{\molar}$ en présence de $[\ce{Cl-}]=\qty{0{,}10}{\molar}$,
        par quel facteur environ la solubilité a-t-elle été divisée ?
\end{enumerate}
\end{exercice}

\begin{exercice}[la formule pour un sel $1{:}2$]
On dissout un sel $\mathrm{MX}_2$ (type $1{:}2$) dans une solution contenant déjà l'anion $\ce{X-}$ à la
concentration $C$ (grande).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris l'expression approchée de $\Kps$ en fonction de $s'$ et de $C$ (l'anion vient surtout de
        la solution).
  \item En déduis $s'$ en fonction de $\Kps$ et de $C$.
\end{enumerate}
\end{exercice}
