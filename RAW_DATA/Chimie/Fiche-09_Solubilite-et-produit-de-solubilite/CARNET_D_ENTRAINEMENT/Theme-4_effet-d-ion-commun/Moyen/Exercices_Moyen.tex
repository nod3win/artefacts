\bandeau{Thème 4 — Effet d'ion commun}%
{Exercices — Niveau Moyen}%
{Ion commun anionique ou cationique, sur différents types de sels}

\noindent\textit{Consigne : identifie d'abord l'ion commun et le type du sel, puis applique
$s'=\Kps/C^{\,n}$. Compare toujours à la solubilité $s_0$ dans l'eau pure.}
\medskip

\begin{exercice}[ion commun anionique sur un sel $1{:}2$]
On dissout du fluorure de calcium \ce{CaF2} ($\Kps=\num{3{,}9e-11}$) dans une solution de fluorure de
sodium \ce{NaF} telle que $[\ce{F-}]=\qty{0{,}10}{\molar}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Quel est l'ion commun ? Écris $s'$ en fonction de $\Kps$ et $[\ce{F-}]$.
  \item Calcule $s'$.
  \item Compare à $s_0=\qty{2{,}1e-4}{\molar}$ (eau pure) : par quel facteur la solubilité chute-t-elle ?
\end{enumerate}
\end{exercice}

\begin{exercice}[ion commun \emph{cationique} sur un sel $1{:}1$]
On dissout du sulfate de baryum \ce{BaSO4} ($\Kps=\num{1{,}1e-10}$) dans une solution de chlorure de
baryum \ce{BaCl2} telle que $[\ce{Ba^2+}]=\qty{1{,}0e-2}{\molar}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Ici l'ion commun est le \textbf{cation}. Écris $\Kps=[\ce{Ba^2+}][\ce{SO4^2-}]$ et exprime
        $s'=[\ce{SO4^2-}]$ en fonction de $\Kps$ et $[\ce{Ba^2+}]$.
  \item Calcule $s'$, puis compare à $s_0=\sqrt{\Kps}$.
\end{enumerate}
\end{exercice}

\begin{exercice}[ion commun \ce{OH-} sur un hydroxyde]
On dissout de l'hydroxyde de magnésium \ce{Mg(OH)2} ($\Kps=\num{9{,}0e-12}$) dans une solution basique
où $[\ce{OH-}]=\qty{1{,}0e-2}{\molar}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Écris $s'$ en fonction de $\Kps$ et $[\ce{OH-}]$ (type $1{:}2$).
  \item Calcule $s'$.
\end{enumerate}
\end{exercice}

\begin{exercice}[dépendance en $C$]
Pour \ce{AgCl} ($\Kps=\num{1{,}8e-10}$), calcule la solubilité $s'$ pour deux concentrations d'ion
commun \ce{Cl-} :
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item $[\ce{Cl-}]=\qty{2{,}0e-2}{\molar}$ ;
  \item $[\ce{Cl-}]=\qty{0{,}10}{\molar}$.
  \item Que remarques-tu sur $s'$ quand on multiplie $[\ce{Cl-}]$ par 5 ?
\end{enumerate}
\end{exercice}

\begin{exercice}[le $\Kps$ ne bouge pas : vérification]
Reprends le cas de l'exercice 1 (\ce{CaF2} avec $[\ce{F-}]=\qty{0{,}10}{\molar}$, $s'=\qty{3{,}9e-9}{\molar}$).
\begin{enumerate}[label=\alph*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item Donne $[\ce{Ca^2+}]$ et $[\ce{F-}]$ dans cette solution.
  \item Calcule le produit $[\ce{Ca^2+}][\ce{F-}]^{2}$ et vérifie qu'il redonne bien $\Kps$. Conclus sur
        ce qui change (la solubilité) et ce qui ne change pas (le $\Kps$).
\end{enumerate}
\end{exercice}
