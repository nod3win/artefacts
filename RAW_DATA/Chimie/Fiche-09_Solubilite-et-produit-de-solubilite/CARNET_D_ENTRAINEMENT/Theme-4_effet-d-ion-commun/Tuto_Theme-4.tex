\bandeau{Thème 4 — L'effet d'ion commun}%
{Tuto — quand un ion est déjà là}%
{Dissoudre un sel dans une solution contenant déjà l'un de ses ions}

\noindent Jusqu'ici on dissolvait les sels dans l'\textbf{eau pure}. Mais souvent, la solution
\emph{contient déjà} l'un des ions du sel (apporté par un autre sel très soluble). C'est l'\textbf{effet
d'ion commun} : la présence de cet ion \textbf{diminue la solubilité} du sel peu soluble.

\section*{1. L'idée : Le Chatelier}
\begin{theoreme}[effet d'ion commun]
Si l'on dissout un sel peu soluble dans une solution qui contient déjà l'un de ses ions, sa
\textbf{solubilité diminue}. En effet, l'équilibre de dissolution
\[ \ce{MX_{n}(s) <=> M^{n+}(aq) + n X-(aq)} \]
se déplace (Le Chatelier) dans le sens qui \emph{consomme} l'ion ajouté, c'est-à-dire vers la
\textbf{formation du solide}.
\end{theoreme}

\section*{2. La formule à appliquer}
\begin{formule}[solubilité en présence d'un ion commun]
Si l'on dissout $\mathrm{MX}_n$ dans une solution où l'anion $\mathrm{X}^-$ est déjà présent à la
concentration $C$ (grande, $C\gg s'$), alors, l'anion étant quasi entièrement apporté par la solution :
\[ \Kps=[\mathrm{M}^{n+}][\mathrm{X}^-]^{n}\approx s'\times C^{\,n}
   \quad\Longrightarrow\quad \boxed{\ s'=\dfrac{\Kps}{C^{\,n}}\ } \]
\end{formule}
\begin{ou}
  \item $s'$ : nouvelle solubilité du sel (en \si{\mole\per\litre}), plus petite que $s_0$ ;
  \item $C$ : concentration (imposée, grande) de l'ion commun ;
  \item $n$ : nombre d'ions $\mathrm{X}^-$ par unité de sel.
\end{ou}

\begin{exemple}[\ce{AgCl} : ajout de \ce{Cl-}]
$\Kps(\ce{AgCl})=\num{1{,}8e-10}$. Dans l'eau pure : $s_0=\sqrt{\Kps}\approx\qty{1{,}3e-5}{\molar}$.
On impose $[\ce{Cl-}]=\qty{0{,}10}{\molar}$ (ion commun) :
\[ s'=[\ce{Ag+}]=\dfrac{\Kps}{[\ce{Cl-}]}=\dfrac{\num{1{,}8e-10}}{0{,}10}=\qty{1{,}8e-9}{\molar}. \]
La solubilité a été divisée par $\sim 7000$ ! La quasi-totalité de l'argent est « chassée » vers le
solide \ce{AgCl}.
\end{exemple}

\begin{attention}[l'ion commun ne change PAS le $\Kps$]
$\Kps$ est une \textbf{constante} à $T$ fixée : il ne bouge pas. L'égalité
$\Kps=[\ce{Ag+}][\ce{Cl-}]$ reste vraie ; mais comme on \emph{impose} $[\ce{Cl-}]$ grand, c'est
$[\ce{Ag+}]$ (donc $s$) qui doit \textbf{chuter} pour conserver le produit.
\end{attention}
