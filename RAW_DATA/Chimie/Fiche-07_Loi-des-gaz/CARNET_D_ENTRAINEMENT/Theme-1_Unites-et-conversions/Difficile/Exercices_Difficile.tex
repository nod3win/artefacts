\begin{exo}[une pression donnée en pascals]
Un capteur mesure une pression $p=\qty{202650}{\pascal}$ dans un réacteur.
\begin{enumerate}[label=\alph*)]
  \item Exprime $p$ en \si{atm} (on prendra $\qty{1}{atm}=\qty{101325}{\pascal}$).
  \item Exprime $p$ en \si{bar}.
  \item Exprime $p$ en \si{mmHg}.
\end{enumerate}
\end{exo}

\begin{exo}[préparer $pV=nRT$ avec $R=\qty{0.082}{atm.L.mol^{-1}.K^{-1}}$]
On souhaite utiliser l'équation d'état avec $R=\num{0,082}$, qui impose $p$ en \si{atm}, $V$ en
\si{\litre} et $T$ en \si{\kelvin}. On dispose des données $p=\qty{760}{mmHg}$, $V=\qty{500}{\milli\litre}$
et $\theta=\qty{25}{\degreeCelsius}$. \emph{Sans calculer $n$}, convertis chaque grandeur dans la bonne
unité.
\begin{enumerate}[label=\alph*)]
  \item la pression $p$, en \si{atm} ;
  \item le volume $V$, en \si{\litre} ;
  \item la température $T$, en \si{\kelvin}.
\end{enumerate}
\end{exo}

\begin{exo}[préparer $pV=nRT$ avec $R=\qty{8.314}{\joule\per\mole\per\kelvin}$]
Cette fois on veut $R=\num{8,314}$, qui impose $p$ en \si{\pascal}, $V$ en \si{\meter\cubed} et $T$ en
\si{\kelvin}. Les données sont $p=\qty{2}{bar}$, $V=\qty{3}{\litre}$ et $\theta=\qty{127}{\degreeCelsius}$.
Convertis chaque grandeur.
\begin{enumerate}[label=\alph*)]
  \item la pression $p$, en \si{\pascal} ;
  \item le volume $V$, en \si{\meter\cubed} ;
  \item la température $T$, en \si{\kelvin}.
\end{enumerate}
\end{exo}

\begin{exo}[compléter la fiche de conversions de la pression standard]
Complète chaque égalité pour la pression standard $\qty{1}{atm}$.
\begin{enumerate}[label=\alph*)]
  \item $\qty{1}{atm}=\ \ldots\ \si{\pascal}$ ;
  \item $\qty{1}{atm}=\ \ldots\ \si{\kilo\pascal}$ ;
  \item $\qty{1}{atm}=\ \ldots\ \si{bar}$ ;
  \item $\qty{1}{atm}=\ \ldots\ \si{mmHg}$.
\end{enumerate}
\end{exo}
