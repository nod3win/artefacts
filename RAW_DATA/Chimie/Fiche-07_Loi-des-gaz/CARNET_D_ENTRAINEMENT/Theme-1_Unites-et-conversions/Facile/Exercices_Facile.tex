\begin{exo}[température : Celsius $\leftrightarrow$ Kelvin]
Convertis chaque température dans l'autre échelle ($T=\theta+273$).
\begin{enumerate}[label=\alph*)]
  \item $\theta=\qty{25}{\degreeCelsius}$ en kelvins.
  \item $\theta=\qty{127}{\degreeCelsius}$ en kelvins.
  \item $T=\qty{373}{\kelvin}$ en degrés Celsius.
  \item $\theta=\qty{-23}{\degreeCelsius}$ en kelvins.
\end{enumerate}
\end{exo}

\begin{exo}[pression : autour de l'atmosphère]
On rappelle $\qty{1}{atm}=\qty{760}{mmHg}=\qty{101325}{\pascal}$. Convertis.
\begin{enumerate}[label=\alph*)]
  \item $\qty{2}{atm}$ en \si{mmHg}.
  \item $\qty{0.5}{atm}$ en \si{mmHg}.
  \item $\qty{760}{mmHg}$ en \si{atm}.
  \item $\qty{1}{atm}$ en \si{\pascal}.
\end{enumerate}
\end{exo}

\begin{exo}[volume : litres, millilitres, mètres cubes]
On rappelle $\qty{1}{\litre}=\qty{1000}{\milli\litre}=\qty{e-3}{\meter\cubed}$. Convertis.
\begin{enumerate}[label=\alph*)]
  \item $\qty{5}{\litre}$ en \si{\milli\litre}.
  \item $\qty{2}{\litre}$ en \si{\meter\cubed}.
  \item $\qty{250}{\milli\litre}$ en \si{\litre}.
  \item $\qty{3}{\meter\cubed}$ en \si{\litre}.
\end{enumerate}
\end{exo}

\begin{exo}[pression : pascals, kilopascals, bars]
On rappelle $\qty{1}{bar}=\qty{e5}{\pascal}=\qty{100}{\kilo\pascal}$ et
$\qty{1}{atm}\approx\qty{101.3}{\kilo\pascal}$. Convertis.
\begin{enumerate}[label=\alph*)]
  \item $\qty{3}{bar}$ en \si{\pascal}.
  \item $\qty{150}{\kilo\pascal}$ en \si{\pascal}.
  \item $\qty{2}{bar}$ en \si{\kilo\pascal}.
  \item $\qty{1}{atm}$ en \si{\kilo\pascal}.
\end{enumerate}
\end{exo}
