\begin{corrige}[température : Celsius $\leftrightarrow$ Kelvin]
On ajoute $273$ pour aller vers les kelvins, on retranche $273$ pour revenir aux \si{\degreeCelsius}.
\begin{enumerate}[label=\alph*)]
  \item $T=25+273=\qty{298}{\kelvin}$.
  \item $T=127+273=\qty{400}{\kelvin}$.
  \item $\theta=373-273=\qty{100}{\degreeCelsius}$.
  \item $T=-23+273=\qty{250}{\kelvin}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[pression : autour de l'atmosphère]
On applique les équivalences $\qty{1}{atm}=\qty{760}{mmHg}=\qty{101325}{\pascal}$.
\begin{enumerate}[label=\alph*)]
  \item $p=2\times760=\qty{1520}{mmHg}$.
  \item $p=0{,}5\times760=\qty{380}{mmHg}$.
  \item $p=\dfrac{760}{760}=\qty{1}{atm}$.
  \item $p=\qty{101325}{\pascal}$ (par définition de l'atmosphère).
\end{enumerate}
\end{corrige}

\begin{corrige}[volume : litres, millilitres, mètres cubes]
Une plus petite unité donne un plus grand nombre, et inversement.
\begin{enumerate}[label=\alph*)]
  \item $V=5\times1000=\qty{5000}{\milli\litre}$.
  \item $V=2\times\num{e-3}=\qty{0.002}{\meter\cubed}$.
  \item $V=\dfrac{250}{1000}=\qty{0.25}{\litre}$.
  \item $V=3\times1000=\qty{3000}{\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[pression : pascals, kilopascals, bars]
On utilise $\qty{1}{bar}=\qty{e5}{\pascal}=\qty{100}{\kilo\pascal}$.
\begin{enumerate}[label=\alph*)]
  \item $p=3\times\num{e5}=\qty{3e5}{\pascal}$.
  \item $p=150\times1000=\qty{150000}{\pascal}=\qty{1.5e5}{\pascal}$.
  \item $p=2\times100=\qty{200}{\kilo\pascal}$.
  \item $p\approx\qty{101.3}{\kilo\pascal}$ (car $\qty{1}{atm}=\qty{101325}{\pascal}$).
\end{enumerate}
\end{corrige}
