\begin{corrige}[masse volumique aux CNTP]
On divise la masse molaire par $\qty{22.4}{\litre\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item $\rho=\dfrac{32}{22{,}4}=\qty{1.43}{\gram\per\litre}$.
  \item $\rho=\dfrac{44}{22{,}4}=\qty{1.96}{\gram\per\litre}$.
  \item $\rho=\dfrac{4}{22{,}4}=\qty{0.179}{\gram\per\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[masse molaire à partir de $\rho$ (CNTP)]
On multiplie par $\qty{22.4}{\litre\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item $M=1{,}96\times22{,}4\approx\qty{44}{\gram\per\mole}$ (\ce{CO2}).
  \item $M=1{,}43\times22{,}4\approx\qty{32}{\gram\per\mole}$ (\ce{O2}).
  \item $M=0{,}71\times22{,}4\approx\qty{16}{\gram\per\mole}$ (\ce{CH4}).
\end{enumerate}
\end{corrige}

\begin{corrige}[densité par rapport à l'air]
On divise la masse molaire par $\qty{28.8}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item $d=\dfrac{44}{28{,}8}=1{,}53$ (plus lourd que l'air).
  \item $d=\dfrac{4}{28{,}8}=0{,}14$ (plus léger, il monte).
  \item $d=\dfrac{28}{28{,}8}=0{,}97$ (quasi celle de l'air).
\end{enumerate}
\end{corrige}

\begin{corrige}[masse molaire à partir de la densité]
On multiplie par $\qty{28.8}{\gram\per\mole}$.
\begin{enumerate}[label=\alph*)]
  \item $M=28{,}8\times1{,}5=\qty{43.2}{\gram\per\mole}$.
  \item $M=28{,}8\times0{,}5=\qty{14.4}{\gram\per\mole}$.
  \item $M=28{,}8\times2{,}0=\qty{57.6}{\gram\per\mole}$.
\end{enumerate}
\end{corrige}
