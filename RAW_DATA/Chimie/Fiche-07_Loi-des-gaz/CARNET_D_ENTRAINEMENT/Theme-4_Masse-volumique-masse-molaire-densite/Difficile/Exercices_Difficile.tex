\begin{exo}[masse molaire par le volume molaire]
Un échantillon de gaz de masse $m=\qty{2.2}{\gram}$ occupe un volume $V=\qty{1.12}{\litre}$ à
\qty{0}{\degreeCelsius} sous \qty{1}{atm} (donc aux CNTP).
\begin{enumerate}[label=\alph*)]
  \item Calcule la quantité de matière $n$ (via $V_{\mathrm{m}}=\qty{22.4}{\litre\per\mole}$).
  \item Déduis la masse molaire $M=\dfrac{m}{n}$.
  \item Propose un gaz compatible.
\end{enumerate}
\end{exo}

\begin{exo}[identification par masse volumique hors CNTP]
La masse volumique d'un gaz inconnu vaut $\rho=\qty{1.30}{\gram\per\litre}$ sous $p=\qty{1.0}{atm}$ à
$T=\qty{300}{\kelvin}$ ($R=\num{0,082}$).
\begin{enumerate}[label=\alph*)]
  \item Écris la relation donnant $M$ à partir de $\rho$, $p$, $T$.
  \item Calcule $M$.
  \item De quel gaz courant s'agit-il vraisemblablement ?
\end{enumerate}
\end{exo}

\begin{exo}[un gaz plus léger que l'air]
Un gaz a une densité $d=0{,}55$ par rapport à l'air ($M_{\text{air}}=\qty{28.8}{\gram\per\mole}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule sa masse molaire $M$.
  \item Ce gaz est un hydrocarbure de formule \ce{CH4}, \ce{C2H6} ou \ce{C3H8} : lequel ?
        (données : $M(\ce{C})=12$, $M(\ce{H})=1$).
  \item Ce gaz monte-t-il ou descend-il dans l'air ? Justifie.
\end{enumerate}
\end{exo}

\begin{exo}[le \ce{CO2} coule-t-il dans l'air ?]
On compare le dioxyde de carbone \ce{CO2} ($M=\qty{44}{\gram\per\mole}$) et l'air
($M_{\text{air}}=\qty{28.8}{\gram\per\mole}$, $\rho_{\text{air}}=\qty{1.29}{\gram\per\litre}$ aux CNTP).
\begin{enumerate}[label=\alph*)]
  \item Calcule la masse volumique du \ce{CO2} aux CNTP.
  \item Calcule sa densité $d$ par rapport à l'air.
  \item Le \ce{CO2} s'accumule-t-il au plafond ou au sol d'une pièce ? Relie ta réponse à son emploi
        dans les extincteurs.
\end{enumerate}
\end{exo}
