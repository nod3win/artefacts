% THEME_TITLE: Masse volumique, masse molaire et densité d'un gaz

L'équation d'état ne contient pas la masse. Or, via $n=\dfrac{m}{M}$, on y fait entrer la masse $m$ et
la masse molaire $M$. On accède alors à la \textbf{masse volumique} $\rho$ et à la \textbf{densité}
d'un gaz — de quoi \emph{identifier} un gaz inconnu.

\begin{cle}[Masse volumique d'un gaz parfait]
Par définition $\rho=\dfrac{m}{V}$ (en \si{\gram\per\litre}). En injectant $n=\dfrac{m}{M}$ dans
$pV=nRT$ :
\[ \boxed{\,\rho=\frac{pM}{RT}\,} \]
À $p$ et $T$ fixés, $\rho$ est \textbf{proportionnelle à $M$} : un gaz « lourd » est plus dense.
\end{cle}

\begin{cle}[Masse molaire d'un gaz]
En isolant $M$ (à partir de $pV=\dfrac{m}{M}RT$) :
\[ \boxed{\,M=\frac{mRT}{pV}\,}
   \qquad\text{et, aux CNTP,}\qquad
   \boxed{\,M=\rho\times V_{\mathrm{m}}=\rho\times\qty{22.4}{\litre\per\mole}\,}. \]
C'est une méthode de \textbf{détermination expérimentale} de $M$ : on pèse un gaz de volume connu.
\end{cle}

\begin{cle}[Air et densité par rapport à l'air]
L'air $\approx 80\,\%$ \ce{N2} $+\ 20\,\%$ \ce{O2} (en volume) : $M_{\text{air}}=\qty{28.8}{\gram\per\mole}$,
$\rho_{\text{air}}\approx\qty{1.29}{\gram\per\litre}$ aux CNTP. La \textbf{densité} $d$ (sans unité) d'un
gaz par rapport à l'air :
\[ d=\frac{\rho_{\text{gaz}}}{\rho_{\text{air}}}=\frac{M_{\text{gaz}}}{M_{\text{air}}}
   =\frac{M_{\text{gaz}}}{\qty{28.8}{\gram\per\mole}}
   \qquad\Longleftrightarrow\qquad
   \boxed{\,M_{\text{gaz}}=\qty{28.8}{\gram\per\mole}\times d\,}. \]
$d<1$ : le gaz \emph{monte} (plus léger que l'air) ; $d>1$ : il \emph{descend}.
\end{cle}

\begin{methode}[Identifier un gaz inconnu]
\begin{enumerate}[leftmargin=1.6em,topsep=2pt,itemsep=1pt]
  \item Mesurer $\rho$ (aux CNTP) ou la densité $d$.
  \item Calculer $M$ : soit $M=\rho\times22{,}4$, soit $M=28{,}8\times d$, soit $M=\dfrac{mRT}{pV}$.
  \item Comparer $M$ aux masses molaires connues pour proposer le gaz.
\end{enumerate}
\end{methode}

\begin{exemplebox}[du $\rho$ à l'identité du gaz]
Aux CNTP, $\rho=\qty{1.25}{\gram\per\litre}$ : $M=1{,}25\times22{,}4=\qty{28}{\gram\per\mole}$. Cela
correspond à \ce{N2} ($28$) ou \ce{CO}\,($12+16=28$). \\[2pt]
\ce{CO2} ($M=44$) : $d=\dfrac{44}{28{,}8}=1{,}53>1$ $\Rightarrow$ plus lourd que l'air, il s'accumule au
sol (extincteurs).
\end{exemplebox}

\begin{piege}[deux conditions à ne pas oublier]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item $M=\rho\times22{,}4$ et $M=28{,}8\,d$ ne valent qu'\textbf{aux CNTP}. Sinon, utiliser
        $M=\dfrac{mRT}{pV}$.
  \item $\rho\propto M$ seulement \textbf{à $p$ et $T$ fixés} ($\rho=\dfrac{pM}{RT}$) : changer $T$ ou
        $p$ change $\rho$ sans changer $M$.
\end{itemize}
\end{piege}
