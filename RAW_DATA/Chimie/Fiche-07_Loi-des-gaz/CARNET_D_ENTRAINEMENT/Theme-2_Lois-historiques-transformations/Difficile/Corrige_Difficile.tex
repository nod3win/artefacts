\begin{corrige}[la loi combinée $pV/T=\text{cste}$]
Toutes les grandeurs sont déjà en unités SI cohérentes.
\begin{enumerate}[label=\alph*)]
  \item À $n$ constant : $\dfrac{p_1 V_1}{T_1}=\dfrac{p_2 V_2}{T_2}$.
  \item $p_2 = p_1\times\dfrac{V_1}{V_2}\times\dfrac{T_2}{T_1}$.
  \item $p_2 = \num{2.0e5}\times\dfrac{0{,}040}{0{,}010}\times\dfrac{400}{200}
        = \num{2.0e5}\times4\times2 = \qty{1.6e6}{\pascal}$.
\end{enumerate}
La compression ($\times4$) et le chauffage ($\times2$) augmentent tous deux la pression : facteur $8$.
\end{corrige}

\begin{corrige}[transformation en deux étapes]
On applique la bonne loi à chaque étape, puis on vérifie globalement.
\begin{enumerate}[label=\alph*)]
  \item Étape isotherme (Boyle) : $V_1=\dfrac{p_0 V_0}{p_1}=\dfrac{1{,}0\times6{,}0}{3{,}0}=\qty{2.0}{\litre}$.
  \item Étape isobare (Charles) : $V_2=V_1\dfrac{T_2}{T_0}=2{,}0\times\dfrac{450}{300}=\qty{3.0}{\litre}$.
  \item Loi combinée depuis l'état initial :
        $V_2=V_0\times\dfrac{p_0}{p_1}\times\dfrac{T_2}{T_0}
        =6{,}0\times\dfrac{1{,}0}{3{,}0}\times\dfrac{450}{300}=\qty{3.0}{\litre}$. Même résultat :
        les deux étapes équivalent à la transformation globale.
\end{enumerate}
\end{corrige}

\begin{corrige}[dilatation isobare d'un gaz chauffé]
$p$ constante $\Rightarrow$ Charles : $V\propto T$ (en kelvins).
\begin{enumerate}[label=\alph*)]
  \item $T_1=127+273=\qty{400}{\kelvin}$, $T_2=527+273=\qty{800}{\kelvin}$.
  \item Facteur $=\dfrac{T_2}{T_1}=\dfrac{800}{400}=2$ : le volume est \textbf{doublé}.
  \item $V_2=V_1\times2=5{,}0\times2=\qty{10}{\litre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[sécurité d'une bombe aérosol (isochore)]
$V$ constant $\Rightarrow$ $\dfrac{p_1}{T_1}=\dfrac{p_2}{T_2}$, températures en kelvins.
\begin{enumerate}[label=\alph*)]
  \item $T_1=300$ K, $T_2=400$ K : $p_2=p_1\dfrac{T_2}{T_1}=3{,}0\times\dfrac{400}{300}=\qty{4.0}{bar}$.
  \item $T=T_1\dfrac{p}{p_1}=300\times\dfrac{6{,}0}{3{,}0}=\qty{600}{\kelvin}$, soit
        $\theta=600-273=\qty{327}{\degreeCelsius}$.
  \item Au feu, la température grimpe fortement ; à volume constant, la pression grimpe proportionnellement
        et peut faire \textbf{éclater} la bombe.
\end{enumerate}
\end{corrige}
