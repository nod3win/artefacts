\begin{center}
{\Large\bfseries\color{primarydark} Thème 2 — Niveau Difficile}\\[2pt]
{\color{primary}\itshape Enchaîner $pV=nRT$, la masse et une transformation}
\end{center}
\textcolor{primary}{\hrule height 1pt}
\medskip

\begin{enonce}[Données]
$R=\qty{0.082}{\atm\litre\per\mole\per\kelvin}$ ; \ $n=\dfrac{m}{M}$ ; \
$M(\ce{N2})=\qty{28}{\gram\per\mole}$. \ Rappel Thème 1 : à $V$ et $n$ constants,
$\dfrac{p_1}{T_1}=\dfrac{p_2}{T_2}$.
\end{enonce}

\begin{exo}[Une bouteille d'azote comprimé au laboratoire]
Une bouteille en acier \textbf{rigide} de volume $V=\qty{20}{\litre}$ contient du diazote \ce{N2} sous une
pression $p_1=\qty{10}{\atm}$ à $\theta_1=\qty{25}{\degreeCelsius}$.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=2pt]
  \item Quelle quantité de matière $n$ de diazote la bouteille contient-elle ?
  \item Quelle masse $m$ de diazote cela représente-t-il ?
  \item On stocke la bouteille près d'un radiateur qui la porte à $\theta_2=\qty{100}{\degreeCelsius}$.
        Le volume et la quantité de gaz ne changent pas. Calcule la nouvelle pression $p_2$, de deux
        façons : \emph{(i)} directement avec $pV=nRT$ ; \emph{(ii)} avec la loi isochore du Thème 1.
        Les deux résultats coïncident-ils ?
  \item En une phrase, explique pourquoi il est déconseillé de stocker une bouteille de gaz comprimé
        près d'une source de chaleur.
\end{enumerate}
\end{exo}

\begin{exo}[Un pneu qui fuit lentement]
Un pneu de volume supposé constant $V=\qty{3.0}{\litre}$ est gonflé à $p_1=\qty{2.5}{\atm}$ à la
température $T=\qty{300}{\kelvin}$. Une petite crevaison le fait fuir lentement ; après une nuit, à la
\emph{même} température, la pression n'est plus que $p_2=\qty{2.0}{\atm}$.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=2pt]
  \item Calcule la quantité de matière $n_1$ d'air initialement contenue, puis $n_2$ après la fuite.
  \item Combien de moles d'air se sont échappées ?
  \item Quelle \textbf{fraction} de l'air initial le pneu a-t-il perdue ?
  \item Ici, quelle est la variable de $pV=nRT$ qui a réellement changé pour expliquer la baisse de
        pression ? (Attention : ce n'est pas une simple transformation à $n$ constant.)
\end{enumerate}
\end{exo}

\begin{exo}[Identifier un gaz inconnu par l'équation d'état]
On isole $\qty{4.4}{\gram}$ d'un gaz inconnu. Placé dans un récipient de volume $V=\qty{2.46}{\litre}$ à
$\theta=\qty{27}{\degreeCelsius}$, il y exerce une pression $p=\qty{1.0}{\atm}$.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=2pt]
  \item À l'aide de $pV=nRT$, détermine la quantité de matière $n$ du gaz.
  \item En déduis sa masse molaire $M$.
  \item Propose une identité possible pour ce gaz. \emph{(On rappelle
        $M(\ce{CO2})=\qty{44}{\gram\per\mole}$, $M(\ce{SO2})=\qty{64}{\gram\per\mole}$,
        $M(\ce{O2})=\qty{32}{\gram\per\mole}$, et le propane $\ce{C3H8}$ a aussi
        $M=\qty{44}{\gram\per\mole}$.)} La seule masse molaire suffit-elle à conclure avec certitude ?
\end{enumerate}
\end{exo}
