\begin{exo}[du nombre de moles aux pressions partielles]
Un récipient de $V=\qty{5.0}{\litre}$ à $T=\qty{300}{\kelvin}$ contient $n(\ce{N2})=\qty{0.40}{\mole}$ et
$n(\ce{O2})=\qty{0.10}{\mole}$ ($R=\num{0,082}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule la quantité de matière totale $n_{\text{tot}}$.
  \item Calcule la pression totale $p_{\text{tot}}$.
  \item Calcule les fractions molaires $\chi(\ce{N2})$ et $\chi(\ce{O2})$.
  \item Calcule les pressions partielles $p(\ce{N2})$ et $p(\ce{O2})$, et vérifie leur somme.
\end{enumerate}
\end{exo}

\begin{exo}[des pressions partielles à la composition]
Un mélange contient du dihydrogène et du diazote : $p(\ce{H2})=\qty{0.60}{atm}$ et
$p(\ce{N2})=\qty{1.80}{atm}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la pression totale $p_{\text{tot}}$.
  \item Calcule la fraction molaire $\chi(\ce{H2})$.
  \item Calcule la fraction molaire $\chi(\ce{N2})$.
  \item Quel est le rapport $n(\ce{H2})/n(\ce{N2})$ ?
\end{enumerate}
\end{exo}

\begin{exo}[dioxygène de l'air en altitude]
L'air contient $20\,\%$ de \ce{O2} en volume. On note $p_{\text{atm}}$ la pression atmosphérique.
\begin{enumerate}[label=\alph*)]
  \item Au niveau de la mer, $p_{\text{atm}}=\qty{1.0}{atm}$ : calcule $p(\ce{O2})$.
  \item En altitude, $p_{\text{atm}}=\qty{0.50}{atm}$ : calcule $p(\ce{O2})$ (même composition).
  \item En une phrase, explique pourquoi il est plus difficile de respirer en altitude.
\end{enumerate}
\end{exo}

\begin{exo}[mélange préparé à partir de masses]
On enferme $\qty{2.8}{\gram}$ de \ce{N2} ($M=28$) et $\qty{3.2}{\gram}$ de \ce{O2} ($M=32$) dans un
récipient où la pression totale s'établit à $p_{\text{tot}}=\qty{2.0}{atm}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule $n(\ce{N2})$ et $n(\ce{O2})$.
  \item Calcule les fractions molaires.
  \item Calcule les pressions partielles $p(\ce{N2})$ et $p(\ce{O2})$.
\end{enumerate}
\end{exo}
