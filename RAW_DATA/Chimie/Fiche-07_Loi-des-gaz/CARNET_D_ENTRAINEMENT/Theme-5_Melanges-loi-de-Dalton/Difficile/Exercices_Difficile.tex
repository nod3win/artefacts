\begin{exo}[récipient scellé : pressions et volumes]
Un récipient rigide de $V=\qty{12.2}{\litre}$, maintenu à $T=\qty{298}{\kelvin}$, contient
$n(\ce{N2})=\qty{0.40}{\mole}$ et $n(\ce{O2})=\qty{0.10}{\mole}$ ($R=\num{0,082}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule les fractions molaires $\chi(\ce{N2})$ et $\chi(\ce{O2})$.
  \item Calcule la pression totale $p_{\text{tot}}$.
  \item Calcule les pressions partielles $p(\ce{N2})$ et $p(\ce{O2})$.
  \item Les deux gaz occupent-ils le même volume dans ce récipient ? Compare aussi leurs pressions
        partielles.
\end{enumerate}
\end{exo}

\begin{exo}[mélange de trois gaz préparé par pesée]
Dans $V=\qty{10.0}{\litre}$ à $T=\qty{300}{\kelvin}$, on introduit $\qty{1.4}{\gram}$ de \ce{N2}
($M=28$), $\qty{1.6}{\gram}$ de \ce{O2} ($M=32$) et $\qty{4.4}{\gram}$ de \ce{CO2} ($M=44$)
($R=\num{0,082}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule les quantités de matière de chaque gaz et $n_{\text{tot}}$.
  \item Calcule la pression totale $p_{\text{tot}}$.
  \item Calcule les fractions molaires des trois gaz.
  \item Déduis les trois pressions partielles et vérifie leur somme.
\end{enumerate}
\end{exo}

\begin{exo}[dihydrogène recueilli sur l'eau]
On recueille du dihydrogène \ce{H2} par déplacement d'eau : le gaz recueilli est en réalité un mélange
de \ce{H2} et de vapeur d'eau. La pression totale mesurée est $p_{\text{tot}}=\qty{760}{mmHg}$ ; à
$\qty{25}{\degreeCelsius}$ ($T=\qty{298}{\kelvin}$), la vapeur d'eau exerce $p(\ce{H2O})=\qty{24}{mmHg}$
($R=\num{0,082}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule la pression partielle du dihydrogène $p(\ce{H2})$ en \si{mmHg}.
  \item Exprime $p(\ce{H2})$ en \si{atm}.
  \item Le volume recueilli est $V=\qty{0.50}{\litre}$. Calcule la quantité de matière de \ce{H2}.
\end{enumerate}
\end{exo}

\begin{exo}[composition de l'air et pressions partielles]
On modélise l'air par $78\,\%$ de \ce{N2}, $21\,\%$ de \ce{O2} et $1\,\%$ d'argon \ce{Ar} \emph{en
volume}, sous $p_{\text{tot}}=\qty{1.0}{atm}$.
\begin{enumerate}[label=\alph*)]
  \item Pour des gaz parfaits, justifie brièvement pourquoi la fraction \emph{volumique} égale la
        fraction \emph{molaire}.
  \item Donne les fractions molaires $\chi(\ce{N2})$, $\chi(\ce{O2})$ et $\chi(\ce{Ar})$.
  \item Calcule les trois pressions partielles et vérifie leur somme.
\end{enumerate}
\end{exo}
