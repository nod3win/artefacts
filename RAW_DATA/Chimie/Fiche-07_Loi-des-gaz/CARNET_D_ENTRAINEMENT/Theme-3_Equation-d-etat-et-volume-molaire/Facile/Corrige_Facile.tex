\begin{corrige}[isoler la bonne grandeur]
On divise ou multiplie les deux membres de $pV=nRT$.
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{pV}{RT}$ ;
  \item $p=\dfrac{nRT}{V}$ ;
  \item $V=\dfrac{nRT}{p}$ ;
  \item $T=\dfrac{pV}{nR}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer la quantité de matière]
$n=\dfrac{pV}{RT}$ avec $RT=0{,}082\times273\approx22{,}4$.
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{1\times44{,}8}{22{,}4}=\qty{2.0}{\mole}$.
  \item $n=\dfrac{2\times11{,}2}{22{,}4}=\qty{1.0}{\mole}$.
  \item $n=\dfrac{0{,}5\times22{,}4}{22{,}4}=\qty{0.5}{\mole}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer $p$, $V$ ou $T$]
On isole puis on remplace ($RT=0{,}082\times273\approx22{,}4$).
\begin{enumerate}[label=\alph*)]
  \item $p=\dfrac{nRT}{V}=\dfrac{1{,}0\times22{,}4}{22{,}4}=\qty{1.0}{atm}$.
  \item $V=\dfrac{nRT}{p}=\dfrac{2{,}0\times22{,}4}{1}=\qty{44.8}{\litre}$.
  \item $T=\dfrac{pV}{nR}=\dfrac{1\times11{,}2}{0{,}5\times0{,}082}=\dfrac{11{,}2}{0{,}041}=\qty{273}{\kelvin}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[volume molaire aux CNTP]
On divise par $\qty{22.4}{\litre\per\mole}$ pour obtenir $n$, on multiplie pour obtenir $V$.
\begin{enumerate}[label=\alph*)]
  \item $n=\dfrac{5{,}6}{22{,}4}=\qty{0.25}{\mole}$.
  \item $V=3{,}0\times22{,}4=\qty{67.2}{\litre}$.
  \item $n=\dfrac{44{,}8}{22{,}4}=\qty{2.0}{\mole}$.
  \item $V=0{,}10\times22{,}4=\qty{2.24}{\litre}$.
\end{enumerate}
\end{corrige}
