\begin{enonce}[Boîte à outils]
Isochore ($V$ cste) : $\dfrac{p_1}{T_1}=\dfrac{p_2}{T_2}$ \quad;\quad
Loi combinée : $\dfrac{p_1V_1}{T_1}=\dfrac{p_2V_2}{T_2}$ ($n$ cste, $T$ en kelvins). \\
Avant de choisir : repère \emph{ce qui est maintenu constant}.
\end{enonce}

\begin{exo}[Une bombe aérosol que l'on approche du feu]
Une bombe aérosol \textbf{rigide} (son volume ne change pas) contient un gaz sous $p_1=\qty{2.0}{\atm}$
à la température ambiante $\theta_1=\qty{27}{\degreeCelsius}$. Un utilisateur imprudent la pose près
d'une source de chaleur qui la porte à $\theta_2=\qty{127}{\degreeCelsius}$.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=1pt]
  \item Quelle loi s'applique ? Justifie en identifiant la variable qui reste constante.
  \item Calcule la nouvelle pression $p_2$ à l'intérieur de la bombe.
\end{enumerate}
\end{exo}

\begin{exo}[Une compression avec changement d'unité]
Une seringue contient $V_1=\qty{500}{\milli\litre}$ d'air sous $p_1=\qty{100}{\kilo\pascal}$, à
température constante. On enfonce le piston jusqu'à $V_2=\qty{250}{\milli\litre}$. Détermine la
nouvelle pression $p_2$, d'abord en \si{\kilo\pascal}, puis exprime-la en atmosphères
(on prendra $\qty{1}{\atm}=\qty{101}{\kilo\pascal}$).
\end{exo}

\begin{exo}[Le piège des degrés Celsius]
Un ballon souple contient $V_1=\qty{2.0}{\litre}$ d'air à $\theta_1=\qty{27}{\degreeCelsius}$, sous
pression atmosphérique constante. On le place dans une étuve à $\theta_2=\qty{87}{\degreeCelsius}$.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=1pt]
  \item Calcule correctement le volume final $V_2$.
  \item Un étudiant écrit $V_2 = 2{,}0\times\dfrac{87}{27}=\qty{6.4}{\litre}$. Explique pourquoi ce
        résultat est absurde et où est l'erreur.
\end{enumerate}
\end{exo}

\begin{exo}[Loi combinée : pression \emph{et} température changent]
Un gaz occupe $V_1=\qty{10.0}{\litre}$ sous $p_1=\qty{1.0}{\atm}$ à $T_1=\qty{300}{\kelvin}$. On le
comprime jusqu'à $V_2=\qty{5.0}{\litre}$ tout en le chauffant jusqu'à $T_2=\qty{450}{\kelvin}$
(la quantité de gaz ne change pas). Quelle est la pression finale $p_2$ ?
\end{exo}

\begin{exo}[Raisonner en facteurs, sans nombre imposé]
Une quantité fixée de gaz subit une transformation au cours de laquelle sa \textbf{température absolue
est multipliée par 3} et sa \textbf{pression est doublée}.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=1pt]
  \item En partant de la loi combinée, exprime le rapport $V_2/V_1$ en fonction des facteurs donnés.
  \item De quel facteur le volume est-il multiplié ? Le gaz s'est-il dilaté ou contracté ?
\end{enumerate}
\end{exo}
