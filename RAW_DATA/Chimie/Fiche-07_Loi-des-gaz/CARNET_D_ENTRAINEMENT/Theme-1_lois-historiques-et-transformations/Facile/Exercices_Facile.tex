\begin{enonce}[Rappels utiles]
$T(\si{\kelvin})=\theta(\si{\degreeCelsius})+273$ \quad;\quad
$\qty{1}{\atm}=\qty{101325}{\pascal}=\qty{760}{\mmHg}=\qty{1.013}{\bar}$ \quad;\quad
$\qty{1}{\bar}=\qty{e5}{\pascal}$. \\
Lois : Boyle $p_1V_1=p_2V_2$ ($T$ cste) ; Charles $\dfrac{V_1}{T_1}=\dfrac{V_2}{T_2}$ ($p$ cste) ;
Avogadro $\dfrac{V_1}{n_1}=\dfrac{V_2}{n_2}$ ($p,T$ cstes).
\end{enonce}

\begin{exo}[Passer des degrés Celsius aux kelvins]
Convertis en kelvins les températures suivantes, indispensables avant tout calcul de gaz :
\[ \text{(a) } \qty{25}{\degreeCelsius}\qquad \text{(b) } \qty{100}{\degreeCelsius}\qquad
   \text{(c) } \qty{-40}{\degreeCelsius}\qquad \text{(d) } \qty{727}{\degreeCelsius}. \]
Convertis ensuite en degrés Celsius : \ (e) $\qty{350}{\kelvin}$, \ (f) $\qty{0}{\kelvin}$
(que représente cette dernière valeur ?).
\end{exo}

\begin{exo}[Jongler avec les unités de pression]
Effectue les conversions suivantes (garde $\qty{1}{\atm}=\qty{101325}{\pascal}=\qty{760}{\mmHg}$) :
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=1pt]
  \item exprime $\qty{2}{\atm}$ en pascals, puis en \si{\mmHg} ;
  \item exprime $\qty{380}{\mmHg}$ en atmosphères ;
  \item exprime $\qty{202650}{\pascal}$ en atmosphères ;
  \item exprime $\qty{1.5}{\bar}$ en kilopascals.
\end{enumerate}
\end{exo}

\begin{exo}[Boyle-Mariotte : une compression à température constante]
Un cylindre contient $\qty{6.0}{\litre}$ d'air sous $p_1=\qty{1.0}{\atm}$. On enfonce le piston,
\emph{sans changer la température}, jusqu'à ce que la pression atteigne $p_2=\qty{3.0}{\atm}$.
Quel est le nouveau volume $V_2$ ? De quel facteur le volume a-t-il été divisé ?
\end{exo}

\begin{exo}[Charles : un ballon que l'on chauffe à pression constante]
Un ballon souple renferme $V_1=\qty{3.0}{\litre}$ de gaz à $T_1=\qty{300}{\kelvin}$, sous la pression
atmosphérique. On le chauffe jusqu'à $T_2=\qty{450}{\kelvin}$, la pression restant celle de
l'atmosphère. Quel volume $V_2$ occupe-t-il alors ?
\end{exo}

\begin{exo}[Avogadro : on ajoute du gaz à $p$ et $T$ constants]
À une certaine température et pression, $n_1=\qty{0.20}{\mole}$ de gaz occupe $V_1=\qty{4.48}{\litre}$.
Toujours dans les mêmes conditions de $p$ et $T$, on injecte du gaz jusqu'à atteindre
$n_2=\qty{0.50}{\mole}$. Quel volume $V_2$ le gaz occupe-t-il ?
\end{exo}
