\begin{exo}[vrai ou faux sur la nature de l'image]
Réponds par \textbf{vrai} ou \textbf{faux} et justifie brièvement.
\begin{enumerate}[label=\alph*)]
  \item Un objet placé entre une lentille convergente et son foyer objet donne une image \emph{droite
        et virtuelle}.
  \item Une lentille convergente ne peut \emph{jamais} donner d'image réelle plus grande que l'objet.
  \item Une lentille convergente peut donner une image à la fois \emph{virtuelle et renversée}.
  \item Pour une lentille divergente, un objet réel donne toujours une image \emph{virtuelle, droite
        et réduite}.
  \item Une image \emph{réelle} peut être droite.
\end{enumerate}
\end{exo}

\begin{exo}[l'objet glisse de $2F$ vers $F$]
On déplace un objet réel du point $2F$ jusqu'au foyer objet $F$ d'une lentille convergente. Parmi les
affirmations suivantes, indique celles qui sont \textbf{correctes}.
\begin{enumerate}[label=\alph*)]
  \item L'image se déplace de $2F'$ jusqu'à l'infini.
  \item L'image reste réelle et renversée.
  \item L'image est de plus en plus petite.
  \item L'image se forme du même côté de la lentille que l'objet.
\end{enumerate}
\end{exo}

\begin{exo}[trois situations, trois instruments]
Une lentille convergente de focale $f$ est utilisée de trois façons. Pour chaque cas, donne la
\textbf{nature complète} de l'image (réelle/virtuelle, sens, taille) et nomme l'\textbf{instrument}
correspondant.
\begin{enumerate}[label=\alph*)]
  \item objet au-delà de $2F$ (objet lointain) ;
  \item objet entre $F$ et $2F$ ;
  \item objet entre $O$ et $F$.
\end{enumerate}
\end{exo}
