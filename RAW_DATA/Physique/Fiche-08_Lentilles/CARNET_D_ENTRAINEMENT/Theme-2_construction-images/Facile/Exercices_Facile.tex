\begin{exo}[à chaque rayon sa règle]
Une lentille convergente reçoit trois rayons incidents. Indique, pour chacun, comment il ressort.
\begin{enumerate}[label=\alph*)]
  \item un rayon qui passe par le centre optique $O$ ;
  \item un rayon parallèle à l'axe optique ;
  \item un rayon qui passe par le foyer objet $F$.
\end{enumerate}
\end{exo}

\begin{exo}[quel rayon émergent est correct ?]
Un rayon incident \textbf{parallèle à l'axe} frappe une lentille convergente. Parmi les quatre tracés
proposés après la lentille, lequel est correct ?
\begin{center}
\begin{tikzpicture}[scale=0.85,>=Stealth,font=\footnotesize]
  \draw[black!35,dash pattern=on 3pt off 2pt] (-4.4,0)--(4.2,0);
  \draw[line width=1.5pt,<->,black!65] (0,-2)--(0,2);
  \fill[black] (0,0) circle (1.2pt); \node[below right] at (0,0) {$O$};
  \fill[primary] (2,0) circle (1.5pt); \node[below] at (2,0) {$F'$};
  \fill[primary] (-2,0) circle (1.5pt); \node[below] at (-2,0) {$F$};
  % incident parallele
  \draw[black!80,line width=1.0pt,->] (-4.2,1)--(0,1);
  \node[black!80,above left] at (-2.6,1) {incident};
  % candidates
  \draw[defcol,line width=0.8pt,->] (0,1)--(3.6,1) node[right]{1};
  \draw[defcol,line width=0.8pt,->] (0,1)--(3.6,-0.8) node[right]{2};
  \draw[defcol,line width=0.8pt,->] (0,1)--(2.4,2.0) node[right]{3};
  \draw[defcol,line width=0.8pt,->] (0,1)--(2.0,-2.0) node[right]{4};
\end{tikzpicture}
\end{center}
\end{exo}

\begin{exo}[compléter deux tracés]
Pour une lentille convergente, indique le trajet du rayon émergent.
\begin{enumerate}[label=\alph*)]
  \item un rayon incident dirigé droit vers le centre optique $O$ ;
  \item un rayon incident qui passe par le foyer objet $F$ avant d'atteindre la lentille.
\end{enumerate}
\end{exo}

\begin{exo}[nature d'une image réelle]
Un objet réel est placé \textbf{au-delà de $2F$} devant une lentille convergente (comme un objet
lointain devant un appareil photo). Sans calcul, précise si l'image est :
\begin{enumerate}[label=\alph*)]
  \item réelle ou virtuelle ?
  \item droite ou renversée ?
  \item agrandie ou réduite ?
\end{enumerate}
\end{exo}

\begin{exo}[l'image d'une lentille divergente]
Un objet réel est placé devant une lentille \textbf{divergente}.
\begin{enumerate}[label=\alph*)]
  \item L'image est-elle réelle ou virtuelle ?
  \item Est-elle droite ou renversée ?
  \item Est-elle plus grande ou plus petite que l'objet ?
\end{enumerate}
\end{exo}
