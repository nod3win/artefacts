\begin{corrige}[à chaque rayon sa règle]
\begin{enumerate}[label=\alph*)]
  \item il ressort \textbf{sans être dévié} (le rayon par $O$ garde sa direction).
  \item il ressort en passant par le \textbf{foyer image $F'$}.
  \item il ressort \textbf{parallèle à l'axe optique}.
\end{enumerate}
\end{corrige}

\begin{corrige}[quel rayon émergent est correct ?]
Un rayon parallèle à l'axe émerge \textbf{en passant par le foyer image $F'$}. Le seul tracé qui passe
par $F'$ est le \textbf{rayon 2}. (1 = non dévié : faux ; 3 et 4 ne passent pas par $F'$.)
\end{corrige}

\begin{corrige}[compléter deux tracés]
\begin{enumerate}[label=\alph*)]
  \item il traverse la lentille \textbf{sans déviation} et continue en ligne droite.
  \item il ressort \textbf{parallèle à l'axe optique} (propriété du foyer objet $F$).
\end{enumerate}
\end{corrige}

\begin{corrige}[nature d'une image réelle]
Objet au-delà de $2F$ : c'est le cas de l'appareil photo.
\begin{enumerate}[label=\alph*)]
  \item \textbf{réelle} (les rayons se croisent de l'autre côté, on peut la recueillir sur un capteur).
  \item \textbf{renversée} (une image réelle l'est toujours).
  \item \textbf{réduite} (plus petite que l'objet).
\end{enumerate}
\end{corrige}

\begin{corrige}[l'image d'une lentille divergente]
Une lentille divergente donne \emph{toujours}, pour un objet réel, le même type d'image.
\begin{enumerate}[label=\alph*)]
  \item \textbf{virtuelle} (les rayons émergents divergent ; seuls leurs prolongements se croisent).
  \item \textbf{droite} (même sens que l'objet).
  \item \textbf{plus petite} que l'objet (réduite).
\end{enumerate}
\end{corrige}
