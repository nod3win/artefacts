\begin{corrige}[repos puis accommodation]
\begin{enumerate}[label=\alph*)]
  \item Repos : $f=\qty{16}{\mm}=\qty{0.016}{\m}$, $C_0=\dfrac{1}{0,016}=62{,}5~\delta$.
  \item Objet à \qty{25}{\cm} ($u=\qty{0.25}{\m}$), image sur la rétine ($v=\qty{0.016}{\m}$) :
        \[ C=\frac{1}{u}+\frac{1}{v}=\frac{1}{0,25}+\frac{1}{0,016}=4+62{,}5=66{,}5~\delta. \]
        Effort d'accommodation : $\Delta C=66{,}5-62{,}5=+4~\delta$.
\end{enumerate}
\end{corrige}

\begin{corrige}[le verre d'un myope]
\begin{enumerate}[label=\alph*)]
  \item L'\oe il voit net jusqu'à \qty{90}{\cm} : le verre doit ramener l'image de l'objet lointain
        \emph{à cette distance}, en image \textbf{virtuelle} $\Rightarrow v=-\qty{0.90}{\m}$.
  \item Objet à l'infini ($1/u\to 0$) :
        $\dfrac{1}{f}=0+\dfrac{1}{-0,90}=-1{,}11\Rightarrow f=-\qty{0.90}{\m}$,
        $C=-1{,}11~\delta$ : verre \textbf{divergent}.
\end{enumerate}
\end{corrige}

\begin{corrige}[le verre d'un hypermétrope]
\begin{enumerate}[label=\alph*)]
  \item $u=+\qty{0.25}{\m}$ (objet réel) et $v=-\qty{1.0}{\m}$ (image virtuelle au point net) :
        \[ \frac{1}{f}=\frac{1}{0,25}+\frac{1}{-1,0}=4-1=3\ \Rightarrow\ f=+\qty{0.33}{\m}. \]
  \item $C=+3~\delta$ : verre \textbf{convergent}, cohérent avec la correction de l'hypermétropie.
\end{enumerate}
\end{corrige}

\begin{corrige}[reconnaître le défaut sur un schéma]
\begin{itemize}[label=--,leftmargin=1.4em,topsep=2pt,itemsep=2pt]
  \item \oe il (1) : l'image se forme \textbf{en avant} de la rétine $\Rightarrow$ \textbf{myopie}
        (\oe il trop long) $\Rightarrow$ verre \textbf{divergent}.
  \item \oe il (2) : l'image se formerait \textbf{en arrière} de la rétine $\Rightarrow$
        \textbf{hypermétropie} (\oe il trop court) $\Rightarrow$ verre \textbf{convergent}.
\end{itemize}
\end{corrige}

\begin{corrige}[de combien accommoder ?]
\begin{enumerate}[label=\alph*)]
  \item Objet à l'infini : $C_0=\dfrac{1}{0,016}=62{,}5~\delta$. Objet à \qty{20}{\cm} :
        $C=\dfrac{1}{0,20}+\dfrac{1}{0,016}=5+62{,}5=67{,}5~\delta$.
  \item $\Delta C=67{,}5-62{,}5=+5~\delta$ d'accommodation.
\end{enumerate}
\end{corrige}
