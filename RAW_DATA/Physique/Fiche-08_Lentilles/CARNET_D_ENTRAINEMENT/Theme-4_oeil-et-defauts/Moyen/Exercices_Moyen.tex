\begin{exo}[repos puis accommodation]
Un \oe il normal a sa rétine à \qty{16}{\mm} du centre optique.
\begin{enumerate}[label=\alph*)]
  \item Calcule sa vergence \textbf{au repos} (objet à l'infini).
  \item L'\oe il fixe maintenant un objet à \qty{25}{\cm}. Calcule la nouvelle vergence nécessaire,
        puis l'effort d'accommodation (variation de vergence).
\end{enumerate}
\end{exo}

\begin{exo}[le verre d'un myope]
Un myope ne voit pas nettement au-delà de \qty{90}{\cm} (son point le plus lointain vu net). On veut
qu'il voie un objet situé \textbf{à l'infini}.
\begin{enumerate}[label=\alph*)]
  \item Où le verre doit-il former l'image de cet objet lointain ? En déduire $v$ (avec son signe).
  \item Calcule la distance focale $f$ et la vergence $C$ du verre. Convergent ou divergent ?
\end{enumerate}
\end{exo}

\begin{exo}[le verre d'un hypermétrope]
Un hypermétrope ne voit net qu'au-delà de \qty{100}{\cm}. Il veut lire un texte placé à \qty{25}{\cm}.
Le verre doit donner de ce texte une image virtuelle à \qty{100}{\cm} (là où l'\oe il voit net).
\begin{enumerate}[label=\alph*)]
  \item Écris $u$ et $v$ (avec leurs signes), puis calcule $f$.
  \item Donne la vergence $C$ et confirme la nature (convergent/divergent) du verre.
\end{enumerate}
\end{exo}

\begin{exo}[reconnaître le défaut sur un schéma]
Pour chacun des deux yeux ci-dessous, un objet est à l'infini. Indique où se forme l'image, le
\textbf{défaut} correspondant et le \textbf{verre} qui le corrige.
\begin{center}
\begin{tikzpicture}[scale=0.9,>=Stealth,font=\scriptsize]
  % oeil 1 : myope, image en avant
  \begin{scope}
    \node[black!70] at (0.2,1.9) {\oe il (1)};
    \draw[black!35,dash pattern=on 2pt off 2pt] (-2.6,0)--(2.2,0);
    \draw[line width=1.4pt,<->,black!65] (0,-1.2)--(0,1.2);
    \fill[attncol!35] (2.0,-1.1) rectangle (2.18,1.1);
    \draw[attncol!80,line width=0.7pt] (2.0,-1.1)--(2.0,1.1);
    \node[attncol!70!black,below] at (2.0,-1.15) {rétine};
    \foreach \y in {0.7,-0.7}{ \draw[black!75,line width=0.7pt] (-2.4,\y)--(0,\y);
       \draw[defcol,line width=0.7pt] (0,\y)--(1.2,0);}
    \draw[defcol,line width=0.7pt] (1.2,0)--(2.0,0.53);
    \draw[defcol,line width=0.7pt] (1.2,0)--(2.0,-0.53);
    \fill[black] (1.2,0) circle (1.3pt); \node[below] at (1.2,-0.06) {image};
  \end{scope}
  % oeil 2 : hypermetrope, image en arriere
  \begin{scope}[shift={(6.4,0)}]
    \node[black!70] at (0.2,1.9) {\oe il (2)};
    \draw[black!35,dash pattern=on 2pt off 2pt] (-2.6,0)--(3.0,0);
    \draw[line width=1.4pt,<->,black!65] (0,-1.2)--(0,1.2);
    \fill[attncol!35] (1.9,-1.1) rectangle (2.08,1.1);
    \draw[attncol!80,line width=0.7pt] (1.9,-1.1)--(1.9,1.1);
    \node[attncol!70!black,below] at (1.9,-1.15) {rétine};
    \foreach \y in {0.7,-0.7}{ \draw[black!75,line width=0.7pt] (-2.4,\y)--(0,\y);

    \fill[black] (2.7,0) circle (1.3pt); \node[below] at (2.7,-0.06) {image};
  \end{scope}
\end{tikzpicture}
\end{center}
\end{exo}

\begin{exo}[de combien accommoder ?]
Un \oe il a sa rétine à \qty{16}{\mm}. On lui présente un objet d'abord à l'infini, puis à
\qty{20}{\cm}.
\begin{enumerate}[label=\alph*)]
  \item Calcule la vergence dans les deux cas.
  \item De combien de dioptries l'\oe il a-t-il dû accommoder ?
\end{enumerate}
\end{exo}
