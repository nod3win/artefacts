\begin{exo}[trouver la position de l'image]
Une lentille convergente a une distance focale $f=+\qty{10}{\cm}$. Un objet réel est placé à
$u=+\qty{15}{\cm}$. En travaillant en centimètres, calcule la distance image $v$.
\end{exo}

\begin{exo}[grandissement et taille de l'image]
Un objet de hauteur $o=\qty{3}{\cm}$ est à $u=+\qty{20}{\cm}$ d'une lentille ; son image se forme à
$v=+\qty{60}{\cm}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le grandissement $\gamma$.
  \item Déduis-en la hauteur $i$ de l'image (avec son signe).
\end{enumerate}
\end{exo}

\begin{exo}[remonter à la distance focale]
Un objet réel est à $u=+\qty{30}{\cm}$ d'une lentille ; son image réelle se forme à $v=+\qty{60}{\cm}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la distance focale $f$.
  \item La lentille est-elle convergente ou divergente ?
\end{enumerate}
\end{exo}

\begin{exo}[lire la nature dans les signes]
Pour chaque résultat de calcul, donne la nature de l'image (réelle/virtuelle, droite/renversée,
agrandie/réduite/même taille).
\begin{enumerate}[label=\alph*)]
  \item $v=+\qty{40}{\cm}$, $\gamma=-0{,}5$ ;
  \item $v=-\qty{20}{\cm}$, $\gamma=+2$ ;
  \item $v=+\qty{15}{\cm}$, $\gamma=-1$ ;
  \item $v=-\qty{30}{\cm}$, $\gamma=+0{,}5$.
\end{enumerate}
\end{exo}

\begin{exo}[un cas complet]
Convergente $f=+\qty{20}{\cm}$, objet réel de hauteur $o=\qty{2}{\cm}$ placé à $u=+\qty{60}{\cm}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule $v$ puis $\gamma$.
  \item Donne la hauteur $i$ et la nature complète de l'image.
\end{enumerate}
\end{exo}
