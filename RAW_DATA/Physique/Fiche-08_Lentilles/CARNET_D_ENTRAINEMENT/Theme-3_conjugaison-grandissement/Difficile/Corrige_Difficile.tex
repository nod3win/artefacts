\begin{corrige}[à quelle distance placer l'objet ?]
\begin{enumerate}[label=\alph*)]
  \item $\gamma=-\dfrac{v}{u}=+\dfrac13\Rightarrow v=-\dfrac{u}{3}$. On reporte :
        \[ \frac{1}{f}=\frac{1}{u}+\frac{1}{v}=\frac{1}{u}-\frac{3}{u}=-\frac{2}{u}
           \ \Rightarrow\ u=-2f=-2\times(-20)=+\qty{40}{\cm}. \]
        L'objet est à \qty{40}{\cm} devant la lentille (l'image virtuelle est alors à
        $v=-\qty{13.3}{\cm}$).
  \item $\gamma=-\dfrac{v}{u}=-4\Rightarrow v=4u$. Alors
        $\dfrac{1}{f}=\dfrac{1}{u}+\dfrac{1}{4u}=\dfrac{5}{4u}\Rightarrow u=\dfrac{5f}{4}=+\qty{12.5}{\cm}$,
        et $v=4u=+\qty{50}{\cm}$. On a bien $f<u<2f$ (régime projecteur).
\end{enumerate}
\end{corrige}

\begin{corrige}[le problème du facteur $x$]
\begin{enumerate}[label=\alph*)]
  \item $\dfrac{1}{f}=\dfrac{1}{u}+\dfrac{1}{v}=\dfrac{1}{xf}+\dfrac{1}{1{,}2\,f}$. On multiplie par
        $f$ : $1=\dfrac{1}{x}+\dfrac{1}{1{,}2}$. Donc $\dfrac{1}{x}=1-0{,}8333=0{,}1667$ et
        \[ x=6. \]
  \item $u=6f>2f$ : l'objet est bien au-delà de $2f$. $\gamma=-\dfrac{v}{u}=-\dfrac{1{,}2f}{6f}=-0{,}2$ :
        image \textbf{réduite} (et renversée), cohérent avec un objet lointain.
\end{enumerate}
\end{corrige}

\begin{corrige}[attention aux unités]
\begin{enumerate}[label=\alph*)]
  \item On convertit : $f=\qty{0.05}{\m}=\qty{5}{\cm}$ et $d=\qty{10}{\cm}$ (même unité !).
        $\dfrac{1}{d'}=\dfrac{1}{5}-\dfrac{1}{10}=\dfrac{2-1}{10}=\dfrac{1}{10}\Rightarrow d'=+\qty{10}{\cm}$.
        Comme $d=10=2f$, on retrouve $d'=2f$ : image \textbf{réelle, renversée, de même taille}.
  \item Objet à l'infini : $v=f=+\qty{3}{\cm}$. Image \textbf{réelle}, ponctuelle, au foyer image $F'$,
        à \qty{3}{\cm} derrière la lentille.
\end{enumerate}
\end{corrige}

\begin{corrige}[calculer pour vérifier une affirmation]
\begin{enumerate}[label=\alph*)]
  \item $\dfrac{1}{v}=\dfrac{1}{5}-\dfrac{1}{15}=\dfrac{3-1}{15}=\dfrac{2}{15}\Rightarrow v=+\qty{7.5}{\cm}$,
        $\gamma=-\dfrac{7{,}5}{15}=-0{,}5$. Réelle, renversée, à \qty{7.5}{\cm} : \textbf{vrai}
        (l'image est en outre réduite).
  \item $\dfrac{1}{v}=\dfrac{1}{5}-\dfrac{1}{3}=\dfrac{3-5}{15}=-\dfrac{2}{15}\Rightarrow v=-\qty{7.5}{\cm}$ :
        $v<0$, image \textbf{virtuelle} (et droite, agrandie). L'affirmation « réelle » est
        \textbf{fausse} : c'est le régime loupe ($u<f$).
  \item $\dfrac{1}{v}=\dfrac{1}{5}-\dfrac{1}{10}=\dfrac{1}{10}\Rightarrow v=+\qty{10}{\cm}=2f$,
        $\gamma=-1$ : même taille, renversée, à \qty{10}{\cm} : \textbf{vrai}.
\end{enumerate}
Affirmations correctes : \textbf{a)} et \textbf{c)}.
\end{corrige}
