\sisetup{per-mode=symbol}

Une lentille convergente devient un \textbf{instrument} selon la position de l'objet. Le même jeu de
formules explique la loupe, l'appareil photo, le projecteur et le microscope.

\begin{cle}[instruments à une lentille convergente]
\begin{center}\small
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{>{\bfseries}l l l}
\toprule
Instrument & Position de l'objet & Image \\
\midrule
\rowcolor{excol!8} Loupe & $0<u<f$ & virtuelle, droite, \textbf{agrandie} \\
Appareil photo & $u>2f$ (objet lointain) & réelle, renversée, \textbf{réduite} (sur le capteur) \\
\rowcolor{defcol!8} Projecteur & $f<u<2f$ & réelle, renversée, \textbf{agrandie} (sur l'écran) \\
\bottomrule
\end{tabular}
\end{center}
Photo et projecteur donnent une image \textbf{réelle} (sur un écran/capteur) ; la loupe une image
\textbf{virtuelle} (regardée à l'\oe il).
\end{cle}

\begin{cle}[le microscope : deux convergentes en série]
\begin{itemize}[leftmargin=1.5em,topsep=2pt,itemsep=1pt]
  \item L'\textbf{objectif} (courte focale $f_1$) : l'objet est \emph{juste au-delà} de $F_1$
        ($f_1<u_1<2f_1$) ; il en donne une image intermédiaire \textbf{réelle, renversée, très
        agrandie}.
  \item L'\textbf{oculaire} (focale $f_2$) agit comme une \textbf{loupe} sur cette image
        intermédiaire : image finale \textbf{virtuelle, agrandie}, observée à l'\oe il.
  \item \textbf{Réglage à l'infini} : l'image intermédiaire est placée dans le \emph{plan focal objet
        de l'oculaire} ($F_2$) $\Rightarrow$ les rayons sortent parallèles, l'\oe il n'accommode pas.
\end{itemize}
Le grossissement total vaut $G\approx |\gamma_1|\times G_{\text{oculaire}}$, avec
$G_{\text{oculaire}}\propto 1/f_2$ (oculaire court $=$ fort grossissement).
\end{cle}

\begin{methode}[reconnaître l'instrument]
Regarde \textbf{où est l'objet} par rapport à $f$ et $2f$, puis lis la \emph{nature} de l'image dans
le tableau : image réelle réduite $\to$ photo ; réelle agrandie $\to$ projecteur ; virtuelle agrandie
$\to$ loupe. Deux lentilles en série $\to$ microscope.
\end{methode}

\begin{piege}[confusions fréquentes]
\begin{itemize}[leftmargin=1.5em,topsep=2pt,itemsep=1pt]
  \item Appareil photo ($u>2f$, image \emph{réduite}) et projecteur ($f<u<2f$, image \emph{agrandie})
        utilisent tous deux un objectif convergent : c'est la \emph{position de l'objet} qui les
        distingue.
  \item L'\textbf{oculaire} d'un microscope est \textbf{convergent} (une loupe), jamais divergent.
  \item L'image intermédiaire de l'objectif est \textbf{renversée} (pas droite).
\end{itemize}
\end{piege}
