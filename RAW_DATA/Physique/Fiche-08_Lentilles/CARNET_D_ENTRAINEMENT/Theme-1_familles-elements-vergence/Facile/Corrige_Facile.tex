\begin{corrige}[reconnaître la famille]
Critère unique : \textbf{bords minces $\Rightarrow$ convergente} ; \textbf{bords épais $\Rightarrow$
divergente}.
\begin{itemize}[label=--,leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item (a) biconvexe : plus épaisse au centre, bords \emph{minces} $\Rightarrow$
        \textcolor{excol}{\textbf{convergente}}.
  \item (b) biconcave : plus mince au centre, bords \emph{épais} $\Rightarrow$
        \textcolor{attncol}{\textbf{divergente}}.
  \item (c) plan-convexe : une face plate, une face bombée, plus épaisse au centre $\Rightarrow$
        bords \emph{minces} $\Rightarrow$ \textcolor{excol}{\textbf{convergente}}.
\end{itemize}
\end{corrige}

\begin{corrige}[symbole et signe de $f$]
Les flèches du symbole donnent le sens du comportement.
\begin{enumerate}[label=\alph*)]
  \item Flèches vers l'\emph{extérieur} : lentille \textbf{convergente}, donc $f>0$.
  \item Flèches vers l'\emph{intérieur} : lentille \textbf{divergente}, donc $f<0$.
\end{enumerate}
\end{corrige}

\begin{corrige}[de la distance focale à la vergence]
On applique $C=\dfrac{1}{f}$ avec $f$ \textbf{en mètres}.
\begin{enumerate}[label=\alph*)]
  \item $f=+\qty{50}{\cm}=+\qty{0.50}{\m}$ : $C=\dfrac{1}{0,50}=+2~\delta$.
  \item $f=+\qty{10}{\cm}=+\qty{0.10}{\m}$ : $C=\dfrac{1}{0,10}=+10~\delta$.
\end{enumerate}
La seconde est \emph{plus convergente} (vergence plus grande, focale plus courte).
\end{corrige}

\begin{corrige}[de la vergence à la distance focale]
On inverse : $f=\dfrac{1}{C}$ (résultat en mètres).
\begin{enumerate}[label=\alph*)]
  \item $f=\dfrac{1}{+4}=+\qty{0.25}{\m}=+\qty{25}{\cm}$ : $C>0\Rightarrow$ \textbf{convergente}.
  \item $f=\dfrac{1}{-5}=-\qty{0.20}{\m}=-\qty{20}{\cm}$ : $C<0\Rightarrow$ \textbf{divergente}.
\end{enumerate}
\end{corrige}

\begin{corrige}[deux lentilles accolées]
Les vergences des lentilles accolées s'\textbf{ajoutent}.
\begin{enumerate}[label=\alph*)]
  \item $C_{\text{éq}}=C_1+C_2=+3+2=+5~\delta$.
  \item $C_{\text{éq}}>0$ : le système est \textbf{convergent}, de distance focale
        $f_{\text{éq}}=\dfrac{1}{C_{\text{éq}}}=\dfrac{1}{5}=+\qty{0.20}{\m}=+\qty{20}{\cm}$.
\end{enumerate}
\end{corrige}
