\begin{exo}[vergences et pièges d'unités]
Calcule la vergence de chaque lentille. Sois attentif aux unités et aux signes.
\begin{enumerate}[label=\alph*)]
  \item lentille convergente $f=+\qty{5}{\cm}$ ;
  \item lentille divergente $f=-\qty{40}{\cm}$ ;
  \item lentille convergente $f=+\qty{2}{\mm}$.
\end{enumerate}
\end{exo}

\begin{exo}[une convergente et une divergente accolées]
On accole une lentille convergente $C_1=+6~\delta$ et une lentille divergente $C_2=-2~\delta$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la vergence équivalente $C_{\text{éq}}$.
  \item Le système global est-il convergent ou divergent ? Donne sa distance focale $f_{\text{éq}}$
        (en cm).
\end{enumerate}
\end{exo}

\begin{exo}[fabriquer un verre neutre (afocal)]
On dispose d'une lentille convergente de vergence $C_1=+3~\delta$. On veut, en lui \emph{accolant} une
seconde lentille, obtenir un système \textbf{afocal} (vergence totale nulle, équivalent à une vitre
plane).
\begin{enumerate}[label=\alph*)]
  \item Quelle vergence $C_2$ doit avoir la seconde lentille ?
  \item Quelle est sa nature et sa distance focale $f_2$ ?
\end{enumerate}
\end{exo}

\begin{exo}[classer par puissance]
Quatre lentilles ont pour distances focales :
\[ L_1:\ f=+\qty{20}{\cm}\quad L_2:\ f=+\qty{5}{\cm}\quad L_3:\ f=-\qty{10}{\cm}\quad L_4:\ f=+\qty{1}{\m}. \]
\begin{enumerate}[label=\alph*)]
  \item Donne la vergence de chacune.
  \item Classe les lentilles \textbf{convergentes} de la plus puissante à la moins puissante.
\end{enumerate}
\end{exo}

\begin{exo}[aller-retour $f \leftrightarrow C$]
\begin{enumerate}[label=\alph*)]
  \item Une lentille a une vergence $C=-2{,}5~\delta$. Quelle est sa distance focale (en cm) et sa
        famille ?
  \item Un objectif « puissant » a une vergence $C=+20~\delta$. Exprime sa distance focale en cm
        \emph{puis} en mm.
\end{enumerate}
\end{exo}
