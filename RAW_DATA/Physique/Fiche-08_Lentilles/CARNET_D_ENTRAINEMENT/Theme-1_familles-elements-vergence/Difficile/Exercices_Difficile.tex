\begin{exo}[vrai ou faux : repérer les affirmations correctes]
Pour chaque affirmation, réponds par \textbf{vrai} ou \textbf{faux} et \emph{justifie en une ligne}.
\begin{enumerate}[label=\alph*)]
  \item La vergence d'une lentille est l'opposé de sa distance focale.
  \item Une loupe est une lentille divergente.
  \item Une lentille à bords épais a une distance focale négative.
  \item Une lentille de vergence $+8~\delta$ a une distance focale de \qty{8}{\cm}.
  \item En accolant deux lentilles, la vergence du système est la somme des vergences.
\end{enumerate}
\end{exo}

\begin{exo}[un jeu de trois lentilles accolées]
Un opticien accole trois lentilles minces de vergences
\[ C_1=+5~\delta,\qquad C_2=-8~\delta,\qquad C_3=+1~\delta. \]
\begin{enumerate}[label=\alph*)]
  \item Calcule la vergence équivalente $C_{\text{éq}}$ du système.
  \item Le système est-il convergent ou divergent ? Donne sa distance focale $f_{\text{éq}}$ (en cm).
  \item Par quelle \emph{unique} lentille pourrait-on remplacer l'ensemble sans rien changer ?
\end{enumerate}
\end{exo}

\begin{exo}[remonter à une lentille inconnue]
On accole une lentille convergente $L_1$ de distance focale $f_1=+\qty{20}{\cm}$ et une lentille
inconnue $L_2$. On mesure que le système équivalent a une vergence $C_{\text{éq}}=+2~\delta$.
\begin{enumerate}[label=\alph*)]
  \item Détermine la vergence $C_1$ de $L_1$.
  \item Déduis-en la vergence $C_2$, la nature et la distance focale $f_2$ de $L_2$.
  \item Par quelle lentille $L_2'$ faudrait-il remplacer $L_2$ pour rendre le système
        \textbf{afocal} ($C_{\text{éq}}=0$) ?
\end{enumerate}
\end{exo}

\begin{exo}[qui plie le plus la lumière ?]
On compare quatre lentilles :
\[ L_1:\ C=+2~\delta\quad L_2:\ f=+\qty{5}{\cm}\quad L_3:\ f=-\qty{25}{\cm}\quad L_4:\ C=-4~\delta. \]
\begin{enumerate}[label=\alph*)]
  \item Exprime toutes ces lentilles par leur \textbf{vergence}, puis sépare les convergentes des
        divergentes.
  \item Range les quatre lentilles par \textbf{pouvoir de déviation décroissant} (c'est-à-dire par
        $|C|$ décroissant) et indique laquelle « plie » le plus fortement la lumière.
\end{enumerate}
\end{exo}
