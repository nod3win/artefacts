\begin{corrige}[vrai ou faux : repérer les affirmations correctes]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Faux} : la vergence est l'\emph{inverse} de $f$ ($C=1/f$), pas son opposé.
  \item \textbf{Faux} : une loupe est \emph{convergente} (bords minces, $f>0$).
  \item \textbf{Vrai} : bords épais $\Rightarrow$ divergente $\Rightarrow f<0$.
  \item \textbf{Faux} : $f=\dfrac{1}{8}=\qty{0.125}{\m}=\qty{12.5}{\cm}$, pas \qty{8}{\cm}.
  \item \textbf{Vrai} : pour des lentilles accolées, $C_{\text{éq}}=\sum_k C_k$.
\end{enumerate}
Bilan : les affirmations correctes sont \textbf{c)} et \textbf{e)}.
\end{corrige}

\begin{corrige}[un jeu de trois lentilles accolées]
\begin{enumerate}[label=\alph*)]
  \item $C_{\text{éq}}=C_1+C_2+C_3=+5-8+1=-2~\delta$.
  \item $C_{\text{éq}}<0$ : système \textbf{divergent},
        $f_{\text{éq}}=\dfrac{1}{-2}=-\qty{0.5}{\m}=-\qty{50}{\cm}$.
  \item Par une \emph{unique} lentille \textbf{divergente} de vergence $-2~\delta$ (focale
        $-\qty{50}{\cm}$) : elle est optiquement équivalente au jeu des trois.
\end{enumerate}
\end{corrige}

\begin{corrige}[remonter à une lentille inconnue]
\begin{enumerate}[label=\alph*)]
  \item $C_1=\dfrac{1}{f_1}=\dfrac{1}{0,20}=+5~\delta$.
  \item $C_2=C_{\text{éq}}-C_1=+2-5=-3~\delta$ : $L_2$ est \textbf{divergente},
        $f_2=\dfrac{1}{-3}\approx-\qty{0.33}{\m}=-\qty{33}{\cm}$.
  \item Afocal : $C_1+C_2'=0\Rightarrow C_2'=-C_1=-5~\delta$, soit une divergente de focale
        $-\qty{20}{\cm}$ (elle annule exactement $L_1$).
\end{enumerate}
\end{corrige}

\begin{corrige}[qui plie le plus la lumière ?]
\begin{enumerate}[label=\alph*)]
  \item $L_1:+2~\delta$ ; $L_2: \dfrac{1}{0,05}=+20~\delta$ ; $L_3:\dfrac{1}{-0,25}=-4~\delta$ ;
        $L_4:-4~\delta$.\\
        Convergentes : $L_1,\,L_2$ ($C>0$). Divergentes : $L_3,\,L_4$ ($C<0$).
  \item Par $|C|$ décroissant :
        \[ L_2\,(20)\ >\ L_3\,(4)\ =\ L_4\,(4)\ >\ L_1\,(2). \]
        $L_2$ « plie » le plus fortement la lumière (plus grande vergence en valeur absolue, focale la
        plus courte). $L_3$ et $L_4$ dévient autant l'une que l'autre, mais en \emph{écartant} les
        rayons (divergentes).
\end{enumerate}
\end{corrige}
