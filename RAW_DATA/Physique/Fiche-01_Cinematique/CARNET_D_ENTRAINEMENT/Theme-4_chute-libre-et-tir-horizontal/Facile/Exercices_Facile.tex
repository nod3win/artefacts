On prend $g=\qty{10}{\metre\per\second\squared}$ et on néglige les frottements de l'air.

\begin{exo}[chute d'un objet lâché]
On lâche un objet, sans vitesse initiale, d'une hauteur $h=\qty{20}{\metre}$.
\begin{enumerate}[label=\alph*)]
  \item Combien de temps met-il pour toucher le sol ?
  \item Avec quelle vitesse arrive-t-il au sol ?
\end{enumerate}
\end{exo}

\begin{exo}[vitesse et hauteur pendant la chute]
Un caillou tombe en chute libre depuis le repos.
\begin{enumerate}[label=\alph*)]
  \item Quelle est sa vitesse à $t=\qty{3}{\second}$ ?
  \item De quelle hauteur est-il tombé à cet instant ?
\end{enumerate}
\end{exo}

\begin{exo}[le temps de chute d'un tir horizontal]
Une bille est lancée \emph{horizontalement} depuis une table située à \qty{45}{\metre} de hauteur.
\begin{enumerate}[label=\alph*)]
  \item Combien de temps met-elle à atteindre le sol ?
  \item Ce temps dépend-il de la vitesse horizontale de lancement ? Justifie.
\end{enumerate}
\end{exo}

\begin{exo}[portée d'un tir horizontal]
Un objet est lancé horizontalement à $v_{0x}=\qty{15}{\metre\per\second}$ ; son vol dure \qty{2}{\second}.
\begin{enumerate}[label=\alph*)]
  \item Quelle est la portée horizontale (distance parcourue au sol) ?
  \item Quelle est sa vitesse verticale au moment de l'impact ?
\end{enumerate}
\end{exo}

\begin{exo}[deux billes lâchées et lancées]
D'une même hauteur, on \emph{lâche} une bille A et on \emph{lance horizontalement} une bille B, au même instant.
\begin{enumerate}[label=\alph*)]
  \item Laquelle touche le sol la première ?
  \item Laquelle parcourt la plus grande distance horizontale ?
\end{enumerate}
\end{exo}
