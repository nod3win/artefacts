\begin{exo}[conversions d'unités]
Passe d'une unité à l'autre (facteur $3{,}6$).
\begin{enumerate}[label=\alph*)]
  \item Convertis \qty{72}{\kilo\metre\per\hour} en \si{\metre\per\second}.
  \item Convertis \qty{30}{\metre\per\second} en \si{\kilo\metre\per\hour}.
\end{enumerate}
\end{exo}

\begin{exo}[vitesse moyenne et signe]
Un mobile se déplace sur un axe $Ox$.
\begin{enumerate}[label=\alph*)]
  \item Il passe de $x_1=\qty{20}{\metre}$ à $x_2=\qty{170}{\metre}$ en \qty{10}{\second}. Calcule $v_{\text{moy}}$.
  \item Il repart ensuite de $x_2=\qty{170}{\metre}$ et revient à $x_3=\qty{70}{\metre}$ en \qty{5}{\second}. Calcule $v_{\text{moy}}$ sur ce second trajet et commente son signe.
\end{enumerate}
\end{exo}

\begin{exo}[accélération moyenne]
Détermine l'accélération moyenne dans chaque cas.
\begin{enumerate}[label=\alph*)]
  \item Une voiture passe de \qty{10}{\metre\per\second} à \qty{25}{\metre\per\second} en \qty{5}{\second}.
  \item Elle freine ensuite de \qty{30}{\metre\per\second} à l'arrêt en \qty{6}{\second}.
\end{enumerate}
\end{exo}

\begin{exo}[distance parcourue et déplacement]
Un coureur part de $O$ ($x=0$), atteint $x=\qty{40}{\metre}$ en \qty{8}{\second}, puis revient
jusqu'à $x=\qty{10}{\metre}$ en \qty{6}{\second} de plus.
\begin{enumerate}[label=\alph*)]
  \item Quel est son \emph{déplacement} total $\Delta x$ ?
  \item Quelle \emph{distance} totale a-t-il parcourue ?
  \item Quelle est sa vitesse moyenne sur l'ensemble ?
\end{enumerate}
\end{exo}

\begin{exo}[dériver une équation horaire]
La position d'un mobile est $x(t)=2t^2+5t+1$ (en \si{\metre}, $t$ en \si{\second}).
\begin{enumerate}[label=\alph*)]
  \item Donne l'expression de la vitesse $v(t)$.
  \item Donne l'expression de l'accélération $a(t)$.
  \item Calcule la vitesse à $t=\qty{3}{\second}$.
\end{enumerate}
\end{exo}
