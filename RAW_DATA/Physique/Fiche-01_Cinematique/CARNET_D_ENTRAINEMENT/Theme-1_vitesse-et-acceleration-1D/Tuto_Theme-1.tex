% THEME_TITLE: Vitesse et accélération à une dimension
\sisetup{per-mode=symbol}

\noindent La cinématique décrit le mouvement \emph{en fonction du temps}. Sur un axe, tout se
ramène à deux \og taux de variation\fg{} : la \textbf{vitesse} (comment change la position) et
l'\textbf{accélération} (comment change la vitesse).

\begin{cle}[vitesse moyenne]
Sur une durée $\Delta t=t_2-t_1$, si l'abscisse passe de $x_1$ à $x_2$ :
\[ v_{\text{moy}}=\frac{\Delta x}{\Delta t}=\frac{x_2-x_1}{t_2-t_1}\qquad(\si{\metre\per\second}). \]
$\Delta x$ est le \emph{déplacement algébrique} : $v_{\text{moy}}>0$ dans le sens de l'axe,
$v_{\text{moy}}<0$ à contre-sens. Pour un trajet fait de plusieurs tronçons,
$v_{\text{moy}}=\dfrac{\text{distance totale}}{\text{durée totale}}$.
\end{cle}

\begin{cle}[accélération moyenne]
\[ a_{\text{moy}}=\frac{\Delta v}{\Delta t}=\frac{v_2-v_1}{t_2-t_1}\qquad(\si{\metre\per\second\squared}). \]
Il suffit de \textbf{deux vitesses} à \textbf{deux instants}. Sur une même direction : $a>0$ si la
vitesse augmente (dans le sens $+$), $a<0$ si elle diminue (souvent un \emph{freinage}).
\end{cle}

\begin{methode}[dériver et intégrer : la chaîne position--vitesse--accélération]
La vitesse et l'accélération \emph{instantanées} s'obtiennent en \textbf{dérivant} par rapport au temps :
\[ v(t)=\frac{\mathrm{d}x}{\mathrm{d}t}=\dot x,\qquad a(t)=\frac{\mathrm{d}v}{\mathrm{d}t}=\ddot x. \]
On \textbf{remonte} la chaîne en \textbf{intégrant} : $\ds v(t)=v_0+\int_0^t a\,\mathrm{d}t'$ et
$\ds x(t)=x_0+\int_0^t v\,\mathrm{d}t'$.

\end{methode}

\begin{exemplebox}[deux gestes réflexes]
\textbf{Conversion.} $\qty{90}{\kilo\metre\per\hour}=\dfrac{90\times1000}{3600}=\qty{25}{\metre\per\second}$
(diviser par $3{,}6$ pour passer de \si{\kilo\metre\per\hour} à \si{\metre\per\second}).\\
\textbf{Dérivation.} Si $x(t)=3t^2+2t$ (en \si{\metre}), alors $v(t)=6t+2$ et $a(t)=6\ \si{\metre\per\second\squared}$ (constante).
\end{exemplebox}

\begin{piege}[deux confusions classiques]
\textbf{(1) Distance $\neq$ déplacement.} Pour un aller-retour, le déplacement $\Delta x$ (donc
$v_{\text{moy}}$) est \emph{nul}, alors que la distance parcourue ne l'est pas.\\
\textbf{(2) Moyenne des vitesses $\neq$ vitesse moyenne.} Faire la moitié d'un trajet à
\qty{40}{\kilo\metre\per\hour} puis l'autre moitié à \qty{60}{\kilo\metre\per\hour} ne donne
\emph{pas} \qty{50}{\kilo\metre\per\hour} : il faut diviser la distance totale par la durée totale.
\end{piege}

\begin{remarque}[lecture graphique]
Sur la courbe $x(t)$, la vitesse est la \textbf{pente de la tangente}. Sur la courbe $v(t)$,
l'\textbf{aire algébrique} sous la courbe donne le déplacement $\Delta x$, et la pente donne l'accélération.
