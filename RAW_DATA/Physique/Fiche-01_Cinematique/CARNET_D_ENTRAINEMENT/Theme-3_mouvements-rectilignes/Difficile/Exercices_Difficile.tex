\begin{exo}[distance d'arrêt et carré de la vitesse]
À \qty{50}{\kilo\metre\per\hour}, la distance d'arrêt d'une voiture (freinage supposé constant) est de \qty{25}{\metre}.
\begin{enumerate}[label=\alph*)]
  \item Montre, à partir de Torricelli, que la distance d'arrêt est proportionnelle au \emph{carré} de la vitesse initiale.
  \item Quelle serait la distance d'arrêt à \qty{70}{\kilo\metre\per\hour}, avec le même freinage ?
  \item Et à \qty{100}{\kilo\metre\per\hour} ?
\end{enumerate}
\end{exo}

\begin{exo}[éviter l'obstacle à temps]
Le conducteur d'un train roulant à \qty{80}{\metre\per\second} aperçoit un obstacle à \qty{1000}{\metre}
et freine aussitôt avec une décélération constante.
\begin{enumerate}[label=\alph*)]
  \item Quelle est la décélération \emph{minimale} (en valeur absolue) pour s'arrêter juste avant l'obstacle ?
  \item Combien de temps dure alors le freinage ?
  \item Si le conducteur met en fait \qty{1}{\second} à réagir avant de freiner, cette décélération suffit-elle encore ? Justifie.
\end{enumerate}
\end{exo}

\begin{exo}[deux véhicules se croisent]
À midi, un véhicule quitte $A$ vers $B$ à \qty{90}{\kilo\metre\per\hour} ; au même instant, un autre
quitte $B$ vers $A$ à \qty{70}{\kilo\metre\per\hour}. Les villes sont distantes de \qty{80}{\kilo\metre}.
\begin{enumerate}[label=\alph*)]
  \item À quel instant se croisent-ils ?
  \item À quelle distance de $A$ a lieu le croisement ?
  \item Vérifie ton résultat en calculant la distance parcourue par le second véhicule.
\end{enumerate}
\end{exo}

\begin{exo}[la poursuite du policier]
Un automobiliste passe à \qty{72}{\kilo\metre\per\hour} (vitesse constante) devant un policier à l'arrêt.
Le policier démarre au même instant avec une accélération constante $a=\qty{5}{\metre\per\second\squared}$.
\begin{enumerate}[label=\alph*)]
  \item Écris la position de chacun en fonction du temps ($x=0$ au point de départ).
  \item Au bout de combien de temps le policier rejoint-il l'automobiliste ?
  \item Quelle distance a-t-il alors parcourue, et à quelle vitesse roule-t-il ?
\end{enumerate}
\end{exo}

\begin{exo}[la Formule 1 et le doublement du temps]
Une Formule 1 roulant à \qty{216}{\kilo\metre\per\hour} subit une décélération constante de \qty{5}{\metre\per\second\squared} pendant \qty{6}{\second}.
\begin{enumerate}[label=\alph*)]
  \item Quelle est sa vitesse au bout des \qty{6}{\second} ? (vérifie qu'elle ne s'est pas arrêtée)
  \item Quelle distance parcourt-elle pendant ce laps de temps ?
  \item \emph{Question de raisonnement.} Deux voitures partent du repos avec la \emph{même} accélération ; l'une accélère deux fois plus longtemps que l'autre. Dans quel rapport sont les distances parcourues ?
\end{enumerate}
\end{exo}
