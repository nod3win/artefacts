% THEME_TITLE: Vecteurs vitesse dans le plan : projection et composition
\sisetup{per-mode=symbol}

\noindent Dans le plan, la vitesse est un \textbf{vecteur} $\vv{v}$. Deux gestes suffisent :
le \emph{projeter} sur les axes ($\vv{v}\to v_x,v_y$) et \emph{composer} plusieurs vitesses
entre elles (loi de Chasles).

\begin{cle}[décomposition et norme]
On projette $\vv{v}$ (norme $v$, angle $\theta$ avec $Ox$) sur les axes :
\[ v_x=v\cos\theta,\qquad v_y=v\sin\theta,\qquad v=\sqrt{v_x^{2}+v_y^{2}},\qquad \theta=\arctan\frac{v_y}{v_x}. \]
La relation $v^2=v_x^2+v_y^2$ est le \textbf{théorème de Pythagore} : $\vv{v}$ est l'hypoténuse du rectangle des composantes.

  \coordinate (V) at (2.6,2);

\end{tikzpicture}
\end{center}
\end{cle}

\begin{cle}[composition des vitesses (loi de Chasles)]
La vitesse résultante est la \textbf{somme vectorielle} $\vv{v}=\vv{v}_1+\vv{v}_2$, obtenue en
plaçant les vecteurs \emph{bout à bout}. Cas très fréquent : $\vv{v}_1\perp\vv{v}_2$, alors
$v=\sqrt{v_1^{2}+v_2^{2}}$ et la déviation vaut $\arctan(v_2/v_1)$.
\end{cle}

\begin{methode}[traverser une rivière / voler dans le vent]
\begin{enumerate}[label=\arabic*.,leftmargin=1.6em,topsep=1pt,itemsep=0pt]
  \item Dessine les deux vitesses bout à bout (propre au mobile + celle du milieu).
  \item Norme résultante par Pythagore si elles sont perpendiculaires.
  \item Déviation : $\arctan(\text{transverse}/\text{longitudinale})$.
  \item Le \emph{temps} de traversée ne dépend que de la composante \emph{vers l'autre rive} ; la dérive $=$ (vitesse du courant)$\times$(temps).
\end{enumerate}

\begin{exemplebox}[nageur perpendiculaire au courant]
Un nageur avance à \qty{3}{\metre\per\second} perpendiculairement à un courant de \qty{4}{\metre\per\second}.
Sa vitesse par rapport à la rive vaut $v=\sqrt{3^2+4^2}=\qty{5}{\metre\per\second}$, déviée de $\arctan(4/3)\approx\ang{53}$ vers l'aval.
\end{exemplebox}

\begin{piege}[« vitesse constante » n'est pas « vecteur vitesse constant »]
La \emph{norme} peut rester constante tandis que la \emph{direction} change : le vecteur vitesse
varie alors, donc il y a \textbf{accélération}. C'est le cas de tout mouvement courbe.
\end{piege}

\begin{remarque}[mouvement circulaire uniforme]
En MCU, $\vv{v}$ est toujours \emph{tangent} au cercle : sa norme est constante mais sa direction
tourne sans cesse. Le mobile subit donc une accélération dirigée vers le centre (dite \emph{centripète}).
\end{remarque}
