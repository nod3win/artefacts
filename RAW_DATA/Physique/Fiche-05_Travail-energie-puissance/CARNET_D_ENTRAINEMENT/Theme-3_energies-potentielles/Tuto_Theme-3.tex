% THEME_TITLE: Les énergies potentielles (pesanteur et élastique)
\sisetup{per-mode=symbol}

Un corps peut stocker de l'énergie \emph{grâce à sa position} : c'est l'\textbf{énergie potentielle}.
Ce thème couvre ses deux formes : de \textbf{pesanteur} (liée à l'altitude) et \textbf{élastique}
(liée à la déformation d'un ressort).

\begin{cle}[énergie potentielle de pesanteur]
\[
\boxed{\;E_p=mgh\;}\qquad(\text{en }\si{\joule})
\]
$m$ en \si{\kilogram}, $g=\qty{9,81}{\metre\per\second\squared}$ (souvent $\approx\qty{10}{\metre\per\second\squared}$),
$h$ l'altitude au-dessus d'un niveau de référence (\si{\metre}). Le « zéro » d'altitude est
\emph{arbitraire} : seule la \emph{variation} $\Delta E_p=mg(h_B-h_A)$ a un sens physique. On choisit
en général $E_p=0$ au point le plus bas.
\end{cle}

\begin{cle}[théorème de l'énergie potentielle]
Le travail du poids est l'\emph{opposé} de la variation d'énergie potentielle :
\[
\boxed{\;W(\vv G)=-\Delta E_p=-\bigl(E_p(B)-E_p(A)\bigr)\;}
\]
\emph{Descente} : $E_p$ diminue, $W(\vv G)>0$ (moteur). \emph{Montée} : $E_p$ augmente, $W(\vv G)<0$
(résistant).
\end{cle}

\begin{cle}[énergie potentielle élastique et loi de Hooke]
Un ressort de raideur $k$ (\si{\newton\per\metre}) écarté de $x$ de sa position d'équilibre exerce une
force de rappel $F=k\,x$ (loi de Hooke) et stocke une énergie
\[
\boxed{\;E_p=\tfrac12\,k\,x^2\;}\qquad(\text{en }\si{\joule}).
\]
Comme $F=kx$ croît linéairement, le travail (donc $E_p$) est l'\emph{aire du triangle} sous la droite
$F(x)$ : $\tfrac12\times x\times(kx)=\tfrac12 kx^2$. La dépendance est \emph{en carré} : doubler $x$
\emph{quadruple} $E_p$.
\end{cle}

\begin{cle}[oscillations d'un ressort]
Écarté puis lâché, un ressort de raideur $k$ portant une masse $m$ oscille avec
\[
\boxed{\;T=2\pi\sqrt{\dfrac{m}{k}}\;}\qquad\qquad\boxed{\;f=\dfrac1T=\dfrac{1}{2\pi}\sqrt{\dfrac{k}{m}}\;}
\]
$T$ en \si{\second}, $f$ en \si{\hertz}. Plus le ressort est \emph{raide} ($k$ grand), plus il oscille
\emph{vite} ; plus la masse est \emph{lourde}, plus il oscille \emph{lentement}.
\end{cle}

\begin{exemplebox}[énergie d'un ressort étiré]
Un ressort tel que $F=\qty{2,5}{\newton}$ pour $x=\qty{10}{\centi\metre}=\qty{0,10}{\metre}$ a une raideur
$k=\dfrac{F}{x}=\dfrac{2,5}{0,10}=\qty{25}{\newton\per\metre}$, et stocke
$E_p=\tfrac12\times25\times(0,10)^2=\qty{0,125}{\joule}$.
\end{exemplebox}

\begin{piege}[erreurs fréquentes]
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item Confondre \textbf{force} $F=kx$ et \textbf{énergie} $E_p=\tfrac12 kx^2$ (facteur $\tfrac12$ et carré).
  \item Oublier de convertir $x$ en \textbf{mètres} avant le calcul de $E_p$.
  \item Croire que doubler l'élongation double $E_p$ : elle \emph{quadruple} ($E_p\propto x^2$).
\end{itemize}
\end{piege}
