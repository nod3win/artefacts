\begin{exo}[vitesse en bas d'une chute libre]
On lâche sans vitesse initiale une bille d'une hauteur $h=\qty{5}{\metre}$ (sans frottement,
$g=\qty{10}{\metre\per\second\squared}$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.85]
  \draw[black!70,line width=0.8pt] (-0.5,0)--(3,0);
  \foreach \x in {-0.3,0.1,...,2.7} \draw[black!45] (\x,0)--(\x-0.18,-0.2);
  \fill[attncol] (1.2,2.6) circle (0.16); \node[right] at (1.4,2.6){\small $m$};
  \draw[->,black!55] (1.2,2.3)--(1.2,0.4);
  \draw[<->,black!55,dashed] (0.2,0)--(0.2,2.6) node[midway,left]{\small $h=\qty{5}{\metre}$};
\end{tikzpicture}
\end{center}
En utilisant la conservation de l'énergie ($\tfrac12 m v^2=mgh$), calcule la vitesse $v$ à l'arrivée
au sol.
\end{exo}

\begin{exo}[hauteur maximale d'un lancer vertical]
On lance verticalement vers le haut un objet avec une vitesse $v_0=\qty{20}{\metre\per\second}$
($g=\qty{10}{\metre\per\second\squared}$, sans frottement). Au sommet, sa vitesse s'annule.
Calcule la hauteur maximale $h_{\max}=\dfrac{v_0^2}{2g}$ qu'il atteint.
\end{exo}

\begin{exo}[conversion de l'énergie potentielle en énergie cinétique]
Une bille de masse $m=\qty{2}{\kilogram}$ est lâchée du repos en haut d'une piste lisse, à une hauteur
$h=\qty{3}{\metre}$ au-dessus du bas de la piste ($g=\qty{10}{\metre\per\second\squared}$).
\begin{enumerate}[label=\alph*)]
  \item Que vaut son énergie potentielle en haut (référence $E_p=0$ en bas) ?
  \item Sans frottement, toute cette énergie devient cinétique en bas. Quelle est l'énergie cinétique
        $E_c$ de la bille en bas de la piste ?
\end{enumerate}
\end{exo}

\begin{exo}[vitesse en bas d'une pente sans frottement]
Un objet part du repos et glisse sans frottement le long d'une pente. La dénivellation (hauteur de
descente) vaut $h=\qty{1,8}{\metre}$ ($g=\qty{10}{\metre\per\second\squared}$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.9]
  \draw[fill=primary!8,draw=black!70] (0,2)--(5,0)--(0,0)--cycle;
  \foreach \x in {0,0.4,...,4.8} \draw[black!45] (\x,0)--(\x-0.2,-0.22);
  \fill[defcol] (0.55,1.72) circle (0.13); \node[above] at (0.4,1.9){\small départ};
  \draw[<->,black!55,dashed] (-0.4,0)--(-0.4,2) node[midway,left]{\small $h=\qty{1,8}{\metre}$};
\end{tikzpicture}
\end{center}
Calcule la vitesse $v=\sqrt{2gh}$ en bas de la pente. Ce résultat dépend-il de l'inclinaison ?
\end{exo}

\begin{exo}[l'énergie mécanique se conserve]
Au sommet d'une piste lisse, un chariot au repos possède une énergie potentielle
$E_p=\qty{80}{\joule}$ et une énergie cinétique nulle. On prend la référence $E_p=0$ en bas.
\begin{enumerate}[label=\alph*)]
  \item Que vaut l'énergie mécanique $E_m$ du chariot au sommet ?
  \item En bas de la piste, l'énergie potentielle est nulle. Déduis-en l'énergie cinétique $E_c$ en
        bas, en utilisant la conservation de $E_m$.
\end{enumerate}
\end{exo}
