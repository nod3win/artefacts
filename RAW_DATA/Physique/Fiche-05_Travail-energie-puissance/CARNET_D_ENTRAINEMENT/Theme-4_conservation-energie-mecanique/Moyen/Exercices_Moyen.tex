\begin{exo}[à quelle altitude le skieur atteint-il sa vitesse ?]
Un skieur, initialement à l'arrêt au sommet d'une piste située à $\qty{2500}{\metre}$ d'altitude, se
laisse glisser \emph{sans frottement}. On prend $g=\qty{10}{\metre\per\second\squared}$.
\begin{enumerate}[label=\alph*)]
  \item Convertis $\qty{72}{\kilo\metre\per\hour}$ en \si{\metre\per\second}, puis calcule la
        dénivellation $\Delta h$ nécessaire pour atteindre cette vitesse ($\tfrac12 v^2=g\,\Delta h$).
  \item À quelle \emph{altitude} le skieur roule-t-il alors à $\qty{72}{\kilo\metre\per\hour}$ ?
\end{enumerate}
\end{exo}

\begin{exo}[vitesse après une descente sur plan incliné]
Un bloc part du repos et glisse \emph{sans frottement} sur un plan incliné à $\beta=\ang{30}$. Il
parcourt $L=\qty{10}{\metre}$ le long de la pente ($g=\qty{10}{\metre\per\second\squared}$,
$\sin\ang{30}=0,5$).
\begin{enumerate}[label=\alph*)]
  \item Calcule la hauteur de descente $h=L\sin\ang{30}$.
  \item Déduis-en la vitesse $v=\sqrt{2gh}$ après ces \qty{10}{\metre}.
\end{enumerate}
\end{exo}

\begin{exo}[la balle qui rebondit]
Une balle rebondit sur une plateforme située à $\qty{3}{\metre}$ de hauteur. Juste après l'impact, sa
vitesse est dirigée vers le haut et vaut $\qty{10}{\metre\per\second}$ (frottements négligés,
$g=\qty{10}{\metre\per\second\squared}$).
\begin{enumerate}[label=\alph*)]
  \item De quelle hauteur \emph{supplémentaire} $\Delta h$ la balle monte-t-elle au-dessus de la
        plateforme ($\Delta h=v^2/2g$) ?
  \item Quelle est la hauteur maximale atteinte par rapport au \emph{sol} ?
\end{enumerate}
\end{exo}

\begin{exo}[le sac lâché d'une montgolfière]
Une montgolfière \emph{s'élève} à $\qty{10}{\metre\per\second}$ lorsqu'elle est à $\qty{120}{\metre}$
d'altitude. On y lâche un sac de sable (frottements négligés, $g=\qty{10}{\metre\per\second\squared}$).
\begin{enumerate}[label=\alph*)]
  \item Au moment du lâcher, quelle est la vitesse du sac, et dans quel sens ? De quelle hauteur
        supplémentaire monte-t-il encore avant de redescendre ?
  \item Quelle est la hauteur maximale du sac par rapport au sol ?
\end{enumerate}
\end{exo}

\begin{exo}[le wagon des montagnes russes]
Un wagon de montagnes russes part du repos au sommet $A$, à $h_A=\qty{45}{\metre}$, sur un rail
\emph{sans frottement} ($g=\qty{10}{\metre\per\second\squared}$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.9]
  \draw[primary,line width=1.3pt] (0,3) .. controls (1.2,3) and (1.6,0) .. (3,0)
       .. controls (4,0) and (4.2,1.6) .. (5.2,1.6);
  \fill[defcol] (0.05,3.0) circle (0.12); \node[above] at (0.2,3.1){\small $A$};
  \fill[defcol] (3,0.05) circle (0.12); \node[below] at (3,-0.1){\small $B$ (bas)};
  \fill[defcol] (5.15,1.65) circle (0.12); \node[right] at (5.25,1.7){\small $C$};
  \draw[<->,black!55,dashed] (-0.45,0)--(-0.45,3) node[midway,left]{\small $h_A=\qty{45}{\metre}$};
  \draw[<->,black!55,dashed] (5.6,0)--(5.6,1.6) node[midway,right]{\small $h_C=\qty{20}{\metre}$};
  \draw[dashed,black!35] (-0.45,0)--(5.6,0);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule la vitesse au point le plus bas $B$ ($h_B=0$).
  \item Calcule la vitesse au sommet suivant $C$, situé à $h_C=\qty{20}{\metre}$.
\end{enumerate}
\end{exo}
