\begin{corrige}[théorème avec deux vitesses non nulles]
\[
\sum W=E_c(B)-E_c(A)=\tfrac12 m v_B^2-\tfrac12 m v_A^2=\tfrac12\times200\times(5^2-2^2)=100\times(25-4)=\qty{2100}{\joule}.
\]
\end{corrige}

\begin{corrige}[distance de freinage]
\begin{enumerate}[label=\alph*)]
  \item Seule la force de freinage travaille (résistante, $\alpha=\ang{180}$). À l'arrêt $v_B=0$ :
        \[
        -F\,d=E_c(B)-E_c(A)=0-\tfrac12 m v^2\quad\Longrightarrow\quad F\,d=\tfrac12 m v^2.
        \]
  \item $d=\dfrac{\tfrac12 m v^2}{F}=\dfrac{\tfrac12\times1200\times20^2}{6000}=\dfrac{240000}{6000}=\qty{40}{\metre}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[force de frottement à partir de la variation d'énergie cinétique]
\begin{enumerate}[label=\alph*)]
  \item À l'arrêt, toute l'énergie cinétique est perdue :
        $\Delta E_c=0-\tfrac12 m v^2=-\tfrac12\times0,5\times4^2=-\tfrac12\times0,5\times16=\qty{-4}{\joule}$.
  \item $-F_f\,d=\Delta E_c=-4$, donc $F_f=\dfrac{4}{d}=\dfrac{4}{10}=\qty{0,4}{\newton}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[l'énergie cinétique ne dépend pas de la pesanteur]
\begin{enumerate}[label=\alph*)]
  \item $E_c=\tfrac12 m v^2=\tfrac12\times3\times10^2=\tfrac12\times3\times100=\qty{150}{\joule}$.
  \item \emph{Même valeur} : $E_c=\qty{150}{\joule}$. La formule $E_c=\tfrac12 m v^2$ ne fait
        intervenir que la masse et la vitesse, \emph{jamais} $g$ : à masse et vitesse identiques,
        l'énergie cinétique est la même sur la Lune que sur la Terre.
\end{enumerate}
\end{corrige}

\begin{corrige}[la distance de freinage varie comme le carré de la vitesse]
\begin{enumerate}[label=\alph*)]
  \item Comme $F\,d=\tfrac12 m v^2$ avec $F$ constante, $d\propto v^2$ :
        $\dfrac{d_2}{d_1}=\left(\dfrac{v_2}{v_1}\right)^2$.
  \item $d_2=d_1\left(\dfrac{70}{50}\right)^2=25\times(1,4)^2=25\times1,96=\qty{49}{\metre}$.
        Rouler à peine \qty{20}{\kilo\metre\per\hour} plus vite double presque la distance d'arrêt.
\end{enumerate}
\end{corrige}
