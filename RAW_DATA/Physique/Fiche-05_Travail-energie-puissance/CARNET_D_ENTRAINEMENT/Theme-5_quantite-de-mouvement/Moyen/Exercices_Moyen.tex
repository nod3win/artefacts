\begin{exo}[une voiture et un camion à la même vitesse]
Une voiture de masse $m_1=\qty{2000}{\kilogram}$ et un camion de masse $m_2=\qty{4000}{\kilogram}$
roulent à la \emph{même} vitesse $v=\qty{80}{\kilo\metre\per\hour}$.
\begin{enumerate}[label=\alph*)]
  \item Convertis la vitesse en \si{\metre\per\second} (arrondis à $\qty{22,2}{\metre\per\second}$).
  \item Calcule la quantité de mouvement de chacun des deux véhicules.
  \item Par quel facteur la quantité de mouvement du camion dépasse-t-elle celle de la voiture ?
        Lequel est le plus difficile à arrêter ?
\end{enumerate}
\end{exo}

\begin{exo}[même quantité de mouvement, vitesses différentes]
Une bille de masse $m_1=\qty{2}{\kilogram}$ et un chariot de masse $m_2=\qty{8}{\kilogram}$ ont la
\emph{même} quantité de mouvement $p=\qty{16}{\kilogram\metre\per\second}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la vitesse de la bille.
  \item Calcule la vitesse du chariot. Que constates-tu par rapport à la bille ?
\end{enumerate}
\end{exo}

\begin{exo}[quantité de mouvement et énergie cinétique quand la vitesse double]
Un mobile de masse $m=\qty{2}{\kilogram}$ se déplace à $v=\qty{10}{\metre\per\second}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule sa quantité de mouvement $p$ et son énergie cinétique $E_c$.
  \item Sa vitesse double et passe à $\qty{20}{\metre\per\second}$. Que valent alors $p$ et $E_c$ ?
        Par quel facteur chacune a-t-elle été multipliée ?
\end{enumerate}
\end{exo}

\begin{exo}[classer trois mobiles]
Trois mobiles sont en mouvement :
\begin{itemize}[leftmargin=1.6em,topsep=1pt,itemsep=1pt]
  \item $A$ : $m_A=\qty{1000}{\kilogram}$ à $v_A=\qty{20}{\metre\per\second}$ ;
  \item $B$ : $m_B=\qty{2000}{\kilogram}$ à $v_B=\qty{15}{\metre\per\second}$ ;
  \item $C$ : $m_C=\qty{1500}{\kilogram}$ à $v_C=\qty{18}{\metre\per\second}$.
\end{itemize}
Calcule la quantité de mouvement de chacun, puis classe-les de la plus grande à la plus petite.
\end{exo}
