\begin{corrige}[une voiture et un camion à la même vitesse]
\begin{enumerate}[label=\alph*)]
  \item $v=\dfrac{80}{3,6}\approx\qty{22,2}{\metre\per\second}$.
  \item $p_1=m_1 v=2000\times22,2\approx\qty{44400}{\kilogram\metre\per\second}$ ;
        $p_2=m_2 v=4000\times22,2\approx\qty{88800}{\kilogram\metre\per\second}$.
  \item $\dfrac{p_2}{p_1}=\dfrac{m_2}{m_1}=\dfrac{4000}{2000}=2$ : le camion a une quantité de mouvement
        \textbf{deux fois} plus grande. C'est donc \emph{le camion} qui est le plus difficile à arrêter.
\end{enumerate}
\end{corrige}

\begin{corrige}[même quantité de mouvement, vitesses différentes]
\begin{enumerate}[label=\alph*)]
  \item $v_1=\dfrac{p}{m_1}=\dfrac{16}{2}=\qty{8}{\metre\per\second}$.
  \item $v_2=\dfrac{p}{m_2}=\dfrac{16}{8}=\qty{2}{\metre\per\second}$. À quantité de mouvement égale, le
        chariot (4 fois plus lourd) va 4 fois moins vite que la bille : la masse et la vitesse se
        \emph{compensent}.
\end{enumerate}
\end{corrige}

\begin{corrige}[quantité de mouvement et énergie cinétique quand la vitesse double]
\begin{enumerate}[label=\alph*)]
  \item $p=m v=2\times10=\qty{20}{\kilogram\metre\per\second}$ ;
        $E_c=\tfrac12 m v^2=\tfrac12\times2\times10^2=\qty{100}{\joule}$.
  \item À $v=\qty{20}{\metre\per\second}$ : $p=2\times20=\qty{40}{\kilogram\metre\per\second}$ et
        $E_c=\tfrac12\times2\times20^2=\qty{400}{\joule}$. La quantité de mouvement a été multipliée
        par $2$ (elle est en $v$), l'énergie cinétique par $4$ (elle est en $v^2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[classer trois mobiles]
\[
p_A=1000\times20=\qty{20000}{\kilogram\metre\per\second},\quad
p_B=2000\times15=\qty{30000}{\kilogram\metre\per\second},\quad
p_C=1500\times18=\qty{27000}{\kilogram\metre\per\second}.
\]
Classement décroissant : $p_B>p_C>p_A$, soit $\mathbf{B}>\mathbf{C}>\mathbf{A}$.
\end{corrige}
