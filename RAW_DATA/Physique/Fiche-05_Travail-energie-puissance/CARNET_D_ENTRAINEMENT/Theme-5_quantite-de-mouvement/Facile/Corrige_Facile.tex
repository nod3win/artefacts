\begin{corrige}[quantité de mouvement d'un ballon]
\[
p=m\,v=0,5\times6=\qty{3}{\kilogram\metre\per\second}.
\]
\end{corrige}

\begin{corrige}[quantité de mouvement avec conversion]
\begin{enumerate}[label=\alph*)]
  \item $v=\dfrac{54}{3,6}=\qty{15}{\metre\per\second}$.
  \item $p=m\,v=1200\times15=\qty{18000}{\kilogram\metre\per\second}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[retrouver la masse]
\[
p=m\,v\ \Rightarrow\ m=\frac{p}{v}=\frac{24}{8}=\qty{3}{\kilogram}.
\]
\end{corrige}

\begin{corrige}[retrouver la vitesse et son sens]
\begin{enumerate}[label=\alph*)]
  \item $v=\dfrac{p}{m}=\dfrac{40}{5}=\qty{8}{\metre\per\second}$.
  \item Le vecteur $\vv p=m\vv v$ a la \emph{même} direction et le \emph{même} sens que $\vv v$
        (car $m>0$) : la vitesse pointe donc, elle aussi, vers l'\textbf{est}.
\end{enumerate}
\end{corrige}
