\begin{exo}[travail du poids sur un plan incliné]
Un objet de masse $m=\qty{20}{\kilogram}$ glisse le long d'un plan incliné d'un angle $\beta=\ang{30}$
sur l'horizontale. Il descend de $A$ à $B$, la longueur de la pente valant $AB=\qty{6}{\metre}$.
On prend $g=\qty{10}{\metre\per\second\squared}$ et $\sin\ang{30}=0,5$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.95]
  \coordinate (A) at (0,3); \coordinate (B) at (5.2,0); \coordinate (C) at (0,0);
  \draw[fill=primary!8,draw=black!70] (A)--(B)--(C)--cycle;
  \foreach \x in {0,0.4,...,5.0} \draw[black!45] (\x,0)--(\x-0.2,-0.22);
  \draw[->,black!60] (1.1,0) arc (180:150:1.1); \node at (1.5,0.3){\small $\beta$};
  \coordinate (M) at (2.6,1.5);
  \draw[fill=defcol!25,draw=black!70,rotate around={-30:(M)}] ($(M)+(-0.28,0)$) rectangle ($(M)+(0.28,0.42)$);
  \draw[->,line width=1.1pt,attncol] (M)--++(0,-1.3) node[right,black]{\small $\vv G$};
  \draw[<->,black!55,dashed] (-0.45,0)--(-0.45,3) node[midway,left]{\small $h$};
  \node[above] at (A){\small $A$}; \node[below right] at (B){\small $B$};
  \node at (3.5,2.1){\small $AB=\qty{6}{\metre}$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule la hauteur de descente $h$ à partir de $AB$ et de $\sin\ang{30}$.
  \item Calcule le travail du poids $W(\vv G)$ de deux façons : par $W=mgh$, puis par la composante
        $W=(G\sin\beta)\times AB$. Les deux résultats coïncident-ils ?
\end{enumerate}
\end{exo}

\begin{exo}[puissance d'un moteur à vitesse constante]
Une voiture roule en mouvement rectiligne \emph{uniforme} (vitesse constante) à
$v=\qty{72}{\kilo\metre\per\hour}$. La résultante des forces de frottement, constante, vaut
$F_f=\qty{400}{\newton}$.
\begin{enumerate}[label=\alph*)]
  \item Convertis la vitesse en \si{\metre\per\second}.
  \item À vitesse constante, la force motrice compense exactement les frottements. Calcule la
        puissance $P$ que le moteur doit développer, avec $P=F\times v$. Exprime-la en \si{\kilo\watt}.
\end{enumerate}
\end{exo}

\begin{exo}[travail comme aire sous la courbe $F(x)$]
Une force parallèle au déplacement varie avec la position $x$ selon le graphe ci-dessous. Le travail
fourni entre $x_A=\qty{0}{\metre}$ et $x_B=\qty{4}{\metre}$ est égal à l'\emph{aire} comprise entre la
courbe $F(x)$ et l'axe des $x$.
\begin{center}

\end{center}
Sachant que la région hachurée est un \emph{trapèze} (bases $F_A=\qty{10}{\newton}$ et
$F_B=\qty{30}{\newton}$, largeur $\qty{4}{\metre}$), calcule le travail $W$.
\end{exo}

\begin{exo}[bilan des travaux sur un trajet horizontal]
Une luge est tirée sur un sol horizontal par une corde exerçant $F=\qty{100}{\newton}$ inclinée de
$\alpha=\ang{60}$. Elle avance de $d=\qty{20}{\metre}$. Une force de frottement $F_f=\qty{25}{\newton}$
s'oppose au mouvement. On prend $\cos\ang{60}=0,5$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le travail $W(\vv F)$ de la force de traction.
  \item Calcule le travail $W(\vv F_f)$ du frottement.
  \item Que valent les travaux du poids et de la réaction normale sur ce trajet ?
  \item Déduis-en le travail total $\sum W$ reçu par la luge.
\end{enumerate}
\end{exo}

\begin{exo}[puissance moyenne d'une grue]
Une grue soulève verticalement une charge de masse $m=\qty{500}{\kilogram}$ jusqu'à une hauteur
$h=\qty{12}{\metre}$, en une durée $\Delta t=\qty{8}{\second}$. On prend $g=\qty{10}{\metre\per\second\squared}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le travail $W$ fourni par la grue pour élever la charge ($W=mgh$).
  \item Déduis-en la puissance moyenne $P$ développée, en \si{\kilo\watt}.
\end{enumerate}
\end{exo}
