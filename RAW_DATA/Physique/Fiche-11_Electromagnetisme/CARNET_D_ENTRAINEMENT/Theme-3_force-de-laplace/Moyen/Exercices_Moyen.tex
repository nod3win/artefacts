\begin{exo}[force avec un angle quelconque]
Un conducteur de longueur $L=\qty{20}{\centi\metre}$, parcouru par $I=\qty{5}{\ampere}$, fait un
angle $\theta=\ang{30}$ avec un champ uniforme $B=\qty{0.2}{\tesla}$. Calcule l'intensité de la
force de Laplace. (On donne $\sin\ang{30}=0{,}5$.)
\end{exo}

\begin{exo}[retrouver le champ magnétique]
Un conducteur de longueur $L=\qty{15}{\centi\metre}$, perpendiculaire à un champ $\vv{B}$
($\theta=\ang{90}$), parcouru par $I=\qty{8}{\ampere}$, subit une force $F=\qty{0.6}{\newton}$.
Détermine l'intensité $B$ du champ magnétique.
\end{exo}

\begin{exo}[force entre deux fils parallèles]
Deux fils rectilignes parallèles distants de $d=\qty{2}{\centi\metre}$ sont parcourus par un même
courant $I=\qty{5}{\ampere}$ de même sens. (On prend $\mu_0/2\pi=2\times10^{-7}$.)
\begin{enumerate}[label=\alph*)]
  \item Calcule la force par unité de longueur $F/L$ qu'exerce un fil sur l'autre.
  \item Ces fils s'attirent-ils ou se repoussent-ils ?
\end{enumerate}
\end{exo}

\begin{exo}[sens de la force avec $\vv{B}$ entrant]
Un conducteur horizontal est parcouru par un courant $I$ vers la \textbf{droite}. Le champ $\vv{B}$
est \textbf{perpendiculaire à la feuille} et \textbf{entrant} ($\otimes$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \foreach \x/\y in {-1.5/0.7, 0/0.7, 1.5/0.7, -1.5/-0.7, 0/-0.7, 1.5/-0.7}{
    \draw[blue!60!black,line width=1pt] (\x,\y) circle (0.15);
    \draw[blue!60!black,line width=1pt] (\x-0.1,\y-0.1)--(\x+0.1,\y+0.1);
    \draw[blue!60!black,line width=1pt] (\x-0.1,\y+0.1)--(\x+0.1,\y-0.1);}
  \node[blue!70!black,right,font=\footnotesize] at (1.75,0.7){$\vv{B}$ (entrant)};
  \draw[red!80!black,->,line width=2.2pt] (-2,0)--(2,0) node[right]{$I$};
\end{tikzpicture}
\end{center}
Dans quel sens pointe la force de Laplace $\vv{F}$ (vers le haut ou vers le bas) ? Justifie par la
règle des trois doigts.
\end{exo}

\begin{exo}[le rail de Laplace]
Une barre conductrice mobile de longueur $L=\qty{8}{\centi\metre}$ repose sur deux rails horizontaux.
Un courant $I=\qty{4}{\ampere}$ la traverse (vers le haut de la figure). Le champ
$B=\qty{0.3}{\tesla}$ est perpendiculaire au plan des rails et \textbf{entrant} ($\otimes$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \draw[black,line width=1pt] (-2,-0.9)--(2,-0.9);
  \draw[black,line width=1pt] (-2,0.9)--(2,0.9);
  \draw[gray,line width=2.5pt] (0.4,-0.9)--(0.4,0.9);
  \draw[red!80!black,->,line width=1.6pt] (0.4,-0.4)--(0.4,0.4); \node[red!80!black,right,font=\footnotesize] at (0.5,0.6){$I$};
  \foreach \x/\y in {-1.2/0.4,-1.2/-0.4,1.3/0.4,1.3/-0.4}{
    \draw[blue!60!black,line width=0.9pt] (\x,\y) circle (0.13);
    \draw[blue!60!black,line width=0.9pt] (\x-0.09,\y-0.09)--(\x+0.09,\y+0.09);
    \draw[blue!60!black,line width=0.9pt] (\x-0.09,\y+0.09)--(\x+0.09,\y-0.09);}
  \node[blue!70!black,font=\footnotesize] at (-1.2,0.85){$\vv{B}$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité de la force de Laplace sur la barre.
  \item Dans quel sens la barre se met-elle en mouvement (vers la gauche ou vers la droite) ?
\end{enumerate}
\end{exo}
