\begin{exo}[le cadre qui pivote (moteur)]
Un cadre rectangulaire conducteur, dans le plan de la feuille, est parcouru par un courant $I$ (sens
horaire). Il est plongé dans un champ uniforme $\vv{B}$ dirigé \textbf{vers la droite}, dans le plan
du cadre.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \foreach \y in {-1.1,-0.55,0,0.55,1.1}{\draw[blue!55!black,line width=0.7pt,->] (-2.9,\y)--(-2.2,\y);}
  \node[blue!70!black] at (-3.2,0){$\vv{B}$};
  \draw[red!80!black,line width=1.8pt] (-1.3,-1.1) rectangle (1.3,1.1);
  \draw[red!80!black,->,line width=1.8pt] (-0.2,1.1)--(0.4,1.1);
  \draw[red!80!black,->,line width=1.8pt] (1.3,0.2)--(1.3,-0.4);
  \draw[red!80!black,->,line width=1.8pt] (0.2,-1.1)--(-0.4,-1.1);
  \draw[red!80!black,->,line width=1.8pt] (-1.3,-0.2)--(-1.3,0.4);
  \node[red!80!black,font=\footnotesize] at (0.9,1.32){$I$};
  \node[font=\footnotesize] at (-1.55,0.75){côté 1};
  \node[font=\footnotesize] at (1.6,0.75){côté 2};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Sur quels côtés du cadre la force de Laplace est-elle \textbf{nulle} ? Pourquoi ?
  \item Que subissent les deux côtés verticaux (côtés 1 et 2), et quel est le mouvement résultant du
        cadre : translation ou rotation ?
\end{enumerate}
\end{exo}

\begin{exo}[faire léviter une barre]
Une barre conductrice horizontale, de masse $m=\qty{20}{\gram}$ et de longueur $L=\qty{25}{\centi\metre}$,
est placée dans un champ $B=\qty{0.8}{\tesla}$ perpendiculaire (entrant, $\otimes$). On veut que la
force de Laplace \textbf{compense exactement son poids} pour la faire léviter ($g=\qty{10}{\metre\per\second\squared}$).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \draw[gray,line width=3pt] (-1.3,0)--(1.3,0);
  \draw[excol,->,line width=1.6pt] (0,0)--(0,1.1) node[above,font=\footnotesize]{$\vv{F}$ (Laplace)};
  \draw[black,->,line width=1.4pt] (0,0)--(0,-1.1) node[below,font=\footnotesize]{$\vv{P}=m\vv{g}$};
  \foreach \x/\y in {-0.9/0.55,0.9/0.55,-0.9/-0.55,0.9/-0.55}{
    \draw[blue!60!black,line width=0.8pt] (\x,\y) circle (0.12);
    \draw[blue!60!black,line width=0.8pt] (\x-0.08,\y-0.08)--(\x+0.08,\y+0.08);
    \draw[blue!60!black,line width=0.8pt] (\x-0.08,\y+0.08)--(\x+0.08,\y-0.08);}
  \node[blue!70!black,font=\footnotesize] at (1.25,0.85){$\vv{B}$};
\end{tikzpicture}
\end{center}
Quelle intensité de courant $I$ faut-il faire passer dans la barre ?
\end{exo}

\begin{exo}[deux fils, courants opposés]
Deux fils parallèles distants de $d=\qty{1}{\centi\metre}$ portent des courants de \textbf{sens
contraires} : $I_1=\qty{10}{\ampere}$ et $I_2=\qty{15}{\ampere}$. (On prend $\mu_0/2\pi=2\times10^{-7}$.)
\begin{enumerate}[label=\alph*)]
  \item Calcule la force par unité de longueur entre les deux fils.
  \item S'attirent-ils ou se repoussent-ils ?
\end{enumerate}
\end{exo}

\begin{exo}[longueur effectivement plongée dans le champ]
Un long conducteur rectiligne de \qty{30}{\centi\metre} traverse une zone de champ uniforme
$B=\qty{0.4}{\tesla}$ (entrant), mais seule une portion de \textbf{longueur} $\ell=\qty{12}{\centi\metre}$
est réellement \emph{dans} le champ ; le reste est hors champ. Le courant vaut $I=\qty{6}{\ampere}$,
perpendiculaire à $\vv{B}$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \draw[gray!30,fill=blue!6] (-0.9,-1) rectangle (0.9,1);
  \foreach \x/\y in {-0.45/0.6,0.45/0.6,-0.45/-0.6,0.45/-0.6,0/0}{
    \draw[blue!55!black,line width=0.8pt] (\x,\y) circle (0.11);
    \draw[blue!55!black,line width=0.8pt] (\x-0.07,\y-0.07)--(\x+0.07,\y+0.07);
    \draw[blue!55!black,line width=0.8pt] (\x-0.07,\y+0.07)--(\x+0.07,\y-0.07);}
  \node[blue!70!black,font=\footnotesize] at (0,1.25){zone de champ, $\ell$};
  \draw[red!80!black,->,line width=2pt] (-2.6,0)--(2.6,0) node[right]{$I$};
  \draw[<->,gray,line width=0.5pt] (-0.9,-1.25)--(0.9,-1.25); \node[gray,below,font=\footnotesize] at (0,-1.28){$\ell$};
\end{tikzpicture}
\end{center}
Calcule la force de Laplace subie par le conducteur. Attention au choix de la longueur !
\end{exo}
