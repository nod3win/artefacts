\begin{corrige}[deux champs perpendiculaires de même norme]
Les deux champs étant orthogonaux, on applique le théorème de Pythagore (surtout pas $B_1+B_2$ !).
\begin{enumerate}[label=\alph*)]
  \item $B_{\text{tot}}=\sqrt{B_1^2+B_2^2}=\sqrt{2}\,B_1=1{,}414\times\num{1.0e-4}
        \approx\qty{1.41e-4}{\tesla}.$
  \item $\tan\theta=\dfrac{B_2}{B_1}=1 \Rightarrow \theta=\ang{45}$ : le champ résultant est la
        \textbf{bissectrice} de l'angle droit (logique, puisque $B_1=B_2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[superposition orthogonale : Pythagore]
\begin{enumerate}[label=\alph*)]
  \item $B_{\text{tot}}=\sqrt{B_1^2+B_2^2}=\sqrt{(3\cdot10^{-4})^2+(4\cdot10^{-4})^2}
        =\sqrt{9+16}\times10^{-4}=\sqrt{25}\times10^{-4}=\qty{5e-4}{\tesla}.$
        (C'est le triangle $3$–$4$–$5$.)
  \item $\tan\theta=\dfrac{B_2}{B_1}=\dfrac{4}{3}\Rightarrow \theta\approx\ang{53}$ au-dessus de
        l'horizontale $\vv{B}_1$.
\end{enumerate}
\end{corrige}

\begin{corrige}[fil horizontal : champ au-dessus et au-dessous]
Main droite : \textbf{pouce vers la droite} (sens de $I$), les doigts s'enroulent autour du fil.
\begin{enumerate}[label=\alph*)]
  \item Au-dessus du fil, les doigts sortent vers l'avant : au point $P$, $\vv{B}$ \textbf{sort} de la
        feuille, $\odot$.
  \item Au-dessous, les doigts plongent vers l'arrière : au point $Q$, $\vv{B}$ \textbf{entre} dans la
        feuille, $\otimes$. Le champ « tourne » bien autour du fil (sens opposés de part et d'autre).
\end{enumerate}
\end{corrige}

\begin{corrige}[deux fils identiques : champ au milieu]
\begin{enumerate}[label=\alph*)]
  \item Chaque courant sort de la feuille : main droite, pouce vers soi, doigts en sens
        trigonométrique. \textbf{Fil 1} (à gauche de $M$) crée en $M$ un champ dirigé \textbf{vers le
        haut} ; \textbf{fil 2} (à droite de $M$) crée en $M$ un champ dirigé \textbf{vers le bas}.
        Les deux champs sont donc \textbf{opposés}.
  \item De même norme ($\qty{4e-5}{\tesla}$) et de sens contraires, ils se \textbf{compensent} :
        \[ B_{\text{tot}}=|B_1-B_2|=0. \]
        Le champ résultant est \textbf{nul} au milieu de deux fils identiques parcourus dans le même sens.
\end{enumerate}
\end{corrige}
