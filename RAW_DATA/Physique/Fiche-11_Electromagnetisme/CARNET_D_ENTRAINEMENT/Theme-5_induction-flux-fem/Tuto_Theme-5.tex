\begin{center}
{\Large\bfseries\color{primarydark} Thème 5 — Induction : flux et loi de Lenz--Faraday}
\end{center}

\noindent Quand le champ magnétique à travers un circuit \textbf{varie}, un courant apparaît : c'est
l'\textbf{induction}. Il faut savoir calculer le \textbf{flux}, la \textbf{f.é.m. induite}, trouver le
\textbf{sens} du courant (loi de Lenz), et manier l'\textbf{alternatif} (valeur efficace).

\begin{cle}[flux magnétique et loi de Lenz--Faraday]
Flux à travers une spire d'aire $S$ : $\ \boxed{\,\phi=B\,S\,\cos\alpha\,}\ $ (en webers, \si{\weber}),
où $\alpha$ est l'angle entre $\vv{B}$ et le \textbf{vecteur surface} $\vv{S}$ (la normale à la spire).\\[2pt]
Pour une bobine de $N$ spires, la f.é.m. induite est
\[ \boxed{\,e=-\,N\,\dfrac{\mathrm{d}\phi}{\mathrm{d} t}\,}\qquad(\text{en volts}). \]
Le signe « $-$ » est la \textbf{loi de Lenz} : le courant induit s'oppose à la \emph{variation} du flux.
\end{cle}

\begin{methode}[trouver le sens du courant induit — loi de Lenz]
« La bobine \textbf{résiste} à ce qu'on lui fait subir. »
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=0pt]
  \item On \textbf{approche} un pôle Nord $\Rightarrow$ le flux augmente $\Rightarrow$ la bobine crée une
        face \textbf{Nord} face à l'aimant pour le \textbf{repousser} ;
  \item On \textbf{éloigne} l'aimant $\Rightarrow$ le flux diminue $\Rightarrow$ la bobine crée une face
        \textbf{Sud} pour le \textbf{retenir}.
\end{itemize}
Le flux $\phi=BS\cos\alpha$ varie si l'on change $B$, $S$ \emph{ou} $\alpha$ (faire tourner la spire
$=$ alternateur).
\end{methode}

\begin{piege}[deux pièges classiques]
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=0pt]
  \item $\alpha$ se mesure entre $\vv{B}$ et la \textbf{normale} $\vv{S}$, \textbf{pas} le plan de la
        spire. Si l'énoncé donne l'angle $\beta$ avec le \emph{plan}, alors $\alpha=\ang{90}-\beta$ et
        $\cos\alpha=\sin\beta$.
  \item Un flux \textbf{constant} (aimant immobile) ne crée \textbf{aucun} courant : il faut une
        \textbf{variation} ($\mathrm{d}\phi/\mathrm{d} t\neq0$).
\end{itemize}
\end{piege}

\begin{exemplebox}[f.é.m. induite]
Une bobine de $N=200$ spires voit son flux passer de $0$ à \qty{5e-3}{\weber} en \qty{0.1}{\second} :
\[ e=N\,\frac{\Delta\phi}{\Delta t}=200\times\frac{5\cdot10^{-3}}{0{,}1}=\qty{10}{\volt}. \]
\end{exemplebox}

\begin{remarque}[alternateur, valeur efficace, courants de Foucault]
Une spire qui tourne à la vitesse $\omega=2\pi f$ produit une f.é.m. \textbf{sinusoïdale}
$e(t)=N\omega BS\sin(\omega t)$. Pour une grandeur alternative, la \textbf{valeur efficace} est
$\boxed{I_{\text{eff}}=I_{\max}/\sqrt2}$ (idem $U_{\text{eff}}=U_{\max}/\sqrt2$). Dans un métal massif
soumis à un champ variable, des \textbf{courants de Foucault} apparaissent : ils dissipent de la chaleur
et \textbf{freinent} le mouvement (loi de Lenz).
\end{remarque}
