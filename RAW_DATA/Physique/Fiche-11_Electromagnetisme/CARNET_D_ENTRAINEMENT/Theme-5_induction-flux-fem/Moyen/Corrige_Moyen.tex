\begin{corrige}[calculer une f.é.m. induite]
Variation de flux : $\Delta\phi=8\cdot10^{-3}-2\cdot10^{-3}=\qty{6e-3}{\weber}$. Alors
\[ |e|=N\,\frac{\Delta\phi}{\Delta t}=100\times\frac{6\cdot10^{-3}}{0{,}05}=100\times0{,}12=\qty{12}{\volt}. \]
\end{corrige}

\begin{corrige}[l'angle avec le plan]
\begin{enumerate}[label=\alph*)]
  \item L'angle se mesure avec la \textbf{normale} $\vv{S}$ : $\alpha=\ang{90}-\beta=\ang{90}-\ang{30}=\ang{60}$.
  \item $\phi=B\,S\,\cos\alpha=0{,}2\times0{,}05\times\cos\ang{60}=0{,}2\times0{,}05\times0{,}5=\qty{5e-3}{\weber}=\qty{5}{\milli\weber}$.
        (Utiliser $\cos\ang{30}$ ici serait le piège.)
\end{enumerate}
\end{corrige}

\begin{corrige}[sens du courant induit]
La bobine « résiste » à la variation de flux (loi de Lenz).
\begin{enumerate}[label=\alph*)]
  \item \textbf{Approche} : le flux augmente ; la bobine crée un champ opposé et présente donc une
        face \textbf{Nord} à l'aimant, ce qui le \textbf{repousse}.
  \item \textbf{Éloignement} : le flux diminue ; la bobine « tente » de le maintenir en présentant une
        face \textbf{Sud}, ce qui \textbf{attire} l'aimant. Le courant induit a donc \textbf{changé de
        sens} entre les deux phases.
\end{enumerate}
\end{corrige}

\begin{corrige}[f.é.m. maximale d'un alternateur]
\begin{enumerate}[label=\alph*)]
  \item $\omega=2\pi f=2\pi\times50\approx\qty{314}{\radian\per\second}$.
  \item $e_{\max}=N\omega BS=100\times314\times0{,}1\times0{,}02=100\times0{,}628\approx\qty{63}{\volt}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[le disque freiné (courants de Foucault)]
\begin{enumerate}[label=\alph*)]
  \item Le disque plein entre dans le champ : le flux à travers le métal \textbf{varie} $\Rightarrow$
        des \textbf{courants de Foucault} apparaissent. Par la loi de Lenz, ils créent une force qui
        \textbf{s'oppose au mouvement} : le disque est \textbf{freiné} et s'arrête presque
        instantanément.
  \item Les fentes \textbf{interrompent les boucles} de courant : les courants de Foucault deviennent
        négligeables, le freinage disparaît, le disque oscille presque librement. C'est pourquoi les
        noyaux des transformateurs sont \textbf{feuilletés} (tôles isolées) : pour limiter ces courants
        parasites et l'échauffement.
\end{enumerate}
\end{corrige}
