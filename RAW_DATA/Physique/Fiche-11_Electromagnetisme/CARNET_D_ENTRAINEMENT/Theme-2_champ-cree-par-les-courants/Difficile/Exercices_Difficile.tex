\begin{exo}[sous une ligne à haute tension]
Une ligne à haute tension transporte $I=\qty{100}{\ampere}$. On prend $\mu_0/2\pi=2\times10^{-7}$
et $B_T=\qty{5e-5}{\tesla}$ pour le champ terrestre.
\begin{enumerate}[label=\alph*)]
  \item Calcule le champ magnétique à $d=\qty{40}{\centi\metre}$ du fil.
  \item Calcule le champ à $d=\qty{30}{\metre}$ (hauteur typique du câble) et compare-le au champ
        terrestre : combien de fois plus faible ?
\end{enumerate}
\end{exo}

\begin{exo}[mettre le solénoïde à l'échelle]
Un solénoïde crée un champ interne $B_0=\qty{2e-3}{\tesla}$. On modifie ses caractéristiques :
on \textbf{double} le nombre de spires, on \textbf{divise la longueur par deux}, et on
\textbf{triple} le courant.
\begin{enumerate}[label=\alph*)]
  \item Sachant que $B=\dfrac{\mu_0 N I}{L}$, par quel facteur le champ est-il multiplié ?
  \item Quelle est la nouvelle valeur du champ interne ?
\end{enumerate}
\end{exo}

\begin{exo}[combien de spires pour la bobine ?]
On veut qu'une bobine plate de rayon $r=\qty{3}{\centi\metre}$, parcourue par $I=\qty{4}{\ampere}$,
crée en son centre un champ $B=\qty{2e-3}{\tesla}$. Combien de spires $N$ faut-il ?
(On prend $\mu_0=\qty{1.26e-6}{}$.)
\end{exo}

\begin{exo}[deux fils : Biot-Savart puis superposition]
Deux fils parallèles, perpendiculaires à la feuille, distants de \qty{6}{\centi\metre}, portent des
courants de \textbf{même sens} (tous deux sortants, $\odot$) : $I_1=\qty{10}{\ampere}$ (fil 1, à
gauche) et $I_2=\qty{20}{\ampere}$ (fil 2, à droite). On étudie le point milieu $M$, à
\qty{3}{\centi\metre} de chaque fil.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \draw[red!80!black,line width=1.4pt] (-2,0) circle (0.16); \fill[red!80!black] (-2,0) circle (0.05);
  \node[red!80!black,below,font=\footnotesize] at (-2,-0.32){fil 1 : $I_1$};
  \draw[red!80!black,line width=1.4pt] (2,0) circle (0.16); \fill[red!80!black] (2,0) circle (0.05);
  \node[red!80!black,below,font=\footnotesize] at (2,-0.32){fil 2 : $I_2$};
  \fill[black] (0,0) circle (1.5pt) node[below,font=\footnotesize]{$M$};
  \draw[<->,gray,line width=0.5pt] (-2,0.7)--(2,0.7); \node[gray,above,font=\footnotesize] at (0,0.7){\qty{6}{\centi\metre}};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule les intensités $B_1$ et $B_2$ créées par chaque fil en $M$ (loi de Biot et Savart).
  \item Donne le sens de chacun (main droite), puis calcule l'intensité du champ résultant en $M$.
\end{enumerate}
\end{exo}
