\begin{center}
{\Large\bfseries\color{primarydark} Thème 2 — Champ créé par les courants}
\end{center}

\noindent Un courant électrique crée toujours un champ magnétique. Selon la \emph{forme} du
conducteur, on dispose de trois formules à connaître par cœur. Ce thème entraîne le calcul de $B$,
l'isolement d'une variable et la vigilance sur les unités.

\begin{cle}[les trois formules de Biot et Savart]
\renewcommand{\arraystretch}{1.6}
\begin{center}
\begin{tabular}{@{}>{\raggedright\arraybackslash}p{4.1cm} >{\centering\arraybackslash}p{4.6cm} >{\raggedright\arraybackslash}p{5.0cm}@{}}
\textbf{Source} & \textbf{Champ} & \textbf{Point où l'on calcule} \\ \hline
Fil rectiligne infini & $B=\dfrac{\mu_0 I}{2\pi d}=\dfrac{2\cdot10^{-7}\,I}{d}$ & à la distance $d$ du fil \\
Spire circulaire ($N$ tours) & $B=\dfrac{N\,\mu_0 I}{2r}$ & au \emph{centre} de la spire \\
Solénoïde ($N$ tours, long. $L$) & $B=\dfrac{\mu_0 N I}{L}$ & à l'\emph{intérieur} (uniforme) \\
\end{tabular}
\end{center}
avec $\mu_0=4\pi\times10^{-7}\ \si{\tesla\metre\per\ampere}$, d'où le raccourci
$\dfrac{\mu_0}{2\pi}=2\times10^{-7}$.
\end{cle}

\begin{methode}[calculer $B$ ou isoler une variable]
\begin{enumerate}[leftmargin=1.5em,topsep=1pt,itemsep=1pt]
  \item Identifier la \textbf{forme} (fil / spire / solénoïde) $\Rightarrow$ choisir la formule.
  \item Convertir les longueurs ($d$, $r$, $L$) en \textbf{mètres} et les courants en \textbf{ampères}.
  \item Appliquer la formule, ou l'\textbf{inverser} pour isoler l'inconnue. Exemples :
        \[ d=\frac{\mu_0 I}{2\pi B}=\frac{2\cdot10^{-7} I}{B},\qquad
           I=\frac{B\,L}{\mu_0 N},\qquad N=\frac{B\,L}{\mu_0 I}. \]
\end{enumerate}
\end{methode}

\begin{exemplebox}[champ dans un solénoïde]
Solénoïde de $N=500$ spires, longueur $L=\qty{25}{\centi\metre}=\qty{0.25}{\metre}$, courant
$I=\qty{4}{\ampere}$ :
\[ B=\frac{\mu_0 N I}{L}=\frac{4\pi\times10^{-7}\times500\times4}{0{,}25}
    =\frac{1{,}26\cdot10^{-6}\times2000}{0{,}25}\approx\qty{1.0e-2}{\tesla}=\qty{10}{\milli\tesla}. \]
\end{exemplebox}

\begin{piege}[ce qui pilote réellement le champ]
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=0pt]
  \item Fil : $B\propto \dfrac{1}{d}$ (et \textbf{non} $1/d^2$ !). Doubler la distance \textbf{divise
        $B$ par 2}, pas par 4.
  \item Solénoïde : c'est la \textbf{densité de spires} $n=N/L$ qui compte, pas $N$ seul. Deux bobines
        de même $N/L$ et même $I$ ont le \textbf{même} champ interne.
\end{itemize}
\end{piege}

\begin{remarque}[le noyau de fer]
Glisser un \textbf{noyau de fer doux} dans un solénoïde remplace $\mu_0$ par $\mu=\mu_r\,\mu_0$ avec
$\mu_r$ pouvant valoir des \emph{centaines à milliers} : le champ est massivement renforcé. C'est le
principe de l'\textbf{électro-aimant} (le champ disparaît dès que le courant est coupé).
\end{remarque}
