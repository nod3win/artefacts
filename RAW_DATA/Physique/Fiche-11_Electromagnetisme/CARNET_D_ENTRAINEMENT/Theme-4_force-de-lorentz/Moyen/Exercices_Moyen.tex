\begin{exo}[force de Lorentz avec un angle]
Un proton ($q=\qty{1.6e-19}{\coulomb}$) se déplace à $v=\qty{2e6}{\metre\per\second}$ en faisant un
angle $\alpha=\ang{30}$ avec un champ $B=\qty{0.4}{\tesla}$. Calcule l'intensité de la force de
Lorentz. (On donne $\sin\ang{30}=0{,}5$.)
\end{exo}

\begin{exo}[rayon et doublement de la vitesse]
Un proton ($m=\qty{1.67e-27}{\kilogram}$, $q=\qty{1.6e-19}{\coulomb}$) décrit un cercle dans un champ
$B=\qty{0.4}{\tesla}$ avec une vitesse $v=\qty{2e6}{\metre\per\second}$ ($\vv{v}\perp\vv{B}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule le rayon $R$ de sa trajectoire.
  \item Quel devient ce rayon si l'on \textbf{double} la vitesse (à $B$ constant) ?
\end{enumerate}
\end{exo}

\begin{exo}[nature de la trajectoire d'un électron]
Un faisceau d'électrons pénètre \textbf{horizontalement} dans une région où règne un champ magnétique
uniforme \textbf{vertical}. On néglige le poids des électrons.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \draw[blue!70!black,->,line width=1.6pt] (0,-1.2)--(0,1.2) node[above]{$\vv{B}$};
  \draw[orange!85!black,->,line width=1.8pt] (-2.4,0)--(-0.4,0) node[above,font=\footnotesize]{$\vv{v}$};
  \fill[red!80!black] (-2.4,0) circle (2pt); \node[below,font=\footnotesize] at (-2.4,-0.1){$e^-$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item La force de Lorentz modifie-t-elle la \emph{norme} de la vitesse ? Justifie.
  \item Quelle est alors la nature de la trajectoire du faisceau ?
\end{enumerate}
\end{exo}

\begin{exo}[le sens pour une charge négative]
Un \textbf{électron} ($q<0$) se déplace vers la \textbf{droite} dans un champ $\vv{B}$ dirigé vers
le \textbf{haut} (plan de la feuille).
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \fill[red!80!black] (0,0) circle (2.5pt); \node[below left,font=\footnotesize] at (0,0){$e^-$};
  \draw[orange!85!black,->,line width=1.8pt] (0,0)--(2,0) node[right]{$\vv{v}$};
  \draw[blue!70!black,->,line width=1.6pt] (0,0)--(0,1.6) node[above]{$\vv{B}$};
\end{tikzpicture}
\end{center}
Dans quel sens pointe la force $\vv{F}$ : sort-elle ($\odot$) ou entre-t-elle ($\otimes$) dans la
feuille ? (Attention au signe de la charge.)
\end{exo}

\begin{exo}[la force qui ne travaille pas]
Une particule chargée tourne dans un champ magnétique uniforme.
\begin{enumerate}[label=\alph*)]
  \item La force magnétique peut-elle modifier la \textbf{vitesse} (en norme) de la particule ? Pourquoi ?
  \item Si l'on \textbf{double} le champ $B$ (à vitesse inchangée), que deviennent le rayon $R$ et
        l'énergie cinétique de la particule ?
\end{enumerate}
\end{exo}
