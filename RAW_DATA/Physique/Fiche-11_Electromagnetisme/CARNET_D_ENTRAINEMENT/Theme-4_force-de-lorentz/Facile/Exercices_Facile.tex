\begin{exo}[calculer la force de Lorentz]
Un proton (charge $q=\qty{1.6e-19}{\coulomb}$) se déplace à la vitesse $v=\qty{1e6}{\metre\per\second}$,
\textbf{perpendiculairement} ($\alpha=\ang{90}$) à un champ $B=\qty{0.5}{\tesla}$. Calcule
l'intensité de la force de Lorentz.
\end{exo}

\begin{exo}[quand la force est-elle nulle ?]
Indique, en justifiant, la force magnétique subie dans chaque cas :
\begin{enumerate}[label=\alph*)]
  \item un électron \textbf{immobile} placé dans un champ $B=\qty{2}{\tesla}$ ;
  \item un \textbf{neutron} (particule neutre) lancé à $v=\qty{1e7}{\metre\per\second}$ dans ce même champ.
\end{enumerate}
\end{exo}

\begin{exo}[sens de la force sur une charge positive]
Une particule de charge \textbf{positive} $q>0$ se déplace vers la \textbf{droite}. Le champ $\vv{B}$
est dirigé \textbf{vers le haut}, dans le plan de la feuille.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1.0]
  \fill[red!80!black] (0,0) circle (2.5pt); \node[below left,font=\footnotesize] at (0,0){$q>0$};
  \draw[orange!85!black,->,line width=1.8pt] (0,0)--(2,0) node[right]{$\vv{v}$};
  \draw[blue!70!black,->,line width=1.6pt] (0,0)--(0,1.6) node[above]{$\vv{B}$};
\end{tikzpicture}
\end{center}
Par la règle de la main droite, la force $\vv{F}$ sort-elle de la feuille ($\odot$) ou y entre-t-elle
($\otimes$) ?
\end{exo}

\begin{exo}[l'effet de la vitesse]
Une particule chargée subit une force de Lorentz $F_0=\qty{8e-14}{\newton}$ à la vitesse $v$
(avec $\vv{v}\perp\vv{B}$).
\begin{enumerate}[label=\alph*)]
  \item Quelle force subit-elle si on \textbf{double} sa vitesse ?
  \item Quelle force subit-elle si elle est \textbf{immobile} ?
\end{enumerate}
\end{exo}

\begin{exo}[rayon de la trajectoire]
Un proton ($m=\qty{1.67e-27}{\kilogram}$, $q=\qty{1.6e-19}{\coulomb}$) entre à
$v=\qty{1e6}{\metre\per\second}$ perpendiculairement à un champ $B=\qty{0.5}{\tesla}$. Calcule le
rayon $R$ du cercle qu'il décrit.
\end{exo}
