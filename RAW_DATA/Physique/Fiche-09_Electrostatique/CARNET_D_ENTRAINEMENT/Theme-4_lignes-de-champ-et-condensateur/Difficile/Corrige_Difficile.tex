\begin{corrige}[quel spectre est correct ?]
\begin{enumerate}[label=\alph*)]
  \item Le schéma \textbf{(C)} : les lignes sont parallèles (champ uniforme) et $\vv{E}$ est bien
        \emph{tangent} à la ligne, dans son sens.
  \item \textbf{(A)} : le vecteur $\vv{E}$ est dessiné \emph{perpendiculaire} à la ligne ; or $\vv{E}$
        doit toujours être \textbf{tangent} à la ligne de champ. \textbf{(B)} : les flèches vont du
        $-$ vers le $+$ ; or les lignes partent \textbf{toujours} du $+$ et vont vers le $-$ : le
        sens est inversé.
\end{enumerate}
\end{corrige}

\begin{corrige}[condensateur : attention au piège]
\begin{enumerate}[label=\alph*)]
  \item Le champ ne dépend que de $U$ et $d$, \emph{pas} du rayon (piège) :
        \[ d = 2R = \SI{20}{\centi\metre} = \SI{0.20}{\metre},\qquad
           E = \frac{|U_{AB}|}{d} = \frac{100}{0{,}20} = \SI{500}{\volt\per\metre}. \]
        Le rayon $R$ est inutile.
  \item Le potentiel décroît linéairement : au milieu, $V_P = (V_A+V_B)/2$, donc
        \[ U_{AP} = V_A - V_P = \frac{V_A - V_B}{2} = \frac{U_{AB}}{2} = \frac{100}{2} = \SI{50}{\volt}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[analyse complète d'un spectre]
\begin{enumerate}[label=\alph*)]
  \item Les lignes \emph{partent} de $A$ (donc $A$ est \textbf{positive}) et \emph{aboutissent} à
        $B$ (donc $B$ est \textbf{négative}).
  \item Le nombre de lignes est proportionnel à la charge : $A$ émet $12$ lignes, $B$ n'en reçoit
        que $6$, donc
        \[ \frac{|q_A|}{|q_B|} = \frac{12}{6} = 2. \]
        $A$ est \textbf{deux fois} plus grande (en valeur absolue) que $B$.
  \item Les $6$ lignes restantes de $A$ ne trouvent pas de charge $-$ à rejoindre : elles
        \textbf{s'échappent vers l'infini} (la charge $-$ est trop petite pour toutes les capter).
\end{enumerate}
\end{corrige}

\begin{corrige}[vrai ou faux : lignes de champ et condensateur]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai}. Par définition, la ligne est orientée dans le sens de $\vv{E}$.
  \item \textbf{Faux}. Les lignes ne se croisent \emph{jamais} (direction unique du champ) ; là où
        $\vv{E}=\vv{0}$, aucune ligne ne passe.
  \item \textbf{Faux}. $\vv{E}$ pointe vers l'armature de potentiel le plus \emph{bas} (du $+$ vers
        le $-$).
  \item \textbf{Vrai}. Densité des lignes $=$ intensité du champ.
  \item \textbf{Vrai}. C'est le point neutre, au milieu du segment (champs opposés égaux).
\end{enumerate}
\textbf{Bilan : 3 propositions correctes} (a, d, e).
\end{corrige}
