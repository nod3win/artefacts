\begin{corrige}[quel est le signe de la charge ?]
\begin{enumerate}[label=\alph*)]
  \item Spectre I : les lignes \emph{divergent} (elles sortent de la charge). Le champ \emph{fuit}
        le $+$ : la charge est \textbf{positive}.
  \item Spectre II : les lignes \emph{convergent} (elles rentrent vers la charge). Le champ
        \emph{pointe vers} le $-$ : la charge est \textbf{négative}.
\end{enumerate}
\end{corrige}

\begin{corrige}[lecture d'un dipôle]
\begin{enumerate}[label=\alph*)]
  \item La charge \textbf{$X$} (à gauche) est positive.
  \item Les lignes \emph{partent} de $X$ et \emph{aboutissent} à $Y$. Elles naissent toujours sur
        le $+$ et se terminent sur le $-$ : donc $X$ est $+$ et $Y$ est $-$.
\end{enumerate}
\end{corrige}

\begin{corrige}[où le champ est-il le plus fort ?]
\begin{enumerate}[label=\alph*)]
  \item En $P$ (proche de la charge) : les lignes y sont plus \emph{resserrées}, et la densité des
        lignes mesure l'intensité du champ. (Cohérent avec $E = K|Q|/r^2$ : $E$ décroît quand $r$
        augmente.)
  \item En doublant la charge, il faut deux fois plus de lignes : elles deviennent plus
        \textbf{denses} (le champ est partout plus intense).
\end{enumerate}
\end{corrige}

\begin{corrige}[le condensateur plan]
\begin{enumerate}[label=\alph*)]
  \item $\vv{E}$ va \emph{de l'armature $+$ vers l'armature $-$}, c'est-à-dire du \textbf{haut vers
        le bas} (de $A$ vers $B$).
  \item Le champ pointe vers le potentiel le plus \emph{bas} : c'est donc l'armature de départ, la
        \textbf{plaque $A$} (positive, en haut), qui est au potentiel le plus élevé.
\end{enumerate}
\end{corrige}

\begin{corrige}[une propriété interdite]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Non}, ce spectre est \emph{impossible}.
  \item En un point, le champ $\vv{E}$ a une \textbf{direction unique} ; si deux lignes se
        croisaient en $X$, le champ y aurait \emph{deux} directions à la fois. Les lignes de champ
        ne se croisent donc jamais.
\end{enumerate}
\end{corrige}
