\begin{corrige}[quelle charge rend le champ horizontal ?]
Les quatre charges sont à la même distance de $O$ (demi-diagonale). Chaque champ est porté par une
diagonale ; on projette sur la verticale.
\begin{enumerate}[label=\alph*)]
  \item Un champ horizontal signifie que la somme des \textbf{composantes verticales} est nulle.
        Les trois charges connues donnent chacune une composante verticale \emph{vers le bas} :
        $\SI{2}{\nano\coulomb}$ ($+$, en haut : fuit vers le bas) $\to 2$ ;
        $\SI{1}{\nano\coulomb}$ ($+$, en haut) $\to 1$ ;
        $\SI{-1}{\nano\coulomb}$ ($-$, en bas : pointe vers le bas) $\to 1$. Total : $4$ (unités de
        $\propto|Q|$) vers le bas.
  \item La quatrième charge (bas-droite) doit fournir une composante verticale \emph{vers le haut}
        de valeur $4$. Un champ vers le haut depuis un coin \emph{bas} exige une charge
        \textbf{positive} (le champ la fuit), de grandeur $4$ :
        \[ \boxed{Q = \SI{+4}{\nano\coulomb}}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[deviner les signes à partir du champ]
\begin{enumerate}[label=\alph*)]
  \item Si $A$ et $B$ avaient le \emph{même} signe, leurs champs au milieu $P$ (mêmes grandeurs,
        mêmes distances) seraient opposés et se compenseraient : $\vv{E}(P) = \vv{0}$. Or le champ
        est non nul : les charges sont de \textbf{signes opposés}.
  \item Signes opposés : les deux champs pointent dans le \emph{même} sens le long de $[AB]$. Le
        champ \emph{fuit} le $+$ et \emph{pointe vers} le $-$. Comme $\vv{E}$ va vers la droite,
        $A$ (à gauche) est \textbf{positif} et $B$ (à droite) est \textbf{négatif}.
\end{enumerate}
\end{corrige}

\begin{corrige}[champ sur la médiatrice]
\begin{enumerate}[label=\alph*)]
  \item En $C$, les deux champs (mêmes charges, même distance $r$) sont opposés :
        $\vv{E}(C) = \vv{0}$.
  \item En $D$, chaque charge est à la distance $AD = \sqrt{r^2 + r^2} = r\sqrt2$. Par symétrie, les
        composantes \emph{horizontales} s'annulent ; les composantes \emph{verticales} s'ajoutent.
        L'angle entre $AD$ et la verticale vérifie $\cos\theta = r/(r\sqrt2) = 1/\sqrt2$, donc
        \[ E_D = 2\,E_A\cos\theta = 2\,E_A\times\frac{1}{\sqrt2} = \sqrt2\,E_A. \]
        (avec $E_A = Kq/(r\sqrt2)^2 = Kq/(2r^2)$.)
  \item En tout point de la médiatrice les composantes horizontales se compensent par symétrie : le
        champ total est toujours \textbf{porté par la médiatrice} (vertical), dirigé en s'éloignant
        du segment $[AB]$.
\end{enumerate}
\end{corrige}

\begin{corrige}[superposition de deux charges inégales]
\begin{enumerate}[label=\alph*)]
  \item Les deux charges sont positives : $\vv{E}_{C/B}$ fuit $C$ (vers la droite) et
        $\vv{E}_{O/B}$ fuit $O$ (vers la droite). Même sens $\Rightarrow$ ils s'\textbf{ajoutent}.
  \item \[ E(B) = K\,\frac{q}{(2r)^2} + K\,\frac{4q}{r^2}
             = \frac{Kq}{4r^2} + \frac{4Kq}{r^2}
             = \frac{Kq}{r^2}\!\left(\frac14 + 4\right) = \frac{17}{4}\,\frac{Kq}{r^2}. \]
  \item $E' = K\,\dfrac{4q}{(2r)^2} = \dfrac{Kq}{r^2}$. Le rapport vaut
        \[ \frac{E(B)}{E'} = \frac{\tfrac{17}{4}\,Kq/r^2}{Kq/r^2} = \frac{17}{4} \approx 4{,}25. \]
        Le champ total en $B$ est donc $17/4$ fois plus intense que celui de la seule charge $4q$
        en $D$.
\end{enumerate}
\end{corrige}
