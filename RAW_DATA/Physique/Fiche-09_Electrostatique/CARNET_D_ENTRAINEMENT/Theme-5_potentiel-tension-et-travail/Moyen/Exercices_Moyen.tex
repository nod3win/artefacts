\begin{exo}[travail et signe]
On déplace une charge $q = \SI{-5}{\micro\coulomb}$ entre deux points tels que
$U_{AB} = V_A - V_B = \SI{+40}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le travail $W_{B\to A} = q\,U_{AB}$ de la force électrique.
  \item Quel est le signe de $W_{B\to A}$ ? Que vaudrait $W_{A\to B}$ ?
\end{enumerate}
\end{exo}

\begin{exo}[énergie dans un condensateur]
Un condensateur plan est soumis à $U_{AB} = \SI{100}{\volt}$ ($V_A > V_B$), ses armatures séparées
de $d = \SI{4}{\centi\metre}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le champ électrique entre les armatures.
  \item Quel travail la force électrique fournit-elle pour déplacer une charge
        $q = \SI{+3}{\micro\coulomb}$ de $B$ vers $A$ ?
\end{enumerate}
\end{exo}

\begin{exo}[la gouttelette en équilibre]
Une petite particule de masse $m = \SI{3.2e-12}{\kilogram}$ reste \textbf{immobile} entre deux
plaques horizontales distantes de $d = \SI{2}{\centi\metre}$, soumises à $U_{AB} = \SI{400}{\volt}$,
la plaque \emph{supérieure} $A$ étant positive. On prend $g = \SI{10}{\newton\per\kilogram}$.
\begin{center}
\begin{tikzpicture}[scale=0.8,>=Stealth,font=\scriptsize]
  \fill[attncol!35] (-2.2,1.3) rectangle (2.2,1.42); \node[attncol,font=\bfseries] at (2.7,1.36){$+A$};
  \fill[defcol!35] (-2.2,-1.42) rectangle (2.2,-1.3); \node[defcol,font=\bfseries] at (2.7,-1.36){$-B$};
  \foreach \x in {-1.4,-0.7,0,0.7,1.4}{\draw[->,excol,line width=0.8pt] (\x,1.18)--(\x,-1.18);}
  \node[excol] at (-1.75,0.5){$\vv{E}$};
  \node[circle,fill=black!25,draw=black!55,minimum size=5.5mm,inner sep=0pt,font=\scriptsize] at (0,0) {$m$};
  \draw[->,black,line width=1pt] (0,-0.1)--(0,-0.95) node[right,font=\scriptsize]{$\vv{P}$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Quel est le signe de la charge de la particule ? Justifie par les sens de $\vv{E}$ et du poids.
  \item Calcule la valeur $|q|$ de sa charge, puis le nombre d'électrons en excès ou en défaut.
\end{enumerate}
\end{exo}

\begin{exo}[énergie faible, tension élevée]
Une charge $q = \SI{1}{\micro\coulomb}$ est déplacée sous une tension $U = \SI{1000}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'énergie $W = q\,U$ mise en jeu.
  \item Cette énergie est très petite. Peut-on en conclure que la tension est petite ? Explique.
\end{enumerate}
\end{exo}

\begin{exo}[le potentiel décroît linéairement]
Dans un condensateur plan, le potentiel varie \emph{linéairement} de $V_A = \SI{100}{\volt}$ (en $A$)
à $V_B = \SI{0}{\volt}$ (en $B$), les armatures étant distantes de $d = \SI{10}{\centi\metre}$.

  \draw[->] (0,0)--(0,2.4) node[above]{$V$ (V)};
  \draw[defcol,line width=1.2pt] (0,2)--(4,0);
  \node[left,font=\scriptsize] at (0,2){$100$};
  \node[below,font=\scriptsize] at (4,0){$10$};
  \node[below left,font=\scriptsize] at (0,0){$A$};
  \node[below,font=\scriptsize] at (4,-0.25){$B$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Quel est le potentiel au point situé à $\SI{2}{\centi\metre}$ de l'armature $A$ ?
  \item Calcule l'intensité du champ électrique entre les armatures.
\end{enumerate}
\end{exo}
