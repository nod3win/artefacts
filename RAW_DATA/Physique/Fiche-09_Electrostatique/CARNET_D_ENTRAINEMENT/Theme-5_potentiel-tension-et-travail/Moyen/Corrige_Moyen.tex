\begin{corrige}[travail et signe]
\begin{enumerate}[label=\alph*)]
  \item $W_{B\to A} = q\,U_{AB} = (\num{-5e-6})\times 40 = \SI{-2e-4}{\joule}$.
  \item $W_{B\to A} < 0$ : la force électrique s'oppose au déplacement de $B$ vers $A$ (il faut
        fournir un travail contre elle). Sur le trajet inverse,
        $W_{A\to B} = q\,U_{BA} = -W_{B\to A} = \SI{+2e-4}{\joule}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[énergie dans un condensateur]
\begin{enumerate}[label=\alph*)]
  \item $E = \dfrac{|U_{AB}|}{d} = \dfrac{100}{0{,}04} = \SI{2500}{\volt\per\metre}$.
  \item $W_{B\to A} = q\,U_{AB} = \num{3e-6}\times 100 = \SI{3e-4}{\joule}$ ($>0$ : la force
        électrique fournit ce travail à la charge positive).
\end{enumerate}
\end{corrige}

\begin{corrige}[la gouttelette en équilibre]
\begin{enumerate}[label=\alph*)]
  \item Le poids $\vv{P}$ est vers le bas ; pour l'équilibre, la force électrique $\vv{F}=q\vv{E}$
        doit être vers le \emph{haut}. Le champ $\vv{E}$ va de $A$ ($+$, en haut) vers $B$ ($-$),
        donc vers le \emph{bas}. Pour que $\vv{F}$ soit à l'opposé de $\vv{E}$, il faut
        $\boxed{q<0}$ : la particule est \textbf{négative}.
  \item Champ : $E = U_{AB}/d = 400/0{,}02 = \SI{2e4}{\volt\per\metre}$. Équilibre $|q|\,E = m g$ :
        \[ |q| = \frac{m g}{E} = \frac{\num{3.2e-12}\times 10}{\num{2e4}}
               = \frac{\num{3.2e-11}}{\num{2e4}} = \SI{1.6e-15}{\coulomb}. \]
        Nombre d'électrons : $N = \dfrac{|q|}{e} = \dfrac{\num{1.6e-15}}{\num{1.6e-19}} = \num{1e4}$
        électrons \emph{en excès} (charge négative).
\end{enumerate}
\end{corrige}

\begin{corrige}[énergie faible, tension élevée]
\begin{enumerate}[label=\alph*)]
  \item $W = q\,U = \num{1e-6}\times 1000 = \SI{1e-3}{\joule}$ (un millijoule, très peu).
  \item \textbf{Non}. Comme $U = W/q$, une petite énergie s'explique ici par la \emph{petitesse de
        la charge} ($\SI{1}{\micro\coulomb}$), pas par une petite tension : la tension vaut bien
        $\SI{1000}{\volt}$. Il faut raisonner sur la d.d.p., pas sur l'énergie brute.
\end{enumerate}
\end{corrige}

\begin{corrige}[le potentiel décroît linéairement]
\begin{enumerate}[label=\alph*)]
  \item Le potentiel chute de $\SI{100}{\volt}$ sur $\SI{10}{\centi\metre}$, soit
        $\SI{10}{\volt\per\centi\metre}$. À $\SI{2}{\centi\metre}$ de $A$ :
        \[ V = 100 - 10\times 2 = \SI{80}{\volt}. \]
  \item $E = \dfrac{|U_{AB}|}{d} = \dfrac{100}{0{,}10} = \SI{1000}{\volt\per\metre}$
        (pente de la droite $V(x)$).
\end{enumerate}
\end{corrige}
