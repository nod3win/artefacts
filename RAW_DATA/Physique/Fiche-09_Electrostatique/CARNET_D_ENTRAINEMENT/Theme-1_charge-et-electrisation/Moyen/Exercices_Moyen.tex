\begin{exo}[frottement : qui gagne, qui perd ?]
On frotte une tige d'ébonite avec un morceau de laine. Après frottement, la tige porte la charge
$q_{\text{tige}} = \SI{-3.2e-9}{\coulomb}$.
\begin{enumerate}[label=\alph*)]
  \item Combien d'électrons ont été transférés lors du frottement, et dans quel sens (laine
        $\to$ tige ou tige $\to$ laine) ?
  \item En déduire la charge portée par la laine, en justifiant par la conservation de la charge.
\end{enumerate}
\end{exo}

\begin{exo}[partages successifs par contact]
Trois sphères métalliques \emph{identiques} sont montées sur des manches isolants. La sphère $A$
porte $q_A = \SI{+12}{\nano\coulomb}$ ; les sphères $B$ et $C$ sont neutres.
\begin{center}
\begin{tikzpicture}[scale=0.85,>=Stealth,
    bille/.style={circle,draw=black!55,line width=0.8pt,minimum size=11mm,inner sep=0pt,font=\scriptsize}]
  \node[bille,fill=attncol!18] at (0,0) {$A$};
  \node[font=\scriptsize] at (0,-0.95) {$+12$};
  \node[bille,fill=black!5] at (1.7,0) {$B$};
  \node[font=\scriptsize] at (1.7,-0.95) {$0$};
  \node[bille,fill=black!5] at (3.4,0) {$C$};
  \node[font=\scriptsize] at (3.4,-0.95) {$0$};
  \node[font=\scriptsize,align=left,right] at (4.4,0)
    {(1) $A$ touche $B$\\ (2) puis $B$ touche $C$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item On met $A$ en contact avec $B$, puis on les sépare. Quelle charge porte chacune ?
  \item On met ensuite $B$ (avec sa nouvelle charge) en contact avec $C$. Quelle charge porte
        finalement $C$ ?
\end{enumerate}
\end{exo}

\begin{exo}[influence puis mise à la terre]
On approche un bâton \textbf{chargé négativement} d'une sphère conductrice neutre, \emph{sans la
toucher}. On relie alors un instant la sphère à la terre par un fil, on retire d'abord le fil,
\emph{puis} on éloigne le bâton.
\begin{enumerate}[label=\alph*)]
  \item Quel est le signe final de la charge de la sphère ? Explique le mécanisme.
  \item Quel serait le signe final si le bâton avait été chargé \emph{positivement} ?
\end{enumerate}
\end{exo}

\begin{exo}[charge et masse : combien d'électrons partis ?]
Une bille métallique perd des électrons jusqu'à porter la charge $q = \SI{+1}{\micro\coulomb}$.
On donne $e = \num{1.6e-19}\ \si{\coulomb}$ et $m_e = \num{9.1e-31}\ \si{\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Combien d'électrons la bille a-t-elle perdus ?
  \item De combien sa masse a-t-elle varié (en valeur et en signe) ?
\end{enumerate}
\end{exo}

\begin{exo}[bilan protons--électrons]
Un grain de poussière, initialement \textbf{neutre}, contient $\num{4.0e14}$ protons. Il capte
ensuite $\num{5.0e10}$ électrons.
\begin{enumerate}[label=\alph*)]
  \item Combien d'électrons contenait-il \emph{avant} de capter les électrons ?
  \item Quelle est sa charge finale (valeur et signe) ?
\end{enumerate}
\end{exo}
