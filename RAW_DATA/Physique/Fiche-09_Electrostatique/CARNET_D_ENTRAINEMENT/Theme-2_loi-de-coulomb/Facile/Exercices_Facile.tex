\begin{exo}[intensité de la force de Coulomb]
Deux charges $q_1 = \SI{+3}{\micro\coulomb}$ et $q_2 = \SI{+2}{\micro\coulomb}$ sont distantes de
$d = \SI{10}{\centi\metre}$ dans l'air ($K = \num{9e9}\ \si{\newton\metre\squared\per\coulomb\squared}$).
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité $F$ de la force d'interaction (pense à convertir les unités).
  \item Cette force est-elle attractive ou répulsive ?
\end{enumerate}
\end{exo}

\begin{exo}[attraction ou répulsion ?]
Pour chacun des couples ci-dessous, indique si la force est \emph{attractive} ou \emph{répulsive},
et dans quel sens pointe la force subie par la charge de gauche.
\begin{enumerate}[label=\alph*)]
  \item $q_1 = \SI{+5}{\nano\coulomb}$ (gauche) et $q_2 = \SI{-2}{\nano\coulomb}$ (droite).
  \item $q_1 = \SI{-4}{\nano\coulomb}$ (gauche) et $q_2 = \SI{-6}{\nano\coulomb}$ (droite).
\end{enumerate}
\end{exo}

\begin{exo}[jouer sur la charge et la distance]
Deux charges se repoussent avec une force d'intensité $F_0 = \SI{8}{\newton}$ à la distance $d$.
\begin{enumerate}[label=\alph*)]
  \item Que devient l'intensité si l'on \emph{double} la distance (charges inchangées) ?
  \item Que devient l'intensité si, à distance $d$ inchangée, on \emph{triple} l'une des deux
        charges ?
\end{enumerate}
\end{exo}

\begin{exo}[retrouver la distance]
Deux charges $q_1 = \SI{+5}{\micro\coulomb}$ et $q_2 = \SI{+2}{\micro\coulomb}$ se repoussent avec
une force d'intensité $F = \SI{9}{\newton}$.
\begin{enumerate}[label=\alph*)]
  \item Exprime la distance $d$ en fonction de $K$, $q_1$, $q_2$ et $F$.
  \item Calcule $d$ (en cm).
\end{enumerate}
\end{exo}

\begin{exo}[la plus grosse charge subit-elle plus ?]
Deux charges positives $q_1 = \SI{+2}{\micro\coulomb}$ et $q_2 = \SI{+8}{\micro\coulomb}$ sont
distantes de $d = \SI{20}{\centi\metre}$.
\begin{center}
\begin{tikzpicture}[scale=1,>=Stealth,font=\footnotesize,
    ch/.style={circle,minimum size=8mm,inner sep=0pt,text=white,font=\bfseries\small,fill=attncol}]
  \node[ch] (a) at (0,0) {$q_1$};
  \node[ch,minimum size=10mm] (b) at (4,0) {$q_2$};
  \draw[<->,black!55,dashed] (0,-0.7) -- (4,-0.7) node[midway,below]{$d=\SI{20}{\centi\metre}$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité de la force subie par $q_1$, puis celle subie par $q_2$.
  \item La charge $q_2$ (quatre fois plus grande) subit-elle une force plus intense ? Conclus.
\end{enumerate}
\end{exo}
