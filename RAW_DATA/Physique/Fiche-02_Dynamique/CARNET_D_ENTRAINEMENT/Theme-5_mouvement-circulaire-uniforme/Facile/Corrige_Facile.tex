\begin{corrige}[période et fréquence]
\begin{enumerate}[label=\alph*)]
  \item $T=\dfrac{\qty{10}{\second}}{5\ \text{tours}}=\qty{2}{\second}$ par tour.
  \item $f=\dfrac{1}{T}=\dfrac{1}{2}=\qty{0,5}{\hertz}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[vitesse linéaire]
$v=\dfrac{2\pi R}{T}=\dfrac{2\times\num{3,14}\times\num{0,5}}{1}\approx\qty{3,14}{\metre\per\second}$.
\end{corrige}

\begin{corrige}[vitesse angulaire]
\begin{enumerate}[label=\alph*)]
  \item $\omega=\dfrac{2\pi}{T}=\dfrac{2\times\num{3,14}}{2}\approx\qty{3,14}{\radian\per\second}$.
  \item $f=\dfrac{1}{T}=\dfrac{1}{2}=\qty{0,5}{\hertz}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[relier $v$ et $\omega$]
$v=\omega R=4\times 2=\qty{8}{\metre\per\second}$.
\end{corrige}

\begin{corrige}[accélération centripète]
$a_c=\dfrac{v^2}{R}=\dfrac{10^2}{5}=\dfrac{100}{5}=\qty{20}{\metre\per\second\squared}$ (dirigée vers le centre).
\end{corrige}
