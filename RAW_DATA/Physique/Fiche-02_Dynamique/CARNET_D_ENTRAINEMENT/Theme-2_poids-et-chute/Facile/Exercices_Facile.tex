\begin{exo}[calculer un poids]
Un sac a une masse $m=\qty{10}{\kilogram}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule son poids avec $g=\qty{9,81}{\metre\per\second\squared}$.
  \item Recalcule-le avec l'approximation $g\approx\qty{10}{\metre\per\second\squared}$.
\end{enumerate}
\end{exo}

\begin{exo}[masse ou poids sur la Lune ?]
Un objet a une masse $m=\qty{5}{\kilogram}$. On donne $g_{\text{Terre}}=\qty{10}{\metre\per\second\squared}$
et $g_{\text{Lune}}=\qty{1,6}{\metre\per\second\squared}$.
\begin{enumerate}[label=\alph*)]
  \item Quel est son poids sur la Terre ?
  \item Quel est son poids sur la Lune ?
  \item Quelle est sa masse sur la Lune ?
\end{enumerate}
\end{exo}

\begin{exo}[l'accélération de chute libre]
On lâche une bille de masse $m=\qty{2}{\kilogram}$ dans le vide (aucun frottement).
\begin{enumerate}[label=\alph*)]
  \item Quelle est son accélération ? (on prend $g=\qty{9,81}{\metre\per\second\squared}$)
  \item Cette accélération dépend-elle de la masse de la bille ?
\end{enumerate}
\end{exo}

\begin{exo}[retrouver la masse à partir du poids]
Le poids d'un objet vaut \qty{50}{\newton} en un lieu où $g=\qty{10}{\metre\per\second\squared}$.
Quelle est sa masse ?
\end{exo}

\begin{exo}[la plume et le marteau]
Dans une chambre à vide (sans air), on lâche exactement en même temps et de la même hauteur une plume
($\qty{5}{\gram}$) et un marteau ($\qty{1}{\kilogram}$).
\begin{enumerate}[label=\alph*)]
  \item Lequel touche le sol en premier ?
  \item Justifie à l'aide de l'expression de l'accélération de chute libre.
\end{enumerate}
\end{exo}
