\begin{exo}[le recul du canon, en chiffres]
Un canon de masse $m_{\text{canon}}=\qty{800}{\kilogram}$ tire un boulet de
$m_{\text{boulet}}=\qty{8}{\kilogram}$. La force exercée sur le boulet vaut $F=\qty{4000}{\newton}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'accélération du boulet.
  \item Calcule l'accélération du canon (même force, en réaction).
  \item Vérifie que le rapport des accélérations est bien $m_{\text{canon}}/m_{\text{boulet}}$.
\end{enumerate}
\end{exo}

\begin{exo}[la feuille de papier frappée]
Une feuille de papier A4 de masse $m=\qty{5}{\gram}$ est suspendue. Un élève la frappe : sa vitesse
passe de $0$ à \qty{30}{\metre\per\second} en \qty{0,06}{\second}.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'accélération de la feuille, puis la force exercée sur elle.
  \item En déduis l'intensité de la force que la feuille exerce en retour sur la main de l'élève.
\end{enumerate}
\end{exo}

\begin{exo}[la propulsion d'une fusée]
Une fusée de masse $m_{\text{fusée}}=\qty{200}{\kilogram}$ éjecte vers le bas $m_{\text{gaz}}=\qty{4}{\kilogram}$
de gaz. Au moment de l'éjection, le gaz est poussé avec une force $F=\qty{8000}{\newton}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'accélération communiquée au gaz.
  \item En déduis l'accélération de la fusée (intensité et sens).
\end{enumerate}
\end{exo}

\begin{exo}[la moto et la Terre]
Une moto avec son conducteur ($m=\qty{200}{\kilogram}$) accélère : la roue pousse la route (Terre,
$M=\qty{6e24}{\kilogram}$) vers l'arrière avec une force $F=\qty{800}{\newton}$ ; en réaction la route
pousse la moto vers l'avant avec la même force.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'accélération de la moto.
  \item Calcule l'accélération de la Terre. Pourquoi ne la voit-on pas bouger ?
\end{enumerate}
\end{exo}

\begin{exo}[sauter d'une barque]
Une personne de masse $m_1=\qty{60}{\kilogram}$ saute d'une barque de masse $m_2=\qty{120}{\kilogram}$,
initialement au repos. Au moment du saut, la personne et la barque se repoussent avec une force
$F=\qty{180}{\newton}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'accélération de la personne et celle de la barque.
  \item Dans quel sens part la barque, et laquelle des deux accélère le plus ?
\end{enumerate}
\end{exo}
