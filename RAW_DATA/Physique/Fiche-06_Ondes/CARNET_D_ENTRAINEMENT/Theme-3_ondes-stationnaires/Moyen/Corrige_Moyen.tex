\begin{corrige}[les trois premiers harmoniques]
On calcule le fondamental ($\lambda_1=2L$, $f_1=c/2L$) puis on multiplie par $n$.
\begin{enumerate}[label=\alph*)]
  \item $\lambda_1=2L=\SI{1.6}{\metre}$ ; $f_1=\dfrac{c}{2L}=\dfrac{160}{1{,}6}=\SI{100}{\hertz}.$
  \item $\lambda_2=\dfrac{2L}{2}=\SI{0.8}{\metre}$ ; $f_2=\dfrac{2c}{2L}=2f_1=\SI{200}{\hertz}.$
  \item $\lambda_3=\dfrac{2L}{3}\approx\SI{0.53}{\metre}$ ; $f_3=3f_1=\SI{300}{\hertz}$ :
        on a bien $f_2=2f_1$ et $f_3=3f_1$.
\end{enumerate}
\end{corrige}

\begin{corrige}[retrouver le fondamental]
Le nombre de fuseaux \emph{est} le numéro de l'harmonique.
\begin{enumerate}[label=\alph*)]
  \item 3 fuseaux $\Rightarrow$ $n=3$ (c'est la 3\textsuperscript{e} harmonique).
  \item $f=n f_1\Rightarrow f_1=\dfrac{f}{n}=\dfrac{180}{3}=\SI{60}{\hertz}.$
  \item $c=2L f_1=2\times 1{,}2\times 60=\SI{144}{\metre\per\second}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[compter les ventres selon la fréquence]
Pour une corde fixée aux deux bouts, le nombre de ventres égale le numéro d'harmonique, et
$f_n=n f_1$.
\begin{enumerate}[label=\alph*)]
  \item 2 ventres $\Rightarrow n=2$, donc $f_1=\dfrac{120}{2}=\SI{60}{\hertz}.$
  \item 5 ventres $\Rightarrow f_5=5 f_1=5\times 60=\SI{300}{\hertz}.$
  \item \SI{360}{\hertz}$=\dfrac{360}{60}f_1=6 f_1$ : on observe \textbf{6 ventres}.
\end{enumerate}
\end{corrige}

\begin{corrige}[de l'écart des nœuds à la fréquence]
Deux nœuds consécutifs sont séparés de $\lambda/2$.
\begin{enumerate}[label=\alph*)]
  \item $\dfrac{\lambda}{2}=\SI{0.25}{\metre}\Rightarrow \lambda=\SI{0.5}{\metre}.$
  \item $f=\dfrac{c}{\lambda}=\dfrac{200}{0{,}5}=\SI{400}{\hertz}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[corde à une extrémité libre]
Le bout libre impose un ventre : seuls les modes impairs existent, $\lambda_n=4L/(2n-1)$.
\begin{enumerate}[label=\alph*)]
  \item $\lambda_1=\dfrac{4L}{1}=4\times 0{,}5=\SI{2}{\metre}$ ;
        $f_1=\dfrac{c}{\lambda_1}=\dfrac{100}{2}=\SI{50}{\hertz}.$
  \item $\lambda_2=\dfrac{4L}{3}=\dfrac{2}{3}\approx\SI{0.67}{\metre}$ ;
        $f_2=\dfrac{c}{\lambda_2}=\dfrac{100}{2/3}=\SI{150}{\hertz}.$
  \item $f_2=150=3\times 50=3f_1$ : les fréquences propres sont seulement les
        \textbf{multiples impairs} de $f_1$ ($f_1, 3f_1, 5f_1,\dots$), pas tous les multiples.
\end{enumerate}
\end{corrige}
