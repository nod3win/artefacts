% THEME_TITLE: Ondes stationnaires et harmoniques d'une corde

\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 3 — Ondes stationnaires}\\[2pt]
{\large\color{primary} Nœuds, ventres et harmoniques d'une corde}
\end{center}

\medskip
Quand une onde se réfléchit au bout d'une corde et revient sur elle-même, l'onde incidente et
l'onde réfléchie se superposent : il apparaît une figure qui \emph{n'avance plus} mais
\textbf{oscille sur place}, l'\textbf{onde stationnaire}. Ce thème apprend à repérer ses
\textbf{nœuds} et \textbf{ventres}, et à calculer les \textbf{fréquences propres} d'une corde.

\begin{cle}[onde stationnaire : nœuds et ventres]
La superposition de l'onde incidente et de l'onde réfléchie crée :
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item des \textbf{nœuds} : points \emph{toujours immobiles} (élongation nulle) ;
  \item des \textbf{ventres} : points qui oscillent avec l'\emph{amplitude maximale}.
\end{itemize}
Un « \textbf{fuseau} » est la portion de corde entre deux nœuds (elle contient un ventre).
\end{cle}

\begin{cle}[les distances à connaître]
\[ \underbrace{\text{nœud} \to \text{nœud}}_{\lambda/2}\qquad
   \underbrace{\text{ventre} \to \text{ventre}}_{\lambda/2}\qquad
   \underbrace{\text{nœud} \to \text{ventre voisin}}_{\lambda/4} \]
Deux nœuds consécutifs (ou deux ventres) sont séparés de $\lambda/2$ ; un nœud et le ventre
voisin, de $\lambda/4$.
\end{cle}

\begin{cle}[corde fixée à ses deux extrémités]
Les deux bouts fixes imposent un \textbf{nœud} à chaque extrémité : la corde contient un nombre
\emph{entier} de fuseaux. Seules certaines longueurs d'onde « rentrent » :
\[ \boxed{\;\lambda_n = \frac{2L}{n}\;}\qquad
   \boxed{\;f_n = \frac{n\,c}{2L}=n\,f_1\;}\qquad (n=1,2,3,\dots) \]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item $L$ : longueur de la corde ; $c$ : célérité sur la corde ;
  \item $n=1$ : \textbf{mode fondamental} ($f_1$) ; $n=2$ : 2\textsuperscript{e} harmonique\ldots
  \item le mode $n$ possède \textbf{$n$ ventres} (fuseaux) et \textbf{$n+1$ nœuds}
        (extrémités comprises).
\end{itemize}
\end{cle}

\begin{methode}[calculer les harmoniques]
\begin{enumerate}[leftmargin=1.6em,topsep=2pt,itemsep=1.5pt,label=\arabic*.]
  \item Calcule le \textbf{fondamental} : $\lambda_1=2L$ puis $f_1=\dfrac{c}{2L}$.
  \item Les harmoniques suivants sont des \emph{multiples entiers} : $f_n=n\,f_1$,
        et $\lambda_n=\dfrac{2L}{n}=\dfrac{\lambda_1}{n}$.
  \item Pour le nombre de ventres, souviens-toi : $n$ ventres $\Leftrightarrow$ harmonique $n$
        $\Leftrightarrow$ fréquence $n\,f_1$.
\end{enumerate}
\end{methode}

\begin{cle}[corde avec une extrémité libre]
Une extrémité \emph{libre} impose un \textbf{ventre} (et non un nœud). Seuls les modes
« impairs » subsistent :
\[ \boxed{\;\lambda_n=\frac{4L}{2n-1}\;}\qquad
   f_n=\frac{(2n-1)\,c}{4L}\qquad (n=1,2,3,\dots). \]
\end{cle}

\begin{piege}[deux erreurs classiques]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item Un nœud et son ventre voisin sont séparés de $\lambda/4$, \emph{pas} de $\lambda/2$.
  \item Le nombre de ventres \emph{est} le numéro de l'harmonique : doubler la fréquence
        (2 bouts fixes) double le nombre de ventres.
\end{itemize}
\end{piege}
