\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 2 — Onde sinusoïdale}\\[2pt]
{\large\color{primary} Équation, pulsation, nombre d'onde, déphasage}
\end{center}

\medskip
Quand la source oscille de façon \emph{sinusoïdale}, chaque point du milieu reproduit ce
mouvement avec du retard. L'onde est alors décrite par une seule équation, fonction du temps
$t$ \emph{et} de la position $x$. Ce thème apprend à \textbf{lire} et \textbf{écrire} cette
équation, à en tirer $\omega$, $k$, $f$, $\lambda$, et à manier le \textbf{déphasage}.

\begin{cle}[l'équation d'une onde sinusoïdale progressive]
\[ \boxed{\;y(x,t)=A\,\sin\!\left(\omega t - k x + \varphi\right)\;} \]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item $A$ : \textbf{amplitude} (élongation maximale, toujours $>0$), en \si{\metre} ;
  \item $\omega$ : \textbf{pulsation}, en \si{\radian\per\second} ;
  \item $k$ : \textbf{nombre d'onde}, en \si{\radian\per\metre} ;
  \item $\varphi$ : \textbf{phase à l'origine}, en \si{\radian} ;
  \item $(\omega t - kx + \varphi)$ : \textbf{phase} de l'onde.
\end{itemize}
Le signe $-kx$ traduit une propagation vers les $x$ \emph{croissants}.
\end{cle}

\begin{cle}[les deux « ponts » vers T, f, $\lambda$, v]
\[ \boxed{\;\omega = 2\pi f = \frac{2\pi}{T}\;}\qquad
   \boxed{\;k=\frac{2\pi}{\lambda}\;}\qquad
   \frac{\omega}{k}=\lambda f = v \]
$\omega$ = « radians balayés par seconde » (un tour $2\pi$ par période)~;
$k$ = « radians par mètre » (un tour $2\pi$ par longueur d'onde).
\end{cle}

\begin{methode}[écrire l'équation à partir de A, f et $\lambda$ (ou v)]
\begin{enumerate}[leftmargin=1.6em,topsep=2pt,itemsep=1.5pt,label=\arabic*.]
  \item Calcule $\omega = 2\pi f$ (si on te donne $T$ : $\omega=2\pi/T$).
  \item Calcule $k = 2\pi/\lambda$ (si on te donne $v$ : $\lambda=v/f$ d'abord).
  \item Reporte : $y = A\sin(\omega t - kx)$ (ajoute $+\varphi$ si une phase à l'origine est imposée).
\end{enumerate}
\emph{Lecture inverse :} dans $y=A\sin(\omega t - kx)$, le facteur devant $t$ est $\omega$,
celui devant $x$ est $k$ ; on remonte à $f=\omega/2\pi$ et $\lambda=2\pi/k$.
\end{methode}

\begin{cle}[vitesse et accélération d'un point du milieu]
En dérivant $y$ \emph{par rapport au temps} (à $x$ fixé) :
\[ v_{\mathrm P}=A\omega\cos(\omega t - kx),\qquad
   a_{\mathrm P}=-A\omega^{2}\sin(\omega t - kx)=-\omega^{2}\,y. \]
D'où les valeurs \textbf{maximales} :
\[ \boxed{\,v_{\max}=A\,\omega\,}\quad\text{(au passage par l'équilibre)},\qquad
   \boxed{\,a_{\max}=A\,\omega^{2}\,}\quad\text{(aux crêtes et aux creux)}. \]
\end{cle}

\begin{cle}[déphasage entre deux points (ou deux ondes)]
\[ \Delta\varphi = 2\pi\,\frac{\Delta t}{T}\quad(\text{décalage dans le temps}),
   \qquad
   \Delta\varphi = 2\pi\,\frac{d}{\lambda}\quad(\text{décalage dans l'espace}). \]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item \textbf{Concordance de phase} : $d=n\lambda$ $\Rightarrow$ $\Delta\varphi=2\pi n$
        (les points vibrent \emph{à l'identique}) ;
  \item \textbf{Opposition de phase} : $d=(2n+1)\dfrac{\lambda}{2}$ $\Rightarrow$
        $\Delta\varphi=(2n+1)\pi$ (l'un monte quand l'autre descend).
\end{itemize}
Repère utile : un décalage de $\lambda/4$ vaut $\pi/2$ ; de $\lambda/2$ vaut $\pi$.
\end{cle}

\begin{piege}[trois confusions fréquentes]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1.5pt]
  \item $\omega\ne f$ : il y a un facteur $2\pi$. Une pulsation \SI{100}{\radian\per\second}
        n'est pas \SI{100}{\hertz} !
  \item La vitesse $v_{\mathrm P}$ d'un point (transversale, elle oscille) n'est pas la
        célérité $v$ de l'onde (propagation, constante). Souvent $v_{\max}\ne v$.
  \item Pour connaître le \emph{sens} de déplacement d'un point, ne devine pas :
        un point « imite » le voisin situé du côté \emph{d'où vient} l'onde.
\end{itemize}
\end{piege}
