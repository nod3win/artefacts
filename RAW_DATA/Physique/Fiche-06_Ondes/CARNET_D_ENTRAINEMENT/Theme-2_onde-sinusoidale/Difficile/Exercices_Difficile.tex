\begin{exo}[dans quel sens le point P se déplace-t-il ?]
Une impulsion (une « bosse ») se propage vers la \textbf{droite} à la vitesse $\vv v$ le long
d'une corde. La figure fige la corde à un instant donné. Le point $P$ est situé sur le
\emph{flanc avant} (à droite) de la bosse.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \draw[black!55,line width=0.8pt](0,0)--(9,0)node[right,font=\footnotesize]{corde};
  \draw[defcol,line width=1.8pt,smooth,samples=120,domain=0.2:8]
        plot(\x,{1.5*exp(-((\x-3.6)/0.95)^2)});
  \draw[->,primary,line width=1.1pt](4.6,1.75)--(5.9,1.75)node[midway,above,black]{$\vv v$};
  \coordinate (P) at (4.72,0.68);
  \draw[black!45,dashed](4.72,0)--(P);
  \fill[primary] (P) circle (2.6pt);
  \node[right=2pt,font=\bfseries] at (P) {$P$};
  \draw[<->,black!45,line width=0.5pt](8.3,0)--(8.3,1.5)node[midway,right,font=\footnotesize]{$y$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item À l'instant suivant, le point $P$ va-t-il monter ou descendre ?
  \item Justifie avec la règle : « un point imite le voisin situé du côté d'où vient l'onde. »
\end{enumerate}
\end{exo}

\begin{exo}[d'un décalage spatial au déphasage]
Deux ondes de même amplitude et de même longueur d'onde $\lambda=\SI{8}{\metre}$ se propagent
vers les $x>0$. La photo ci-dessous les montre au même instant : l'onde 2 est décalée d'un
\textbf{quart de longueur d'onde} vers la droite par rapport à l'onde 1.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=13cm,height=5cm,
  xmin=0,xmax=12,ymin=-1.5,ymax=1.5,
  xtick={0,2,4,6,8,10,12},ytick=\empty,
  grid=both,grid style={black!12},axis lines=middle,
  xlabel={$x$ (m)},
  xlabel style={at={(axis description cs:0.5,-0.03)},anchor=north},
  tick label style={font=\footnotesize},samples=200,clip=false,
  legend style={at={(0.5,1.18)},anchor=north,legend columns=2,
     draw=black!30,font=\footnotesize},legend cell align=left]
  \addplot[defcol,line width=1.3pt,domain=0:12]{sin(deg(2*pi*x/8))};
  \addlegendentry{onde 1}
  \addplot[primary,line width=1.3pt,domain=0:12]{sin(deg(2*pi*(x-2)/8))};
  \addlegendentry{onde 2}
\end{axis}
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule le déphasage $\Delta\varphi$ entre les deux ondes.
  \item Quel décalage spatial faudrait-il pour qu'elles soient en \emph{opposition} de phase ?
\end{enumerate}
\end{exo}

\begin{exo}[de l'enregistrement à l'équation, puis aux vitesses]
On enregistre le mouvement d'un point d'une corde. La longueur d'onde est par ailleurs
$\lambda=\SI{0.5}{\metre}$.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=11cm,height=4.6cm,
  xmin=0,xmax=0.75,ymin=-5,ymax=5,
  xtick={0,0.125,0.25,0.375,0.5,0.625,0.75},
  xticklabels={$0$,,$0{,}25$,,$0{,}5$,,$0{,}75$},
  ytick={-4,0,4},
  grid=both,grid style={black!12},axis lines=left,
  xlabel={$t$ (s)},ylabel={$y$ (cm)},
  xlabel style={at={(axis description cs:0.5,-0.14)}},
  tick label style={font=\footnotesize},samples=200,clip=false]
  \addplot[primary,line width=1.3pt,domain=0:0.75]{4*sin(deg(2*pi*x/0.25))};
\end{axis}
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Lis l'amplitude $A$ et la période $T$ sur le graphe ; déduis-en $\omega$ et $k$.
  \item Écris l'équation $y(x,t)$ de l'onde (SI, propagation vers les $x>0$, phase nulle).
  \item Calcule la vitesse maximale $v_{\max}$ et l'accélération maximale $a_{\max}$ d'un point.
\end{enumerate}
\end{exo}

\begin{exo}[déphasage supérieur à un tour]
Une onde de longueur d'onde $\lambda=\SI{4}{\metre}$ se propage sur une corde. On considère deux
points de la corde séparés par la distance $d=\SI{10}{\metre}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le déphasage $\Delta\varphi=2\pi\,d/\lambda$ (laisse le résultat en fonction de $\pi$).
  \item Ramène ce déphasage dans l'intervalle $[0,2\pi[$ et conclus : concordance, opposition,
        ou ni l'un ni l'autre ?
\end{enumerate}
\end{exo}

\begin{exo}[le point peut-il dépasser l'onde ?]
Une onde sinusoïdale a pour amplitude $A=\SI{2}{\centi\metre}$, fréquence $f=\SI{50}{\hertz}$ et
longueur d'onde $\lambda=\SI{0.2}{\metre}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la célérité $v$ de l'onde, puis la vitesse maximale $v_{\max}$ d'un point du milieu.
  \item Lequel est le plus grand ?
  \item À partir de quelle amplitude $A$ aurait-on $v_{\max}=v$ ? (exprime $A$ en fonction de
        $\lambda$, puis calcule.)
\end{enumerate}
\end{exo}
