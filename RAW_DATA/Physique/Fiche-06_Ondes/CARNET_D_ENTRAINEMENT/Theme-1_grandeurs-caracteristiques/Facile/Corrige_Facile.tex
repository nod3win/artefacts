\begin{corrige}[période et fréquence]
Période et fréquence sont inverses l'une de l'autre : $T=1/f$ et $f=1/T$.
\begin{enumerate}[label=\alph*)]
  \item $T=\dfrac{1}{f}=\dfrac{1}{\SI{50}{\hertz}}=\SI{0.02}{\second}=\SI{20}{\milli\second}.$
  \item On convertit d'abord : $T=\SI{4}{\milli\second}=\SI{4e-3}{\second}$, puis
        $f=\dfrac{1}{T}=\dfrac{1}{\num{4e-3}}=\SI{250}{\hertz}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[lire la période sur un graphe]
Le graphe est en fonction du \emph{temps} : on y lit une période (durée d'un motif complet).
\begin{enumerate}[label=\alph*)]
  \item Le motif se répète tous les \SI{0.5}{\second} : $T=\SI{0.5}{\second}$.
  \item $f=\dfrac{1}{T}=\dfrac{1}{0{,}5}=\SI{2}{\hertz}.$
  \item L'élongation maximale lue sur l'axe vertical est $A=\SI{3}{\centi\metre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire la longueur d'onde sur un graphe]
Le graphe est en fonction de la \emph{position} : on y lit une longueur d'onde (période spatiale).
\begin{enumerate}[label=\alph*)]
  \item Le motif se répète tous les \SI{4}{\metre} : $\lambda=\SI{4}{\metre}.$
  \item L'amplitude est $A=\SI{2}{\centi\metre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[relier $\lambda$, $v$ et $f$]
On part de la relation unique $\lambda=v/f$, soit encore $v=\lambda f$.
\begin{enumerate}[label=\alph*)]
  \item $\lambda=\dfrac{v}{f}=\dfrac{\SI{340}{\metre\per\second}}{\SI{200}{\hertz}}
        =\SI{1.7}{\metre}.$
  \item $v=\lambda f=\SI{2}{\metre}\times\SI{5}{\hertz}=\SI{10}{\metre\per\second}.$
\end{enumerate}
\end{corrige}

\begin{corrige}[longitudinale ou transversale ?]
On compare la direction de \emph{vibration} à la direction de \emph{propagation} (portée par
$\vv v$, horizontale dans les deux cas).
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=3pt]
  \item \textbf{(1) Ressort} : les spires se resserrent puis s'écartent \emph{le long} du
        ressort, donc la vibration est \emph{parallèle} à $\vv v$ $\Rightarrow$ onde
        \textbf{longitudinale} (zones de compression / raréfaction, comme le son).
  \item \textbf{(2) Corde} : chaque point monte et descend \emph{perpendiculairement} à
        $\vv v$ $\Rightarrow$ onde \textbf{transversale}.
\end{itemize}
\end{corrige}
