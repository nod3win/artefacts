\begin{exo}[fentes de Young : interfrange et frange]
Le schéma représente un dispositif de fentes de Young. Données :
$\lambda=\SI{500}{\nano\metre}$, $a=\SI{1}{\milli\metre}$, $D=\SI{3}{\metre}$.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \draw[black,line width=1pt](0,-1.8)--(0,1.8);
  \draw[black,line width=1.4pt](6.4,-2)--(6.4,2.1)node[above,font=\footnotesize]{écran};
  \fill[black](0,0.5)circle(2pt)node[left,font=\footnotesize]{$S_2$};
  \fill[black](0,-0.5)circle(2pt)node[left,font=\footnotesize]{$S_1$};
  \draw[<->,black](-0.5,-0.5)--(-0.5,0.5)node[midway,left,font=\footnotesize]{$a$};
  \draw[dashed,black!55](0,0)--(6.4,0);
  \draw[<->,black](0,-1.75)--(6.4,-1.75)node[midway,below,font=\footnotesize]{$D$};
  \fill[primary](6.4,1.1)circle(2pt)node[right,font=\footnotesize]{$M$};
  \draw[defcol,line width=0.8pt](0,-0.5)--(6.4,1.1);
  \draw[defcol,line width=0.8pt](0,0.5)--(6.4,1.1);
  \draw[<->,black](6.7,0)--(6.7,1.1)node[midway,right,font=\footnotesize]{$x$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'interfrange $i$.
  \item Donne la position $x_2$ de la 2\textsuperscript{e} frange brillante.
\end{enumerate}
\end{exo}

\begin{exo}[nature de l'interférence en un point]
Deux sources cohérentes $S_1$ et $S_2$ émettent des ondes de longueur d'onde
$\lambda=\SI{3}{\centi\metre}$. On étudie un point $M$ défini par ses distances aux sources
$d_1=S_1M$ et $d_2=S_2M$.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[black](0,0.8)circle(2.2pt)node[left,font=\footnotesize]{$S_1$};
  \fill[black](0,-0.8)circle(2.2pt)node[left,font=\footnotesize]{$S_2$};
  \fill[primary](5.5,0.3)circle(2.4pt)node[right,font=\footnotesize]{$M$};
  \draw[defcol,line width=0.9pt](0,0.8)--(5.5,0.3)node[midway,above,font=\footnotesize,black]{$d_1$};
  \draw[excol,line width=0.9pt](0,-0.8)--(5.5,0.3)node[midway,below,font=\footnotesize,black]{$d_2$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Cas 1 : $d_1=\SI{15}{\centi\metre}$, $d_2=\SI{12}{\centi\metre}$. Calcule $\delta$ et
        donne la nature de l'interférence.
  \item Cas 2 : $d_1=\SI{13.5}{\centi\metre}$, $d_2=\SI{12}{\centi\metre}$. Même question.
\end{enumerate}
\end{exo}

\begin{exo}[remonter à la longueur d'onde]
Dans une expérience de Young, on mesure un interfrange $i=\SI{3}{\milli\metre}$ avec des fentes
distantes de $a=\SI{0.5}{\milli\metre}$ et un écran à $D=\SI{2.5}{\metre}$.
\begin{enumerate}[label=\alph*)]
  \item À partir de $i=\lambda D/a$, exprime $\lambda$ en fonction de $i$, $a$ et $D$.
  \item Calcule $\lambda$ (en \si{\metre} puis en \si{\nano\metre}).
\end{enumerate}
\end{exo}

\begin{exo}[du déphasage à la différence de marche]
On donne le déphasage $\Delta\varphi$ entre deux ondes cohérentes de longueur d'onde $\lambda$.
On rappelle $\Delta\varphi=2\pi\,\delta/\lambda$.
\begin{enumerate}[label=\alph*)]
  \item $\Delta\varphi=\pi$ : quelle est la nature de l'interférence, et que vaut $\delta$
        (en fonction de $\lambda$) ?
  \item $\Delta\varphi=2\pi$ : mêmes questions.
  \item $\Delta\varphi=\pi/2$ : mêmes questions.
\end{enumerate}
\end{exo}

\begin{exo}[éloigner l'écran]
Un dispositif de Young utilise $\lambda=\SI{600}{\nano\metre}$ et des fentes distantes de
$a=\SI{0.4}{\milli\metre}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'interfrange $i$ pour un écran à $D=\SI{1}{\metre}$.
  \item Calcule $i$ pour $D=\SI{2}{\metre}$.
  \item Comment varie l'interfrange lorsqu'on \emph{double} la distance $D$ ?
\end{enumerate}
\end{exo}
