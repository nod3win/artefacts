\begin{exo}[attraction Terre--Soleil]
La Terre ($M_T=\qty{6.0e24}{\kilogram}$) gravite autour du Soleil
($M_S=\qty{2.0e30}{\kilogram}$) à la distance moyenne $d=\qty{1.5e11}{\metre}$.
Calcule l'intensité de la force de gravitation exercée par le Soleil sur la Terre.
Détaille numérateur puis dénominateur.
\end{exo}

\begin{exo}[deux satellites de même masse]
Deux satellites $A$ et $B$ ont la \emph{même} masse. Leurs centres sont à
$d_A=\qty{9.0e6}{\metre}$ et $d_B=\qty{6.0e6}{\metre}$ du centre de la Terre.
\begin{enumerate}[label=\alph*)]
  \item Écris le rapport $\dfrac{F_A}{F_B}$ des forces subies, en fonction de $d_A$ et $d_B$.
  \item Calcule sa valeur numérique.
  \item Lequel des deux satellites subit la force la plus intense ?
\end{enumerate}
\end{exo}

\begin{exo}[retrouver la distance]
Deux masses identiques $m_A=m_B=\qty{1000}{\kilogram}$ s'attirent avec une force
d'intensité $F=\qty{6.67e-3}{\newton}$. Détermine la distance $d$ entre leurs centres.
\end{exo}

\begin{exo}[masses et distance modifiées]
Deux corps s'attirent avec une force $F_0=\qty{8.0e-9}{\newton}$. On multiplie la
masse $m_A$ par $4$ et, en même temps, la distance $d$ par $2$.
\begin{enumerate}[label=\alph*)]
  \item Par quel facteur la force est-elle multipliée à cause des masses seules ?
  \item Par quel facteur est-elle divisée à cause de la distance seule ?
  \item Quelle est la nouvelle intensité de la force ?
\end{enumerate}
\end{exo}

\begin{exo}[retrouver une masse]
Une masse $m_A=\qty{100}{\kilogram}$ et une masse inconnue $m_B$, distantes de
$d=\qty{1.0}{\metre}$, s'attirent avec une force $F=\qty{6.67e-7}{\newton}$.
Détermine la masse $m_B$.
\end{exo}
