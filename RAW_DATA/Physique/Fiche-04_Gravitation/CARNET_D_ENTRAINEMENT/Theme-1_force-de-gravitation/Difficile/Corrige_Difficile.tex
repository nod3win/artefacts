\begin{corrige}[la pomme et la Terre --- vrai ou faux ?]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai.} La loi $F=\mathcal{G}\,m_A m_B/d^2$ est \emph{symétrique} :
        elle donne la même intensité pour la force Terre$\to$pomme et pomme$\to$Terre.
  \item \textbf{Faux.} Les deux intensités sont rigoureusement \emph{égales}. Ce qui
        diffère, c'est l'\emph{effet} : la même force donne à la pomme une grande
        accélération, mais à la Terre (énorme) une accélération imperceptible.
  \item \textbf{Faux.} Ces deux forces s'appliquent sur \emph{deux corps différents}
        (l'une sur la pomme, l'autre sur la Terre) : elles ne se compensent jamais. On
        n'additionne que des forces agissant sur un \emph{même} corps.
  \item \textbf{Vrai.} C'est exactement la paire action--réaction gravitationnelle
        (3\up{e} loi de Newton).
\end{enumerate}
Bilan : deux propositions correctes (a et d).
\end{corrige}

\begin{corrige}[deux satellites --- la racine du rapport]
\begin{enumerate}[label=\alph*)]
  \item Mêmes masses : $\ds \frac{F_A}{F_B}=\left(\frac{d_B}{d_A}\right)^2$.
  \item $\ds \sqrt{\frac{F_A}{F_B}}=\frac{d_B}{d_A}=\frac{6000}{8000}=\frac34=0{,}75$.
  \item $0{,}75<1$ donc $F_A<F_B$ : c'est \textbf{$B$}, le plus proche du centre, qui
        subit la force la plus intense.
\end{enumerate}
\end{corrige}

\begin{corrige}[le Soleil nous attire-t-il plus que la Terre ?]
\begin{enumerate}[label=\alph*)]
  \item En formant le rapport, $\mathcal{G}$ et $m$ se simplifient :
        \[
        \frac{F_S}{F_T}=\frac{\mathcal{G}M_S m/D^2}{\mathcal{G}M_T m/R_T^2}
        =\frac{M_S}{M_T}\times\left(\frac{R_T}{D}\right)^2.
        \]
  \item $\ds \frac{M_S}{M_T}=\frac{\num{2.0e30}}{\num{6.0e24}}=\num{3.3e5}$ ;
        $\ds \left(\frac{R_T}{D}\right)^2=\left(\frac{\num{6.4e6}}{\num{1.5e11}}\right)^2
        =(\num{4.3e-5})^2=\num{1.8e-9}$. D'où
        \[
        \frac{F_S}{F_T}\approx\num{3.3e5}\times\num{1.8e-9}\approx\num{6.1e-4}.
        \]
  \item $F_S/F_T\approx\num{6e-4}\ll 1$ : la \textbf{Terre} attire la personne environ
        $1600$ fois plus fort que le Soleil. Le Soleil est pourtant bien plus massif,
        mais il est très loin, et la loi en $1/d^2$ l'emporte.
\end{enumerate}
\end{corrige}

\begin{corrige}[combien de propositions correctes ?]
On sépare l'effet des masses (produit, au numérateur) et de la distance (au carré,
au dénominateur).
\begin{enumerate}[label=\alph*)]
  \item \textbf{Correcte.} Chaque masse $\times 2$ : produit $\times 4$, donc $F\times 4$.
  \item \textbf{Fausse.} $d\times 3\Rightarrow F\div 3^2=\div\,9$ (et non $\div 3$).
  \item \textbf{Correcte.} Une masse $\times 2$ et $d\times 2$ : $F\to\dfrac{2}{2^2}F
        =\dfrac{F}{2}$, donc divisée par $2$.
  \item \textbf{Correcte.} $d\div 3\Rightarrow F\times 3^2=\times\,9$.
\end{enumerate}
\textbf{Trois} propositions sont correctes (a, c, d).
\end{corrige}
