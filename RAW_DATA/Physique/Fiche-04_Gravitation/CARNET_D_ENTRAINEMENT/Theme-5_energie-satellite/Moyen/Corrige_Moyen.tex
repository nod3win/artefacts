% ================= CORRIGÉ MOYEN =================

\begin{corrige}[le bilan énergétique complet]
\begin{enumerate}[label=\alph*)]
  \item $E_c=\dfrac{\mathcal{G}M_T m}{2r}=\dfrac{\num{8.0e16}}{2\times\num{8.0e6}}
        =\dfrac{\num{8.0e16}}{\num{1.6e7}}=\qty{5.0e9}{\joule}$.
  \item $E_p=-\dfrac{\mathcal{G}M_T m}{r}=-\dfrac{\num{8.0e16}}{\num{8.0e6}}
        =-\qty{1.0e10}{\joule}$.
  \item $E_m=E_c+E_p=\num{5.0e9}-\num{1.0e10}=-\qty{5.0e9}{\joule}=-E_c$. \checkmark
\end{enumerate}
\end{corrige}

\begin{corrige}[énergie cinétique par $\tfrac12 m v^2$]
\begin{enumerate}[label=\alph*)]
  \item $v=\sqrt{\dfrac{\num{4.0e14}}{\num{1.0e7}}}=\sqrt{\num{4.0e7}}
        \approx\qty{6.3e3}{\metre\per\second}$ (donc $v^2=\num{4.0e7}$).
  \item $E_c=\tfrac12 m v^2=\tfrac12\times 1000\times\num{4.0e7}=\qty{2.0e10}{\joule}$.
        (On retrouve bien $\mathcal{G}M_T m/2r=\num{4.0e14}\times 1000/(2\times\num{1.0e7})$.)
\end{enumerate}
\end{corrige}

\begin{corrige}[comparer deux énergies cinétiques]
\begin{enumerate}[label=\alph*)]
  \item $E_c\propto\dfrac1r$, donc $\ds \frac{E_{cA}}{E_{cB}}=\frac{r_B}{r_A}$.
  \item $\ds \frac{\num{2.0e7}}{\num{1.0e7}}=2$ : le satellite $A$ (le plus bas) a une
        énergie cinétique \emph{deux fois} plus grande que $B$.
\end{enumerate}
\end{corrige}

\begin{corrige}[retrouver le rayon]
On isole $r=\dfrac{\mathcal{G}M_T m}{|E_p|}$ :
\[
r=\frac{\num{4.0e14}\times 400}{\num{2.0e10}}=\frac{\num{1.6e17}}{\num{2.0e10}}
 =\qty{8.0e6}{\metre}.
\]
\end{corrige}

\begin{corrige}[retrouver la masse]
On isole $m=\dfrac{2r\,E_c}{\mathcal{G}M_T}$ :
\[
m=\frac{2\times\num{8.0e6}\times\num{1.0e10}}{\num{4.0e14}}
 =\frac{\num{1.6e17}}{\num{4.0e14}}=\qty{400}{\kilogram}.
\]
\end{corrige}
