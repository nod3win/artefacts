\begin{corrige}[l'énergie d'un satellite --- vrai ou faux]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vrai.} Il se déplace, donc $E_c>0$.
  \item \textbf{Vrai.} Orbite circulaire $\Rightarrow$ vitesse constante $\Rightarrow$
        $E_c$ constante.
  \item \textbf{Vrai.} $r$ constant $\Rightarrow$ $E_p$ constante ; la somme
        $E_c+E_p=E_m$ est donc constante sur toute l'orbite.
  \item \textbf{Faux.} $E_m=-E_c<0$ : l'énergie mécanique est \emph{négative} (satellite lié).
\end{enumerate}
Correctes : a, b, c.
\end{corrige}

\begin{corrige}[bilan d'un satellite géostationnaire]
\begin{enumerate}[label=\alph*)]
  \item $E_c=\dfrac{\mathcal{G}M_T m}{2r}=\dfrac{\num{4.0e17}}{2\times\num{4.22e7}}
        =\dfrac{\num{4.0e17}}{\num{8.44e7}}\approx\qty{4.74e9}{\joule}$.
  \item $E_p=-\dfrac{\mathcal{G}M_T m}{r}=-\dfrac{\num{4.0e17}}{\num{4.22e7}}
        \approx-\qty{9.48e9}{\joule}$.
  \item $E_m=E_c+E_p\approx\num{4.74e9}-\num{9.48e9}=-\qty{4.74e9}{\joule}=-E_c$. \checkmark
\end{enumerate}
\end{corrige}

\begin{corrige}[libérer un satellite]
\begin{enumerate}[label=\alph*)]
  \item $E_m=-\dfrac{\mathcal{G}M_T m}{2r}=-\dfrac{\num{4.0e14}\times 500}{2\times\num{1.0e7}}
        =-\dfrac{\num{2.0e17}}{\num{2.0e7}}=-\qty{1.0e10}{\joule}$.
  \item Pour amener $E_m$ à $0$, il faut fournir $\Delta E=0-E_m=+\qty{1.0e10}{\joule}$
        (soit exactement $E_c$). C'est le coût énergétique pour « libérer » le satellite.
\end{enumerate}
\end{corrige}

\begin{corrige}[lequel est le plus fortement lié ?]
\begin{enumerate}[label=\alph*)]
  \item $E_m\propto-\dfrac1r$, donc $\ds \frac{E_{mA}}{E_{mB}}=\frac{r_B}{r_A}$.
  \item $\ds \frac{\num{1.0e7}}{\num{2.0e7}}=0{,}5$ : $E_{mA}=0{,}5\,E_{mB}$, donc
        $E_{mB}$ est \emph{deux fois plus négative} que $E_{mA}$.
  \item C'est \textbf{$B$} (le plus bas) qui est le plus fortement lié : son $E_m$ est la
        plus négative, il faut lui fournir plus d'énergie pour le libérer.
\end{enumerate}
\end{corrige}
