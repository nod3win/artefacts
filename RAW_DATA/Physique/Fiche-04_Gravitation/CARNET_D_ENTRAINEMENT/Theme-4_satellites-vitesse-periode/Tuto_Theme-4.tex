% THEME_TITLE: Les satellites --- vitesse, période et orbite géostationnaire
\sisetup{per-mode=symbol}

Un satellite en orbite circulaire est en mouvement circulaire uniforme : la gravité
joue le rôle de \textbf{force centripète}. En égalant force de gravitation et force
centripète $\dfrac{mv^2}{r}$ (avec $r=R_T+h$), la masse du satellite se simplifie.

\begin{cle}[vitesse d'un satellite]
\[
\mathcal{G}\,\frac{M_T m}{(R_T+h)^2}=\frac{m\,v^2}{R_T+h}
\quad\Longrightarrow\quad
\boxed{\,v=\sqrt{\dfrac{\mathcal{G}\,M_T}{R_T+h}}\,}
\]
$v$ (en \si{\metre\per\second}) ne dépend \emph{pas} de la masse du satellite. Plus
l'orbite est \textbf{haute} (grand $r=R_T+h$), plus $v$ est \textbf{faible}.
\end{cle}

\begin{cle}[période --- 3\up{e} loi de Kepler]
En remplaçant $v=\dfrac{2\pi r}{T}$ dans $v^2=\dfrac{\mathcal{G}M_T}{r}$ :
\[
\boxed{\,T^2=\dfrac{4\pi^2\,(R_T+h)^3}{\mathcal{G}\,M_T}\,}
\qquad\text{soit}\qquad \frac{T^2}{r^3}=\frac{4\pi^2}{\mathcal{G}M_T}=\text{constante}.
\]
Le carré de la période est proportionnel au cube du rayon : plus un satellite est
loin, plus sa période est longue.
\end{cle}

\begin{cle}[satellite géostationnaire]
Il paraît \emph{immobile} vu du sol. Trois conditions : orbite dans le \textbf{plan de
l'équateur}, \textbf{même sens} que la Terre, et \textbf{période} $T\approx\qty{24}{\hour}$
(un jour). Il en résulte une \emph{unique} altitude : $h\approx\qty{35800}{\kilo\metre}$.
On a alors $\omega_{\text{sat}}=\omega_{\text{Terre}}$, avec
$v=\omega\,(R_T+h)$ et $\omega=\dfrac{2\pi}{T}$.
\end{cle}

\begin{methode}[calculer vitesse et période]
1) $r=R_T+h$ (en mètres). \quad 2) $v=\sqrt{\mathcal{G}M_T/r}=\sqrt{\num{4.0e14}/r}$.
\quad 3) $T=\dfrac{2\pi r}{v}$ (durée d'un tour).
\end{methode}

\begin{exemplebox}[orbite basse à $h=\qty{800}{\kilo\metre}$]
$r=\qty{7.2e6}{\metre}$ : \;
$v=\sqrt{\dfrac{\num{4.0e14}}{\num{7.2e6}}}=\sqrt{\num{5.56e7}}\approx\qty{7455}{\metre\per\second}
\approx\qty{7.5}{\kilo\metre\per\second}$ ; \;
$T=\dfrac{2\pi r}{v}=\dfrac{\num{4.52e7}}{7455}\approx\qty{6068}{\second}\approx\qty{101}{\minute}$.
\end{exemplebox}

\begin{piege}[« même $\omega$ » n'est pas « même $v$ » ; « plus haut » = « plus lent »]
Deux satellites géostationnaires et une maison sur l'équateur ont la \emph{même}
vitesse angulaire $\omega$ (même $T$), mais \emph{pas} la même vitesse linéaire
$v=\omega r$ : plus on est loin du centre, plus $v$ est grande. Attention aussi :
contrairement à l'intuition, un satellite \textbf{plus haut} va \textbf{moins vite}
($v$ décroît quand $r$ augmente).
\end{piege}
