% ================= CORRIGÉ FACILE =================

\begin{corrige}[vitesse d'un satellite]
\[
v=\sqrt{\frac{\mathcal{G}M_T}{r}}=\sqrt{\frac{\num{4.0e14}}{\num{1.0e7}}}
 =\sqrt{\num{4.0e7}}\approx\qty{6.3e3}{\metre\per\second}\;(\approx\qty{6.3}{\kilo\metre\per\second}).
\]
\end{corrige}

\begin{corrige}[vitesse angulaire]
\[
\omega=\frac{2\pi}{T}=\frac{2\pi}{6000}\approx\qty{1.05e-3}{\radian\per\second}.
\]
\end{corrige}

\begin{corrige}[de la vitesse angulaire à la vitesse linéaire]
\[
v=\omega\,r=\num{1.0e-3}\times\num{7.0e6}=\qty{7.0e3}{\metre\per\second}.
\]
\end{corrige}

\begin{corrige}[les conditions du géostationnaire]
\begin{enumerate}[label=\alph*)]
  \item \textbf{Vraie} : l'orbite doit être équatoriale.
  \item \textbf{Vraie} : il doit tourner dans le même sens que la Terre.
  \item \textbf{Vraie} : $T\approx\qty{24}{\hour}$ (période de rotation de la Terre).
  \item \textbf{Fausse} : ces conditions imposent une \emph{unique} altitude,
        $h\approx\qty{35800}{\kilo\metre}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[classer trois satellites]
$v=\sqrt{\mathcal{G}M_T/r}$ \emph{décroît} quand $r$ augmente. Comme $r_3<r_2<r_1$ :
\[
\boxed{v_3>v_2>v_1}.
\]
Le satellite le plus bas est le plus rapide.
\end{corrige}
