% THEME_TITLE: Le poids apparent --- l'expérience de l'ascenseur
\sisetup{per-mode=symbol}

Une balance n'affiche pas votre poids réel $mg$, mais la \textbf{force de réaction}
$N$ du support sur vous : c'est le \textbf{poids apparent}. Il change avec le
mouvement, alors que le poids réel $mg$, lui, ne change pas (si $g$ est constante).

\begin{cle}[poids apparent dans un ascenseur]
Sur l'individu (masse $m$) s'exercent le poids $mg$ (vers le bas) et la force normale
$N$ du plancher (vers le haut). Avec l'axe vertical orienté \emph{vers le haut}, la
2\up{e} loi de Newton donne $N-mg=ma$, d'où
\[
\boxed{\,N=mg+ma=m\,(g+a)\,}
\]
\begin{itemize}[leftmargin=1.5em,topsep=1pt,itemsep=0pt]
  \item $N$ : poids apparent (valeur lue sur la balance), en \si{\newton} ;
  \item $m$ : masse, en \si{\kilogram} ; \quad $g$ : pesanteur, en \si{\metre\per\second\squared} ;
  \item $a$ : accélération de l'ascenseur, comptée $>0$ \emph{vers le haut}, $<0$ vers le bas.
\end{itemize}
\end{cle}

apparent $N$ (bleu) varie avec l'accélération $\vv a$.}

\begin{methode}[résoudre un problème d'ascenseur]
1) Orienter l'axe vers le haut. \quad 2) Repérer le signe de $a$ (vers le haut $>0$,
vers le bas $<0$, chute libre $a=-g$). \quad 3) Appliquer $N=m(g+a)$.
\end{methode}

\begin{exemplebox}[l'ascenseur démarre vers le haut]
$m=\qty{70}{\kilogram}$, $g=\qty{9.8}{\metre\per\second\squared}$,
$a=+\qty{2}{\metre\per\second\squared}$ :
\[
N=m(g+a)=70\times(9{,}8+2)=70\times 11{,}8=\qty{826}{\newton}.
\]
La balance « voit » $826/9{,}8\approx\qty{84}{\kilogram}$ : on se sent plus lourd au
démarrage vers le haut.
\end{exemplebox}

\begin{piege}[« impesanteur » ne veut pas dire « plus de gravité »]
En chute libre ($a=-g$), $N=m(g-g)=0$ : la balance affiche $0$ et l'on flotte. Mais
le poids \emph{réel} $mg$ est intact ! C'est seulement le poids \emph{apparent} qui
s'annule, car support et personne tombent \emph{ensemble}. De même, un objet suspendu
dans une cabine en chute libre ne tend plus son fil : la \textbf{tension devient nulle}.
\end{piege}
