\begin{exo}[ascenseur à l'arrêt]
Une personne de masse $m=\qty{80}{\kilogram}$ est debout sur une balance dans un
ascenseur \emph{immobile}. On prend $g=\qty{9.8}{\metre\per\second\squared}$.
Quelle valeur $N$ la balance affiche-t-elle ?
\end{exo}

\begin{exo}[l'ascenseur accélère vers le haut]
Une personne de masse $m=\qty{60}{\kilogram}$ est dans un ascenseur qui accélère
\emph{vers le haut} avec $a=+\qty{2}{\metre\per\second\squared}$. On prend
$g=\qty{10}{\metre\per\second\squared}$. Calcule son poids apparent $N=m(g+a)$.
\end{exo}

\begin{exo}[l'ascenseur accélère vers le bas]
Même personne ($m=\qty{60}{\kilogram}$, $g=\qty{10}{\metre\per\second\squared}$), mais
l'ascenseur accélère maintenant \emph{vers le bas} : $a=-\qty{2}{\metre\per\second\squared}$.
Calcule son poids apparent.
\end{exo}

\begin{exo}[le câble casse]
La cabine tombe en chute libre : $a=-g$. Pour une personne de masse
$m=\qty{50}{\kilogram}$, que vaut le poids apparent $N=m(g+a)$ ? Que ressent-elle ?
\end{exo}

\begin{exo}[deviner l'état de l'ascenseur]
Une personne de masse $m=\qty{70}{\kilogram}$ ($g=\qty{10}{\metre\per\second\squared}$,
donc $mg=\qty{700}{\newton}$) lit sa balance. Pour chaque valeur, dis si l'ascenseur
est au repos/MRU, accélère vers le haut, ou accélère vers le bas.
\begin{enumerate}[label=\alph*)]
  \item $N=\qty{700}{\newton}$ ;
  \item $N=\qty{840}{\newton}$ ;
  \item $N=\qty{560}{\newton}$.
\end{enumerate}
\end{exo}
