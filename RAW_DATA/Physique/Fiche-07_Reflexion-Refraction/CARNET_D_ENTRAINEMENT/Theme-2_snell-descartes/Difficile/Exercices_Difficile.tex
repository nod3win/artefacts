\begin{exo}[la lame de verre à faces parallèles]
Un rayon lumineux traverse une lame de verre ($n=1{,}5$) à \textbf{faces parallèles}, d'épaisseur
$e=\qty{4}{cm}$, entourée d'air. Il arrive sur la première face sous un angle d'incidence
$\hat{\imath}=\ang{60}$.
\begin{center}
\begin{tikzpicture}[scale=1.05,>=Stealth]
  \fill[defcol!16] (-2.8,-0.8) rectangle (2.8,0.8);
  \draw[primary,line width=1.4pt] (-2.8,0.8) -- (2.8,0.8);
  \draw[primary,line width=1.4pt] (-2.8,-0.8) -- (2.8,-0.8);
  \node[font=\scriptsize] at (-2.3,0){verre};
  \node[font=\scriptsize] at (2.2,1.15){air}; \node[font=\scriptsize] at (2.2,-1.15){air};
  \draw[<->,black!55] (-3.05,-0.8) -- (-3.05,0.8); \node[left,font=\scriptsize] at (-3.05,0){$e$};
  \coordinate (E) at (-1.1,0.8); \coordinate (S) at (0.05,-0.8);
  \draw[->,line width=1pt] (-2.25,1.55) -- (E);
  \draw[line width=1pt,primary] (E) -- (S);
  \draw[->,line width=1pt] (S) -- (1.1,-1.55);
  \draw[dash pattern=on 3pt off 2pt,black!60] (E) -- (-1.1,1.55);
  \draw[dash pattern=on 3pt off 2pt,black!60] (E) -- (-1.1,0.05);
  \draw[dash pattern=on 3pt off 2pt,black!60] (S) -- (0.05,-0.05);
  \draw[dash pattern=on 3pt off 2pt,black!60] (S) -- (0.05,-1.55);
  \node[font=\scriptsize] at (-0.78,1.2){$\hat{\imath}$}; \node[font=\scriptsize] at (-0.78,0.35){$\hat r$};
  \fill (E) circle (1pt); \fill (S) circle (1pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Montre que le rayon émergent est \textbf{parallèle} au rayon incident ($\hat{\imath}'=\hat{\imath}$).
  \item L'angle de réfraction dans le verre est $\hat r$ tel que $\sin\hat r=\sin\hat{\imath}/n$.
        Sachant que le décalage latéral du rayon vaut $d=\dfrac{e\,\sin(\hat{\imath}-\hat r)}{\cos\hat r}$,
        calcule $d$.
\end{enumerate}
\end{exo}

\begin{exo}[un empilement de couches parallèles]
Un rayon issu de l'air ($n_0=1$) sous une incidence $\hat{\imath}_0=\ang{50}$ traverse une couche
d'eau ($n_1=1{,}33$), puis un bloc de verre ($n_2=1{,}5$). Les trois interfaces sont \textbf{parallèles}.
\begin{center}
\begin{tikzpicture}[scale=0.95,>=Stealth]
  \fill[defcol!8] (-2.8,1.4) rectangle (2.8,2.6);
  \fill[defcol!20] (-2.8,0) rectangle (2.8,1.4);
  \fill[defcol!34] (-2.8,-1.4) rectangle (2.8,0);
  \draw[primary,line width=1.2pt] (-2.8,1.4) -- (2.8,1.4);
  \draw[primary,line width=1.2pt] (-2.8,0) -- (2.8,0);
  \node[font=\scriptsize] at (2.25,2.0){air $n_0$};
  \node[font=\scriptsize] at (2.25,0.7){eau $n_1$};
  \node[font=\scriptsize] at (2.25,-0.7){verre $n_2$};
  \draw[dash pattern=on 3pt off 2pt,black!55] (-1.0,-1.4) -- (-1.0,2.6);
  \draw[->,line width=1pt] (-2.1,2.5) -- (-1.0,1.4);
  \node[font=\scriptsize] at (-0.72,1.75){$\hat{\imath}_0$};
  \fill (-1.0,1.4) circle (1pt); \fill (-1.0,0) circle (1pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle du rayon dans l'eau.
  \item Calcule l'angle du rayon dans le verre.
  \item Montre que l'angle dans le verre ne dépend \emph{pas} de la présence de la couche d'eau.
        Que faut-il connaître pour prévoir l'angle final ?
\end{enumerate}
\end{exo}

\begin{exo}[imposer la déviation]
Un rayon passe de l'air ($n_1=1$) au verre ($n_2=1{,}5$). On souhaite que la déviation
$D=\hat{\imath}_1-\hat{\imath}_2$ du rayon soit exactement égale à $\ang{20}$.

En posant $\hat{\imath}_2=\hat{\imath}_1-\ang{20}$ dans la loi de Snell–Descartes, détermine l'angle
d'incidence $\hat{\imath}_1$. \emph{(Indication : développe $\sin(\hat{\imath}_1-\ang{20})$ puis
regroupe pour obtenir $\tan\hat{\imath}_1$.)}
\end{exo}
