% THEME_TITLE: Réfraction — la loi de Snell–Descartes

\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 2 — La loi de Snell–Descartes}\\[2pt]
{\color{primary}\itshape Calculer un angle de réfraction, prévoir le sens de la déviation}
\end{center}

\begin{cle}[la loi de la réfraction]
Quand un rayon passe d'un milieu 1 (indice $n_1$) à un milieu 2 (indice $n_2$), il traverse le
\emph{dioptre} en \textbf{changeant de direction}. L'angle d'incidence $\hat{\imath}_1$ et l'angle
de réfraction $\hat{\imath}_2$ (tous deux mesurés \textbf{par rapport à la normale}) vérifient :
\[ \boxed{\,n_1\sin\hat{\imath}_1 = n_2\sin\hat{\imath}_2\,} \]
Le rayon incident, le rayon réfracté et la normale sont dans un même plan.
\end{cle}

\begin{cle}[dans quel sens le rayon est-il dévié ?]
Réécrivons la loi sous la forme $\sin\hat{\imath}_2=\dfrac{n_1}{n_2}\sin\hat{\imath}_1$ :
\begin{itemize}[leftmargin=1.6em,topsep=2pt,itemsep=1pt]
  \item si $n_2>n_1$ (on entre dans un milieu \emph{plus} réfringent) : $\hat{\imath}_2<\hat{\imath}_1$,
        le rayon \textbf{se rapproche de la normale} ;
  \item si $n_2<n_1$ (milieu \emph{moins} réfringent) : $\hat{\imath}_2>\hat{\imath}_1$, le rayon
        \textbf{s'éloigne de la normale}.
\end{itemize}
\end{cle}

\begin{methode}[calculer un angle de réfraction]
\begin{enumerate}[leftmargin=1.7em,topsep=2pt,itemsep=1pt]
  \item \textbf{Tracer la normale} et vérifier que l'angle donné est bien mesuré par rapport à elle
        (sinon convertir : $\hat{\imath}=90^\circ-(\text{angle avec le dioptre})$).
  \item \textbf{Isoler} l'inconnue : $\sin\hat{\imath}_2=\dfrac{n_1\sin\hat{\imath}_1}{n_2}$.
  \item \textbf{Calculer} $\hat{\imath}_2=\arcsin(\cdots)$.
  \item \textbf{Vérifier le sens} : le rayon doit se rapprocher de la normale si $n_2>n_1$, s'en
        éloigner si $n_2<n_1$.
\end{enumerate}
\end{methode}

\begin{exemplebox}[air $\to$ verre]
Air ($n_1=1$) vers verre ($n_2=1{,}5$), incidence $\hat{\imath}_1=\ang{50}$ :
\[ \sin\hat{\imath}_2=\frac{1\times\sin\ang{50}}{1{,}5}=\frac{0{,}766}{1{,}5}=0{,}511
   \;\Rightarrow\;\hat{\imath}_2=\arcsin(0{,}511)\approx\ang{30.7}. \]
On entre dans un milieu plus réfringent : $\hat{\imath}_2<\hat{\imath}_1$, cohérent.
\end{exemplebox}

\begin{piege}[deux fautes classiques]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item \textbf{Angle mal repéré} : un angle « à $\ang{40}$ du dioptre » vaut $\hat{\imath}=\ang{50}$
        par rapport à la normale. On mesure toujours \emph{par rapport à la normale}.
  \item \textbf{Rapport à l'envers} : la loi s'écrit $n_1\sin\hat{\imath}_1=n_2\sin\hat{\imath}_2$.
        Une valeur $\sin\hat{\imath}_2>1$ n'est pas une erreur de calcul : elle signale qu'il n'y a
        \emph{pas} de réfraction (réflexion totale, voir Thème 4).
\end{itemize}
\end{piege}
