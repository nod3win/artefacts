\begin{corrige}[air vers eau]
Loi de Snell–Descartes : $n_1\sin\hat{\imath}_1=n_2\sin\hat{\imath}_2$.
\begin{enumerate}[label=\alph*)]
  \item $\sin\hat{\imath}_2=\dfrac{n_1\sin\hat{\imath}_1}{n_2}=\dfrac{1\times\sin\ang{40}}{1{,}33}
        =\dfrac{0{,}643}{1{,}33}=0{,}483$, d'où $\hat{\imath}_2=\arcsin(0{,}483)\approx\ang{28.9}$.
  \item L'eau est plus réfringente que l'air ($n_2>n_1$) : le rayon \textbf{se rapproche} de la
        normale ($\hat{\imath}_2<\hat{\imath}_1$). Cohérent : $\ang{28.9}<\ang{40}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[air vers verre]
\[ \sin\hat{\imath}_2=\frac{1\times\sin\ang{60}}{1{,}5}=\frac{0{,}866}{1{,}5}=0{,}577
   \;\Rightarrow\;\hat{\imath}_2=\arcsin(0{,}577)\approx\ang{35.3}. \]
Le rayon se rapproche de la normale ($n_2>n_1$).
\end{corrige}

\begin{corrige}[eau vers air]
\begin{enumerate}[label=\alph*)]
  \item $\sin\hat{\imath}_2=\dfrac{n_1\sin\hat{\imath}_1}{n_2}=\dfrac{1{,}33\times\sin\ang{30}}{1}
        =1{,}33\times0{,}5=0{,}665$, d'où $\hat{\imath}_2=\arcsin(0{,}665)\approx\ang{41.7}$.
  \item L'air est moins réfringent que l'eau ($n_2<n_1$) : le rayon \textbf{s'éloigne} de la normale
        ($\hat{\imath}_2>\hat{\imath}_1$). Cohérent : $\ang{41.7}>\ang{30}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[retrouver l'angle d'incidence]
On isole $\hat{\imath}_1$ : $n_1\sin\hat{\imath}_1=n_2\sin\hat{\imath}_2$, donc
\[ \sin\hat{\imath}_1=\frac{n_2\sin\hat{\imath}_2}{n_1}=\frac{1{,}5\times\sin\ang{28}}{1}
   =1{,}5\times0{,}469=0{,}704 \;\Rightarrow\;\hat{\imath}_1=\arcsin(0{,}704)\approx\ang{44.8}. \]
On vérifie $\hat{\imath}_1>\hat{\imath}_2$ : normal, on entrait dans un milieu plus réfringent.
\end{corrige}

\begin{corrige}[angle donné par rapport au dioptre]
\begin{enumerate}[label=\alph*)]
  \item La normale est perpendiculaire au dioptre : $\hat{\imath}_1=\ang{90}-\ang{35}=\ang{55}$.
  \item $\sin\hat{\imath}_2=\dfrac{1\times\sin\ang{55}}{1{,}5}=\dfrac{0{,}819}{1{,}5}=0{,}546
        \;\Rightarrow\;\hat{\imath}_2=\arcsin(0{,}546)\approx\ang{33.1}$.
\end{enumerate}
\end{corrige}
