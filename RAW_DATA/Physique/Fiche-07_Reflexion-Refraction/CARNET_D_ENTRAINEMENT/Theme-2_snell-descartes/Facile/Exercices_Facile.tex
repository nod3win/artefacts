\begin{exo}[air vers eau]
Un rayon lumineux passe de l'air ($n_1=1$) à l'eau ($n_2=1{,}33$) sous un angle d'incidence
$\hat{\imath}_1=\ang{40}$ (mesuré par rapport à la normale).
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[defcol!12] (-2.4,-1.9) rectangle (2.4,0);
  \draw[primary,line width=1.5pt] (-2.4,0) -- (2.4,0);
  \node[primary,right,font=\scriptsize] at (2.4,0){dioptre};
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,-1.9) -- (0,1.9);
  \node[black!60,above,font=\scriptsize] at (0,1.9){normale};
  \node[font=\scriptsize] at (-1.9,1.2){air $n_1$};
  \node[font=\scriptsize] at (-1.9,-1.0){eau $n_2$};
  \draw[->,line width=1pt] ({-1.9*sin(40)},{1.9*cos(40)}) -- (0,0);
  \draw[line width=0.6pt] (0,0.8) arc (90:130:0.8);
  \node[font=\scriptsize] at (-0.42,0.95){$\hat{\imath}_1$};
  \fill (0,0) circle (1.2pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle de réfraction $\hat{\imath}_2$.
  \item Le rayon se rapproche-t-il ou s'éloigne-t-il de la normale ?
\end{enumerate}
\end{exo}

\begin{exo}[air vers verre]
Un rayon passe de l'air ($n_1=1$) au verre ($n_2=1{,}5$) sous un angle d'incidence
$\hat{\imath}_1=\ang{60}$.

Calcule l'angle de réfraction $\hat{\imath}_2$.
\end{exo}

\begin{exo}[eau vers air]
Un rayon part du fond d'un bassin et passe de l'eau ($n_1=1{,}33$) à l'air ($n_2=1$) sous un angle
d'incidence $\hat{\imath}_1=\ang{30}$.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[defcol!12] (-2.4,-1.9) rectangle (2.4,0);
  \draw[primary,line width=1.5pt] (-2.4,0) -- (2.4,0);
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,-1.9) -- (0,1.9);
  \node[black!60,above,font=\scriptsize] at (0,1.9){normale};
  \node[font=\scriptsize] at (-1.9,1.0){air $n_2$};
  \node[font=\scriptsize] at (-1.9,-1.2){eau $n_1$};
  \draw[->,line width=1pt] ({-1.9*sin(30)},{-1.9*cos(30)}) -- (0,0);
  \draw[line width=0.6pt] (0,-0.8) arc (270:240:0.8);
  \node[font=\scriptsize] at (-0.4,-0.95){$\hat{\imath}_1$};
  \fill (0,0) circle (1.2pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle de réfraction $\hat{\imath}_2$ dans l'air.
  \item Le rayon se rapproche-t-il ou s'éloigne-t-il de la normale ?
\end{enumerate}
\end{exo}

\begin{exo}[retrouver l'angle d'incidence]
Un rayon passe de l'air ($n_1=1$) au verre ($n_2=1{,}5$). On mesure un angle de réfraction
$\hat{\imath}_2=\ang{28}$.

Quel était l'angle d'incidence $\hat{\imath}_1$ ?
\end{exo}

\begin{exo}[angle donné par rapport au dioptre]
Un rayon arrive sur la surface d'un bloc de verre ($n_2=1{,}5$) en faisant un angle de $\ang{35}$
\emph{avec le dioptre} (la surface). Le milieu de départ est l'air ($n_1=1$).
\begin{enumerate}[label=\alph*)]
  \item Quel est l'angle d'incidence $\hat{\imath}_1$ (par rapport à la normale) ?
  \item Calcule l'angle de réfraction $\hat{\imath}_2$.
\end{enumerate}
\end{exo}
