% THEME_TITLE: Réflexion totale et angle limite

\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 4 — Réflexion totale et angle limite}\\[2pt]
{\color{primary}\itshape Fibres optiques, prismes à réflexion totale, périscopes}
\end{center}

\begin{cle}[l'angle limite]
Quand la lumière passe d'un milieu \textbf{plus} réfringent vers un milieu \textbf{moins} réfringent
($n_1>n_2$), le rayon réfracté s'éloigne de la normale. Il existe un angle d'incidence pour lequel
l'angle de réfraction atteint $\ang{90}$ (le rayon réfracté rase le dioptre) : c'est l'\textbf{angle
limite} $\hat{\imath}_{\text{lim}}$. En posant $\sin\hat{\imath}_2=1$ dans $n_1\sin\hat{\imath}_1=n_2\sin\hat{\imath}_2$ :
\[ \boxed{\;\hat{\imath}_{\text{lim}}=\arcsin\!\left(\dfrac{n_2}{n_1}\right)\;}\qquad(n_1>n_2). \]
\end{cle}

\begin{cle}[la condition de réflexion totale]
La \textbf{réflexion totale} (toute la lumière est réfléchie, il n'y a plus de rayon réfracté) se
produit si \emph{et seulement si} les \textbf{deux} conditions sont réunies :
\begin{enumerate}[label=(\arabic*),leftmargin=2em,topsep=2pt,itemsep=1pt]
  \item on part du milieu \textbf{le plus réfringent} vers le moins réfringent ($n_1>n_2$) ;
  \item l'angle d'incidence dépasse l'angle limite : $\hat{\imath}_1>\hat{\imath}_{\text{lim}}$.
\end{enumerate}
\end{cle}

\begin{methode}[tester la réflexion totale]
\begin{enumerate}[leftmargin=1.7em,topsep=2pt,itemsep=1pt]
  \item Vérifier qu'on part bien du milieu \textbf{le plus réfringent} (sinon : jamais de réflexion
        totale, la réfraction a toujours lieu).
  \item Calculer $\hat{\imath}_{\text{lim}}=\arcsin(n_2/n_1)$.
  \item Comparer : $\hat{\imath}_1>\hat{\imath}_{\text{lim}}\Rightarrow$ réflexion totale ;
        $\hat{\imath}_1<\hat{\imath}_{\text{lim}}\Rightarrow$ réfraction (le rayon sort).
  \item \emph{Astuce} : la réflexion est totale dès que $\dfrac{n_1}{n_2}\sin\hat{\imath}_1>1$ (le
        « $\sin\hat{\imath}_2$ » deviendrait $>1$, impossible).
\end{enumerate}
\end{methode}

\begin{exemplebox}[angle limite verre $\to$ air]
Verre ($n_1=1{,}5$) vers air ($n_2=1$) : $\hat{\imath}_{\text{lim}}=\arcsin(1/1{,}5)=\arcsin(0{,}667)
\approx\ang{41.8}$. Tout rayon frappant la surface verre/air à plus de $\ang{41.8}$ est
\emph{intégralement} réfléchi dans le verre.
\end{exemplebox}

\begin{exemplebox}[le prisme à $\ang{45}$ des jumelles]
Dans un prisme en verre à section triangle rectangle isocèle ($\ang{45}$–$\ang{45}$–$\ang{90}$), un
rayon entrant perpendiculairement frappe l'hypoténuse à $\ang{45}$. Comme $\ang{45}>\ang{41.8}$, la
réflexion y est totale : l'hypoténuse agit comme un \emph{miroir parfait}. C'est le principe des
jumelles et des périscopes.
\end{exemplebox}

\begin{piege}[le sens compte, et $n_2/n_1<1$]
La réflexion totale n'existe que du milieu \emph{dense} vers le \emph{moins dense}. De l'air vers le
verre, jamais de réflexion totale. Dans la formule, $n_2/n_1<1$ (petit indice divisé par grand) :
un rapport $>1$ signalerait une erreur de sens.
\end{piege}
