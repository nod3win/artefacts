\begin{exo}[angle limite verre $\to$ air]
Un rayon se propage dans du verre ($n_1=1{,}5$) et arrive sur la surface verre/air ($n_2=1$).

Calcule l'angle limite $\hat{\imath}_{\text{lim}}$ de ce dioptre.
\end{exo}

\begin{exo}[angle limite eau $\to$ air]
Un plongeur éclaire vers le haut la surface de l'eau ($n_1=1{,}33$), qui sépare l'eau de l'air
($n_2=1$).
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle limite $\hat{\imath}_{\text{lim}}$.
  \item Que se passe-t-il pour un rayon frappant la surface sous $\ang{60}$ ?
\end{enumerate}
\end{exo}

\begin{exo}[réflexion totale, oui ou non ?]
Un rayon dans le verre ($n_1=1{,}5$) frappe la surface verre/air ($n_2=1$) sous un angle d'incidence
$\hat{\imath}_1=\ang{50}$.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[defcol!30] (-2.6,-1.7) rectangle (2.6,0);
  \draw[primary,line width=1.5pt] (-2.6,0) -- (2.6,0);
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,-1.7) -- (0,1.5);
  \node[font=\scriptsize] at (-2.1,0.5){air};
  \node[font=\scriptsize] at (-2.1,-0.7){verre};
  \draw[->,line width=1pt] ({-1.6*sin(50)},{-1.6*cos(50)}) -- (0,0);
  \draw[line width=0.6pt] (0,-0.8) arc (270:230:0.8);
  \node[font=\scriptsize] at (0.55,-0.95){$\hat{\imath}_1$};
  \fill (0,0) circle (1.2pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Rappelle l'angle limite verre/air ($\hat{\imath}_{\text{lim}}=\arcsin(1/1{,}5)\approx\ang{41.8}$).
  \item Le rayon est-il réfracté (il sort) ou totalement réfléchi (il reste dans le verre) ?
\end{enumerate}
\end{exo}

\begin{exo}[dans quel sens ?]
On considère les deux passages possibles à travers la surface air/eau :
\begin{itemize}[leftmargin=1.5em,topsep=2pt]
  \item cas (A) : de l'air ($n=1$) vers l'eau ($n=1{,}33$) ;
  \item cas (B) : de l'eau ($n=1{,}33$) vers l'air ($n=1$).
\end{itemize}
Dans lequel de ces deux cas une \textbf{réflexion totale} peut-elle se produire ? Justifie en une
phrase.
\end{exo}

\begin{exo}[angle limite verre $\to$ eau]
Un rayon passe du verre ($n_1=1{,}5$) à l'eau ($n_2=1{,}33$).
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle limite $\hat{\imath}_{\text{lim}}$ de ce dioptre.
  \item Est-il plus grand ou plus petit que l'angle limite verre/air ($\approx\ang{41.8}$) ? Explique.
\end{enumerate}
\end{exo}
