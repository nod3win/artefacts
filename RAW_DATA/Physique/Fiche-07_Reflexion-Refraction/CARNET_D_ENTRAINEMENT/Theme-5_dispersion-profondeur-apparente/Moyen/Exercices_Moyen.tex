\begin{exo}[dispersion : rouge et bleu dans le verre]
Une lumière blanche passe de l'air ($n=1$) au verre sous un angle d'incidence $\hat{\imath}=\ang{60}$.
Le verre est dispersif : $n_{\text{rouge}}=1{,}51$ et $n_{\text{bleu}}=1{,}53$.
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[defcol!22] (-2.6,-1.8) rectangle (2.6,0);
  \draw[primary,line width=1.4pt] (-2.6,0)--(2.6,0);
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,-1.8)--(0,1.7);
  \node[font=\scriptsize] at (-2.05,0.55){air}; \node[font=\scriptsize] at (-2.05,-0.6){verre};
  \draw[->,line width=1.2pt] ({-1.75*sin(60)},{1.75*cos(60)})--(0,0);
  \node[font=\scriptsize,align=center] at (-1.5,1.15){lumière\\blanche};
  \draw[->,line width=1pt,red] (0,0)--({1.9*sin(35)},{-1.9*cos(35)});
  \draw[->,line width=1pt,blue] (0,0)--({1.9*sin(34.5)},{-1.9*cos(34.5)});
  \node[red,font=\scriptsize] at (1.45,-1.5){rouge}; \node[blue,font=\scriptsize] at (0.5,-1.75){bleu};
  \fill (0,0) circle (1.2pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'angle de réfraction du rouge, puis celui du bleu.
  \item Déduis-en l'écart angulaire entre les deux et indique quelle couleur est la plus déviée.
\end{enumerate}
\end{exo}

\begin{exo}[profondeur apparente d'une piscine]
Une piscine a une profondeur réelle $h=\qty{2.0}{m}$ (eau $n_1=1{,}33$). Un maître-nageur la regarde
presque à la verticale depuis le bord ($n_2=1$).
\begin{enumerate}[label=\alph*)]
  \item Calcule la profondeur apparente $h'$ du fond.
  \item De combien le fond paraît-il « remonté » ?
\end{enumerate}
\end{exo}

\begin{exo}[le mécanisme de l'arc-en-ciel]
L'arc-en-ciel se forme quand la lumière du Soleil traverse des gouttes d'eau.
\begin{enumerate}[label=\alph*)]
  \item Cite, dans l'ordre, les trois phénomènes optiques subis par un rayon dans une goutte.
  \item Pourquoi les couleurs se séparent-elles ? (Quelle propriété de l'eau est en jeu ?)
  \item L'arc-en-ciel pourrait-il exister si l'eau n'était pas dispersive ?
\end{enumerate}
\end{exo}

\begin{exo}[une marque sous une plaque de verre]
Une petite marque est gravée sous une plaque de verre ($n=1{,}5$) d'épaisseur $e=\qty{6.0}{cm}$. On
la regarde d'au-dessus, presque à la verticale (air $n=1$ au-dessus).
\begin{enumerate}[label=\alph*)]
  \item À quelle profondeur apparente $h'$ voit-on la marque ?
  \item De combien paraît-elle remontée ?
\end{enumerate}
\end{exo}

\begin{exo}[vitesses des couleurs dans le verre]
Dans un verre dispersif, $n_{\text{rouge}}=1{,}51$ et $n_{\text{bleu}}=1{,}53$. On prend
$c=\qty{3.0e8}{m/s}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule les célérités $v_{\text{rouge}}$ et $v_{\text{bleu}}$ dans ce verre.
  \item Pour une même source, quelle couleur a la plus \emph{courte} longueur d'onde \emph{dans le
        verre} ?
\end{enumerate}
\end{exo}
