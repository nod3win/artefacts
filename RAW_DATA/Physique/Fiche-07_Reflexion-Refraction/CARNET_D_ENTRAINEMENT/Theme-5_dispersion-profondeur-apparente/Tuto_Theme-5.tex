\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 5 — Dispersion et profondeur apparente}\\[2pt]
{\color{primary}\itshape Décomposer la lumière blanche ; l'objet immergé qui paraît remonté}
\end{center}

\begin{cle}[la dispersion : l'indice dépend de la couleur]
Un milieu est \textbf{dispersif} lorsque son indice $n$ \emph{dépend de la longueur d'onde} (donc de
la couleur). Dans le verre :
\[ n_{\text{bleu}}>n_{\text{vert}}>n_{\text{rouge}}
   \;\Rightarrow\; v_{\text{bleu}}<v_{\text{vert}}<v_{\text{rouge}}
   \;\Rightarrow\; \hat r_{\text{bleu}}<\hat r_{\text{vert}}<\hat r_{\text{rouge}}. \]
Le \textbf{bleu} est le \emph{plus} dévié (le plus proche de la normale) ; le \textbf{rouge} le
\emph{moins} dévié (le plus éloigné de la normale).
\end{cle}

\begin{cle}[la profondeur apparente]
Un objet immergé paraît \textbf{plus proche de la surface} qu'il ne l'est : ses rayons, en passant de
l'eau à l'air, s'éloignent de la normale, et l'\oe il en prolonge la direction. Pour une observation
quasi verticale :
\[ \boxed{\,h'=\dfrac{n_2}{n_1}\,h\,}\qquad
   \text{(de l'eau } n_1=1{,}33 \text{ vers l'air } n_2=1 : h'=h/1{,}33). \]
$h$ est la profondeur réelle, $h'$ la profondeur apparente ($h'<h$).
\end{cle}

\begin{methode}[les réflexes du thème]
\begin{enumerate}[leftmargin=1.7em,topsep=2pt,itemsep=1pt]
  \item \textbf{Dispersion} : classer les couleurs par indice ($n_{\text{bleu}}$ le plus grand) puis
        en déduire vitesses et déviations (le bleu tourne le plus).
  \item \textbf{Déviation $\ne$ dispersion} : \emph{tout} rayon est dévié ; seule une lumière
        \emph{polychromatique} est \emph{décomposée} (étalée en couleurs).
  \item \textbf{Profondeur apparente} : appliquer $h'=(n_2/n_1)\,h$ ; ne pas oublier que l'objet
        paraît \emph{remonté} de $h-h'$.
\end{enumerate}
\end{methode}

\begin{exemplebox}[profondeur apparente d'un caillou]
Caillou à $h=\qty{1.20}{m}$ sous l'eau ($n_1=1{,}33$), vu presque à la verticale :
$h'=\dfrac{1}{1{,}33}\times1{,}20=\qty{0.90}{m}$. Il paraît remonté de $\qty{30}{cm}$.
\end{exemplebox}

\begin{piege}[deux confusions fréquentes]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item « Le rouge est le plus éloigné de la normale » signifie qu'il fait le plus \emph{grand} angle
        $\hat r$, donc qu'il est le \emph{moins} dévié. C'est le \textbf{bleu} qui « tourne » le plus.
  \item Une lumière \textbf{rouge monochromatique} traverse un prisme : elle est \emph{déviée} mais
        \textbf{pas décomposée} (une seule couleur). Sans plusieurs longueurs d'onde, pas de spectre.
  \item L'arc-en-ciel n'est pas une simple réflexion : c'est \emph{réfraction + réflexion + réfraction}
        dans une goutte \emph{dispersive}.
\end{itemize}
\end{piege}
