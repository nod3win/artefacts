\begin{corrige}[la vitesse double]
La fréquence $f$ est la même dans les deux milieux, donc $\lambda=c_i/f\propto c_i$.
\begin{enumerate}[label=\alph*)]
  \item $\dfrac{\lambda_1}{\lambda_2}=\dfrac{c_1}{c_2}=\dfrac{c_1}{2c_1}=\dfrac12$, donc
        $\boxed{\lambda_1=\dfrac{\lambda_2}{2}}$.
  \item L'indice $n=c/c_i$ est le plus grand là où la vitesse est la plus \emph{petite} : c'est le
        \textbf{milieu 1} (où $c_1<c_2$) qui a le plus grand indice.
\end{enumerate}
\end{corrige}

\begin{corrige}[la longueur d'onde dans le verre]
\begin{enumerate}[label=\alph*)]
  \item $c_1=\dfrac{c}{n_1}=\qty{3.0e8}{m/s}$ ; $c_2=\dfrac{c}{n_2}=\dfrac{\qty{3.0e8}{m/s}}{1{,}5}
        =\qty{2.0e8}{m/s}$.
  \item $\dfrac{\lambda_2}{\lambda_1}=\dfrac{c_2}{c_1}=\dfrac{n_1}{n_2}=\dfrac{1}{1{,}5}$, donc
        $\lambda_2=\dfrac{\lambda_1}{1{,}5}=\dfrac{\qty{600}{nm}}{1{,}5}=\qty{400}{nm}$.
        (La longueur d'onde raccourcit dans le milieu plus réfringent.)
\end{enumerate}
\end{corrige}

\begin{corrige}[lire les vitesses sur un schéma]
Le rayon se rapproche de la normale : $\hat{\imath}_2<\hat{\imath}_1$.
\begin{enumerate}[label=\alph*)]
  \item D'après $n_1\sin\hat{\imath}_1=n_2\sin\hat{\imath}_2$, si $\hat{\imath}_2<\hat{\imath}_1$ alors
        $n_2>n_1$ : le \textbf{milieu 2 est le plus réfringent}.
  \item Comme $c_i=c/n$, un plus grand indice donne une plus petite vitesse :
        $\boxed{c_2<c_1}$. La lumière est plus lente dans le milieu 2.
\end{enumerate}
\end{corrige}

\begin{corrige}[bilan complet air $\to$ eau]
\begin{enumerate}[label=\alph*)]
  \item $c_2=\dfrac{c}{n_2}=\dfrac{\qty{3.0e8}{m/s}}{1{,}33}=\qty{2.26e8}{m/s}$.
  \item $\lambda_1=\dfrac{c_1}{f}=\dfrac{\qty{3.0e8}{m/s}}{\qty{5.0e14}{Hz}}=\qty{600}{nm}$ ;
        $\lambda_2=\dfrac{c_2}{f}=\dfrac{\qty{2.26e8}{m/s}}{\qty{5.0e14}{Hz}}=\qty{4.51e-7}{m}
        \approx\qty{451}{nm}$.
  \item La fréquence \textbf{ne change pas} : $f=\qty{5.0e14}{Hz}$ dans l'eau comme dans l'air.
\end{enumerate}
\end{corrige}

\begin{corrige}[la couleur change-t-elle sous l'eau ?]
\begin{enumerate}[label=\alph*)]
  \item $\lambda_{\text{eau}}=\dfrac{\lambda_0}{n}=\dfrac{\qty{589}{nm}}{1{,}33}\approx\qty{443}{nm}$.
  \item \textbf{Non}, la couleur ne change pas : elle dépend de la \emph{fréquence}, qui reste
        constante d'un milieu à l'autre. Seule la longueur d'onde a diminué ; l'\oe il du plongeur
        perçoit la même couleur (jaune du sodium).
\end{enumerate}
\end{corrige}
