\begin{corrige}[l'indice à partir de la vitesse]
\[ n=\frac{c}{c_i}=\frac{\qty{3.0e8}{m/s}}{\qty{2.0e8}{m/s}}=1{,}5. \]
L'indice est sans unité et bien supérieur à $1$.
\end{corrige}

\begin{corrige}[la vitesse à partir de l'indice]
\[ c_i=\frac{c}{n}=\frac{\qty{3.0e8}{m/s}}{2{,}42}=\qty{1.24e8}{m/s}. \]
Le diamant, très réfringent, ralentit beaucoup la lumière (moins de la moitié de $c$).
\end{corrige}

\begin{corrige}[longueur d'onde et fréquence]
\[ \lambda=\frac{c}{f}=\frac{\qty{3.0e8}{m/s}}{\qty{5.0e14}{Hz}}=\qty{6.0e-7}{m}=\qty{600}{nm}. \]
\end{corrige}

\begin{corrige}[ce qui change, ce qui ne change pas]
\begin{enumerate}[label=\alph*)]
  \item $f$ : \textbf{ne change pas} (imposée par la source).
  \item $c_i$ : \textbf{diminue} (l'eau, plus réfringente, ralentit la lumière : $c_i=c/n$).
  \item $\lambda$ : \textbf{diminue} (car $\lambda=c_i/f$ et $c_i$ diminue tandis que $f$ est fixe).
\end{enumerate}
\end{corrige}

\begin{corrige}[classer trois milieux]
La célérité vaut $c_i=c/n$ : plus $n$ est grand, plus $c_i$ est petite.
\begin{enumerate}[label=\alph*)]
  \item Vitesse décroissante : \textbf{air} ($n=1$) $>$ \textbf{eau} ($n=1{,}33$) $>$ \textbf{verre}
        ($n=1{,}5$).
  \item $\lambda=c_i/f$ suit la vitesse : $\lambda$ est la plus \emph{courte} là où la vitesse est la
        plus faible, c'est-à-dire dans le \textbf{verre} (indice le plus grand).
\end{enumerate}
\end{corrige}
