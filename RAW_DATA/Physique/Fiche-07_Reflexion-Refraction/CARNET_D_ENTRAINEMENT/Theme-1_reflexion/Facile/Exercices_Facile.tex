\begin{exo}[le rebond de base]
Un rayon lumineux frappe un miroir plan avec un angle d'incidence $\hat{\imath}=\ang{28}$, mesuré
\emph{par rapport à la normale}.
\begin{center}
\begin{tikzpicture}[scale=1.05,>=Stealth]
  \fill[pattern=north east lines,pattern color=black!40] (-2.8,-0.28) rectangle (2.8,0);
  \draw[black,line width=1.4pt] (-2.8,0) -- (2.8,0);
  \node[black!70,right,font=\footnotesize] at (2.8,-0.16){miroir};
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,0) -- (0,2.4);
  \node[black!60,above,font=\footnotesize] at (0,2.4){normale};
  \draw[->,line width=1.1pt] (-1.7,2.2) -- (0,0);
  \node[left,font=\footnotesize] at (-1.7,2.05){rayon incident};
  \draw[line width=0.7pt] (0,1) arc (90:127:1);
  \node[font=\footnotesize] at (-0.42,1.2){$\hat{\imath}$};
  \fill (0,0) circle (1.3pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Donne la valeur de l'angle de réflexion $\hat{r}$.
  \item Calcule l'angle formé \emph{entre} le rayon incident et le rayon réfléchi.
\end{enumerate}
\end{exo}

\begin{exo}[angle donné par rapport au miroir]
Un rayon lumineux arrive sur un miroir plan en faisant un angle de $\ang{35}$ \emph{avec la surface
du miroir} (et non avec la normale).
\begin{center}
\begin{tikzpicture}[scale=1.05,>=Stealth]
  \fill[pattern=north east lines,pattern color=black!40] (-2.8,-0.28) rectangle (2.8,0);
  \draw[black,line width=1.4pt] (-2.8,0) -- (2.8,0);
  \node[black!70,right,font=\footnotesize] at (2.8,-0.16){miroir};
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,0) -- (0,2.4);
  \node[black!60,above,font=\footnotesize] at (0,2.4){normale};
  \draw[->,line width=1.1pt] (-2.2,1.54) -- (0,0);
  \draw[line width=0.7pt] (-1,0) arc (180:145:1);
  \node[font=\footnotesize] at (-1.2,0.28){$\ang{35}$};
  \fill (0,0) circle (1.3pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Quel est l'angle d'incidence $\hat{\imath}$ (par rapport à la normale) ?
  \item Quel est l'angle de réflexion $\hat{r}$ ?
\end{enumerate}
\end{exo}

\begin{exo}[miroir posé au sol]
Un miroir plan est posé horizontalement sur le sol. Un rayon lumineux descend et le frappe en faisant
un angle d'incidence $\hat{\imath}=\ang{60}$ avec la normale (verticale).
\begin{center}
\begin{tikzpicture}[scale=1.0,>=Stealth]
  \fill[pattern=north east lines,pattern color=black!40] (-2.8,-0.28) rectangle (2.8,0);
  \draw[black,line width=1.4pt] (-2.8,0) -- (2.8,0);
  \node[black!70,right,font=\footnotesize] at (2.8,-0.16){sol (miroir)};
  \draw[dash pattern=on 3pt off 2pt,black!60] (0,0) -- (0,2.2);
  \node[black!60,above,font=\footnotesize] at (0,2.2){normale};
  \draw[->,line width=1.1pt] ({-2.2*sin(60)},{2.2*cos(60)}) -- (0,0);
  \draw[line width=0.7pt] (0,0.9) arc (90:150:0.9);
  \node[font=\footnotesize] at (-0.62,0.95){$\hat{\imath}$};
  \fill (0,0) circle (1.3pt);
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Donne l'angle de réflexion $\hat{r}$.
  \item Quel angle le rayon réfléchi fait-il avec le sol ?
\end{enumerate}
\end{exo}

\begin{exo}[incidence perpendiculaire]
Un rayon lumineux arrive sur un miroir plan \emph{perpendiculairement} à sa surface, c'est-à-dire le
long de la normale ($\hat{\imath}=\ang{0}$).
\begin{enumerate}[label=\alph*)]
  \item Que vaut l'angle de réflexion $\hat{r}$ ?
  \item Décris la direction du rayon réfléchi par rapport au rayon incident.
\end{enumerate}
\end{exo}

\begin{exo}[rayon réfléchi perpendiculaire à l'incident]
On souhaite que le rayon réfléchi reparte \textbf{perpendiculairement} au rayon incident, c'est-à-dire
que l'angle entre les deux rayons soit exactement $\ang{90}$.

Quel doit être l'angle d'incidence $\hat{\imath}$ ? (On rappelle que l'angle entre le rayon incident et
le rayon réfléchi vaut $2\hat{\imath}$.)
\end{exo}
