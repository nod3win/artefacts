\begin{corrige}[le miroir qui tourne]
\begin{enumerate}[label=\alph*)]
  \item Faire tourner le miroir de $\alpha$ fait tourner la normale de $\alpha$. Le rayon incident
        étant fixe, l'angle d'incidence augmente de $\alpha$ : $\hat{\imath}'=\ang{30}+\ang{12}=\ang{42}$.
  \item Le rayon réfléchi fait toujours l'angle $\hat{r}'=\hat{\imath}'$ de l'autre côté de la normale ;
        comme la normale a tourné de $\alpha$ et que l'angle réfléchi a augmenté de $\alpha$, le rayon
        réfléchi tourne au total de $2\alpha=\ang{24}$.
        \[ \boxed{\text{Règle : un miroir qui tourne de }\alpha\text{ fait tourner le rayon réfléchi de }2\alpha.} \]
\end{enumerate}
\end{corrige}

\begin{corrige}[réflecteur en coin : deux miroirs perpendiculaires]
\begin{enumerate}[label=\alph*)]
  \item Loi de la réflexion en $A$ : $\hat{r}_1=\hat{\imath}_1=\ang{35}$.
  \item Les deux miroirs sont perpendiculaires, donc leurs normales aussi. Le rayon $AB$ fait
        $\ang{35}$ avec la normale de $M_1$, donc $\ang{90}-\ang{35}=\ang{55}$ avec la normale de $M_2$ :
        \[ \hat{\imath}_2=\ang{55}. \]
        (On retrouve la règle $\hat{\imath}_1+\hat{\imath}_2=\ang{90}$ pour deux miroirs perpendiculaires.)
  \item Prenons les axes $Ox$ (miroir $M_2$) et $Oy$ (miroir $M_1$). La réflexion sur $M_1$ (vertical)
        inverse la composante horizontale de la direction : $(u_x,u_y)\to(-u_x,u_y)$. La réflexion sur
        $M_2$ (horizontal) inverse la composante verticale : $(-u_x,u_y)\to(-u_x,-u_y)=-\vv{u}$. Le rayon
        sortant a donc la direction $-\vv{u}$ : il est \textbf{antiparallèle} au rayon entrant, quel que
        soit l'angle d'entrée. C'est le principe du catadioptre.
\end{enumerate}
\end{corrige}

\begin{corrige}[le périscope]
\begin{enumerate}[label=\alph*)]
  \item Le miroir est incliné à $\ang{45}$ ; sa normale fait donc $\ang{45}$ avec l'horizontale. Le
        rayon horizontal arrive sous un angle d'incidence $\hat{\imath}=\ang{45}$.
  \item Réflexion sur le miroir 1 : $\hat{r}=\ang{45}$, le rayon est dévié de $2\hat{\imath}=\ang{90}$
        et repart \textbf{verticalement vers le bas}. Il frappe le miroir 2 (aussi à $\ang{45}$) sous
        $\ang{45}$, subit une seconde déviation de $\ang{90}$ et repart \textbf{horizontalement}. La
        direction finale est donc \textbf{parallèle} au rayon entrant (et de même sens), simplement
        décalée vers le bas : c'est ce qui permet de « voir par-dessus » un obstacle.
\end{enumerate}
\end{corrige}

\begin{corrige}[se voir en entier dans un miroir]
On applique $\hat{\imath}=\hat{r}$ aux rayons issus du sommet du crâne et des pieds, qui doivent tous
deux parvenir à l'\oe il après réflexion.
\begin{enumerate}[label=\alph*)]
  \item Le rayon venant du sommet du crâne se réfléchit à mi-hauteur entre le crâne et les yeux ; le
        rayon venant des pieds se réfléchit à mi-hauteur entre les pieds et les yeux. La portion utile
        du miroir s'étend donc du milieu [pieds–yeux] au milieu [crâne–yeux], soit une hauteur égale à
        la \textbf{moitié de la taille} :
        \[ h_{\min}=\frac{H}{2}=\frac{\qty{1.80}{m}}{2}=\qty{0.90}{m}. \]
  \item \textbf{Non} : le calcul ne fait intervenir que les hauteurs, jamais la distance au miroir. La
        hauteur minimale reste $H/2$, que la personne soit près ou loin du miroir.
\end{enumerate}
\end{corrige}

\begin{corrige}[retrouver l'angle entre deux miroirs]
Par la loi de la réflexion, le rayon interne (entre les deux miroirs) fait le même angle avec chaque
miroir que, respectivement, le rayon entrant et le rayon sortant : il fait donc $\ang{30}$ avec le
miroir inférieur et $\ang{40}$ avec le miroir supérieur. Ces deux angles sont ceux du triangle formé
par les deux miroirs et le trajet interne, dont le troisième sommet ($O$) porte l'angle $\theta$. La
somme des angles d'un triangle vaut $\ang{180}$ :
\[ \theta+\ang{30}+\ang{40}=\ang{180}\;\Longrightarrow\;\boxed{\theta=\ang{110}}. \]
\end{corrige}
