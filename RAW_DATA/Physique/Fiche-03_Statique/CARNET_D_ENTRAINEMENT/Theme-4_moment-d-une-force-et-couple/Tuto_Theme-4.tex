\begin{center}
{\LARGE\bfseries\color{primarydark}Thème 4 — Le moment d'une force et le couple}\\[2pt]
{\color{primary}\itshape Statique — Fiche 3}
\end{center}
\vspace{1mm}

Une force ne fait pas que déplacer un solide : elle peut le faire \textbf{tourner}. La grandeur qui
mesure cet effet de rotation autour d'un axe $\Delta$ est le \textbf{moment} de la force.

\begin{cle}[moment d'une force]
\[
\boxed{\;\mathcal{M}_\Delta(\vv F) = \pm\, F\times b\;}
\]
$\mathcal{M}$ en newtons-mètres (\si{\newton\metre}) ; $F$ en \si{\newton} ; $b$ = \textbf{bras de
levier} = distance \emph{perpendiculaire} de l'axe $\Delta$ à la \textbf{ligne d'action} de la force,
en \si{\metre}. Le signe $\pm$ indique le sens de rotation.
\end{cle}

\begin{piege}[le bras de levier est la distance \emph{perpendiculaire}]
$b$ n'est \emph{pas} la distance « de l'axe au point d'application », mais la distance la plus courte
(perpendiculaire) de l'axe à la \emph{ligne d'action}. Si cette ligne \textbf{passe par l'axe}, alors
$b=0$ et le moment est \textbf{nul} : la force ne fait pas tourner le solide (c'est le cas de la
réaction au pivot).
\end{piege}

\begin{remarque}[convention de signe]
On choisit un sens positif, par exemple : moment $+$ si la force tend à faire tourner dans le
\textbf{sens horaire}, $-$ dans le sens anti-horaire. Le choix inverse est valable — l'essentiel est de
rester \emph{cohérent} dans tout le problème.
\end{remarque}

\begin{cle}[couple de forces]
Un \textbf{couple} est un ensemble de \emph{deux} forces parallèles, de \textbf{même intensité}, de
\textbf{sens opposé}, à lignes d'action \emph{distinctes}. Sa résultante est \emph{nulle} (aucune
translation) mais il fait \textbf{tourner} le solide. Son moment :
\[
\boxed{\;\mathcal{M} = F\times d\;}
\]
où $d$ est la distance \emph{entre les deux lignes d'action}. Le solide tourne jusqu'à ce que les deux
forces deviennent colinéaires ($d=0$, moment nul).
\end{cle}

\begin{exemplebox}[moment autour d'un pivot]
Une clé serre un boulon : on pousse avec $F=\qty{40}{N}$ à $b=\qty{0.25}{m}$ de l'axe.
$\mathcal{M}=F\times b = 40\times 0{,}25 = \qty{10}{N.m}$. En poussant \emph{plus loin} (grand $b$), on
obtient le même effet avec \emph{moins} de force : c'est le principe du bras de levier.
\end{exemplebox}

\begin{piege}[ne pas confondre travail et moment]
Le \textbf{travail} $W=F\times d$ (déplacement $d$ \emph{parallèle} à la force) est une \emph{énergie},
en \textbf{joules}. Le \textbf{moment} $\mathcal{M}=F\times b$ (bras $b$ \emph{perpendiculaire}) n'est
\emph{pas} une énergie, on le laisse en \si{\newton\metre}. Même unité $\si{\newton}\times\si{\metre}$,
mais grandeurs \textbf{différentes}.
\end{piege}
