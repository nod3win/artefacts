\begin{exo}[moment algébrique de plusieurs forces]
Une barre peut tourner autour d'un axe $\Delta$. Deux forces s'y appliquent : $\vv F_1$
($F_1=\qty{10}{N}$, bras $b_1=\qty{0.4}{m}$) tend à la faire tourner dans le sens \textbf{horaire} ;
$\vv F_2$ ($F_2=\qty{20}{N}$, bras $b_2=\qty{0.15}{m}$) dans le sens \textbf{anti-horaire}. On compte
positivement le sens horaire.
\begin{enumerate}[label=\alph*)]
  \item Calcule le moment algébrique de chaque force.
  \item Détermine le moment résultant et le sens de rotation de la barre.
\end{enumerate}
\end{exo}

\begin{exo}[bras de levier ou distance au point d'application ?]
Une force $F=\qty{50}{N}$ est appliquée à l'extrémité d'un levier, à une distance $L=\qty{0.4}{m}$ de
l'axe. Mais la force n'est pas perpendiculaire au levier : la distance \emph{perpendiculaire} de l'axe
à sa ligne d'action ne vaut que $b=\qty{0.3}{m}$.
\begin{enumerate}[label=\alph*)]
  \item Lequel de $L$ ou $b$ faut-il utiliser pour le moment ? Calcule ce moment.
  \item Quel moment obtiendrait-on si la force était \emph{perpendiculaire} au levier (bras $=L$) ?
        Conclus sur l'orientation la plus efficace.
\end{enumerate}
\end{exo}

\begin{exo}[est-ce un couple ?]
Sur un solide, deux forces verticales sont appliquées : $\qty{6}{N}$ vers le haut au point $x=0$ et
$\qty{6}{N}$ vers le bas au point $x=\qty{0.5}{m}$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1]
  \draw[primary!60,line width=1.4pt] (-0.6,0)--(3.1,0);
  \draw[->,line width=1.2pt,defcol] (0,0)--(0,1.1) node[above]{\scriptsize \qty{6}{N}};
  \draw[->,line width=1.2pt,attncol] (2.5,0)--(2.5,-1.1) node[below]{\scriptsize \qty{6}{N}};
  \draw[<->,black!55] (0,1.3)--(2.5,1.3) node[midway,fill=white,inner sep=1pt]{\scriptsize $d=\qty{0.5}{m}$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Ces deux forces forment-elles un couple ? Justifie (résultante, sens, lignes d'action).
  \item Calcule le moment de ce couple.
\end{enumerate}
\end{exo}

\begin{exo}[le volant]
On tourne un volant de rayon $r=\qty{0.2}{m}$ en poussant \emph{tangentiellement} avec chaque main :
une force $F=\qty{15}{N}$ vers le haut d'un côté, $F=\qty{15}{N}$ vers le bas de l'autre (aux deux
extrémités d'un diamètre).
\begin{enumerate}[label=\alph*)]
  \item Quelle est la distance $d$ entre les deux lignes d'action ?
  \item Calcule le moment du couple appliqué au volant.
\end{enumerate}
\end{exo}

\begin{exo}[travail et moment : même nombre, sens différent]
Une force d'intensité $F=\qty{25}{N}$ est utilisée de deux façons.
\begin{enumerate}[label=\alph*)]
  \item Elle déplace une caisse de \qty{2}{m} \emph{dans sa propre direction} : calcule le travail $W$
        (et donne son unité).
  \item Elle est appliquée à un bras de levier \emph{perpendiculaire} de \qty{2}{m} : calcule le moment
        $\mathcal{M}$ (et donne son unité).
  \item Les deux résultats sont numériquement égaux : s'agit-il pour autant de la même grandeur ?
\end{enumerate}
\end{exo}
