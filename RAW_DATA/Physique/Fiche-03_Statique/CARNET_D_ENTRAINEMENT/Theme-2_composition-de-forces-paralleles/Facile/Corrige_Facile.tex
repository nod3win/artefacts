\begin{corrige}[résultante de deux forces de même sens]
\begin{enumerate}[label=\alph*)]
  \item Forces de même sens $\Rightarrow$ les intensités s'ajoutent : $R = F_1+F_2 = 3+5 = \qty{8}{N}$.
  \item $\vv R$ est \textbf{parallèle} à $\vv F_1$ et $\vv F_2$, et de \textbf{même sens} qu'elles (vers le
        bas).
\end{enumerate}
\end{corrige}

\begin{corrige}[résultante de deux forces de sens opposé]
\begin{enumerate}[label=\alph*)]
  \item Sens opposés $\Rightarrow$ on soustrait : $F_r = F_2 - F_1 = 6 - 2 = \qty{4}{N}$.
  \item $\vv F_r$ a le sens de la \textbf{plus grande} force, c'est-à-dire celui de $\vv F_2$.
\end{enumerate}
\end{corrige}

\begin{corrige}[rapport des bras de levier]
\begin{enumerate}[label=\alph*)]
  \item De $F_1\times\overline{AO}=F_2\times\overline{BO}$ on tire
        $\dfrac{\overline{AO}}{\overline{BO}} = \dfrac{F_2}{F_1} = \dfrac{4}{2} = 2$.
  \item $O$ est plus proche de \textbf{$B$}. En effet $\overline{AO}=2\,\overline{BO}$ : la distance à $A$
        est le double de la distance à $B$. Le point d'application est toujours plus proche de la force
        la plus intense ($F_2$).
\end{enumerate}
\end{corrige}

\begin{corrige}[position du point d'application]
\begin{enumerate}[label=\alph*)]
  \item Même sens $\Rightarrow$ $O$ est entre $A$ et $B$, donc $\overline{AO}+\overline{BO}=\qty{30}{cm}$.
        La règle des moments donne $\dfrac{\overline{AO}}{\overline{BO}}=\dfrac{F_2}{F_1}=\dfrac{2}{1}=2$,
        soit $\overline{AO}=2\,\overline{BO}$. En reportant :
        \[ 2\,\overline{BO}+\overline{BO}=30 \;\Rightarrow\; 3\,\overline{BO}=30
        \;\Rightarrow\; \overline{BO}=\qty{10}{cm},\quad \overline{AO}=\qty{20}{cm}. \]
\end{enumerate}
\end{corrige}

\begin{corrige}[force équilibrante]
\begin{enumerate}[label=\alph*)]
  \item La \textbf{force équilibrante} : appliquée au même point $O$, elle doit annuler la résultante,
        donc $\sum\vv F=\vv 0$.
  \item Même intensité que $\vv R$ : $\qty{10}{N}$ ; sens \textbf{opposé} : dirigée \emph{vers le haut}.
\end{enumerate}
\end{corrige}
