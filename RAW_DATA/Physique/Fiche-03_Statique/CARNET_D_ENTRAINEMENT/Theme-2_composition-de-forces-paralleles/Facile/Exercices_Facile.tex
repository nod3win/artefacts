\begin{exo}[résultante de deux forces de même sens]
Sur une barre, deux forces parallèles \textbf{de même sens} sont appliquées : $F_1=\qty{3}{N}$ en $A$
et $F_2=\qty{5}{N}$ en $B$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1]
  \draw[primary!70,line width=1.6pt,fill=primary!12] (-2.5,-0.1) rectangle (2.5,0.1);
  \fill (-1.6,0) circle (1.4pt) node[above=3pt]{$A$};
  \fill (1.6,0) circle (1.4pt) node[above=3pt]{$B$};
  \draw[->,line width=1.2pt,defcol] (-1.6,0)--(-1.6,-1.2) node[below]{$\vv F_1$};
  \draw[->,line width=1.2pt,attncol] (1.6,0)--(1.6,-1.8) node[below]{$\vv F_2$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité $R$ de la résultante.
  \item Précise sa direction et son sens.
\end{enumerate}
\end{exo}

\begin{exo}[résultante de deux forces de sens opposé]
Deux forces parallèles \textbf{de sens opposé} agissent sur un solide : $F_1=\qty{2}{N}$ (en $A$) et
$F_2=\qty{6}{N}$ (en $B$).
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité $F_r$ de la résultante.
  \item Quel est son sens : celui de $\vv F_1$ ou celui de $\vv F_2$ ?
\end{enumerate}
\end{exo}

\begin{exo}[rapport des bras de levier]
Deux forces parallèles de \textbf{même sens} sont appliquées en $A$ ($F_1=\qty{2}{N}$) et en $B$
($F_2=\qty{4}{N}$). Le point d'application $O$ de la résultante vérifie
$F_1\times\overline{AO}=F_2\times\overline{BO}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule le rapport $\dfrac{\overline{AO}}{\overline{BO}}$.
  \item Le point $O$ est-il plus proche de $A$ ou de $B$ ? Pourquoi ?
\end{enumerate}
\end{exo}

\begin{exo}[position du point d'application]
Sur une tige, $F_1=\qty{1}{N}$ est appliquée en $A$ et $F_2=\qty{2}{N}$ en $B$, \textbf{de même sens},
avec $\overline{AB}=\qty{30}{cm}$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1]
  \draw[primary!70,line width=1.6pt] (-2.4,0)--(2.4,0);
  \fill (-2,0) circle (1.4pt) node[above=3pt]{$A$};
  \fill (2,0) circle (1.4pt) node[above=3pt]{$B$};
  \fill (0.5,0) circle (1.4pt) node[above=3pt]{$O$};
  \draw[->,line width=1pt,defcol] (-2,0)--(-2,-0.9) node[below]{$\vv F_1$};
  \draw[->,line width=1pt,attncol] (2,0)--(2,-1.4) node[below]{$\vv F_2$};
  \draw[<->,black!60] (-2,0.6)--(2,0.6) node[midway,fill=white,inner sep=1pt]{\scriptsize \qty{30}{cm}};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Écris la relation entre $\overline{AO}$ et $\overline{BO}$, puis calcule ces deux distances.
\end{enumerate}
\end{exo}

\begin{exo}[force équilibrante]
Deux forces parallèles de même sens ont pour résultante $\vv R$, d'intensité $R=\qty{10}{N}$, dirigée
\emph{vers le bas} et appliquée en un point $O$.
\begin{enumerate}[label=\alph*)]
  \item Quelle force unique, appliquée en $O$, maintient le solide en équilibre de translation ?
  \item Donne son intensité et son sens.
\end{enumerate}
\end{exo}
