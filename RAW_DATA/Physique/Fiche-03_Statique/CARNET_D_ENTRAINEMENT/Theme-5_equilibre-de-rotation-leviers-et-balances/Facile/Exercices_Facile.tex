\begin{exo}[levier équilibré]
Un levier repose sur un pivot central. À gauche, un poids $G_A=\qty{20}{N}$ est placé à
$d_A=\qty{0.3}{m}$ du pivot. À droite, un poids $G_B$ est à $d_B=\qty{0.2}{m}$.
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=1]
  \fill[primary!75] (0,0)--(-0.4,-0.7)--(0.4,-0.7)--cycle;
  \draw[primarydark,line width=2pt] (-2.6,0.12)--(2.6,0.12);
  \draw[->,line width=1.1pt,orange] (-2,0.12)--(-2,-0.9) node[below]{\scriptsize $G_A$};
  \draw[->,line width=1.1pt,orange] (1.4,0.12)--(1.4,-0.9) node[below]{\scriptsize $G_B$};
  \draw[<->,black!55] (-2,0.6)--(0,0.6) node[midway,fill=white,inner sep=1pt]{\scriptsize $d_A$};
  \draw[<->,black!55] (0,0.9)--(1.4,0.9) node[midway,fill=white,inner sep=1pt]{\scriptsize $d_B$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Écris la condition d'équilibre de rotation.
  \item Calcule l'intensité $G_B$.
\end{enumerate}
\end{exo}

\begin{exo}[à quelle distance ?]
Sur un levier à pivot, un poids $G_A=\qty{12}{N}$ est à $d_A=\qty{0.5}{m}$ du pivot. On veut
l'équilibrer avec un poids $G_B=\qty{30}{N}$.
\begin{enumerate}[label=\alph*)]
  \item À quelle distance $d_B$ du pivot faut-il placer $G_B$ ?
  \item Ce point est-il plus proche ou plus loin du pivot que $G_A$ ? Est-ce logique ?
\end{enumerate}
\end{exo}

\begin{exo}[deux moments opposés]
Une barre peut tourner autour d'un axe. Une force $\vv F_1$ ($F_1=\qty{6}{N}$, bras
$b_1=\qty{0.4}{m}$) crée un moment dans le sens horaire.
\begin{enumerate}[label=\alph*)]
  \item Calcule ce moment.
  \item Quelle intensité $F_2$, appliquée avec un bras $b_2=\qty{0.3}{m}$ dans le sens anti-horaire,
        maintient la barre en équilibre ?
\end{enumerate}
\end{exo}

\begin{exo}[la balançoire]
Sur une balançoire (pivot au centre), un enfant de poids \qty{300}{N} est assis à \qty{1.5}{m} du
pivot.
\begin{enumerate}[label=\alph*)]
  \item Quel moment son poids crée-t-il par rapport au pivot ?
  \item Un adulte de poids \qty{750}{N} veut l'équilibrer. À quelle distance du pivot doit-il s'asseoir ?
\end{enumerate}
\end{exo}

\begin{exo}[réaction et bras nul]
Une poutre légère horizontale repose sur deux appuis $A$ et $B$. Pour calculer les réactions, on
choisit d'écrire l'équilibre de rotation \emph{autour de l'appui $A$}.
\begin{enumerate}[label=\alph*)]
  \item Que vaut le bras de levier de la réaction $R_A$ par rapport à l'axe passant par $A$ ?
  \item Quel est alors le moment de $R_A$ autour de $A$ ? Quel est l'intérêt de choisir cet axe ?
\end{enumerate}
\end{exo}
