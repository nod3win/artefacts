\begin{corrige}[trouver une masse sur un levier]
\begin{enumerate}[label=\alph*)]
  \item Égalité des moments : $m_1 g\, d_1 = m_2 g\, d_2$. On simplifie par $g$ (présent des deux
        côtés) : $m_1 d_1 = m_2 d_2$.
  \item $m_2 = \dfrac{m_1 d_1}{d_2} = \dfrac{2\times 0{,}4}{0{,}5} = \qty{1.6}{kg}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[poutre sur deux appuis]
\begin{enumerate}[label=\alph*)]
  \item Charge au milieu : par symétrie, $R_A = R_B = \dfrac{100}{2} = \qty{50}{N}$.
  \item Moments autour de $A$ : $R_B\times 2 = 100\times 0{,}5 = 50 \Rightarrow R_B = \qty{25}{N}$.
        Translation : $R_A = 100 - R_B = \qty{75}{N}$. L'appui $A$, plus proche de la charge, en
        supporte la plus grande part.
\end{enumerate}
\end{corrige}

\begin{corrige}[la balance romaine]
\begin{enumerate}[label=\alph*)]
  \item Équilibre des moments autour du pivot : $W\times d_1 = 5\times d_2$, donc
        \[ W = \frac{5\times d_2}{d_1} = \frac{5\times 0{,}4}{0{,}1} = \qty{20}{N}. \]
        Un petit contrepoids éloigné équilibre un objet plus lourd placé près du pivot.
\end{enumerate}
\end{corrige}

\begin{corrige}[la brouette]
\begin{enumerate}[label=\alph*)]
  \item Moments autour de l'axe de la roue : $F\times 1{,}2 = 200\times 0{,}4 = 80$, donc
        \[ F = \frac{80}{1{,}2} \approx \qty{66.7}{N}. \]
  \item Le bras des poignées ($\qty{1.2}{m}$) est bien plus grand que celui de la charge
        ($\qty{0.4}{m}$) : un petit effort à grand bras équilibre une grande charge à petit bras.
        C'est l'avantage mécanique du levier.
\end{enumerate}
\end{corrige}

\begin{corrige}[où placer le pivot ?]
\begin{enumerate}[label=\alph*)]
  \item Le pivot doit être au point où les moments des deux poids se compensent :
        $m_1\,x = m_2\,(0{,}9-x)$, soit $3x = 6(0{,}9-x) \Rightarrow 3x = 5{,}4-6x \Rightarrow 9x=5{,}4
        \Rightarrow x=\qty{0.6}{m}$.
  \item Ce point est le \textbf{centre de gravité} de l'ensemble : posée sur son centre de gravité, la
        tige reste horizontale (le poids total passe par le pivot, moment nul).
\end{enumerate}
\end{corrige}
