\begin{exo}[quelles barres sont en équilibre ?]
Trois barres rigides identiques reposent chacune sur un \textbf{pivot} central (support ponctuel). Leur
centre de gravité $G$ (point rouge) occupe une position différente dans chaque cas.
\begin{center}
\begin{tikzpicture}[scale=0.8]
  \foreach \i/\gx/\lab in {0/0/1, 4.6/1.3/2, 9.2/-1.3/3}{
    \begin{scope}[shift={(\i,0)}]
      \draw[primary!70,line width=1.5pt] (-1.9,0)--(1.9,0);
      \fill[primary!75] (0,-0.08)--(-0.5,-0.9)--(0.5,-0.9)--cycle;
      \draw[black!55] (-0.9,-0.9)--(0.9,-0.9);
      \fill[attncol] (\gx,0.28) circle (2.2pt);
      \node[attncol,above] at (\gx,0.34){\scriptsize $G$};
      \node at (0,-1.4){\footnotesize Barre \lab};
    \end{scope}}
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Rappelle la condition d'équilibre d'une barre posée sur un pivot, en termes de position de $G$.
  \item Parmi les barres 1, 2 et 3, laquelle (lesquelles) est (sont) en équilibre ? Pour les autres,
        indique dans quel sens elles basculent.
\end{enumerate}
\end{exo}

\begin{exo}[centre de gravité d'une plaque en L]
Une plaque homogène mince a la forme d'un \textbf{L}, décomposable en deux rectangles : $R_1$
(base, $\qty{6}{cm}\times\qty{2}{cm}$) et $R_2$ (montant, $\qty{2}{cm}\times\qty{4}{cm}$). On place
l'origine au coin inférieur gauche.
\begin{center}
\begin{tikzpicture}[scale=0.6]
  \draw[primary!70,line width=1pt,fill=primary!12] (0,0)--(6,0)--(6,2)--(2,2)--(2,6)--(0,6)--cycle;
  \draw[black!35,densely dashed] (2,0)--(2,2);
  \node at (4,1){\scriptsize $R_1$}; \node at (1,4){\scriptsize $R_2$};
  \draw[->,black!60] (0,-0.6)--(6.3,-0.6) node[right]{\scriptsize $x$};
  \draw[->,black!60] (-0.6,0)--(-0.6,6.3) node[above]{\scriptsize $y$};
  \fill (3,1) circle (2pt) node[above right=-1pt]{\scriptsize $G_1$};
  \fill (1,4) circle (2pt) node[above right=-1pt]{\scriptsize $G_2$};
\end{tikzpicture}
\end{center}
La masse d'une partie est proportionnelle à son \textbf{aire}.
\begin{enumerate}[label=\alph*)]
  \item Donne l'aire et le centre $(x,y)$ de chaque rectangle.
  \item Calcule les coordonnées $(x_G,\,y_G)$ du centre de gravité de la plaque.
\end{enumerate}
\end{exo}

\begin{exo}[à partir de quelle inclinaison bascule-t-il ?]
Un bloc homogène (base $b=\qty{30}{cm}$, hauteur $h=\qty{60}{cm}$) est posé sur un plan que l'on
incline progressivement d'un angle $\theta$. Son centre de gravité est en son centre.
\begin{center}
\begin{tikzpicture}[scale=0.9]
  \begin{scope}[rotate=18]
    \draw[black!55,line width=1pt] (-0.6,0)--(3.4,0);
    \draw[fill=defcol!18,draw=defcol] (0.6,0) rectangle (2.2,1.6);
    \fill (1.4,0.8) circle (1.6pt) node[above right=-1pt]{\scriptsize $G$};
    \draw[orange,line width=0.8pt,densely dashed] (1.4,0.8)--++(0,-2);
  \end{scope}
  \draw[black!45] (2.6,-0.2) arc (0:18:2.2); \node at (3.1,0.15){\scriptsize $\theta$};
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*)]
  \item Le bloc bascule lorsque la verticale passant par $G$ dépasse l'arête basse. Montre que cela se
        produit quand $\tan\theta = \dfrac{b}{h}$.
  \item Calcule l'angle de basculement $\theta_c$.
  \item Pour rendre le bloc plus stable, vaut-il mieux l'élargir ou le rehausser ?
\end{enumerate}
\end{exo}

\begin{exo}[placer une masse pour viser un centre de gravité]
Sur une règle légère graduée, une masse $m_1=\qty{1}{kg}$ est fixée à l'origine ($x=0$). On souhaite
que le centre de gravité de l'ensemble se trouve exactement en $x_G=\qty{40}{cm}$.
\begin{enumerate}[label=\alph*)]
  \item Où faut-il placer une masse $m_2=\qty{4}{kg}$ pour y parvenir ?
  \item Si l'on suspendait la règle par ce point $x_G$, resterait-elle horizontale ? Justifie.
\end{enumerate}
\end{exo}
