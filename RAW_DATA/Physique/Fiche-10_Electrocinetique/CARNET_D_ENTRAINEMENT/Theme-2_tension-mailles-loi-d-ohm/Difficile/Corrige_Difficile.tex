\begin{corrige}[trois caractéristiques : lequel suit la loi d'Ohm ?]
\begin{enumerate}[label=\alph*)]
  \item Le conducteur \textbf{(1)} : sa caractéristique est une droite \emph{passant par l'origine},
        donc $U/I$ est constant (loi d'Ohm). (2) est une courbe (non ohmique) ; (3) est une droite mais
        qui \emph{ne passe pas par l'origine}.
  \item Pente de (1) : elle passe par $(2\,\si{\ampere},\,4\,\si{\volt})$, donc
        $R=\dfrac{4}{2}=\qty{2}{\ohm}$.
  \item \textbf{Faux}. Une droite ne suffit pas : pour être ohmique, elle doit \emph{passer par
        l'origine}. Celle de (3) a une ordonnée à l'origine non nulle, donc $U/I$ n'est pas constant.
\end{enumerate}
\end{corrige}

\begin{corrige}[loi des mailles avec deux sources]
\begin{enumerate}[label=\alph*)]
  \item En parcourant la maille, $E_1$ est motrice ($+$), $E_2$ est en opposition ($-$) et $R$ subit une
        chute $U_R=RI$ : \[ E_1-E_2-R\,I=0. \]
  \item $I=\dfrac{E_1-E_2}{R}=\dfrac{9-3}{2}=\qty{3}{\ampere}$.
  \item $U_R=R\,I=2\times3=\qty{6}{\volt}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[même courant, quelle tension ?]
Les deux résistances sont en série : elles sont traversées par le \emph{même} courant $I$. La loi
d'Ohm donne $U_1=R_1 I$ et $U_2=R_2 I$, donc à même $I$ la tension est proportionnelle à la
résistance. Comme $R_1<R_2$ :
\[ \boxed{U_1<U_2.} \]
(Les valeurs $\qty{1}{\volt}$ et $\qty{2}{\volt}$ des sources ne servent qu'à fixer le courant ; elles
n'inversent pas cette comparaison.)
\end{corrige}

\begin{corrige}[la lampe : une résistance qui change]
\begin{enumerate}[label=\alph*)]
  \item Point 1 : $\dfrac{U_1}{I_1}=\dfrac{2}{0{,}5}=\qty{4}{\ohm}$. \quad
        Point 2 : $\dfrac{U_2}{I_2}=\dfrac{6}{1}=\qty{6}{\ohm}$.
  \item \textbf{Non} : le rapport $U/I$ n'est pas constant ($4\,\Omega$ puis $6\,\Omega$), donc la lampe
        n'est pas ohmique.
  \item Quand $I$ augmente, le filament chauffe ; or la résistance d'un métal \emph{croît} avec la
        température. La « résistance » de la lampe augmente donc avec l'intensité : sa caractéristique
        s'incurve vers le haut.
\end{enumerate}
\end{corrige}
