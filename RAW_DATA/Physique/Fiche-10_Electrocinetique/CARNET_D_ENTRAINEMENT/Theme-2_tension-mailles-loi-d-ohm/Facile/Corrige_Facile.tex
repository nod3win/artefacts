\begin{corrige}[loi d'Ohm : trouver la tension]
Loi d'Ohm $U=R\,I$ :
\[ U=R\,I=20\times0{,}3=\qty{6}{\volt}. \]
\end{corrige}

\begin{corrige}[loi d'Ohm : trouver l'intensité]
$I=\dfrac{U}{R}=\dfrac{12}{40}=\qty{0.3}{\ampere}$.
\end{corrige}

\begin{corrige}[loi d'Ohm : déduire la résistance]
$R=\dfrac{U}{I}=\dfrac{9}{0{,}5}=\qty{18}{\ohm}$.
\end{corrige}

\begin{corrige}[brancher un voltmètre]
Le voltmètre se branche \textbf{en parallèle} (en dérivation) aux bornes de $L$ : il mesure la
différence de potentiel entre les deux bornes. Un ampèremètre, lui, se branche \textbf{en série}
(le courant doit le traverser) pour mesurer l'intensité.
\end{corrige}

\begin{corrige}[additivité des tensions en série]
En série, les tensions s'ajoutent (loi des mailles) :
\[ U_{AB}=U_1+U_2=4+6=\qty{10}{\volt}. \]
\end{corrige}
