\begin{center}
{\LARGE\bfseries\color{primarydark} Thème 4 — Puissance et énergie (loi de Joule)}\\[2pt]
{\color{primary}\itshape $P=UI=RI^2=U^2/R$, énergie de Joule et effet thermique}
\end{center}

\begin{cle}[puissance électrique]
La \textbf{puissance} est l'énergie échangée par unité de temps : $P=\dfrac{W}{t}=U\,I$, en
\textbf{watts} (\si{\watt}). Pour un conducteur \emph{ohmique}, trois écritures équivalentes :
\[ \boxed{\;P=U\,I=R\,I^{2}=\dfrac{U^{2}}{R}\;} \]
On passe de l'une à l'autre avec la loi d'Ohm $U=RI$.
\end{cle}

\begin{cle}[loi de Joule (énergie thermique)]
La chaleur libérée dans un conducteur de résistance $R$ parcouru par $I$ pendant $t$ vaut :
\[ \boxed{\;W_{\text{th}}=R\,I^{2}\,t = P\,t\;} \]
en \textbf{joules} (\si{\joule}). Elle provient des chocs des électrons contre les atomes : c'est
l'\textbf{effet Joule}.
\end{cle}

\begin{cle}[énergie et unités]
$W=P\,t$ : penser à mettre $t$ \textbf{en secondes} ($\qty{1}{\hour}=\qty{3600}{\second}$). Sur une
facture on utilise le \textbf{kilowattheure} : $\qty{1}{\kilo\watt\hour}=\qty{3.6e6}{\joule}$.
\end{cle}

\begin{methode}[exploiter une plaque signalétique ($P$, $U$)]
\begin{enumerate}[leftmargin=1.7em,topsep=1pt,itemsep=0pt]
  \item \textbf{Intensité absorbée} : $I=\dfrac{P}{U}$.
  \item \textbf{Résistance équivalente} : $R=\dfrac{U^{2}}{P}$ (ou $R=U/I$).
\end{enumerate}
\end{methode}

\begin{remarque}[fusible et transport à haute tension]
Le \textbf{fusible} exploite l'effet Joule : son fil \emph{fond} dès que $I$ dépasse un seuil (ex.
\qty{16}{\ampere}), coupant le circuit. Pour \textbf{transporter} une puissance $P=UI$ donnée, on
\emph{augmente $U$} : à $P$ constante $I=P/U$ diminue, et comme les pertes en ligne valent
$R_{\text{ligne}}I^{2}$, \textbf{doubler $U$ divise les pertes par 4}.
\end{remarque}

\begin{piege}[ne pas confondre les formes]
$P=R\,I^{2}$ (et non $R^2 I$) ; $P=U^{2}/R$ (et non $U/R^2$). Les trois formes $UI$, $RI^2$,
$U^2/R$ ne sont \emph{équivalentes} que pour un dipôle \textbf{ohmique} (où $U=RI$).
\end{piege}
