\begin{exo}[énergie de Joule dégagée]
Une résistance $R=\qty{20}{\ohm}$ est parcourue par un courant $I=\qty{3}{\ampere}$ pendant
$t=\qty{5}{\minute}$. Quelle énergie thermique dégage-t-elle ? (Donne le résultat en joules puis en
kilojoules.)
\end{exo}

\begin{exo}[plaque signalétique d'un appareil]
Un radiateur porte les indications $\qty{2000}{\watt}$ et $\qty{230}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle intensité absorbe-t-il en fonctionnement ?
  \item Quelle est sa résistance ?
\end{enumerate}
\end{exo}

\begin{exo}[trois écritures, un seul résultat]
Une résistance $R=\qty{12}{\ohm}$ est soumise à une tension $U=\qty{24}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'intensité $I$ qui la traverse.
  \item Calcule la puissance dissipée par les \emph{trois} formes $UI$, $RI^2$ et $U^2/R$, et vérifie
        qu'elles donnent la même valeur.
\end{enumerate}
\end{exo}

\begin{exo}[énergie et facture]
Un chauffage de puissance $P=\qty{2}{\kilo\watt}$ fonctionne $\qty{3}{\hour}$ par jour.
\begin{enumerate}[label=\alph*)]
  \item Quelle énergie consomme-t-il par jour, en kilowattheures ?
  \item Au prix de \num{0.15} euro le kilowattheure, quel est le coût journalier ?
\end{enumerate}
\end{exo}

\begin{exo}[à même tension, qui chauffe le plus ?]
Deux résistances $R_1=\qty{10}{\ohm}$ et $R_2=\qty{40}{\ohm}$ sont soumises \emph{séparément} à la
même tension $U=\qty{20}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule la puissance dissipée par chacune.
  \item Laquelle chauffe le plus, la plus grande ou la plus petite résistance ? Explique.
\end{enumerate}
\end{exo}
