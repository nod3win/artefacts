\begin{exo}[puissance $P=UI$]
Une lampe fonctionne sous une tension $U=\qty{12}{\volt}$ en absorbant un courant
$I=\qty{0.5}{\ampere}$. Quelle est sa puissance $P$ ?
\end{exo}

\begin{exo}[puissance par effet Joule $P=RI^2$]
Une résistance $R=\qty{10}{\ohm}$ est traversée par un courant $I=\qty{2}{\ampere}$. Quelle puissance
dissipe-t-elle par effet Joule ?
\end{exo}

\begin{exo}[puissance $P=U^2/R$]
Une résistance $R=\qty{40}{\ohm}$ est soumise à une tension $U=\qty{20}{\volt}$. Calcule la puissance
dissipée en utilisant la forme $P=U^2/R$.
\end{exo}

\begin{exo}[de la puissance à l'énergie]
Un appareil de puissance $P=\qty{60}{\watt}$ fonctionne pendant $t=\qty{2}{\hour}$.
\begin{enumerate}[label=\alph*)]
  \item Calcule l'énergie consommée en joules (penser à convertir la durée en secondes).
  \item Exprime cette énergie en kilowattheures ($\qty{1}{\kilo\watt\hour}=\qty{3.6e6}{\joule}$).
\end{enumerate}
\end{exo}

\begin{exo}[lire une plaque signalétique]
Un appareil porte les indications $\qty{1000}{\watt}$ et $\qty{200}{\volt}$.
\begin{enumerate}[label=\alph*)]
  \item Quelle intensité absorbe-t-il ?
  \item Quelle est sa résistance équivalente ?
\end{enumerate}
\end{exo}
