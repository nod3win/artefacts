\begin{corrige}[puissance $P=UI$]
$P=U\,I=12\times0{,}5=\qty{6}{\watt}$.
\end{corrige}

\begin{corrige}[puissance par effet Joule $P=RI^2$]
$P=R\,I^{2}=10\times2^{2}=10\times4=\qty{40}{\watt}$.
\end{corrige}

\begin{corrige}[puissance $P=U^2/R$]
$P=\dfrac{U^{2}}{R}=\dfrac{20^{2}}{40}=\dfrac{400}{40}=\qty{10}{\watt}$.
\end{corrige}

\begin{corrige}[de la puissance à l'énergie]
\begin{enumerate}[label=\alph*)]
  \item $t=\qty{2}{\hour}=7200\,\si{\second}$, donc $W=P\,t=60\times7200=\qty{432000}{\joule}$.
  \item $W=\dfrac{432000}{3{,}6\times10^{6}}=\qty{0.12}{\kilo\watt\hour}$.
\end{enumerate}
\end{corrige}

\begin{corrige}[lire une plaque signalétique]
\begin{enumerate}[label=\alph*)]
  \item $I=\dfrac{P}{U}=\dfrac{1000}{200}=\qty{5}{\ampere}$.
  \item $R=\dfrac{U^{2}}{P}=\dfrac{200^{2}}{1000}=\dfrac{40000}{1000}=\qty{40}{\ohm}$
        (ou $R=U/I=200/5=\qty{40}{\ohm}$).
\end{enumerate}
\end{corrige}
