\begin{corrige}[réseau mixte : série et parallèle]
On simplifie d'abord le bloc parallèle $R_2\,\|\,R_3$ (deux résistances égales) :
\[ R_{23}=\frac{R_2}{2}=\frac{12}{2}=\qty{6}{\ohm}. \]
Ce bloc est en série avec $R_1$ :
\[ R_{\text{eq}}=R_1+R_{23}=4+6=\qty{10}{\ohm}. \]
\end{corrige}

\begin{corrige}[diviseur de courant]
\begin{enumerate}[label=\alph*)]
  \item $R_{\text{eq}}=\dfrac{4\times1}{4+1}=\qty{0.8}{\ohm}$, donc la tension commune est
        $U=R_{\text{eq}}\,I=0{,}8\times5=\qty{4}{\volt}$.
  \item $I_{4\Omega}=\dfrac{U}{4}=\qty{1}{\ampere}$ ; \; $I_{1\Omega}=\dfrac{U}{1}=\qty{4}{\ampere}$
        (vérif. $1+4=5=I$ \checkmark). C'est la \textbf{plus petite résistance} ($\qty{1}{\ohm}$) qui
        prend le plus de courant.
\end{enumerate}
\end{corrige}

\begin{corrige}[générateur réel débitant dans une résistance]
\begin{enumerate}[label=\alph*)]
  \item La maille donne $E=(R+r)\,I$, donc
        \[ I=\frac{E}{R+r}=\frac{12}{5{,}5+0{,}5}=\frac{12}{6}=\qty{2}{\ampere}. \]
  \item $U=E-rI=12-0{,}5\times2=\qty{11}{\volt}$ (c'est aussi $U=RI=5{,}5\times2=\qty{11}{\volt}$ \checkmark).
\end{enumerate}
\end{corrige}

\begin{corrige}[court-circuit d'une pile]
\begin{enumerate}[label=\alph*)]
  \item En court-circuit $U\approx0$, donc $0=E-rI_{\text{cc}}$ :
        \[ I_{\text{cc}}=\frac{E}{r}=\frac{4{,}5}{0{,}5}=\qty{9}{\ampere}. \]
  \item $P=r\,I_{\text{cc}}^{2}=0{,}5\times9^{2}=\qty{40.5}{\watt}$ dissipés dans la pile : elle
        \textbf{chauffe fortement} (risque d'échauffement dangereux).
\end{enumerate}
\end{corrige}

\begin{corrige}[trois résistances en parallèle : intensité totale]
\textbf{Par branche} : $I_1=\dfrac{2{,}5}{10}=\qty{0.25}{\ampere}$,
$I_2=\dfrac{2{,}5}{100}=\qty{0.025}{\ampere}$, $I_3=\dfrac{2{,}5}{20}=\qty{0.125}{\ampere}$, donc
$I=I_1+I_2+I_3=\qty{0.40}{\ampere}$.

\textbf{Par $R_{\text{eq}}$} : $\dfrac{1}{R_{\text{eq}}}=\dfrac{1}{10}+\dfrac{1}{100}+\dfrac{1}{20}=0{,}16$,
soit $R_{\text{eq}}=\qty{6.25}{\ohm}$ et $I=\dfrac{2{,}5}{6{,}25}=\qty{0.40}{\ampere}$. Les deux voies
concordent.
\end{corrige}
