\begin{corrige}[résistance équivalente en série]
En série, les résistances s'ajoutent :
\[ R_{\text{eq}}=R_1+R_2+R_3=10+20+30=\qty{60}{\ohm}. \]
\end{corrige}

\begin{corrige}[deux résistances en parallèle]
\[ R_{\text{eq}}=\frac{R_1 R_2}{R_1+R_2}=\frac{6\times3}{6+3}=\frac{18}{9}=\qty{2}{\ohm}. \]
On vérifie : $R_{\text{eq}}=\qty{2}{\ohm}$ est bien \emph{inférieure} à la plus petite ($\qty{3}{\ohm}$).
\end{corrige}

\begin{corrige}[$n$ résistances égales en parallèle]
Pour $n$ résistances égales, $R_{\text{eq}}=\dfrac{R}{n}=\dfrac{30}{3}=\qty{10}{\ohm}$.
\end{corrige}

\begin{corrige}[générateur réel]
\begin{enumerate}[label=\alph*)]
  \item À vide, $I=0$ : $U=E-r\times0=E=\qty{9}{\volt}$.
  \item En charge : $U=E-rI=9-1\times2=\qty{7}{\volt}$ (la résistance interne fait « perdre »
        $\qty{2}{\volt}$).
\end{enumerate}
\end{corrige}

\begin{corrige}[intensité dans un circuit série]
$R_{\text{eq}}=R_1+R_2=\qty{30}{\ohm}$, puis
\[ I=\frac{U}{R_{\text{eq}}}=\frac{30}{30}=\qty{1}{\ampere}. \]
\end{corrige}
