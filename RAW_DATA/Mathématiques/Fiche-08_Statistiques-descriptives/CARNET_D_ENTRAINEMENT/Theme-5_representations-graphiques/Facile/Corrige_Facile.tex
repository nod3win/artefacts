\bandeau{Thème 5 — Représentations graphiques}{Niveau Facile — corrigé}

\begin{corbox}{Corrigé 1 — Lire un diagramme en bâtons}
En lisant les hauteurs : $38\to5$, $39\to12$, $40\to9$, $41\to3$.\\
(a) Effectif de la pointure $39$ : $\mathbf{12}$. \quad
(b) Mode : la pointure la plus vendue, $M=\mathbf{39}$. \quad
(c) Total : $N=5+12+9+3=\mathbf{29}$ paires.
\end{corbox}

\begin{corbox}{Corrigé 2 — Choisir la représentation}
(a) Groupe sanguin (qualitatif) $\to$ \textbf{diagramme circulaire}.\\
(b) Nombre d'enfants (discret) $\to$ \textbf{diagramme en bâtons}.\\
(c) Taille en classes (continu) $\to$ \textbf{histogramme}.
\end{corbox}

\begin{corbox}{Corrigé 3 — Angle d'un secteur}
\[ \theta_i=f_i\times360^\circ=0{,}25\times360^\circ=\mathbf{90^\circ}. \]
(Un quart du disque, cohérent avec $25\,\%$.)
\end{corbox}

\begin{corbox}{Corrigé 4 — Lire un histogramme}
La barre la plus haute (en rouge) est la classe $[20,30[$, d'effectif $14$ : classe modale
$M=\mathbf{[20,30[}$. Les barres sont \emph{jointives} et le caractère est une durée : il est
\textbf{continu}.
\end{corbox}

\begin{corbox}{Corrigé 5 — Lire la médiane sur l'ogive}
La droite $f_c=0{,}50$ coupe l'ogive entre les points d'abscisses $20$ (où $f_c=0{,}40$) et $30$
(où $f_c=0{,}70$). La médiane est donc située \textbf{entre $20$ et $30$}.
\end{corbox}
