\bandeau{Thème 5 — Représentations graphiques}{Niveau Facile — exercices}

\noindent\textit{Objectif : lire un graphique et choisir la bonne représentation.}\medskip

\begin{exobox}{Exercice 1 — Lire un diagramme en bâtons}
\begin{center}
\begin{tikzpicture}[scale=0.8]
  \draw[->,>=Stealth] (0,0)--(0,4) node[above,font=\scriptsize]{$n_i$};
  \draw[->,>=Stealth] (0,0)--(6,0) node[right,font=\scriptsize]{pointure};
  \foreach \y/\v in {1/4,2/8,3/12}{\draw[black!12] (0,\y)--(5.6,\y);
     \draw (0,\y)--(-0.06,\y) node[left,font=\tiny]{\v};}
  \foreach \x/\h/\v in {1/1.25/38,2/3/39,3/2.25/40,4/0.75/41}{
     \draw[barcol,line width=4pt] (\x,0)--(\x,\h);
     \fill[barcol] (\x,\h) circle (2.5pt);
     \node[below,font=\scriptsize] at (\x,0){\v};}
\end{tikzpicture}
\end{center}
(a) Quel est l'effectif de la pointure $39$ ? \quad (b) Quel est le mode ? \quad
(c) Combien de paires ont été vendues au total ?
\end{exobox}

\begin{exobox}{Exercice 2 — Choisir la représentation}
Pour chaque caractère, indiquer le graphique adapté (diagramme en bâtons, histogramme, ou
diagramme circulaire) :
(a) le groupe sanguin (qualitatif) ; (b) le nombre d'enfants par famille (discret) ;
(c) la taille en cm regroupée en classes (continu).
\end{exobox}

\begin{exobox}{Exercice 3 — Angle d'un secteur}
Dans un diagramme circulaire, une modalité a une fréquence $f_i=0{,}25$.
Quel est l'angle $\theta_i=f_i\times360^\circ$ du secteur correspondant ?
\end{exobox}

\begin{exobox}{Exercice 4 — Lire un histogramme}
\begin{center}
\begin{tikzpicture}[scale=0.8]
  \draw[->,>=Stealth] (0,0)--(0,4) node[above,font=\scriptsize]{$n_i$};
  \draw[->,>=Stealth] (0,0)--(6.6,0) node[right,font=\scriptsize]{durée (min)};
  \foreach \y/\v in {1/5,2/10,3/15}{\draw[black!12] (0,\y)--(6.2,\y);
     \draw (0,\y)--(-0.06,\y) node[left,font=\tiny]{\v};}
  \foreach \x/\h/\c in {0/0.8/barcol,1.2/2.0/barcol,2.4/2.8/modecol,3.6/1.6/barcol,4.8/0.8/barcol}{
     \fill[\c,opacity=0.85] (\x,0) rectangle (\x+1.2,\h);
     \draw[\c!70!black] (\x,0) rectangle (\x+1.2,\h);}
  \foreach \x/\lab in {0/0,1.2/10,2.4/20,3.6/30,4.8/40,6/50}
     \node[below,font=\tiny] at (\x,0){\lab};
\end{tikzpicture}
\end{center}
Quelle est la classe modale ? Le caractère est-il discret ou continu ?
\end{exobox}

\begin{exobox}{Exercice 5 — Lire la médiane sur l'ogive}
\begin{center}
\begin{tikzpicture}[scale=0.8]
  \draw[->,>=Stealth] (0,0)--(6.6,0) node[below right,font=\scriptsize]{valeur};
  \draw[->,>=Stealth] (0,0)--(0,4.4) node[left,font=\scriptsize]{$f_c$};
  \foreach \y/\v in {1/0{,}25,2/0{,}50,3/0{,}75,4/1{,}00}{\draw[black!12] (0,\y)--(6.2,\y);
     \draw (0,\y)--(-0.06,\y) node[left,font=\tiny]{\v};}
  \foreach \x/\lab in {1/10,2/20,3/30,4/40,5/50}\node[below,font=\tiny] at (\x,0){\lab};
  \draw[ogivecol,line width=1.1pt] (0,0)--(1,0.4)--(2,1.6)--(3,2.8)--(4,3.6)--(5,4);
  \foreach \x/\y in {1/0.4,2/1.6,3/2.8,4/3.6,5/4}\fill[ogivecol] (\x,\y) circle (1.4pt);
  \draw[medcol,dashed] (0,2)--(6.2,2);
\end{tikzpicture}
\end{center}
La ligne pointillée est à $f_c=0{,}50$. Entre quelles deux valeurs se situe la médiane ?
\end{exobox}
