\bandeau{Thème 2 — Mode et moyenne}{Niveau Moyen — corrigé}

\begin{corbox}{Corrigé 1 — Moyenne d'une série en classes}
Centres : $22,26,30,34,38$. Somme des produits :
\[
\sum n_i x_i=4(22)+10(26)+14(30)+8(34)+4(38)=88+260+420+272+152=1192.
\]
\[ m=\frac{1192}{40}=\mathbf{29{,}8\ \text{g}}. \]
\end{corbox}

\begin{corbox}{Corrigé 2 — Moyenne d'une série discrète}
\[
\sum n_i x_i=4(0)+9(1)+8(2)+3(3)+1(4)=0+9+16+9+4=38,\qquad N=25.
\]
\[ m=\frac{38}{25}=\mathbf{1{,}52\ \text{enfant/famille}}. \]
\end{corbox}

\begin{corbox}{Corrigé 3 — Changement d'unité}
$F=1{,}8\,C+32$, donc $\alpha=1{,}8$ et $\beta=32$ :
\[ m_F=1{,}8\times m_C+32=1{,}8\times20+32=36+32=\mathbf{68\ ^\circ\mathrm{F}}. \]
\end{corbox}

\begin{corbox}{Corrigé 4 — Retrouver la transformation}
\textbf{(a)} Ajouter $\beta$ donne $m'=m+\beta$. On veut $50+\beta=55$, donc $\beta=\mathbf{5}$\,kg.\\[2pt]
\textbf{(b)} Multiplier par $\alpha$ donne $m'=\alpha m$. On veut $50\alpha=55$, donc
$\alpha=\dfrac{55}{50}=\mathbf{1{,}1}$.
\end{corbox}

\begin{corbox}{Corrigé 5 — Mode et moyenne ensemble}
Classe modale : plus grand effectif $15$, soit $M=\mathbf{[20,30[}$.\\
Centres $15,25,35,45$, $N=6+15+12+7=40$ :
\[
\sum n_i x_i=6(15)+15(25)+12(35)+7(45)=90+375+420+315=1200,\quad m=\frac{1200}{40}=\mathbf{30}.
\]
La moyenne ($30$) tombe à la \emph{frontière} de la classe modale $[20,30[$ : mode et moyenne
\textbf{ne coïncident pas} en général. Le mode repère le « pic » d'effectif, la moyenne le centre de gravité.
\end{corbox}
