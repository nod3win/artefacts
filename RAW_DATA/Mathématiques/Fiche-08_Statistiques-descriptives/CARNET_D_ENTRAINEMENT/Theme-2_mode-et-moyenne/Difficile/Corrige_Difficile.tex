\bandeau{Thème 2 — Mode et moyenne}{Niveau Difficile — corrigé}

\begin{corbox}{Corrigé 1 — Barème et rééchelonnage}
\textbf{(a)} Transformation $x\mapsto x/2$ ($\alpha=\tfrac12,\ \beta=0$) :
$m'=\dfrac{27}{2}=\mathbf{13{,}5/20}$.\\[2pt]
\textbf{(b)} On ajoute $2$ : $m''=13{,}5+2=\mathbf{15{,}5/20}$.\\[2pt]
\textbf{(c)} Enchaîner « diviser par $2$ » puis « ajouter $2$ » donne
\[ x\ \longmapsto\ \frac{x}{2}+2=\mathbf{0{,}5\,x+2}\qquad(\alpha=0{,}5,\ \beta=2). \]
Vérification : $0{,}5\times27+2=13{,}5+2=15{,}5$. \checkmark\ (même résultat qu'en (b)).\\[2pt]
\textbf{(d)} On résout $0{,}5\,x+2=10$ : $0{,}5\,x=8$, donc $x=\mathbf{16/40}$.
(Une note de $16/40$ devient bien $8/20+2=10/20$.)
\end{corbox}

\begin{corbox}{Corrigé 2 — Moyenne de groupes réunis}
\textbf{(a)} Total des points $=$ effectif $\times$ moyenne :
\[ S_A=20\times12=240,\qquad S_B=30\times14=420. \]
\textbf{(b)} Moyenne d'ensemble = total des points $\div$ effectif total :
\[ m=\frac{240+420}{20+30}=\frac{660}{50}=\mathbf{13{,}2/20}. \]
\textbf{(c)} \textbf{Non}, ce n'est pas $13$. La moyenne d'ensemble est \emph{pondérée} par les
effectifs : le groupe B, plus nombreux ($30$ contre $20$), tire la moyenne vers sa valeur $14$.
La demi-somme $13$ ne serait juste que si les deux groupes avaient le même effectif.\\[2pt]
\textbf{(d)} On écrit la moyenne des trois groupes :
\[
\frac{240+420+10n}{50+n}=12 \;\Longrightarrow\; 660+10n=12(50+n)=600+12n.
\]
Donc $660-600=12n-10n$, soit $60=2n$ et $n=\mathbf{30}$ étudiants.
\end{corbox}
