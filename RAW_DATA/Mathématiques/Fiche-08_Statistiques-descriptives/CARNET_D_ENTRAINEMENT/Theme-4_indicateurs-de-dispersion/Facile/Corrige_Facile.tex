\bandeau{Thème 4 — Les indicateurs de dispersion}{Niveau Facile — corrigé}

\begin{corbox}{Corrigé 1 — Étendue}
La plus petite valeur est la borne inférieure de la première classe ($10$), la plus grande est la
borne supérieure de la dernière ($50$) :
\[ \text{étendue}=x_{\max}-x_{\min}=50-10=\mathbf{40}. \]
\end{corbox}

\begin{corbox}{Corrigé 2 — Écart à la moyenne}
\[
|15-10|=5,\quad |15-12|=3,\quad |15-18|=3,\quad |15-20|=5.
\]
(La valeur absolue rend chaque écart positif, quel que soit le côté de la moyenne.)
\end{corbox}

\begin{corbox}{Corrigé 3 — Écart moyen}
Écarts : $|4-2|=2,\ |4-4|=0,\ |4-6|=2$. Pondérés :
\[ \sum n_i|m-x_i|=1(2)+2(0)+1(2)=4,\qquad \ecm=\frac{4}{4}=\mathbf{1}. \]
\end{corbox}

\begin{corbox}{Corrigé 4 — Variance par König--Huygens}
\[ V=\frac{1}{N}\sum n_i x_i^{\,2}-m^2=169-12^2=169-144=\mathbf{25}. \]
\end{corbox}

\begin{corbox}{Corrigé 5 — Écart type}
\[ \ect=\sqrt{V}=\sqrt{144}=\mathbf{12}. \]
\end{corbox}
