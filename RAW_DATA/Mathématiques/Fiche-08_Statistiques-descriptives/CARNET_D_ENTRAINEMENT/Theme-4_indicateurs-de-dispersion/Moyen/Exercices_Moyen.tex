\bandeau{Thème 4 — Les indicateurs de dispersion}{Niveau Moyen — exercices}

\noindent\textit{Les exercices 1 à 4 portent sur la série des notes (/20) de $N=40$ élèves, de
moyenne $m=10{,}8$ :}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
Classes (/20) & $[0,4[$ & $[4,8[$ & $[8,12[$ & $[12,16[$ & $[16,20[$ \\\hline
$x_i$ & 2 & 6 & 10 & 14 & 18 \\\hline
$n_i$ & 4 & 8 & 10 & 12 & 6 \\\hline
\end{tabular}
\end{center}

\begin{exobox}{Exercice 1 — Écart moyen}
Dresser la colonne $|m-x_i|$ puis $n_i|m-x_i|$, et calculer l'écart moyen $\ecm$.
\end{exobox}

\begin{exobox}{Exercice 2 — Variance (formule directe)}
Calculer la variance par la formule directe $V=\dfrac{1}{N}\sum n_i(m-x_i)^2$.
\end{exobox}

\begin{exobox}{Exercice 3 — Variance (König--Huygens)}
Recalculer la variance par $V=\dfrac{1}{N}\sum n_i x_i^{\,2}-m^2$ et vérifier qu'on retrouve le même
résultat.
\end{exobox}

\begin{exobox}{Exercice 4 — Écart type et cohérence}
En déduire l'écart type $\ect$. Vérifier que $\ect\geqslant\ecm$.
\end{exobox}

\begin{exobox}{Exercice 5 — Même moyenne, dispersions différentes}
On considère deux séries de moyenne $10$ :
\[ A=\{8,\ 10,\ 12\}\qquad B=\{2,\ 10,\ 18\}. \]
Calculer la variance et l'écart type de chacune. Laquelle est la plus dispersée ?
\end{exobox}
