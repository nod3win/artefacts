\bandeau{Thème 3 — Médiane et quartiles}{Niveau Difficile — corrigé}

\begin{corbox}{Corrigé 1 — Résumé complet d'une série}
\textbf{(a)} Effectifs cumulés : $8;\ 28;\ 56;\ 72;\ 80$.\\[2pt]
\textbf{(b)} Médiane, rang $\tfrac{N}{2}=40$. Premier cumul $\geqslant40$ : $[40,60[$ ($n_c=56$),
$n_c^{\text{préc}}=28$, $r=40-28=12$, $n_{\text{classe}}=28$ :
\[ \med=40+\frac{12}{28}\times20=40+8{,}57=\mathbf{48{,}57\ \text{min}}. \]
\textbf{(c)} $Q_1$, rang $\tfrac{N}{4}=20$ : classe $[20,40[$ ($n_c=28$), $r=20-8=12$, $n=20$ :
\[ Q_1=20+\frac{12}{20}\times20=20+12=\mathbf{32\ \text{min}}. \]
$Q_3$, rang $\tfrac{3N}{4}=60$ : classe $[60,80[$ ($n_c=72$), $r=60-56=4$, $n=16$ :
\[ Q_3=60+\frac{4}{16}\times20=60+5=\mathbf{65\ \text{min}}. \]
\textbf{(d)} Écart interquartile $Q_3-Q_1=65-32=\mathbf{33\ \text{min}}$. Interprétation : la moitié
« centrale » des patients attend entre $32$ et $65$ minutes ; c'est une dispersion importante, utile
au service pour dimensionner l'accueil.
\end{corbox}

\begin{corbox}{Corrigé 2 — Médiane, moyenne et symétrie}
\textbf{(a)} Effectifs cumulés : $3;\ 10;\ 20;\ 28;\ 35;\ 40$.\\[2pt]
\textbf{(b)} $n=40$ pair : rangs $\tfrac{n}{2}=20$ et $21$. Le cumul atteint $20$ à la valeur $x=2$
(rang $20$), et le rang $21$ correspond à $x=3$. Donc $\med=\dfrac{2+3}{2}=\mathbf{2{,}5}$.\\[2pt]
\textbf{(c)} $\sum n_i x_i=0+7+20+24+28+25=104$, donc $m=\dfrac{104}{40}=\mathbf{2{,}6}$.\\[2pt]
\textbf{(d)} On a $m=2{,}6>\med=2{,}5$ : la distribution est \emph{presque} symétrique, très
légèrement \textbf{étalée vers le haut} (les patients à $4$–$5$ consultations tirent la moyenne
un peu au-dessus de la médiane).
\end{corbox}
