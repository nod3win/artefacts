\bandeau{Thème 3 — Médiane et quartiles}{Niveau Moyen — corrigé}

\begin{corbox}{Corrigé 1 — Médiane par interpolation}
Rang visé $\tfrac{N}{2}=25$. Premier cumul $\geqslant25$ : $[20,30[$ ($n_c=32$).
Donc $a=20,\ b=30,\ n_{\text{classe}}=16$ et $r=25-16=9$ :
\[ \med=20+\frac{9}{16}\times10=20+5{,}625=\mathbf{25{,}63\ \text{min}}. \]
\end{corbox}

\begin{corbox}{Corrigé 2 — Premier quartile $Q_1$}
Rang visé $\tfrac{N}{4}=12{,}5$. Premier cumul $\geqslant12{,}5$ : $[10,20[$ ($n_c=16$).
$a=10,\ b=20,\ n_{\text{classe}}=10$, $r=12{,}5-6=6{,}5$ :
\[ Q_1=10+\frac{6{,}5}{10}\times10=10+6{,}5=\mathbf{16{,}5\ \text{min}}. \]
\end{corbox}

\begin{corbox}{Corrigé 3 — Troisième quartile $Q_3$}
Rang visé $\tfrac{3N}{4}=37{,}5$. Premier cumul $\geqslant37{,}5$ : $[30,40[$ ($n_c=44$).
$a=30,\ b=40,\ n_{\text{classe}}=12$, $r=37{,}5-32=5{,}5$ :
\[ Q_3=30+\frac{5{,}5}{12}\times10=30+4{,}58=\mathbf{34{,}58\ \text{min}}. \]
\end{corbox}

\begin{corbox}{Corrigé 4 — Médiane d'une série discrète tabulée}
Effectifs cumulés : $4;\ 13;\ 21;\ 24;\ 25$. $n=25$ impair, rang $\tfrac{25+1}{2}=13$.
Le $13^{\text{e}}$ individu tombe là où le cumul atteint $13$, soit la valeur $x=1$ :
$\med=\mathbf{1}$.
\end{corbox}

\begin{corbox}{Corrigé 5 — Écart interquartile}
\[ Q_3-Q_1=34{,}58-16{,}5=\mathbf{18{,}08\ \text{min}}. \]
Interprétation : les $50\,\%$ « centraux » des trajets durent entre $16{,}5$ et $34{,}58$ minutes ;
plus cet écart est petit, plus les durées du milieu sont resserrées.
\end{corbox}
