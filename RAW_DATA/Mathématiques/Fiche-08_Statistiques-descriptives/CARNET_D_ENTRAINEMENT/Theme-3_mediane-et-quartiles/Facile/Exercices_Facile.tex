\bandeau{Thème 3 — Médiane et quartiles}{Niveau Facile — exercices}

\noindent\textit{Objectif : décomposer l'interpolation en gestes élémentaires. Arrondir à $0{,}01$.}\medskip

\begin{exobox}{Exercice 1 — Médiane discrète ($n$ impair)}
Série : $12,\ 7,\ 3,\ 15,\ 9$. Ranger par ordre croissant puis donner la médiane.
\end{exobox}

\begin{exobox}{Exercice 2 — Médiane discrète ($n$ pair)}
Série : $8,\ 3,\ 5,\ 10,\ 6,\ 1$. Donner la médiane.
\end{exobox}

\noindent\textit{Les exercices 3 à 5 utilisent le même tableau ($N=40$) :}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{|c|c|c|c|c|}
\hline
Classes & $[0,5[$ & $[5,10[$ & $[10,15[$ & $[15,20[$ \\\hline
$n_i$ & 5 & 13 & 12 & 10 \\\hline
$n_c$ & 5 & 18 & 30 & 40 \\\hline
\end{tabular}
\end{center}

\begin{exobox}{Exercice 3 — Localiser la classe médiane}
Calculer le rang visé $\tfrac{N}{2}$, puis, à l'aide des effectifs cumulés, désigner la classe
médiane.
\end{exobox}

\begin{exobox}{Exercice 4 — Rang intérieur}
En déduire le rang intérieur $r=\tfrac{N}{2}-n_c^{\text{préc}}$ dans la classe médiane.
\end{exobox}

\begin{exobox}{Exercice 5 — Appliquer l'interpolation}
La classe médiane est $[10,15[$ ($a=10$, $b=15$, $n_{\text{classe}}=12$) et $r=2$.
Calculer $\med=a+\dfrac{r}{n_{\text{classe}}}(b-a)$.
\end{exobox}
