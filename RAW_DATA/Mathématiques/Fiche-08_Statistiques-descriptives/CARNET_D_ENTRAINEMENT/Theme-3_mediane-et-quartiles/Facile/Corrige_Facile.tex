\bandeau{Thème 3 — Médiane et quartiles}{Niveau Facile — corrigé}

\begin{corbox}{Corrigé 1 — Médiane discrète ($n$ impair)}
Tri : $3,\ 7,\ \underline{9},\ 12,\ 15$. $n=5$ impair, rang $\tfrac{5+1}{2}=3$ : $\med=\mathbf{9}$.
\end{corbox}

\begin{corbox}{Corrigé 2 — Médiane discrète ($n$ pair)}
Tri : $1,\ 3,\ \underline{5,\ 6},\ 8,\ 10$. $n=6$ pair, rangs $3$ et $4$ :
$\med=\dfrac{5+6}{2}=\mathbf{5{,}5}$.
\end{corbox}

\begin{corbox}{Corrigé 3 — Localiser la classe médiane}
$\dfrac{N}{2}=\dfrac{40}{2}=20$. On cherche le premier cumul $\geqslant 20$ : la classe $[5,10[$
s'arrête à $18$ (trop tôt), la classe $[10,15[$ monte à $30$. Classe médiane : $\mathbf{[10,15[}$.
\end{corbox}

\begin{corbox}{Corrigé 4 — Rang intérieur}
Le cumul de la classe précédente est $n_c^{\text{préc}}=18$, donc
$r=\dfrac{N}{2}-n_c^{\text{préc}}=20-18=\mathbf{2}$.
\end{corbox}

\begin{corbox}{Corrigé 5 — Appliquer l'interpolation}
\[
\med=10+\frac{2}{12}\,(15-10)=10+\frac{2}{12}\times5=10+\frac{10}{12}=10+0{,}83=\mathbf{10{,}83}.
\]
Interprétation : la moitié des individus ont une valeur inférieure à $10{,}83$.
\end{corbox}
