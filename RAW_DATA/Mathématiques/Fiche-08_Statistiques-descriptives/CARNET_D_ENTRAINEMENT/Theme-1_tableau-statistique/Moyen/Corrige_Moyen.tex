\bandeau{Thème 1 — Le tableau statistique}{Niveau Moyen — corrigé}

\begin{corbox}{Corrigé 1 — Compléter un tableau}
Centres $x_i=(a+b)/2$ ; fréquences $f_i=n_i/40$ ; cumuls par additions successives.
\begin{center}
\renewcommand{\arraystretch}{1.2}
\rowcolors{2}{primary!5}{white}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
\rowcolor{primary!18}
Classes (g) & $x_i$ & $n_i$ & $f_i$ & $n_c$ & $f_c$ \\\hline
$[20,24[$ & 22 & 4  & 0{,}10 & 4  & 0{,}10 \\
$[24,28[$ & 26 & 10 & 0{,}25 & 14 & 0{,}35 \\
$[28,32[$ & 30 & 14 & 0{,}35 & 28 & 0{,}70 \\
$[32,36[$ & 34 & 8  & 0{,}20 & 36 & 0{,}90 \\
$[36,40[$ & 38 & 4  & 0{,}10 & 40 & 1{,}00 \\\hline
\multicolumn{2}{|r|}{Totaux} & $N=40$ & $1{,}00$ & & \\\hline
\end{tabular}
\end{center}
Contrôles : $\sum n_i=40=N$ et le dernier $f_c=1{,}00$. \checkmark
\end{corbox}

\begin{corbox}{Corrigé 2 — Lecture des cumuls}
(a) « moins de $32$\,g » = effectif cumulé de la classe $[28,32[$ : $n_c=\mathbf{28}$ souris.\\
(b) « moins de $28$\,g » = fréquence cumulée de la classe $[24,28[$ : $f_c=0{,}35=\mathbf{35\,\%}$.
\end{corbox}

\begin{corbox}{Corrigé 3 — Des pourcentages aux effectifs}
On applique $n_i=f_i\times N$ avec $N=300$ :
\[
\text{O}:0{,}45\times300=135;\quad \text{A}:0{,}40\times300=120;\quad
\text{B}:0{,}10\times300=30;\quad \text{AB}:0{,}05\times300=15.
\]
Contrôle : $135+120+30+15=\mathbf{300}=N$. \checkmark
\end{corbox}

\begin{corbox}{Corrigé 4 — Série discrète}
On compte les apparitions de chaque valeur (caractère discret : $x_i$ = la valeur).
\begin{center}
\renewcommand{\arraystretch}{1.2}
\rowcolors{2}{primary!5}{white}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
\rowcolor{primary!18}
$x_i$ (enfants) & 0 & 1 & 2 & 3 & 4 \\\hline
$n_i$ & 4 & 9 & 8 & 3 & 1 \\\hline
$f_i$ & 0{,}16 & 0{,}36 & 0{,}32 & 0{,}12 & 0{,}04 \\\hline
\end{tabular}
\end{center}
Contrôle : $4+9+8+3+1=25=N$ et $\sum f_i=1{,}00$. \checkmark
\end{corbox}

\begin{corbox}{Corrigé 5 — Effectif manquant}
La somme des effectifs vaut $N$, donc
\[ n_3=N-n_1-n_2-n_4=60-9-21-12=\mathbf{18}. \]
Puis $f_3=\dfrac{18}{60}=\mathbf{0{,}30}=30\,\%$.
\end{corbox}
