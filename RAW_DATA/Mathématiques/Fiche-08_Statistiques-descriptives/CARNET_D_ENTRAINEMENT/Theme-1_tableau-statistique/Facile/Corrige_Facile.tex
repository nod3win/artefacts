\bandeau{Thème 1 — Le tableau statistique}{Niveau Facile — corrigé}

\begin{corbox}{Corrigé 1 — Centre de classe}
On applique $x_i=\dfrac{a+b}{2}$ (milieu de l'intervalle).
\[
\text{(a) } \frac{55+65}{2}=\mathbf{60}\quad
\text{(b) } \frac{100+110}{2}=\mathbf{105}\quad
\text{(c) } \frac{0+4}{2}=\mathbf{2}\quad
\text{(d) } \frac{12{,}5+17{,}5}{2}=\mathbf{15}.
\]
\end{corbox}

\begin{corbox}{Corrigé 2 — Effectif total}
On additionne toute la colonne :
\[ N=3+7+12+15+8+5=\mathbf{50}. \]
\end{corbox}

\begin{corbox}{Corrigé 3 — Fréquence}
\[ f_i=\frac{n_i}{N}=\frac{18}{50}=\mathbf{0{,}36}\ =\ \mathbf{36\,\%}. \]
\emph{Lecture :} $36\,\%$ des individus appartiennent à cette classe.
\end{corbox}

\begin{corbox}{Corrigé 4 — Effectifs cumulés}
On additionne de proche en proche : $6$ ; $6+11=17$ ; $17+9=26$ ; $26+4=30$.
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{|c|c|c|c|c|}
\hline
$n_i$ & 6 & 11 & 9 & 4 \\\hline
$n_c$ & 6 & 17 & 26 & \textbf{30} \\\hline
\end{tabular}
\end{center}
La dernière case vaut $\mathbf{30}$ : c'est l'effectif total $N=6+11+9+4=30$. La dernière valeur
cumulée doit \emph{toujours} être égale à $N$ (contrôle).
\end{corbox}

\begin{corbox}{Corrigé 5 — Retrouver un effectif}
On inverse $f_i=\dfrac{n_i}{N}$ en $n_i=f_i\times N$ :
\[ n_i=0{,}15\times 200=\mathbf{30\ \text{patients}}. \]
\end{corbox}
