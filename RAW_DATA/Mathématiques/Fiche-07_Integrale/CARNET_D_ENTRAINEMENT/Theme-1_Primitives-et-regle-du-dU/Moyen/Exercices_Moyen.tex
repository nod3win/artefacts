\section*{Niveau Moyen — Composition affine et compensation d'une constante}

\begin{center}
\itshape Repérer $U$, calculer $\dd U$, et compenser la constante manquante si besoin.
\end{center}
\medskip

\begin{exo}[1 — puissance d'une expression affine]
Calculer $\displaystyle J_1=\int (x+3)^3\,\dd x$.
\end{exo}

\begin{exo}[2 — cosinus composé]
Calculer $\displaystyle J_2=\int \cos(2x)\,\dd x$.
\end{exo}

\begin{exo}[3 — exponentielle composée]
Calculer $\displaystyle J_3=\int e^{2x}\,\dd x$.
\end{exo}

\begin{exo}[4 — le $\dd U$ presque prêt]
Calculer $\displaystyle J_4=\int \frac{2x}{x^2+3}\,\dd x$.
\end{exo}

\begin{exo}[5 — la forme $1/\cos^2$ composée]
Calculer $\displaystyle J_5=\int \frac{\dd x}{\cos^2(3x)}$.
\end{exo}
