\section*{Niveau Difficile — Corrigé détaillé}

\begin{corr}[1 — mettre à l'échelle $\arcsin$ et $\arctan$]
\textbf{(a)} On pose $U=\tfrac{x}{3}$, donc $\dd U=\tfrac13\,\dd x$, soit $\dd x=3\,\dd U$. Forme 5 :
\[
\int \frac{\dd x}{\sqrt{1-\bigl(\tfrac{x}{3}\bigr)^2}}
=\int \frac{3\,\dd U}{\sqrt{1-U^2}}=3\arcsin U+C
=\boxed{\,3\arcsin\!\Bigl(\frac{x}{3}\Bigr)+C.\,}
\]
\emph{Vérification :} $\Bigl[3\arcsin\!\tfrac{x}{3}\Bigr]'=3\cdot\dfrac{1/3}{\sqrt{1-(x/3)^2}}=\dfrac{1}{\sqrt{1-(x/3)^2}}$. \checkmark

\medskip
\textbf{(b)} On factorise le dénominateur pour retrouver la forme $1+U^2$ :
\[
\frac{1}{3+x^2}=\frac{1}{3}\cdot\frac{1}{1+\bigl(\tfrac{x}{\sqrt3}\bigr)^2}.
\]
On pose $U=\tfrac{x}{\sqrt3}$, d'où $\dd x=\sqrt3\,\dd U$ :
\[
\int \frac{\dd x}{3+x^2}=\frac13\int\frac{\sqrt3\,\dd U}{1+U^2}
=\frac{\sqrt3}{3}\arctan U+C
=\boxed{\,\frac{1}{\sqrt3}\arctan\!\Bigl(\frac{x}{\sqrt3}\Bigr)+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac{1}{\sqrt3}\arctan\tfrac{x}{\sqrt3}\Bigr]'
=\tfrac{1}{\sqrt3}\cdot\dfrac{1/\sqrt3}{1+x^2/3}=\dfrac{1/3}{1+x^2/3}=\dfrac{1}{3+x^2}$. \checkmark
\end{corr}

\begin{corr}[2 — puissance composée]
On pose $U=5x^2+3$, donc $U'=10x$ et $\dd U=10x\,\dd x$.

\medskip
\textbf{(a)} Le facteur $10x\,\dd x$ est \emph{exactement} $\dd U$. Forme 1 avec $n=3$ :
\[
\int (5x^2+3)^3\,10x\,\dd x=\int U^3\,\dd U=\frac{U^4}{4}+C=\boxed{\,\frac{(5x^2+3)^4}{4}+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac{(5x^2+3)^4}{4}\Bigr]'=\tfrac{4(5x^2+3)^3\cdot 10x}{4}=(5x^2+3)^3\,10x$. \checkmark

\medskip
\textbf{(b)} Ici on a $5x\,\dd x$, c'est-à-dire $\tfrac12\,\dd U$ (il manque le facteur $2$). Donc
\[
\int (5x^2+3)^3\,5x\,\dd x=\int U^3\cdot\tfrac12\,\dd U=\tfrac12\cdot\frac{U^4}{4}+C
=\boxed{\,\frac{(5x^2+3)^4}{8}+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac{(5x^2+3)^4}{8}\Bigr]'=\tfrac{4(5x^2+3)^3\cdot 10x}{8}=(5x^2+3)^3\,5x$. \checkmark
\end{corr}

\begin{corr}[3 — factoriser le dénominateur / la racine]
\textbf{(a)} On écrit $9+x^2=9\bigl(1+(\tfrac{x}{3})^2\bigr)$. En posant $U=\tfrac{x}{3}$, $\dd x=3\,\dd U$ :
\[
\int \frac{\dd x}{9+x^2}=\frac19\int\frac{3\,\dd U}{1+U^2}
=\frac{3}{9}\arctan U+C=\boxed{\,\frac13\arctan\!\Bigl(\frac{x}{3}\Bigr)+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac13\arctan\tfrac{x}{3}\Bigr]'=\tfrac13\cdot\dfrac{1/3}{1+x^2/9}=\dfrac{1/9}{1+x^2/9}=\dfrac{1}{9+x^2}$. \checkmark

\medskip
\textbf{(b)} On écrit $\sqrt{4-x^2}=\sqrt{4}\,\sqrt{1-(\tfrac{x}{2})^2}=2\sqrt{1-(\tfrac{x}{2})^2}$.
En posant $U=\tfrac{x}{2}$, $\dd x=2\,\dd U$ :
\[
\int \frac{\dd x}{\sqrt{4-x^2}}=\int\frac{2\,\dd U}{2\sqrt{1-U^2}}
=\int\frac{\dd U}{\sqrt{1-U^2}}=\arcsin U+C=\boxed{\,\arcsin\!\Bigl(\frac{x}{2}\Bigr)+C.\,}
\]
\emph{Vérification :} $\Bigl[\arcsin\tfrac{x}{2}\Bigr]'=\dfrac{1/2}{\sqrt{1-x^2/4}}=\dfrac{1/2}{\tfrac12\sqrt{4-x^2}}=\dfrac{1}{\sqrt{4-x^2}}$. \checkmark
\end{corr}

\begin{corr}[4 — exponentielle de base $a\neq e$]
\textbf{(a)} On pose $U=3x$, donc $\dd x=\tfrac13\,\dd U$. Forme 6 avec $a=2$ :
\[
\int 2^{3x}\,\dd x=\frac13\int 2^{U}\,\dd U=\frac13\cdot\frac{2^{U}}{\ln 2}+C
=\boxed{\,\frac{2^{3x}}{3\ln 2}+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac{2^{3x}}{3\ln 2}\Bigr]'=\tfrac{1}{3\ln 2}\cdot 2^{3x}\ln 2\cdot 3=2^{3x}$. \checkmark

\medskip
\textbf{(b)} Forme 6 directe avec $a=10$, $U=x$ :
\[
\int 10^{x}\,\dd x=\boxed{\,\frac{10^{x}}{\ln 10}+C.\,}
\]
\emph{Vérification :} $\Bigl[\tfrac{10^{x}}{\ln 10}\Bigr]'=\tfrac{1}{\ln 10}\cdot 10^{x}\ln 10=10^{x}$. \checkmark
\end{corr}

\begin{corr}[5 — forme $\dd U/U$ trigonométrique]
On reconnaît la forme $\dd U/U$ en posant $U=3-\cos x$. Alors $U'=\sin x$ et
$\dd U=\sin x\,\dd x$ : le numérateur est exactement $\dd U$. Forme 2 :
\[
\int \frac{\sin x\,\dd x}{3-\cos x}=\int\frac{\dd U}{U}=\ln|U|+C=\boxed{\,\ln(3-\cos x)+C.\,}
\]
La valeur absolue est superflue car $3-\cos x\geqslant 2>0$.\\
\emph{Vérification :} $\bigl[\ln(3-\cos x)\bigr]'=\dfrac{\sin x}{3-\cos x}$. \checkmark
\end{corr}
