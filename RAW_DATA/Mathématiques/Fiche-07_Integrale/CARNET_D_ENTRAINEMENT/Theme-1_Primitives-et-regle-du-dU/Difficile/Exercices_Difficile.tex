\section*{Niveau Difficile — Multiplier-diviser, factoriser, $\arcsin$/$\arctan$}

\begin{center}
\itshape Ajuster le $\dd U$ par une constante, ou factoriser le dénominateur pour retrouver
la forme du tableau. Chaque intégrale se termine par $+\,C$.
\end{center}
\medskip

\begin{exo}[1 — mettre à l'échelle $\arcsin$ et $\arctan$]
Calculer :
\[
\text{(a) } \int \frac{\dd x}{\sqrt{1-\bigl(\tfrac{x}{3}\bigr)^2}}\ ;\qquad\qquad
\text{(b) } \int \frac{\dd x}{3+x^2}.
\]
\end{exo}

\begin{exo}[2 — puissance composée, avec et sans le bon facteur]
Calculer :
\[
\text{(a) } \int (5x^2+3)^3\,10x\,\dd x\ ;\qquad\qquad
\text{(b) } \int (5x^2+3)^3\,5x\,\dd x.
\]
\end{exo}

\begin{exo}[3 — factoriser le dénominateur / la racine]
Calculer :
\[
\text{(a) } \int \frac{\dd x}{9+x^2}\ ;\qquad\qquad
\text{(b) } \int \frac{\dd x}{\sqrt{4-x^2}}.
\]
\end{exo}

\begin{exo}[4 — exponentielle de base $a\neq e$]
Calculer :
\[
\text{(a) } \int 2^{3x}\,\dd x\ ;\qquad\qquad
\text{(b) } \int 10^{x}\,\dd x.
\]
\end{exo}

\begin{exo}[5 — forme $\dd U/U$ trigonométrique]
Calculer $\displaystyle \int \frac{\sin x}{3-\cos x}\,\dd x$.
\end{exo}
