\begin{corr}[1 ã polynûÇme, forme de base]
Par linûˋaritûˋ, une primitive est $F(x)=\dfrac{3x^3}{3}+\dfrac{2x^2}{2}+x=x^3+x^2+x$. Donc
\[
I=\bigl[x^3+x^2+x\bigr]_0^1=(1+1+1)-(0+0+0)=\boxed{\,3\,}\ua.
\]
\textbf{Vûˋrification} : $(x^3+x^2+x)'=3x^2+2x+1$. \checkmark
\end{corr}

\begin{corr}[2 ã bornes nûˋgatives]
Primitive : $F(x)=\dfrac{x^3}{3}-\dfrac{x^2}{2}$. On ûˋvalue chaque borne sûˋparûˋment.
\[
F(2)=\frac{8}{3}-\frac{4}{2}=\frac{8}{3}-2=\frac{8-6}{3}=\frac{2}{3},
\qquad
F(-1)=\frac{-1}{3}-\frac{1}{2}=-\frac{2}{6}-\frac{3}{6}=-\frac{5}{6}.
\]
Donc
\[
J=F(2)-F(-1)=\frac{2}{3}-\Bigl(-\frac{5}{6}\Bigr)=\frac{4}{6}+\frac{5}{6}=\frac{9}{6}=\boxed{\,\frac{3}{2}\,}\ua.
\]
\textbf{Attention} au double signe : $-\,F(-1)=+\tfrac56$.
\end{corr}

\begin{corr}[3 ã bien poser le crochet]
Primitive : $F(x)=x^2-x$. On l'ûˋcrit entre crochets avant de substituer :
\[
K=\bigl[\,x^2-x\,\bigr]_1^3=\underbrace{(3^2-3)}_{F(3)}-\underbrace{(1^2-1)}_{F(1)}=(9-3)-(1-1)=6-0=\boxed{\,6\,}\ua.
\]
\end{corr}

\begin{corr}[4 ã intûˋgrale nulle et inversion des bornes]
\textbf{a)} Les deux bornes sont \emph{ûˋgales} : l'intervalle est rûˋduit û  un point, sans ô¨ largeur ô£,
donc
\[
\int_{5}^{5}\bigl(x^3+7\bigr)\,\dd x=0
\]
quelle que soit la fonction (propriûˋtûˋ $\int_a^a f=0$).

\smallskip
\textbf{b)} Primitive de $x$ : $F(x)=\dfrac{x^2}{2}$.
\[
L=\int_{1}^{3} x\,\dd x=\left[\frac{x^2}{2}\right]_1^3=\frac{9}{2}-\frac{1}{2}=\frac{8}{2}=\boxed{\,4\,}\ua.
\]
Inverser les bornes change le signe, donc \emph{sans recalcul} :
\[
\int_{3}^{1} x\,\dd x=-\int_{1}^{3} x\,\dd x=-L=\boxed{\,-4\,}.
\]
\end{corr}

\begin{corr}[5 ã une puissance fractionnaire]
Avec $\sqrt{x}=x^{1/2}$ et la formule $\int x^n\,\dd x=\dfrac{x^{n+1}}{n+1}$ pour $n=\tfrac12$ :
\[
\int x^{1/2}\,\dd x=\frac{x^{3/2}}{3/2}=\frac{2}{3}\,x^{3/2}.
\]
Donc, en utilisant $4^{3/2}=(\sqrt4)^3=2^3=8$ et $1^{3/2}=1$ :
\[
M=\left[\frac{2}{3}\,x^{3/2}\right]_1^4=\frac{2}{3}\cdot 8-\frac{2}{3}\cdot 1=\frac{16}{3}-\frac{2}{3}=\frac{14}{3}
\approx 4{,}67\ua.
\]
\end{corr}
