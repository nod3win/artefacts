\section*{Niveau Moyen — Compensation du $\dd U$, bornes, Chasles et linéarité}

\begin{center}
\itshape\small Ces intégrales définies demandent de compenser une constante pour faire apparaître le
bon $\dd U$, ou d'exploiter les propriétés (Chasles, linéarité) avant de conclure par
Newton--Leibniz.
\end{center}

\medskip

\begin{exo}[1 — composée trigonométrique]
Calculer
\[
A=\int_0^{\pi/6}\sin(2x)\,\dd x.
\]
\emph{Indication} : une primitive de $\sin(2x)$ est $-\tfrac12\cos(2x)$ ; faire apparaître le
facteur $2$ manquant.
\end{exo}

\begin{exo}[2 — exponentielle composée]
Calculer
\[
B=\int_0^{1/2} e^{2x}\,\dd x.
\]
Donner la valeur exacte, puis une valeur approchée au centième.
\end{exo}

\begin{exo}[3 — le $\dd U$ presque présent]
Calculer
\[
C=\int_0^{1} x\,e^{x^2+1}\,\dd x.
\]
On pourra poser $U=x^2+1$ et adapter les bornes, ou reconnaître directement la forme
$U'\,e^{U}$.
\end{exo}

\begin{exo}[4 — relation de Chasles et inversion]
Soit $f$ une fonction continue sur $[1,4]$. On donne
\[
\int_1^4 f(x)\,\dd x=7\qquad\text{et}\qquad \int_1^2 f(x)\,\dd x=2.
\]
\begin{enumerate}[label=\alph*),leftmargin=1.8em]
  \item En déduire $\displaystyle\int_2^4 f(x)\,\dd x$.
  \item En déduire $\displaystyle\int_4^2 f(x)\,\dd x$.
\end{enumerate}
\end{exo}

\begin{exo}[5 — linéarité pour simplifier]
Calculer
\[
D=\int_0^{\pi/2}\bigl(\cos x+2\sin x\bigr)\,\dd x.
\]
Séparer l'intégrale par linéarité, puis appliquer Newton--Leibniz à chaque terme.
\end{exo}
