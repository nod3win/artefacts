\section*{Niveau Difficile — Corrigé détaillé}

\begin{corr}[1 — comparer et encadrer sans (presque) calculer]
\textbf{a)} Sur $[0,1]$, on a $0\leqslant x\leqslant 1$, donc $x^2\leqslant x$ (multiplier
$x\leqslant 1$ par $x\geqslant 0$). Par la propriété de \textbf{croissance} de l'intégrale (mêmes
bornes, $0<1$) :
\[
\int_0^1 x^2\,\dd x\leqslant\int_0^1 x\,\dd x.
\]
\textbf{Vérification} : $\displaystyle\int_0^1 x^2\,\dd x=\Bigl[\tfrac{x^3}{3}\Bigr]_0^1=\tfrac13$ et
$\displaystyle\int_0^1 x\,\dd x=\Bigl[\tfrac{x^2}{2}\Bigr]_0^1=\tfrac12$ ; on a bien
$\tfrac13\leqslant\tfrac12$. \checkmark

\smallskip
\textbf{b)} Sur $[1,2]$, $x\geqslant 1$ donc $\ln x\geqslant \ln 1=0$. La fonction est positive sur
l'intervalle d'intégration (avec $1<2$) : par \textbf{positivité},
$\displaystyle\int_1^2 \ln x\,\dd x\geqslant 0$. L'intégrale est donc \textbf{positive} (et même
strictement positive car $\ln x>0$ sur $]1,2]$).

\smallskip
\textbf{c)} Sur $[0,1]$, $0\leqslant x^2\leqslant 1$, et comme l'exponentielle est croissante,
$e^{0}\leqslant e^{x^2}\leqslant e^{1}$, soit $1\leqslant e^{x^2}\leqslant e$. On intègre cet
encadrement sur $[0,1]$ (longueur $1$), par croissance :
\[
\int_0^1 1\,\dd x\leqslant\int_0^1 e^{x^2}\,\dd x\leqslant\int_0^1 e\,\dd x
\;\Longleftrightarrow\;
1\leqslant\int_0^1 e^{x^2}\,\dd x\leqslant e.
\]
(On n'a jamais eu besoin d'une primitive de $e^{x^2}$, qui n'est d'ailleurs pas élémentaire.)
\end{corr}

\begin{corr}[2 — combiner Chasles, inversion et linéarité]
\textbf{a)} Chasles sur $[0,4]$ coupé en $1$ :
\[
\int_0^4 f=\int_0^1 f+\int_1^4 f=3+(-2)=\boxed{\,1\,}.
\]
\textbf{b)} Inversion des bornes : $\displaystyle\int_4^0 f=-\int_0^4 f=-1=\boxed{\,-1\,}.$

\smallskip
\textbf{c)} Linéarité (somme et constantes) :
\[
\int_0^4\bigl(2f+3g\bigr)=2\int_0^4 f+3\int_0^4 g=2\cdot 1+3\cdot 5=2+15=\boxed{\,17\,}.
\]
\textbf{d)} De même :
\[
\int_0^4\bigl(f-g\bigr)=\int_0^4 f-\int_0^4 g=1-5=\boxed{\,-4\,}.
\]
\end{corr}

\begin{corr}[3 — intégrale définie par une IPP guidée]
\textbf{a)} La règle de priorité place le \emph{logarithme} en $U$ : on pose $U=\ln x$ et
$V'=x$. Alors
\[
U'=\frac1x,\qquad V=\int x\,\dd x=\frac{x^2}{2}.
\]
\textbf{b)} Formule d'IPP sur $[1,e]$ :
\[
E=\int_1^e x\ln x\,\dd x=\left[\frac{x^2}{2}\ln x\right]_1^e-\int_1^e \frac{x^2}{2}\cdot\frac1x\,\dd x
=\left[\frac{x^2}{2}\ln x\right]_1^e-\frac12\int_1^e x\,\dd x.
\]
\textbf{c)} Terme tout intégré : $\left[\dfrac{x^2}{2}\ln x\right]_1^e
=\dfrac{e^2}{2}\underbrace{\ln e}_{=1}-\dfrac{1}{2}\underbrace{\ln 1}_{=0}=\dfrac{e^2}{2}.$
Intégrale restante : $\dfrac12\displaystyle\int_1^e x\,\dd x=\dfrac12\left[\dfrac{x^2}{2}\right]_1^e
=\dfrac12\cdot\dfrac{e^2-1}{2}=\dfrac{e^2-1}{4}.$ Donc
\[
E=\frac{e^2}{2}-\frac{e^2-1}{4}=\frac{2e^2-(e^2-1)}{4}=\frac{e^2+1}{4}=\boxed{\,\frac{e^2+1}{4}\,}\approx 2{,}10\ua.
\]
\end{corr}

\begin{corr}[4 — quand aire algébrique et aire géométrique diffèrent]
\textbf{a)} Primitive $\dfrac{x^2}{2}-x$ :
\[
\int_0^2 (x-1)\,\dd x=\left[\frac{x^2}{2}-x\right]_0^2=(2-2)-(0-0)=0.
\]
Le résultat est \textbf{nul} : sur $[0,1]$ la fonction est négative, sur $[1,2]$ elle est positive,
et les deux aires algébriques se \emph{compensent} exactement.

\smallskip
\textbf{b)} La racine de $f$ est $x=1$ ; $f\leqslant 0$ sur $[0,1]$ et $f\geqslant 0$ sur $[1,2]$, donc
$|x-1|=1-x$ sur $[0,1]$ et $|x-1|=x-1$ sur $[1,2]$. Par Chasles :
\[
\int_0^2|x-1|\,\dd x=\int_0^1(1-x)\,\dd x+\int_1^2(x-1)\,\dd x.
\]
\[
\int_0^1(1-x)\,\dd x=\left[x-\frac{x^2}{2}\right]_0^1=1-\frac12=\frac12,\qquad
\int_1^2(x-1)\,\dd x=\left[\frac{x^2}{2}-x\right]_1^2=(2-2)-\Bigl(\frac12-1\Bigr)=\frac12.
\]
D'où $\displaystyle\int_0^2|x-1|\,\dd x=\frac12+\frac12=\boxed{\,1\,}\ua.$

\smallskip
\textbf{c)} On compare : $\Bigl|\displaystyle\int_0^2 f\Bigr|=|0|=0$ et
$\displaystyle\int_0^2|f|=1$. L'inégalité $0\leqslant 1$ est bien vérifiée. \checkmark

\emph{Interprétation.} $\int_0^2 f=0$ est l'aire \textbf{algébrique} (les portions au-dessus et
au-dessous de l'axe se compensent), tandis que $\int_0^2|f|=1$ est l'aire \textbf{géométrique}
totale (les deux triangles de la figure, chacun d'aire $\tfrac12$). L'écart entre les deux membres
mesure exactement la partie « sous l'axe » comptée négativement dans la première intégrale.
\end{corr}
