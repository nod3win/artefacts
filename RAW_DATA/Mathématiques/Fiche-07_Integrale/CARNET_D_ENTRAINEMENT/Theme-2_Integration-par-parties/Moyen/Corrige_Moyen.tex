\section*{Niveau Moyen — Corrigé}

\begin{corr}[1]
$\displaystyle\int \ln x\,\dd x$. On pose $U=\ln x$, $V'=1$, donc $U'=\tfrac1x$, $V=x$.
\[
\int \ln x\,\dd x=x\ln x-\int x\cdot\frac1x\,\dd x=x\ln x-\int 1\,\dd x=\boxed{x\ln x-x+C.}
\]
\textbf{Vérif.} $\bigl[x\ln x-x\bigr]'=\ln x+1-1=\ln x$. \checkmark
\end{corr}

\begin{corr}[2]
$\displaystyle\int x\ln x\,\dd x$. Règle L.I.A.T.E. : on dérive le $\ln$. $U=\ln x$, $V'=x$, donc $U'=\tfrac1x$, $V=\tfrac{x^2}{2}$.
\[
\int x\ln x\,\dd x=\frac{x^2}{2}\ln x-\int \frac{x^2}{2}\cdot\frac1x\,\dd x=\frac{x^2}{2}\ln x-\frac12\int x\,\dd x
=\boxed{\frac{x^2}{2}\ln x-\frac{x^2}{4}+C.}
\]
\textbf{Vérif.} $\Bigl[\tfrac{x^2}{2}\ln x-\tfrac{x^2}{4}\Bigr]'=x\ln x+\tfrac{x^2}{2}\cdot\tfrac1x-\tfrac{x}{2}=x\ln x+\tfrac x2-\tfrac x2=x\ln x$. \checkmark
\end{corr}

\begin{corr}[3]
$\displaystyle\int \arctan x\,\dd x$. On pose $U=\arctan x$, $V'=1$, donc $U'=\dfrac{1}{1+x^2}$, $V=x$.
\[
\int \arctan x\,\dd x=x\arctan x-\int \frac{x}{1+x^2}\,\dd x.
\]
Pour la dernière intégrale, forme $\dd W/W$ avec $W=1+x^2$, $\dd W=2x\,\dd x$ :
$\displaystyle\int\frac{x\,\dd x}{1+x^2}=\frac12\ln(1+x^2)$. D'où
\[
\boxed{\int \arctan x\,\dd x=x\arctan x-\frac12\ln(1+x^2)+C.}
\]
\textbf{Vérif.} $\bigl[x\arctan x-\tfrac12\ln(1+x^2)\bigr]'=\arctan x+\dfrac{x}{1+x^2}-\dfrac12\cdot\dfrac{2x}{1+x^2}=\arctan x$. \checkmark
\end{corr}

\begin{corr}[4]
$\displaystyle\int x\,e^{2x}\,\dd x$. $U=x$, $V'=e^{2x}$, donc $U'=1$, $V=\tfrac12 e^{2x}$.
\[
\int x\,e^{2x}\,\dd x=x\cdot\frac12 e^{2x}-\int \frac12 e^{2x}\,\dd x=\frac{x}{2}e^{2x}-\frac12\cdot\frac12 e^{2x}+C
=\boxed{\frac{2x-1}{4}\,e^{2x}+C.}
\]
\textbf{Vérif.} $\Bigl[\tfrac{x}{2}e^{2x}-\tfrac14 e^{2x}\Bigr]'=\tfrac12 e^{2x}+\tfrac x2\cdot 2e^{2x}-\tfrac14\cdot 2e^{2x}=\tfrac12 e^{2x}+x e^{2x}-\tfrac12 e^{2x}=x e^{2x}$. \checkmark
\end{corr}

\begin{corr}[5]
$\displaystyle\int x^{2}e^{x}\,\dd x$. \textbf{1\iere{} IPP :} $U=x^2$, $V'=e^x$ ($U'=2x$, $V=e^x$) :
\[
\int x^2 e^x\,\dd x=x^2 e^x-\int 2x\,e^x\,\dd x=x^2 e^x-2\int x\,e^x\,\dd x.
\]
\textbf{2\ieme{} IPP} (déjà connue) : $\int x e^x\,\dd x=(x-1)e^x+C$. Donc
\[
\int x^2 e^x\,\dd x=x^2 e^x-2(x-1)e^x+C=\boxed{(x^2-2x+2)\,e^x+C.}
\]
\textbf{Vérif.} $\bigl[(x^2-2x+2)e^x\bigr]'=(2x-2)e^x+(x^2-2x+2)e^x=(x^2)e^x=x^2 e^x$. \checkmark
\end{corr}
