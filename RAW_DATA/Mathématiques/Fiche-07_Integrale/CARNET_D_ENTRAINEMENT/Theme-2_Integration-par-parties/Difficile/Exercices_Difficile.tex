\section*{Niveau Difficile — IPP qui boucle, IPP doubles et récurrence}

\emph{On combine les gestes : deux IPP, résolution d'une équation en $I$, intégrale définie et
relation de récurrence. Chaque sous-question prépare la suivante.}

\begin{exo}[1 — l'IPP qui « boucle »]
On pose $\displaystyle I=\int e^{x}\sin x\,\dd x$ et $\displaystyle J=\int e^{x}\cos x\,\dd x$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item En intégrant \emph{deux fois} par parties (en gardant toujours $V'=e^x$), établir une relation
        entre $I$ et $J$, puis une seconde entre $J$ et $I$.
  \item En déduire la valeur de $I$, puis celle de $J$.
  \item Retrouver alors sans nouveau calcul $\displaystyle\int e^{x}(\sin x+\cos x)\,\dd x$, et interpréter le résultat.
\end{enumerate}
\end{exo}

\begin{exo}[2 — double IPP]
Calculer $\displaystyle \int x^{2}\cos x\,\dd x$ à l'aide de deux intégrations par parties successives.
\end{exo}

\begin{exo}[3 — IPP sur un intervalle]
Soit $\displaystyle A=\int_{0}^{1}(x+1)\,e^{x}\,\dd x$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item Déterminer une primitive de $x\mapsto(x+1)e^x$ par intégration par parties.
  \item En déduire la valeur exacte de $A$.
\end{enumerate}
\end{exo}

\begin{exo}[4 — une relation de récurrence]
Pour tout entier $n\geqslant 0$, on pose $\displaystyle I_{n}=\int x^{n}\,e^{x}\,\dd x$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2.2em,topsep=2pt,itemsep=2pt]
  \item Par une IPP, montrer que pour tout $n\geqslant 1$ : $\displaystyle I_{n}=x^{n}e^{x}-n\,I_{n-1}$.
  \item Sachant que $I_{0}=e^{x}+C$, calculer successivement $I_{1}$, $I_{2}$ et $I_{3}$.
\end{enumerate}
\end{exo}
