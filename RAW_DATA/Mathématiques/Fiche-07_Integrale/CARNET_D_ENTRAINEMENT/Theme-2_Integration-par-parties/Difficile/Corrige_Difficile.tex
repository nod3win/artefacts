\section*{Niveau Difficile — Corrigé}

\begin{corr}[1 — l'IPP qui « boucle »]
\textbf{(a)} On garde \emph{toujours} $V'=e^x$ (donc $V=e^x$).

\emph{Calcul de $I$} avec $U=\sin x$ ($U'=\cos x$) :
\[
I=\int e^{x}\sin x\,\dd x=e^{x}\sin x-\int e^{x}\cos x\,\dd x=e^{x}\sin x-J.
\]
\emph{Calcul de $J$} avec $U=\cos x$ ($U'=-\sin x$) :
\[
J=\int e^{x}\cos x\,\dd x=e^{x}\cos x-\int e^{x}(-\sin x)\,\dd x=e^{x}\cos x+I.
\]

\textbf{(b)} On substitue la 2\ieme{} relation dans la 1\iere{} :
\[
I=e^{x}\sin x-\bigl(e^{x}\cos x+I\bigr)\ \Longrightarrow\ 2I=e^{x}(\sin x-\cos x),
\]
d'où $\displaystyle \boxed{I=\tfrac12\,e^{x}(\sin x-\cos x)+C.}$ Puis
$J=e^{x}\cos x+I=\boxed{\tfrac12\,e^{x}(\sin x+\cos x)+C.}$

\textbf{(c)} Par linéarité, $\displaystyle\int e^{x}(\sin x+\cos x)\,\dd x=I+J=e^{x}\sin x+C.$
\textbf{Interprétation :} on retrouve directement que $\bigl(e^{x}\sin x\bigr)'=e^{x}\sin x+e^{x}\cos x
=e^{x}(\sin x+\cos x)$ : l'intégrande était exactement une dérivée de produit.
\end{corr}

\begin{corr}[2 — double IPP]
$\displaystyle\int x^{2}\cos x\,\dd x$. \textbf{1\iere{} IPP :} $U=x^{2}$, $V'=\cos x$ ($U'=2x$, $V=\sin x$) :
\[
\int x^{2}\cos x\,\dd x=x^{2}\sin x-\int 2x\sin x\,\dd x=x^{2}\sin x-2\int x\sin x\,\dd x.
\]
Or $\displaystyle\int x\sin x\,\dd x=-x\cos x+\sin x$ (niveau Facile). Donc
\[
\int x^{2}\cos x\,\dd x=x^{2}\sin x-2(-x\cos x+\sin x)+C=\boxed{x^{2}\sin x+2x\cos x-2\sin x+C.}
\]
\textbf{Vérif.} $\bigl[x^{2}\sin x+2x\cos x-2\sin x\bigr]'
=2x\sin x+x^{2}\cos x+2\cos x-2x\sin x-2\cos x=x^{2}\cos x$. \checkmark
\end{corr}

\begin{corr}[3 — IPP sur un intervalle]
\textbf{(a)} $\displaystyle\int (x+1)e^{x}\,\dd x$ avec $U=x+1$, $V'=e^{x}$ ($U'=1$, $V=e^{x}$) :
\[
\int (x+1)e^{x}\,\dd x=(x+1)e^{x}-\int e^{x}\,\dd x=(x+1)e^{x}-e^{x}+C=x\,e^{x}+C.
\]
Une primitive est donc $F(x)=x\,e^{x}$. \textbf{Vérif.} $(x e^{x})'=e^{x}+x e^{x}=(x+1)e^{x}$. \checkmark

\textbf{(b)} Par Newton--Leibniz,
\[
A=\int_{0}^{1}(x+1)e^{x}\,\dd x=\bigl[x\,e^{x}\bigr]_{0}^{1}=1\cdot e^{1}-0\cdot e^{0}=\boxed{e.}
\]
\end{corr}

\begin{corr}[4 — une relation de récurrence]
\textbf{(a)} Pour $n\geqslant 1$, IPP avec $U=x^{n}$, $V'=e^{x}$ ($U'=n x^{n-1}$, $V=e^{x}$) :
\[
I_{n}=\int x^{n}e^{x}\,\dd x=x^{n}e^{x}-\int n x^{n-1}e^{x}\,\dd x=\boxed{x^{n}e^{x}-n\,I_{n-1}.}
\]

\textbf{(b)} Avec $I_{0}=e^{x}+C$ :
\begin{align*}
I_{1}&=x e^{x}-1\cdot I_{0}=x e^{x}-e^{x}+C=(x-1)e^{x}+C,\\
I_{2}&=x^{2}e^{x}-2 I_{1}=x^{2}e^{x}-2(x-1)e^{x}+C=(x^{2}-2x+2)\,e^{x}+C,\\
I_{3}&=x^{3}e^{x}-3 I_{2}=x^{3}e^{x}-3(x^{2}-2x+2)e^{x}+C=(x^{3}-3x^{2}+6x-6)\,e^{x}+C.
\end{align*}
\textbf{Vérif.\ ($I_3$).} $\bigl[(x^{3}-3x^{2}+6x-6)e^{x}\bigr]'=(3x^{2}-6x+6)e^{x}+(x^{3}-3x^{2}+6x-6)e^{x}=x^{3}e^{x}$. \checkmark
\end{corr}
