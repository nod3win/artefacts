\section*{Thème 3 — Fractions rationnelles \quad\textbullet\quad Corrigé Facile}

\begin{corr}[1]
$\dfrac{2}{(x-1)(x+3)}=\dfrac{A}{x-1}+\dfrac{B}{x+3}$. Astuce «\,multiplier–évaluer\,» :
$A=\left.\dfrac{2}{x+3}\right|_{x=1}=\dfrac24=\tfrac12$,
$B=\left.\dfrac{2}{x-1}\right|_{x=-3}=\dfrac{2}{-4}=-\tfrac12$.
\emph{Contrôle :} $\tfrac12(x+3)-\tfrac12(x-1)=\tfrac12\cdot4=2$ \checkmark. Donc
\[
\int \frac{2}{(x-1)(x+3)}\,\dd x
=\frac12\ln|x-1|-\frac12\ln|x+3|+C.
\]
\end{corr}

\begin{corr}[2]
$\dfrac{1}{(x-2)(x-4)}=\dfrac{A}{x-2}+\dfrac{B}{x-4}$.
$A=\left.\dfrac{1}{x-4}\right|_{x=2}=\dfrac{1}{-2}=-\tfrac12$,
$B=\left.\dfrac{1}{x-2}\right|_{x=4}=\dfrac12$.
\emph{Contrôle :} $-\tfrac12(x-4)+\tfrac12(x-2)=-\tfrac12x+2+\tfrac12x-1=1$ \checkmark. Donc
\[
\int \frac{1}{(x-2)(x-4)}\,\dd x
=-\frac12\ln|x-2|+\frac12\ln|x-4|+C.
\]
\end{corr}

\begin{corr}[3]
$\dfrac{3}{(x+1)(x-2)}=\dfrac{A}{x+1}+\dfrac{B}{x-2}$.
$A=\left.\dfrac{3}{x-2}\right|_{x=-1}=\dfrac{3}{-3}=-1$,
$B=\left.\dfrac{3}{x+1}\right|_{x=2}=\dfrac{3}{3}=1$.
\emph{Contrôle :} $-1(x-2)+1(x+1)=-x+2+x+1=3$ \checkmark. Donc
\[
\int \frac{3}{(x+1)(x-2)}\,\dd x
=-\ln|x+1|+\ln|x-2|+C.
\]
\end{corr}

\begin{corr}[4]
$\dfrac{5}{(x-3)(x+2)}=\dfrac{A}{x-3}+\dfrac{B}{x+2}$.
$A=\left.\dfrac{5}{x+2}\right|_{x=3}=\dfrac{5}{5}=1$,
$B=\left.\dfrac{5}{x-3}\right|_{x=-2}=\dfrac{5}{-5}=-1$.
\emph{Contrôle :} $1(x+2)-1(x-3)=5$ \checkmark. Donc
\[
\int \frac{5}{(x-3)(x+2)}\,\dd x
=\ln|x-3|-\ln|x+2|+C.
\]
\end{corr}

\begin{corr}[5]
$\dfrac{4}{x(x-4)}=\dfrac{A}{x}+\dfrac{B}{x-4}$.
$A=\left.\dfrac{4}{x-4}\right|_{x=0}=\dfrac{4}{-4}=-1$,
$B=\left.\dfrac{4}{x}\right|_{x=4}=\dfrac{4}{4}=1$.
\emph{Contrôle :} $-1(x-4)+1\cdot x=4$ \checkmark. Donc
\[
\int \frac{4}{x(x-4)}\,\dd x
=-\ln|x|+\ln|x-4|+C.
\]
\end{corr}
