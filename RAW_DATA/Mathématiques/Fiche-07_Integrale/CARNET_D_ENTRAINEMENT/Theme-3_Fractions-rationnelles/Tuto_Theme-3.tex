\begin{center}
{\LARGE\bfseries\color{primarydark}Tuto — Thème 3}\\[2pt]
{\large\color{primary}Intégration par décomposition en fractions simples}
\end{center}

\vspace{-2mm}

Aucun tableau ne donne directement une primitive de $\dfrac{2}{(x-1)(x+3)}$. L'idée est de
\textbf{casser} ce quotient de polynômes en une \emph{somme} de fractions plus simples, chacune de
la forme $\dfrac{A}{x-a}$, que l'on sait intégrer aussitôt puisque
$\displaystyle\int\frac{\dd x}{x-a}=\ln|x-a|+C$.

\section*{1. Le principe}

\begin{idee}[décomposer avant d'intégrer]
Si le dénominateur se factorise en deux facteurs du premier degré \emph{distincts}
$(x-\alpha)(x-\beta)$, on cherche deux constantes $A,B$ telles que
\[
\frac{N(x)}{(x-\alpha)(x-\beta)}=\frac{A}{x-\alpha}+\frac{B}{x-\beta}.
\]
Chaque morceau s'intègre en $\ln$ : la difficulté se réduit à \textbf{trouver $A$ et $B$}.
\end{idee}

\section*{2. Deux méthodes pour trouver $A$ et $B$}

\begin{methode}[méthode 1 — identification des coefficients]
On réduit le membre de droite au même dénominateur, puis on \textbf{identifie} numérateur à numérateur :
\[
\frac{A}{x-\alpha}+\frac{B}{x-\beta}=\frac{A(x-\beta)+B(x-\alpha)}{(x-\alpha)(x-\beta)}.
\]
On regroupe en puissances de $x$ et on résout le petit système obtenu.
\end{methode}

\begin{methode}[méthode 2 — « multiplier–évaluer » (la plus rapide)]
Valable pour des pôles \textbf{simples}. Pour obtenir $A$, on multiplie par $(x-\alpha)$ puis on
pose $x=\alpha$ (ce qui annule l'autre terme) :
\[
A=\left.\frac{N(x)}{x-\beta}\right|_{x=\alpha},\qquad
B=\left.\frac{N(x)}{x-\alpha}\right|_{x=\beta}.
\]
\end{methode}

\begin{exemple}[l'exemple de référence, déroulé par les deux méthodes]
Calculer $\displaystyle\int\frac{2}{(x-1)(x+3)}\,\dd x$. On pose
$\dfrac{2}{(x-1)(x+3)}=\dfrac{A}{x-1}+\dfrac{B}{x+3}$.

\smallskip
\textbf{Méthode 1.} $\dfrac{A(x+3)+B(x-1)}{(x-1)(x+3)}=\dfrac{(A+B)x+(3A-B)}{(x-1)(x+3)}$.
Par identification : $A+B=0$ et $3A-B=2$. D'où $4A=2$, soit $A=\tfrac12$, $B=-\tfrac12$.

\smallskip
\textbf{Méthode 2.} $A=\left.\dfrac{2}{x+3}\right|_{x=1}=\dfrac24=\tfrac12$ ;\quad
$B=\left.\dfrac{2}{x-1}\right|_{x=-3}=\dfrac{2}{-4}=-\tfrac12$. \checkmark (même résultat)

\smallskip
\textbf{Intégration.}
\[
\int\frac{2}{(x-1)(x+3)}\,\dd x=\frac12\int\frac{\dd x}{x-1}-\frac12\int\frac{\dd x}{x+3}
=\frac12\ln|x-1|-\frac12\ln|x+3|+C=\frac12\ln\left|\frac{x-1}{x+3}\right|+C.
\]
\end{exemple}

\begin{exemple}[numérateur non constant]
$\displaystyle\frac{2x}{(x+1)(x+3)}=\frac{A}{x+1}+\frac{B}{x+3}$ avec
$A=\left.\dfrac{2x}{x+3}\right|_{x=-1}=\dfrac{-2}{2}=-1$ et
$B=\left.\dfrac{2x}{x+1}\right|_{x=-3}=\dfrac{-6}{-2}=3$. Donc
\[
\int\frac{2x}{(x+1)(x+3)}\,\dd x=-\ln|x+1|+3\ln|x+3|+C.
\]
\end{exemple}

\section*{3. La procédure complète}

\begin{methode}[intégrer une fraction rationnelle à deux pôles simples]
\begin{enumerate}[leftmargin=1.8em,topsep=2pt,itemsep=1pt]
  \item \textbf{Factoriser} le dénominateur en $(x-\alpha)(x-\beta)$ (résoudre le trinôme si besoin).
  \item \textbf{Poser} la décomposition $\dfrac{A}{x-\alpha}+\dfrac{B}{x-\beta}$.
  \item \textbf{Calculer} $A,B$ (multiplier–évaluer, ou identification).
  \item \textbf{Contrôler} en recombinant : $A(x-\beta)+B(x-\alpha)$ doit redonner $N(x)$.
  \item \textbf{Intégrer} chaque morceau : $\int\frac{\dd x}{x-a}=\ln|x-a|$, puis regrouper les $\ln$.
\end{enumerate}
\end{methode}

\begin{attention}[les réflexes qui sauvent]
\textbf{(1)} Toujours la \textbf{valeur absolue} : $\ln|x-a|$. \textbf{(2)} Pour une intégrale
\emph{définie}, vérifier que les pôles $\alpha,\beta$ ne sont pas dans l'intervalle (sinon la
fonction n'y est pas continue). \textbf{(3)} Sur un intervalle où $x-a>0$, on peut retirer la valeur
absolue et regrouper : $p\ln u+q\ln v=\ln(u^{p}v^{q})$.
\end{attention}
