\section*{Thème 3 — Fractions rationnelles \quad\textbullet\quad Corrigé Difficile}

\begin{corr}[1]
\textbf{a) Décomposition.} $\dfrac{2x}{(x+1)(x+3)}=\dfrac{A}{x+1}+\dfrac{B}{x+3}$ avec
$A=\left.\dfrac{2x}{x+3}\right|_{x=-1}=\dfrac{-2}{2}=-1$,
$B=\left.\dfrac{2x}{x+1}\right|_{x=-3}=\dfrac{-6}{-2}=3$.
\emph{Contrôle :} $-1(x+3)+3(x+1)=2x$ \checkmark. Donc
$\dfrac{2x}{(x+1)(x+3)}=\dfrac{-1}{x+1}+\dfrac{3}{x+3}$.

\smallskip
\textbf{b) Primitive.} Les pôles $-1$ et $-3$ ne sont pas dans $[2,3]$ ; sur cet intervalle
$x+1>0$ et $x+3>0$, donc
\[
F(x)=-\ln(x+1)+3\ln(x+3).
\]

\smallskip
\textbf{c) Évaluation.}
\begin{align*}
I=F(3)-F(2)&=\bigl(-\ln4+3\ln6\bigr)-\bigl(-\ln3+3\ln5\bigr)\\
&=(\ln3-\ln4)+3(\ln6-\ln5)
=\ln\tfrac34+3\ln\tfrac65.
\end{align*}
On regroupe : $3\ln\tfrac65=\ln\bigl(\tfrac65\bigr)^3=\ln\tfrac{216}{125}$, d'où
\[
I=\ln\!\Big(\frac34\cdot\frac{216}{125}\Big)=\boxed{\,\ln\frac{162}{125}\,}\approx 0{,}259.
\]
\end{corr}

\begin{corr}[2]
$\dfrac{1}{(x-1)(x-2)}=\dfrac{A}{x-1}+\dfrac{B}{x-2}$,
$A=\left.\dfrac{1}{x-2}\right|_{x=1}=\dfrac{1}{-1}=-1$,
$B=\left.\dfrac{1}{x-1}\right|_{x=2}=1$.
\emph{Contrôle :} $-1(x-2)+1(x-1)=1$ \checkmark.
Les pôles $1$ et $2$ ne sont pas dans $[3,4]$ ; sur $[3,4]$, $x-1>0$ et $x-2>0$, donc
$F(x)=-\ln(x-1)+\ln(x-2)=\ln\dfrac{x-2}{x-1}$.
\[
\int_{3}^{4}\frac{\dd x}{(x-1)(x-2)}=F(4)-F(3)=\ln\frac{2}{3}-\ln\frac{1}{2}
=\ln\!\Big(\frac23\cdot 2\Big)=\boxed{\,\ln\frac43\,}\approx 0{,}288.
\]
\end{corr}

\begin{corr}[3]
$\dfrac{4}{(x+1)(x+3)}=\dfrac{A}{x+1}+\dfrac{B}{x+3}$,
$A=\left.\dfrac{4}{x+3}\right|_{x=-1}=\dfrac42=2$,
$B=\left.\dfrac{4}{x+1}\right|_{x=-3}=\dfrac{4}{-2}=-2$.
\emph{Contrôle :} $2(x+3)-2(x+1)=4$ \checkmark.
Les pôles $-1$ et $-3$ ne sont pas dans $[0,1]$ ; sur $[0,1]$ les deux facteurs sont positifs,
donc $F(x)=2\ln(x+1)-2\ln(x+3)=2\ln\dfrac{x+1}{x+3}$.
\[
\int_{0}^{1}\frac{4\,\dd x}{(x+1)(x+3)}=F(1)-F(0)
=2\ln\frac{2}{4}-2\ln\frac{1}{3}
=2\Big(\ln\tfrac12+\ln 3\Big)=2\ln\frac32=\boxed{\,\ln\frac94\,}\approx 0{,}811.
\]
\end{corr}

\begin{corr}[4]
\textbf{a) Factorisation et décomposition.} $x^2+3x+2=(x+1)(x+2)$ (racines $-1$ et $-2$).
$\dfrac{x+3}{(x+1)(x+2)}=\dfrac{A}{x+1}+\dfrac{B}{x+2}$,
$A=\left.\dfrac{x+3}{x+2}\right|_{x=-1}=\dfrac{2}{1}=2$,
$B=\left.\dfrac{x+3}{x+1}\right|_{x=-2}=\dfrac{1}{-1}=-1$.
\emph{Contrôle :} $2(x+2)-1(x+1)=2x+4-x-1=x+3$ \checkmark. Donc
$\dfrac{x+3}{x^2+3x+2}=\dfrac{2}{x+1}-\dfrac{1}{x+2}$.

\smallskip
\textbf{b) Primitive.} Les pôles $-1,-2$ ne sont pas dans $[1,2]$ ; sur cet intervalle les facteurs
sont positifs, donc $G(x)=2\ln(x+1)-\ln(x+2)$.

\smallskip
\textbf{c) Évaluation.}
\begin{align*}
J=G(2)-G(1)&=\bigl(2\ln3-\ln4\bigr)-\bigl(2\ln2-\ln3\bigr)\\
&=3\ln3-\ln4-2\ln2
=3\ln3-2\ln2-2\ln2=3\ln3-4\ln2.
\end{align*}
On regroupe : $3\ln3=\ln27$ et $4\ln2=\ln16$, d'où
\[
J=\ln27-\ln16=\boxed{\,\ln\frac{27}{16}\,}\approx 0{,}523.
\]
\end{corr}
