\begin{exo}[1 --- fonction racine]
Soit $f(x)=\sqrt{x}$ avec $x_0>0$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=1pt]
  \item Écrire le taux d'accroissement $\tau(h)$.
  \item Multiplier numérateur et dénominateur par la quantité conjuguée
        $\sqrt{x_0+h}+\sqrt{x_0}$.
  \item En déduire $\ds f'(x_0)=\frac{1}{2\sqrt{x_0}}$.
  \item Calculer $f'(4)$.
\end{enumerate}
\end{exo}

\begin{exo}[2 --- mobile MRUA]
Un mobile a pour position $y=f(x)=\tfrac{1}{2}x^2+2x$ (vitesse initiale $2$~m/s,
accélération $1$~m/s$^2$).
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=1pt]
  \item Écrire le taux d'accroissement $\tau(h)$.
  \item Développer et simplifier.
  \item En déduire $f'(x_0)=x_0+2$.
  \item Calculer la vitesse instantanée à l'instant $t=3$~s.
\end{enumerate}
\end{exo}

\begin{exo}[3 --- fonction homographique]
Soit $f(x)=\dfrac{1}{x+1}$ avec $x_0\neq -1$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=1pt]
  \item Écrire le taux d'accroissement $\tau(h)$.
  \item Réduire le numérateur au même dénominateur.
  \item En déduire $\ds f'(x_0)=-\frac{1}{(x_0+1)^2}$.
\end{enumerate}
\end{exo}

\begin{exo}[4 --- valeur absolue, point anguleux]
Soit $f(x)=|x|$ que l'on étudie en $x_0=0$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=1pt]
  \item Calculer le taux d'accroissement $\tau(h)=\dfrac{f(0+h)-f(0)}{h}$
        d'abord pour $h>0$, puis pour $h<0$.
  \item Que valent les limites de $\tau(h)$ à droite ($h\to 0^+$) et à gauche
        ($h\to 0^-$) ?
  \item Conclure : $f$ est-elle dérivable en $0$ ? Interpréter géométriquement.
\end{enumerate}
\end{exo}
