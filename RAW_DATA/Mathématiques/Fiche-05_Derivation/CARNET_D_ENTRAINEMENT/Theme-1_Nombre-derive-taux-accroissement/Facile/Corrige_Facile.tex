{\small\itshape Taux d'accroissement : variation moyenne \& pente de la sécante}
\end{center}
\vspace{2mm}

\begin{corr}[1 --- $f(x)=x^2$ entre $x_0=2$ et $2+h$]
On applique la définition avec $x_0=2$ :
\[
\tau(h)=\frac{f(2+h)-f(2)}{h}=\frac{(2+h)^2-2^2}{h}.
\]
On développe le numérateur : $(2+h)^2-4=4+4h+h^2-4=4h+h^2=h(4+h)$. D'où
\[
\tau(h)=\frac{h(4+h)}{h}=\boxed{4+h}\qquad(h\neq 0).
\]
\end{corr}

\begin{corr}[2 --- $f(x)=3x+1$ en $x_0$ quelconque]
\[
\tau(h)=\frac{f(x_0+h)-f(x_0)}{h}
=\frac{\big(3(x_0+h)+1\big)-\big(3x_0+1\big)}{h}.
\]
Le numérateur vaut $3x_0+3h+1-3x_0-1=3h$, donc
\[
\tau(h)=\frac{3h}{h}=\boxed{3}.
\]
Le taux est \textbf{constant}, égal à $3$ : c'est normal, la courbe est une droite de
pente $3$, toutes ses sécantes ont la même pente.
\end{corr}

\begin{corr}[3 --- $f(x)=x^2-x$ entre $x_0=1$ et $1+h$]
\[
\tau(h)=\frac{f(1+h)-f(1)}{h}
=\frac{\big[(1+h)^2-(1+h)\big]-\big[1^2-1\big]}{h}.
\]
Or $f(1)=1-1=0$ et $(1+h)^2-(1+h)=1+2h+h^2-1-h=h+h^2=h(1+h)$. Donc
\[
\tau(h)=\frac{h(1+h)}{h}=\boxed{1+h}\qquad(h\neq 0).
\]
\end{corr}

\begin{corr}[4 --- vitesse moyenne, $y=f(x)=x^2$]
La vitesse moyenne entre $t=2$~s et $t=5$~s est le taux d'accroissement entre ces deux
instants :
\[
v_{\text{moy}}=\frac{f(5)-f(2)}{5-2}=\frac{5^2-2^2}{3}=\frac{25-4}{3}=\frac{21}{3}
=\boxed{7\ \text{m/s}}.
\]
\end{corr}

\begin{corr}[5 --- pente de la sécante $(AB)$]
La pente de la droite passant par $A(1\,;2)$ et $B(4\,;8)$ est le rapport de la variation
des ordonnées sur celle des abscisses :
\[
\text{pente}=\frac{y_B-y_A}{x_B-x_A}=\frac{8-2}{4-1}=\frac{6}{3}=\boxed{2}.
\]
\end{corr}
