{\small\itshape Exercices --- Niveau Facile : calculer $f'$, résoudre $f'=0$, lire le signe}
\end{center}
\vspace{2mm}

\begin{exo}[1]
Soit $f(x)=x^2-4x+3$. Calculer $f'$, résoudre $f'(x)=0$, puis en déduire la nature
et la valeur de l'extremum.
\end{exo}

\begin{exo}[2]
Soit $f(x)=-x^2+2x$. Déterminer la nature de l'extremum de $f$.
\end{exo}

\begin{exo}[3]
Soit $f(x)=x^3-3x$. Trouver les racines de $f'$ et étudier leur signe.
\end{exo}

\begin{exo}[4]
Soit $f(x)=x^4-2x^2$. Déterminer les racines de $f'$.
\end{exo}

\begin{exo}[5]
Soit $f(x)=x^3$. Montrer que $f'(x)\geq 0$ sur $\R$ et que $f'$ s'annule en $0$
\textbf{sans changer de signe}. Qu'en conclure sur l'existence d'un extremum ?
\end{exo}
