\begin{corr}[1 --- $f(x)=x^2-4x+3$]
$f'(x)=2x-4$. \quad $f'(x)=0 \iff 2x-4=0 \iff x=2$.\\
Signe : $f'(x)=2(x-2)$ est du signe de $x-2$ : négatif avant $2$, positif après.
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) rectangle (7,1.6);
\draw[thick] (0,0.8) -- (7,0.8);
\draw[thick] (1.4,0) -- (1.4,1.6);
\node at (0.7,1.2) {$x$}; \node at (0.7,0.4) {$f'(x)$};
\node at (1.9,1.2) {$-\infty$}; \node at (4.0,1.2) {$2$}; \node at (6.6,1.2) {$+\infty$};
\node[moins] at (2.8,0.4) {$-$}; \node at (4.0,0.4) {$0$}; \node[plus] at (5.4,0.4) {$+$};
\end{tikzpicture}
\end{center}
$f'$ passe de $-$ à $+$ : \textbf{minimum} en $x=2$, $\ f(2)=4-8+3=\mathbf{-1}$.
\end{corr}

\begin{corr}[2 --- $f(x)=-x^2+2x$]
$f'(x)=-2x+2=-2(x-1)$. \quad $f'(x)=0 \iff x=1$.\\
Signe de $-2(x-1)$ : positif avant $1$, négatif après.
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) rectangle (7,1.6);
\draw[thick] (0,0.8) -- (7,0.8);
\draw[thick] (1.4,0) -- (1.4,1.6);
\node at (0.7,1.2) {$x$}; \node at (0.7,0.4) {$f'(x)$};
\node at (1.9,1.2) {$-\infty$}; \node at (4.0,1.2) {$1$}; \node at (6.6,1.2) {$+\infty$};
\node[plus] at (2.8,0.4) {$+$}; \node at (4.0,0.4) {$0$}; \node[moins] at (5.4,0.4) {$-$};
\end{tikzpicture}
\end{center}
$f'$ passe de $+$ à $-$ : \textbf{maximum} en $x=1$, $\ f(1)=-1+2=\mathbf{1}$.
\end{corr}

\begin{corr}[3 --- $f(x)=x^3-3x$]
$f'(x)=3x^2-3=3(x^2-1)=3(x-1)(x+1)$. \quad Racines : $x=-1$ et $x=1$.\\
Signe du produit $(x-1)(x+1)$ (parabole $\smile$) : positif à l'extérieur, négatif entre $-1$ et $1$.
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) rectangle (8,1.6);
\draw[thick] (0,0.8) -- (8,0.8);
\draw[thick] (1.4,0) -- (1.4,1.6);
\node at (0.7,1.2) {$x$}; \node at (0.7,0.4) {$f'(x)$};
\node at (1.9,1.2) {$-\infty$}; \node at (3.6,1.2) {$-1$}; \node at (5.6,1.2) {$1$}; \node at (7.5,1.2) {$+\infty$};
\node[plus] at (2.7,0.4) {$+$}; \node at (3.6,0.4) {$0$}; \node[moins] at (4.6,0.4) {$-$};
\node at (5.6,0.4) {$0$}; \node[plus] at (6.6,0.4) {$+$};
\end{tikzpicture}
\end{center}
$f'$ change de signe en $-1$ ($+\to-$ : \textbf{maximum}) et en $1$ ($-\to+$ : \textbf{minimum}).
\end{corr}

\begin{corr}[4 --- $f(x)=x^4-2x^2$]
$f'(x)=4x^3-4x=4x(x^2-1)=4x(x-1)(x+1)$. \quad Racines : $x=-1$, $x=0$, $x=1$.\\
Règle des signes du produit de trois facteurs (change de signe à chaque racine simple) :
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) rectangle (9,1.6);
\draw[thick] (0,0.8) -- (9,0.8);
\draw[thick] (1.4,0) -- (1.4,1.6);
\node at (0.7,1.2) {$x$}; \node at (0.7,0.4) {$f'(x)$};
\node at (1.9,1.2) {$-\infty$}; \node at (3.4,1.2) {$-1$}; \node at (5.0,1.2) {$0$};
\node at (6.6,1.2) {$1$}; \node at (8.4,1.2) {$+\infty$};
\node[moins] at (2.6,0.4) {$-$}; \node at (3.4,0.4) {$0$}; \node[plus] at (4.2,0.4) {$+$};
\node at (5.0,0.4) {$0$}; \node[moins] at (5.8,0.4) {$-$}; \node at (6.6,0.4) {$0$};
\node[plus] at (7.5,0.4) {$+$};
\end{tikzpicture}
\end{center}
Trois valeurs critiques, avec changement de signe à chacune : minimum en $-1$, maximum en $0$, minimum en $1$.
\end{corr}

\begin{corr}[5 --- $f(x)=x^3$]
$f'(x)=3x^2$. Comme un carré, $x^2\geq 0$, donc $f'(x)=3x^2\geq 0$ pour tout $x$ : $f$ est croissante partout.
De plus $f'(x)=0 \iff x=0$, mais $f'$ reste \textbf{positive} de part et d'autre de $0$ (pas de changement de signe).
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) rectangle (7,1.6);
\draw[thick] (0,0.8) -- (7,0.8);
\draw[thick] (1.4,0) -- (1.4,1.6);
\node at (0.7,1.2) {$x$}; \node at (0.7,0.4) {$f'(x)$};
\node at (1.9,1.2) {$-\infty$}; \node at (4.0,1.2) {$0$}; \node at (6.6,1.2) {$+\infty$};
\node[plus] at (2.8,0.4) {$+$}; \node at (4.0,0.4) {$0$}; \node[plus] at (5.4,0.4) {$+$};
\end{tikzpicture}
\end{center}
Pas de changement de signe $\Rightarrow$ \textbf{aucun extremum}. En $x=0$ la tangente est horizontale
mais la courbe traverse : c'est un \textbf{point d'inflexion à tangente horizontale}.
\end{corr}
