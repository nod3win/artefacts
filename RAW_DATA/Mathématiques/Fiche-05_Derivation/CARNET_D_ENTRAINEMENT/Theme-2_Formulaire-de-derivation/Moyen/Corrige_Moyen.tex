\begin{center}
{\LARGE\bfseries\color{primarydark} Produit et quotient --- corrigé}\\[2pt]
{\color{primary}\rule{0.5\textwidth}{1.2pt}}\\[4pt]
{\small\itshape Niveau Moyen --- règles $(UV)'$ et $(U/V)'$}
\end{center}
\vspace{2mm}

\begin{corr}[1 --- règle du produit]
On pose $U=x^{2}+1$ et $V=x-3$, donc $U'=2x$ et $V'=1$.
\[
y'=U'V+UV'=2x\,(x-3)+(x^{2}+1)\cdot 1
=2x^{2}-6x+x^{2}+1=3x^{2}-6x+1.
\]
\end{corr}

\begin{corr}[2 --- produit avec racine]
On pose $U=x$ et $V=\sqrt{x}=x^{1/2}$, donc $U'=1$ et $V'=\dfrac{1}{2\sqrt{x}}$.
\[
y'=U'V+UV'=\sqrt{x}+x\cdot\frac{1}{2\sqrt{x}}
=\sqrt{x}+\frac{x}{2\sqrt{x}}=\sqrt{x}+\frac{\sqrt{x}}{2}=\frac{3}{2}\sqrt{x}.
\]
\textbf{Vérification :} $y=x\cdot x^{1/2}=x^{3/2}\Rightarrow y'=\tfrac32 x^{1/2}=\tfrac32\sqrt{x}$. \checkmark
\end{corr}

\begin{corr}[3 --- quotient affine]
On pose $U=2x-1$ et $V=x+3$, donc $U'=2$ et $V'=1$.
\[
y'=\frac{U'V-UV'}{V^{2}}
=\frac{2\,(x+3)-(2x-1)\cdot 1}{(x+3)^{2}}
=\frac{2x+6-2x+1}{(x+3)^{2}}=\frac{7}{(x+3)^{2}}.
\]
\end{corr}

\begin{corr}[4 --- quotient avec carré]
On pose $U=x^{2}$ et $V=x^{2}+1$, donc $U'=2x$ et $V'=2x$.
\[
y'=\frac{U'V-UV'}{V^{2}}
=\frac{2x\,(x^{2}+1)-x^{2}\cdot 2x}{(x^{2}+1)^{2}}
=\frac{2x^{3}+2x-2x^{3}}{(x^{2}+1)^{2}}=\frac{2x}{(x^{2}+1)^{2}}.
\]
\end{corr}

\begin{corr}[5 --- quotient affine]
On pose $U=x+1$ et $V=x-1$, donc $U'=1$ et $V'=1$.
\[
y'=\frac{U'V-UV'}{V^{2}}
=\frac{1\cdot(x-1)-(x+1)\cdot 1}{(x-1)^{2}}
=\frac{x-1-x-1}{(x-1)^{2}}=\frac{-2}{(x-1)^{2}}.
\]
\end{corr}
