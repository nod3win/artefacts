\begin{corr}[1 --- produit avec racine et réduction]
\textbf{a)} $U=x^{2}-1$ et $V=\sqrt{x}=x^{1/2}$, donc $U'=2x$ et $V'=\dfrac{1}{2\sqrt{x}}$.\\[2pt]
\textbf{b)} Règle du produit :
\[
y'=U'V+UV'=2x\sqrt{x}+(x^{2}-1)\cdot\frac{1}{2\sqrt{x}}
=2x\sqrt{x}+\frac{x^{2}-1}{2\sqrt{x}}.
\]
\textbf{c)} On réduit au même dénominateur $2\sqrt{x}$, en écrivant $2x\sqrt{x}=\dfrac{2x\sqrt{x}\cdot 2\sqrt{x}}{2\sqrt{x}}=\dfrac{4x^{2}}{2\sqrt{x}}$ (car $\sqrt{x}\cdot\sqrt{x}=x$) :
\[
y'=\frac{4x^{2}}{2\sqrt{x}}+\frac{x^{2}-1}{2\sqrt{x}}
=\frac{4x^{2}+x^{2}-1}{2\sqrt{x}}=\frac{5x^{2}-1}{2\sqrt{x}}.
\]
\end{corr}

\begin{corr}[2 --- quotient avec développement]
\textbf{a)} $U=x^{2}+2x$ et $V=x^{2}+1$, donc $U'=2x+2$ et $V'=2x$.\\[2pt]
\textbf{b)} Règle du quotient :
\[
y'=\frac{U'V-UV'}{V^{2}}
=\frac{(2x+2)(x^{2}+1)-(x^{2}+2x)(2x)}{(x^{2}+1)^{2}}.
\]
\textbf{c)} On développe le numérateur :
\[
(2x+2)(x^{2}+1)=2x^{3}+2x^{2}+2x+2,\qquad (x^{2}+2x)(2x)=2x^{3}+4x^{2},
\]
\[
\text{numérateur}=2x^{3}+2x^{2}+2x+2-2x^{3}-4x^{2}=-2x^{2}+2x+2,
\]
d'où
\[
y'=\frac{-2x^{2}+2x+2}{(x^{2}+1)^{2}}.
\]
\end{corr}

\begin{corr}[3 --- quotient affine à simplifier]
\textbf{a)} $U=3x-2$ et $V=2x+1$, donc $U'=3$ et $V'=2$ :
\[
y'=\frac{U'V-UV'}{V^{2}}=\frac{3\,(2x+1)-(3x-2)\cdot 2}{(2x+1)^{2}}.
\]
\textbf{b)} On simplifie le numérateur :
\[
3(2x+1)-2(3x-2)=6x+3-6x+4=7,
\qquad\text{donc}\qquad
y'=\frac{7}{(2x+1)^{2}}.
\]
\end{corr}

\begin{corr}[4 --- produit vs développement]
\textbf{a)} \emph{Par le produit :} $U=x$ et $V=x^{2}-4$, donc $U'=1$ et $V'=2x$ :
\[
y'=U'V+UV'=(x^{2}-4)+x\cdot 2x=x^{2}-4+2x^{2}=3x^{2}-4.
\]
\textbf{b)} \emph{Après développement :} $y=x(x^{2}-4)=x^{3}-4x$, donc
\[
y'=3x^{2}-4.
\]
\textbf{c)} Les deux méthodes donnent le même résultat $3x^{2}-4$ : cohérence confirmée. \checkmark
\end{corr}
