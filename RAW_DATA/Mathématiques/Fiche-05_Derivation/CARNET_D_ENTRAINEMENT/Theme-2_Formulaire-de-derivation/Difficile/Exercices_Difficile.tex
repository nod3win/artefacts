\begin{methode}[démarche en sous-questions]
Pose toujours $U$, $V$, $U'$, $V'$ avant d'appliquer la formule. Ensuite seulement, on développe le numérateur, on réduit au même dénominateur, et on vérifie le résultat par une autre voie quand c'est possible.
\end{methode}

\begin{exo}[1 --- produit avec racine et réduction]
On considère $\;y=(x^{2}-1)\sqrt{x}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
\item Poser $U$, $V$ et calculer $U'$, $V'$.
\item Appliquer la règle du produit.
\item Réduire au même dénominateur pour obtenir $\dfrac{5x^{2}-1}{2\sqrt{x}}$.
\end{enumerate}
\end{exo}

\begin{exo}[2 --- quotient avec développement]
On considère $\;y=\dfrac{x^{2}+2x}{x^{2}+1}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
\item Poser $U$, $V$, $U'$, $V'$.
\item Appliquer la règle du quotient.
\item Développer le numérateur pour obtenir $\dfrac{-2x^{2}+2x+2}{(x^{2}+1)^{2}}$.
\end{enumerate}
\end{exo}

\begin{exo}[3 --- quotient affine à simplifier]
On considère $\;y=\dfrac{3x-2}{2x+1}$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
\item Dériver par la règle du quotient.
\item Simplifier le numérateur pour obtenir $\dfrac{7}{(2x+1)^{2}}$.
\end{enumerate}
\end{exo}

\begin{exo}[4 --- produit vs développement]
On considère $\;y=x(x^{2}-4)$.
\begin{enumerate}[label=\alph*),leftmargin=1.6em,itemsep=1pt]
\item Dériver par la règle du produit ; montrer que $y'=3x^{2}-4$.
\item Dériver après développement (en écrivant d'abord $y=x^{3}-4x$).
\item Vérifier que les deux résultats coïncident.
\end{enumerate}
\end{exo}
