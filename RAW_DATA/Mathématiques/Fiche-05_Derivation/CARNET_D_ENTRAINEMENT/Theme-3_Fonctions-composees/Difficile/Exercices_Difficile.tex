\begin{cle}[consigne]
Ici l'oignon a \emph{plusieurs couches}, ou bien la fonction intérieure est \emph{inconnue}
($f$, $u$). Les sous-questions te guident : traite une couche à la fois, puis assemble.
\end{cle}

\begin{exo}[1 --- trois couches : carré $\circ$ sinus $\circ$ racine]
Soit $y=\sin^2\!\big(\sqrt{x}+1\big)$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Couche externe (le carré) : justifier que $y'=2\sin(\sqrt x+1)\cdot\big(\sin(\sqrt x+1)\big)'$.
\item Couche sinus : montrer que $\big(\sin(\sqrt x+1)\big)'=\dfrac{1}{2\sqrt x}\cos(\sqrt x+1)$.
\item Assembler puis, à l'aide de $2\sin\theta\cos\theta=\sin 2\theta$, montrer que
      $y'=\dfrac{1}{2\sqrt x}\,\sin\!\big(2(\sqrt x+1)\big)$. Préciser le domaine.
\end{enumerate}
\end{exo}

\begin{exo}[2 --- racine $\circ$ logarithme $\circ$ cosinus]
Soit $y=\sqrt{\ln(\cos x+2)}$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Réécrire $y=\big(\ln(\cos x+2)\big)^{1/2}$ et dériver la couche puissance $\tfrac12$.
\item Dériver la couche logarithme (poser $U=\cos x+2$, donner $U'$).
\item Assembler et montrer que
      $y'=\dfrac{-\sin x}{2(\cos x+2)\sqrt{\ln(\cos x+2)}}$.
\end{enumerate}
{\footnotesize\itshape Remarque : $\cos x+2\ge 1>0$, la racine et le $\ln$ sont bien définis.}
\end{exo}

\begin{exo}[3 --- fonction inconnue $f$]
Soit $f$ une fonction dérivable.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Montrer que $\big(f^2\big)'=2f'f$ (par la règle du produit ou la formule $(U^2)'$).
\item Montrer que $\big(3\ln|f|\big)'=\dfrac{3f'}{f}$ (là où $f\neq 0$).
\item Application : $f(x)=x^2+1$. Calculer $\big(f^2\big)'$ et $\big(3\ln|f|\big)'$.
\end{enumerate}
\end{exo}

\begin{exo}[4 --- fonction inconnue $u$ et changement de variable]
Soit $u$ une fonction dérivable.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item On pose $v(x)=u(3x)$. Montrer que $v'(x)=3u'(3x)$. Si $u(x)=x^2$, vérifier que $v'(x)=18x$.
\item On pose $v(x)=u\big(\ln(x+1)\big)$. Exprimer $v'(x)$.
\end{enumerate}
\end{exo}
