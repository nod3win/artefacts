\begin{exo}[1 --- Lecture graphique d'une cubique]
On considère la fonction $f(x)=x^3-3x$, dont la courbe $\mathcal{C}_f$ est représentée ci-dessous. Elle monte, atteint un \textbf{maximum} en $x=-1$, descend jusqu'à un \textbf{minimum} en $x=1$, puis remonte.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm,height=6.4cm,axis lines=middle,
  xlabel={$x$},ylabel={$y$},xlabel style={right},ylabel style={above},
  xmin=-2.4,xmax=2.4,ymin=-3.2,ymax=3.2,
  xtick={-2,-1,0,1,2},ytick={-2,2},
  grid=both,grid style={grille!55,very thin},tick label style={font=\scriptsize},
]
\addplot[courbe,very thick,domain=-2.05:2.05,samples=120]{x^3-3*x};
\addplot[only marks,mark=*,mark size=1.8pt,black]coordinates{(-1,2)(1,-2)(0,0)};
\node[anchor=south,font=\scriptsize] at (axis cs:-1,2){max};
\node[anchor=north,font=\scriptsize] at (axis cs:1,-2){min};
\node[anchor=south west,font=\scriptsize] at (axis cs:0,0){inflexion};
\end{axis}
\end{tikzpicture}
\end{center}
Répondre par \textbf{vrai/faux} en \textbf{justifiant} par la lecture graphique.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Donner le \textbf{signe de $f'$} sur chacun des intervalles $]-\infty;-1[$, $]-1;1[$ et $]1;+\infty[$.
\item Que vaut $f'(-1)$ et $f'(1)$ ? Justifier.
\item Où a-t-on $f''>0$ et où a-t-on $f''<0$ ? En déduire l'abscisse du \textbf{point d'inflexion}.
\end{enumerate}
\end{exo}

\begin{exo}[2 --- Reconnaître la courbe de $f'$]
Soit $f$ une fonction \textbf{cubique} (de la forme $ax^3+bx^2+cx+d$) qui admet un \textbf{maximum} en $x=0$ et un \textbf{minimum} en $x=2$.
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Que valent $f'(0)$ et $f'(2)$ ?
\item Déterminer le \textbf{signe de $f'$} avant $0$, entre $0$ et $2$, puis après $2$.
\item En déduire que la courbe de $f'$ est une \textbf{parabole tournée vers le haut} passant par les points $(0;0)$ et $(2;0)$.
\end{enumerate}
\end{exo}

\begin{exo}[3 --- Reconstituer les paramètres]
Soit $f(x)=ax^2+bx+c$ (avec $a\neq 0$). On sait que :
\begin{itemize}[leftmargin=7mm,itemsep=1pt]
\item[(i)] la courbe $\mathcal{C}_f$ passe par le point $(0;2)$ ;
\item[(ii)] la tangente en $x=1$ est \textbf{horizontale} ;
\item[(iii)] $f(1)=0$.
\end{itemize}
\begin{enumerate}[label=\alph*),leftmargin=7mm,itemsep=2pt]
\item Traduire chacune des trois conditions (i), (ii), (iii) par une équation portant sur $a$, $b$, $c$.
\item Résoudre le système obtenu.
\item Donner l'expression de $f$ et la factoriser.
\end{enumerate}
\end{exo}
