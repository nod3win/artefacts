\begin{center}
{\LARGE\bfseries\color{primarydark}Thème 4 — Produit scalaire et orthogonalité}\\[2pt]
{\color{primary}\rule{0.66\linewidth}{1.2pt}}\\[4pt]
{\small\itshape Une opération entre deux vecteurs dont le résultat est un \emph{nombre} — et le test
imparable de la perpendicularité.}
\end{center}
\vspace{1mm}

\begin{rappel}[les deux visages du produit scalaire]
Pour $\vec u\,(x;y)$ et $\vec v\,(x';y')$ :
\[
  \boxed{\;\vec u\cdot\vec v=x x'+y y'\;}\ \text{(analytique)}
  \qquad\qquad
  \boxed{\;\vec u\cdot\vec v=\norm{\vec u}\,\norm{\vec v}\,\cos(\vec u,\vec v)\;}\ \text{(géométrique)}
\]
Le résultat est un \textbf{réel} (positif, négatif ou nul), \emph{pas} un vecteur.
\end{rappel}

\begin{rappel}[propriétés à connaître]
\[
  \vec u\cdot\vec v=\vec v\cdot\vec u\ \text{(symétrie)},\qquad
  \vec u\cdot\vec u=\norm{\vec u}^2\ \text{(carré scalaire)},
\]
\[
  (k\vec u)\cdot\vec v=k(\vec u\cdot\vec v),\qquad
  (\vec u+\vec w)\cdot\vec v=\vec u\cdot\vec v+\vec w\cdot\vec v\ \text{(linéarité)}.
\]
\end{rappel}

\begin{idee}[le critère d'orthogonalité, LE réflexe du thème]
Deux vecteurs \emph{non nuls} sont \textbf{orthogonaux (perpendiculaires)} si et seulement si leur
produit scalaire est \textbf{nul} :
\[
  \boxed{\;\vec u\perp\vec v \iff \vec u\cdot\vec v=0 \iff x x'+y y'=0\;}
\]
C'est le test le plus rapide pour la perpendicularité : on calcule $xx'+yy'$ et on regarde s'il vaut
$0$ — pas besoin de l'angle. Le \emph{signe} renseigne d'ailleurs sur l'angle :
$\vec u\cdot\vec v>0$ (angle aigu), $=0$ (droit), $<0$ (angle obtus).
\end{idee}

\begin{methode}[tester si deux droites sont perpendiculaires]
\begin{enumerate}[leftmargin=1.7em,topsep=2pt,itemsep=2pt]
  \item Trouver un \textbf{vecteur directeur} de chaque droite ($\vc{AB}$ si elle passe par $A$ et $B$,
        ou $(-b;a)$ à partir de $ax+by+c=0$).
  \item Calculer le \textbf{produit scalaire} des deux vecteurs directeurs.
  \item S'il est \textbf{nul}, les droites sont perpendiculaires. (De façon équivalente : $m\times m'=-1$.)
\end{enumerate}
\end{methode}

\begin{exemple}[calcul et orthogonalité]
$\vec u\,(3;2)$ et $\vec v\,(-2;3)$ : $\vec u\cdot\vec v=3\times(-2)+2\times3=-6+6=0$. Les vecteurs sont
\textbf{orthogonaux}.\\
$\vec a\,(1;4)$ et $\vec b\,(2;3)$ : $\vec a\cdot\vec b=2+12=14>0$ : angle \emph{aigu}, non orthogonaux.
\end{exemple}

\begin{exemple}[carré scalaire]
$\vec u\,(3;4)$ : $\vec u\cdot\vec u=3^2+4^2=25=\norm{\vec u}^2$, ce qui redonne $\norm{\vec u}=5$.
Le produit scalaire d'un vecteur par lui-même est le carré de sa norme.
\end{exemple}

\begin{exemple}[une application : le travail d'une force]
Une force $\vec F\,(6;8)$ (en newtons) déplace un objet selon $\vec d\,(3;0)$ (en mètres). Le
\emph{travail} est $W=\vec F\cdot\vec d=6\times3+8\times0=18$ joules. Si le déplacement était
perpendiculaire à la force, le travail serait \emph{nul}.
\end{exemple}

\begin{piege}[à ne pas confondre]
\begin{itemize}[leftmargin=1.5em,topsep=2pt,itemsep=1pt]
  \item Produit scalaire $=xx'+yy'$ (on \emph{additionne}). Colinéarité $=xy'-x'y$ (on soustrait,
        \emph{en croix}). Deux formules différentes !
  \item Le résultat d'un produit scalaire est un \textbf{nombre}, jamais un vecteur.
  \item $\vec u\cdot\vec v=0$ signifie \textbf{perpendiculaires}, pas « colinéaires » ni « vecteur nul ».
\end{itemize}
\end{piege}
