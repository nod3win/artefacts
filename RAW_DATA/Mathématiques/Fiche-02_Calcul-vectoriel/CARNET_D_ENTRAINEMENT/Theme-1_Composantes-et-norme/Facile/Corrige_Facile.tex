\cor{1}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\vc{AB}\,(4-1\,;\,6-2)=(3;4)$.
  \item $\vc{CD}\,(5-0\,;\,3-3)=(5;0)$. (Vecteur horizontal.)
  \item $\vc{EF}\,(2-2\,;\,1-7)=(0;-6)$. (Vecteur vertical, dirigé vers le bas.)
  \item $\vc{GH}\,(0-3\,;\,0-5)=(-3;-5)$.
\end{enumerate}
\rep On applique partout « extrémité moins origine ».

\cor{2}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\norm{\vec u}=\sqrt{3^2+4^2}=\sqrt{9+16}=\sqrt{25}=5$.
  \item $\norm{\vec v}=\sqrt{6^2+8^2}=\sqrt{36+64}=\sqrt{100}=10$.
  \item $\norm{\vec w}=\sqrt{5^2+0^2}=\sqrt{25}=5$.
  \item $\norm{\vec t}=\sqrt{2^2+3^2}=\sqrt{4+9}=\sqrt{13}$ (forme exacte ; $\approx 3{,}61$).
\end{enumerate}
\rep Les triplets $(3;4;5)$ et $(6;8;10)$ donnent des normes entières : ce sont des cas à reconnaître.

\cor{3}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\vc{AB}\,(4-1\,;\,5-1)=(3;4)$.
  \item $\norm{\vc{AB}}=\sqrt{3^2+4^2}=\sqrt{25}=5$.
\end{enumerate}

\cor{4}
On lit sur la figure : $\vec a$ part de $(1;1)$ et arrive en $(4;1)$, donc $\vec a\,(3;0)$ et
$\norm{\vec a}=\sqrt{3^2+0^2}=3$.\\
$\vec b$ part de $(1;1)$ et arrive en $(1;4)$, donc $\vec b\,(0;3)$ et
$\norm{\vec b}=\sqrt{0^2+3^2}=3$.\\
\rep Un vecteur horizontal a sa seconde composante nulle ; un vecteur vertical a sa première
composante nulle.

\cor{5}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\vc{AB}\,(7-2\,;\,1-5)=(5;-4)$ \quad et \quad $\vc{BA}\,(2-7\,;\,5-1)=(-5;4)$.
  \item Les composantes sont \emph{opposées} : $\vc{BA}=-\vc{AB}$. Les deux vecteurs ont la même
        longueur : $\norm{\vc{AB}}=\norm{\vc{BA}}=\sqrt{5^2+(-4)^2}=\sqrt{25+16}=\sqrt{41}$.
        Échanger origine et extrémité ne change pas la norme, seulement le sens.
\end{enumerate}
