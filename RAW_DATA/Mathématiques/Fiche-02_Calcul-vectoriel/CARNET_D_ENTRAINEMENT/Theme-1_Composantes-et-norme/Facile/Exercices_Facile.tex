\vspace{1mm}
{\small\itshape Objectif : automatiser le calcul « extrémité moins origine » et la formule de la norme
sur des nombres simples. Laissez les normes sous forme exacte.}
\vspace{2mm}

\exo{1} \textbf{Composantes « extrémité moins origine ».}
Dans un repère, on donne plusieurs couples de points. Calculer les composantes du vecteur demandé.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $A\,(1;2)$ et $B\,(4;6)$ : composantes de $\vc{AB}$.
  \item $C\,(0;3)$ et $D\,(5;3)$ : composantes de $\vc{CD}$.
  \item $E\,(2;7)$ et $F\,(2;1)$ : composantes de $\vc{EF}$.
  \item $G\,(3;5)$ et $H\,(0;0)$ : composantes de $\vc{GH}$.
\end{enumerate}

\exo{2} \textbf{Norme à partir des composantes.}
Calculer la norme de chacun des vecteurs suivants (donnés par leurs composantes).
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\vec{u}\,(3;4)$
  \item $\vec{v}\,(6;8)$
  \item $\vec{w}\,(5;0)$
  \item $\vec{t}\,(2;3)$
\end{enumerate}

\exo{3} \textbf{Le calcul complet.}
On donne $A\,(1;1)$ et $B\,(4;5)$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item Déterminer les composantes de $\vc{AB}$.
  \item En déduire la norme $\norm{\vc{AB}}$ (forme exacte).
\end{enumerate}

\exo{4} \textbf{Lecture sur une figure.}
Lire directement sur la figure les composantes des vecteurs $\vec{a}$ et $\vec{b}$, puis calculer leurs normes.

\exo{5} \textbf{Vecteur opposé.}
On donne $A\,(2;5)$ et $B\,(7;1)$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item Calculer les composantes de $\vc{AB}$, puis celles de $\vc{BA}$.
  \item Que remarque-t-on entre $\vc{AB}$ et $\vc{BA}$ ? Comparer $\norm{\vc{AB}}$ et $\norm{\vc{BA}}$.
\end{enumerate}
