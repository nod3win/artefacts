\vspace{1mm}
{\small\itshape Objectif : gérer les coordonnées négatives, simplifier une norme sous la forme
$a\sqrt{b}$, et remonter d'une norme (ou d'un vecteur) vers un point inconnu.}
\vspace{2mm}

\exo{1} \textbf{Composantes et norme avec des coordonnées négatives.}
Pour chaque couple, donner les composantes de $\vc{AB}$ puis sa norme (forme exacte).
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $A\,(-2;3)$ et $B\,(1;-1)$.
  \item $A\,(4;-2)$ et $B\,(-1;-2)$.
  \item $A\,(-3;-1)$ et $B\,(-3;5)$.
\end{enumerate}

\exo{2} \textbf{Simplifier une norme sous la forme $a\sqrt{b}$.}
Calculer et \emph{simplifier} la norme des vecteurs suivants.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item $\vec{u}\,(4;-2)$
  \item $\vec{v}\,(-6;3)$
  \item $\vec{w}\,(2\sqrt3;2)$
\end{enumerate}

\exo{3} \textbf{Retrouver un point à partir d'un vecteur.}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item On sait que $A\,(2;-1)$ et que $\vc{AB}\,(3;5)$. Déterminer les coordonnées de $B$.
  \item On sait que $\vc{CD}\,(-2;4)$ et que $D\,(1;3)$. Déterminer les coordonnées de $C$.
\end{enumerate}

\exo{4} \textbf{Trouver une coordonnée imposée par une norme.}
On donne $A\,(1;2)$ et $B\,(4;y)$, où $y$ est un réel.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item Exprimer $\norm{\vc{AB}}^2$ en fonction de $y$.
  \item Déterminer les valeurs de $y$ pour lesquelles $\norm{\vc{AB}}=5$.
\end{enumerate}

\exo{5} \textbf{Nature d'un triangle par les longueurs des côtés.}
Dans un repère orthonormé, on considère $A\,(1;1)$, $B\,(5;3)$ et $C\,(3;5)$.
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=2pt]
  \item Calculer les longueurs $AB$, $BC$ et $AC$ (formes exactes).
  \item Le triangle $ABC$ est-il isocèle ? Justifier.
\end{enumerate}
