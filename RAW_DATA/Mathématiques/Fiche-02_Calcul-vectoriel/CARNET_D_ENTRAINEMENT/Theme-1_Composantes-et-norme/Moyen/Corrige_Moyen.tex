\cor{1}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=3pt]
  \item $\vc{AB}\,(1-(-2)\,;\,-1-3)=(3;-4)$, \quad $\norm{\vc{AB}}=\sqrt{3^2+(-4)^2}=\sqrt{25}=5$.
  \item $\vc{AB}\,(-1-4\,;\,-2-(-2))=(-5;0)$, \quad $\norm{\vc{AB}}=\sqrt{(-5)^2+0^2}=5$.
  \item $\vc{AB}\,(-3-(-3)\,;\,5-(-1))=(0;6)$, \quad $\norm{\vc{AB}}=\sqrt{0^2+6^2}=6$.
\end{enumerate}
\rep Bien soustraire les négatifs : $-1-(-2)=+3$. Et $(-4)^2=+16$.

\cor{2}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=3pt]
  \item $\norm{\vec u}=\sqrt{4^2+(-2)^2}=\sqrt{16+4}=\sqrt{20}=\sqrt{4\cdot5}=2\sqrt5$.
  \item $\norm{\vec v}=\sqrt{(-6)^2+3^2}=\sqrt{36+9}=\sqrt{45}=\sqrt{9\cdot5}=3\sqrt5$.
  \item $\norm{\vec w}=\sqrt{(2\sqrt3)^2+2^2}=\sqrt{12+4}=\sqrt{16}=4$.
\end{enumerate}
\rep Pour simplifier $\sqrt{20}$, on repère le plus grand carré parfait facteur : $20=4\times5$,
donc $\sqrt{20}=2\sqrt5$. Et $(2\sqrt3)^2=4\times3=12$.

\cor{3}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=3pt]
  \item $B=A+\vc{AB}$ composante par composante : $x_B=2+3=5$, $y_B=-1+5=4$. Donc $B\,(5;4)$.
  \item Comme $\vc{CD}=(x_D-x_C\,;\,y_D-y_C)$, on a $x_C=x_D-(x_D-x_C)=1-(-2)=3$ et
        $y_C=3-4=-1$. Donc $C\,(3;-1)$.
\end{enumerate}
\rep Retenir : « extrémité $=$ origine $+$ vecteur », donc « origine $=$ extrémité $-$ vecteur ».

\cor{4}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=3pt]
  \item $\vc{AB}\,(4-1\,;\,y-2)=(3;\,y-2)$, donc $\norm{\vc{AB}}^2=3^2+(y-2)^2=9+(y-2)^2$.
  \item $\norm{\vc{AB}}=5 \iff \norm{\vc{AB}}^2=25 \iff 9+(y-2)^2=25 \iff (y-2)^2=16$.
        D'où $y-2=\pm4$, soit $\boxed{y=6 \text{ ou } y=-2}$.
\end{enumerate}
\rep Une norme imposée donne une équation du \emph{second degré} : il y a en général \textbf{deux}
solutions (deux positions symétriques pour $B$).

\cor{5}
\begin{enumerate}[label=\textbf{\alph*)},leftmargin=2em,itemsep=3pt]
  \item $\vc{AB}\,(4;2)$ donc $AB=\sqrt{16+4}=\sqrt{20}=2\sqrt5$.\\
        $\vc{BC}\,(-2;2)$ donc $BC=\sqrt{4+4}=\sqrt8=2\sqrt2$.\\
        $\vc{AC}\,(2;4)$ donc $AC=\sqrt{4+16}=\sqrt{20}=2\sqrt5$.
  \item $AB=AC=2\sqrt5$ : le triangle $ABC$ est \textbf{isocèle} en $A$.
\end{enumerate}
\rep Pour la nature d'un triangle, on compare les longueurs des côtés, calculées comme des normes de
vecteurs.
