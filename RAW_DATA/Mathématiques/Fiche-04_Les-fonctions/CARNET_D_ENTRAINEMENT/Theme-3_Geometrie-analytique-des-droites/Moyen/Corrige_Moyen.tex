\titrecarnet{Thème 3 — Géométrie analytique des droites}{Corrigé — niveau Moyen}{Détail des calculs.}

\exercice{1}
Parallèle $\Rightarrow$ même pente $a=2$. La droite $y=2x+b$ passe par $(0;4)$, donc $b=4$.
\[ \boxed{y=2x+4} \]

\exercice{2}
Perpendiculaire à une pente $3$ : $a'=-\dfrac13$. La droite $y=-\dfrac13 x+b$ passe par $(0;0)$, donc $b=0$.
\[ \boxed{y=-\tfrac13 x} \]

\exercice{3}
La droite est déjà en forme canonique $3x+4y-5=0$, donc $a=3,\ b=4,\ c=-5$.
\[ \operatorname{dist}(P,d)=\frac{\lvert 3\times 1+4\times 1-5\rvert}{\sqrt{3^2+4^2}}=\frac{\lvert 2\rvert}{\sqrt{25}}=\boxed{\dfrac25}. \]

\exercice{4}
Système $\begin{cases} y=2x\\ y=-2x+3\end{cases}$. Par addition : $2y=3\Rightarrow y=\dfrac32$. Puis $2x=\dfrac32\Rightarrow x=\dfrac34$.
\[ \boxed{\Big(\tfrac34;\tfrac32\Big)} \]

\exercice{5}
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item $a=\dfrac{4-2}{5-1}=\dfrac{2}{4}=\dfrac12$ ; $b=y_A-a\,x_A=2-\dfrac12\times 1=\dfrac32$. Donc $\boxed{y=\tfrac12 x+\tfrac32}$.
  \item $M\Big(\dfrac{1+5}{2};\dfrac{2+4}{2}\Big)=\boxed{(3;3)}$.
        (Vérification : $(3;3)$ appartient bien à $(AB)$ car $\tfrac12\times 3+\tfrac32=3$.)
\end{enumerate}
