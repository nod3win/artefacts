\exercice{1}
$f(x)=3x-5$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item $f(2)=3\times 2-5=\boxed{1}$ ;\quad $f(-1)=3\times(-1)-5=\boxed{-8}$ ;\quad $f(0)=3\times 0-5=\boxed{-5}$.
  \item On résout $f(x)=4$ : $3x-5=4\Leftrightarrow 3x=9\Leftrightarrow x=3$. L'antécédent de $4$ est $\boxed{3}$.
\end{enumerate}

\exercice{2}
$g(x)=x^2-2x$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item $g(3)=3^2-2\times 3=9-6=\boxed{3}$ ;\quad $g(-2)=(-2)^2-2\times(-2)=4+4=\boxed{8}$.
  \item On résout $g(x)=0$ : $x^2-2x=0\Leftrightarrow x(x-2)=0\Leftrightarrow x=0 \text{ ou } x=2$.
        Les antécédents de $0$ sont $\boxed{0 \text{ et } 2}$ (un même nombre peut avoir \emph{plusieurs} antécédents).
\end{enumerate}

\exercice{3}
Zone à risque : le dénominateur $x-3$. Condition d'existence : $x-3\neq 0\Leftrightarrow x\neq 3$.
\[ \boxed{\Df=\R\setminus\{3\}} \]

\exercice{4}
Zone à risque : la racine carrée. Condition : $x-2\geq 0\Leftrightarrow x\geq 2$. On garde le crochet fermé en $2$ car $\sqrt{0}=0$ existe.
\[ \boxed{\Df=[2,+\infty[} \]

\exercice{5}
Zone à risque : le logarithme. Condition : $x+1>0\Leftrightarrow x>-1$ (strict : $\ln 0$ n'existe pas).
\[ \boxed{\Df=\,]-1,+\infty[} \]
