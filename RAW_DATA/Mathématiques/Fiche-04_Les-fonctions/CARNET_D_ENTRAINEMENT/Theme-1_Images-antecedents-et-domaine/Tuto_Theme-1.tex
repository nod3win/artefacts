\section*{1. Image et antécédent}

Une fonction $f$ associe à un nombre $x$ \emph{une unique} image $f(x)$.

\begin{idee}[deux gestes inverses]
\begin{itemize}[leftmargin=1.5em,topsep=2pt]
  \item \textbf{Image de $a$} : on \emph{remplace} $x$ par $a$ dans l'expression et on calcule $f(a)$. Le résultat est unique.
  \item \textbf{Antécédent de $k$} : on \emph{résout l'équation} $f(x)=k$. Il peut y avoir $0$, $1$ ou \emph{plusieurs} antécédents.
\end{itemize}
\end{idee}

Graphiquement : l'image se lit \emph{verticalement} (on monte de $x$ jusqu'à la courbe), l'antécédent se lit \emph{horizontalement} (on part d'une hauteur $k$ et on rejoint la courbe).

\begin{exemple}[avec $f(x)=2x+1$]
Image de $1$ : $f(1)=2\times 1+1=3$. \quad Antécédent de $7$ : $2x+1=7\Rightarrow x=3$.
\end{exemple}

\section*{2. Domaine de définition : les trois interdits}

Le \textbf{domaine de définition} $\Df$ est l'ensemble des réels $x$ pour lesquels $f(x)$ \emph{existe}. Le déterminer, c'est repérer puis \emph{exclure} les valeurs interdites.

\begin{idee}[les trois interdits à connaître par cœur]
\begin{enumerate}[leftmargin=1.8em,topsep=2pt,itemsep=1pt]
  \item \textbf{Division} : un dénominateur doit être $\neq 0$.
  \item \textbf{Racine carrée} (et toute racine d'indice \emph{pair}) : la quantité sous la racine doit être $\geq 0$.
  \item \textbf{Logarithme} : l'argument d'un $\ln$ ou $\log$ doit être $>0$ (strictement).
\end{enumerate}
\end{idee}

\begin{methode}[déterminer un domaine de définition]
\begin{enumerate}[leftmargin=1.8em,topsep=2pt,itemsep=1pt]
  \item \textbf{Repérer} les zones à risque : dénominateurs, racines paires, logarithmes.
  \item \textbf{Écrire} une condition d'existence pour chacune ($\neq 0$ ; $\geq 0$ ; $>0$).
  \item \textbf{Résoudre} chaque (in)équation.
  \item \textbf{Combiner} (intersection) et écrire $\Df$ en intervalles. Sans zone à risque : $\Df=\R$.
\end{enumerate}
\end{methode}

Notations utiles : $\R\setminus\{0\}=\R_0$ ; $[0,+\infty[\,=\R^{+}$ ; $\,]0,+\infty[\,=\R_0^{+}$.

\begin{exemple}[trois cas types]
$\dfrac{1}{x-3}$ : dénominateur $\neq 0\Rightarrow x\neq 3$, donc $\Df=\R\setminus\{3\}$. \quad
$\sqrt{x-2}$ : $x-2\geq 0\Rightarrow \Df=[2,+\infty[$. \quad
$\ln(x+1)$ : $x+1>0\Rightarrow \Df=\,]-1,+\infty[$.
\end{exemple}

\begin{piege}[trois réflexes qui font gagner des points]
\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt]
  \item Racine carrée : crochet \emph{fermé} là où la quantité vaut $0$ ($\sqrt{0}=0$ existe) ; mais racine \emph{au dénominateur} $\Rightarrow$ crochet \emph{ouvert} (il faut $>0$).
  \item Racine \emph{cubique} (indice \emph{impair}) : définie sur $\R$ tout entier — pas de condition !
  \item Un quotient sous une racine ou dans un log : dresser un \textbf{tableau de signes}.
\end{itemize}
\end{piege}
