\exercice{1}
$f(x)=-x^2+4x-3$, $a=-1,\ b=4,\ c=-3$.
\begin{itemize}[leftmargin=1.6em,topsep=2pt]
  \item $a=-1<0$ : parabole ouverte vers le \textbf{bas}, sommet = maximum.
  \item $\Delta=16-4(-1)(-3)=16-12=4$. Sommet : $\Big(-\dfrac{b}{2a};-\dfrac{\Delta}{4a}\Big)=\Big(-\dfrac{4}{-2};-\dfrac{4}{-4}\Big)=\boxed{(2;1)}$.
  \item Axe de symétrie : $x=2$.
  \item Racines : $x=\dfrac{-4\pm2}{-2}$, soit $x=\dfrac{-2}{-2}=1$ et $x=\dfrac{-6}{-2}=3$ : $\boxed{x_1=1,\ x_2=3}$.
\end{itemize}

\exercice{2}
$\Delta=(-1)^2-4(1)(-6)=1+24=25$, racines $\dfrac{1\pm5}{2}=3$ et $-2$. Donc
\[ \boxed{x^2-x-6=(x-3)(x+2)}. \]

\exercice{3}
On réécrit $f(x)=-2x^2+36x-34$, donc $a=-2,\ b=36,\ c=-34$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item $a=-2<0$ : le sommet est un \textbf{maximum}.
  \item $\Delta=36^2-4(-2)(-34)=1296-272=1024$. Ordonnée du sommet : $-\dfrac{\Delta}{4a}=-\dfrac{1024}{4\times(-2)}=-\dfrac{1024}{-8}=128$. Elle est \textbf{positive}. Le sommet est donc un maximum d'ordonnée $\boxed{128>0}$.
\end{enumerate}

\exercice{4}
$x^2-4=(x-2)(x+2)$, racines $-2$ et $2$. Comme $a=1>0$ : positif à l'extérieur, négatif entre les racines.
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $-2$ & & $2$ \hspace{1.2cm} $+\infty$ \\
\hline
$x^2-4$ & & $+$ & $0$ & $-$ & $0\ \ +$ \\
\hline
\end{tabular}
\end{center}
Donc $x^2-4>0$ sur $\,]-\infty,-2[\cup\,]2,+\infty[$, \ $x^2-4<0$ sur $\,]-2,2[$, \ et $=0$ en $x=\pm2$.

\exercice{5}
Racine double $\Leftrightarrow \Delta=0$. Ici $\Delta=m^2-4(1)(1)=m^2-4$.
\[ m^2-4=0\Leftrightarrow m^2=4\Leftrightarrow \boxed{m=-2 \text{ ou } m=2}. \]
