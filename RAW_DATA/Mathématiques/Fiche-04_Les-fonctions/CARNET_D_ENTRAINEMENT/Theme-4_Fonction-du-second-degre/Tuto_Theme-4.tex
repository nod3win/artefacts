\titrecarnet{Thème 4 — Fonction du second degré}{Tutoriel}{La parabole $ax^2+bx+c$ : discriminant, racines, sommet, signe.}

\section*{1. Le discriminant décide de tout}

Pour $f(x)=ax^2+bx+c$ ($a\neq 0$, $\Df=\R$), on calcule le \textbf{discriminant}
\[ \boxed{\Delta=b^2-4ac}. \]

\begin{idee}[nombre de racines selon $\Delta$]
\begin{itemize}[leftmargin=1.5em,topsep=2pt]
  \item $\Delta>0$ : \textbf{deux} racines $x_{1,2}=\dfrac{-b\pm\sqrt{\Delta}}{2a}$ ; \ factorisation $a(x-x_1)(x-x_2)$.
  \item $\Delta=0$ : \textbf{une} racine double $x_1=-\dfrac{b}{2a}$ ; \ factorisation $a(x-x_1)^2$.
  \item $\Delta<0$ : \textbf{aucune} racine réelle ; $f$ garde un signe constant, celui de $a$.
\end{itemize}
\end{idee}

\section*{2. Sommet, axe, orientation}

\begin{idee}
\[ \text{Sommet } S\Big(-\dfrac{b}{2a};\,-\dfrac{\Delta}{4a}\Big), \qquad \text{axe de symétrie } x=-\dfrac{b}{2a}, \qquad \text{intersection } Oy:(0;c). \]
\begin{itemize}[leftmargin=1.5em,topsep=2pt]
  \item $a>0$ : parabole ouverte vers le \emph{haut}, sommet = \textbf{minimum} ;
  \item $a<0$ : parabole ouverte vers le \emph{bas}, sommet = \textbf{maximum}.
\end{itemize}
\end{idee}

\begin{methode}[étudier une parabole]
$\Delta$ (nombre de racines) $\to$ racines $\to$ sommet et axe $\to$ orientation ($a$) $\to$ intersection $Oy$ $\to$ tracé.
\end{methode}

\begin{piege}[signe du trinôme]
« \emph{Du signe de $a$ à l'extérieur des racines, du signe opposé entre les racines.} » Si $\Delta<0$, pas de racine : le signe est constant (celui de $a$) sur tout $\R$.
\end{piege}

\begin{exemple}[$f(x)=x^2-4x+3$]
$\Delta=16-12=4>0$ ; racines $\dfrac{4\pm2}{2}=1$ et $3$ ; sommet $(2;-1)$ ; axe $x=2$ ; $Oy:(0;3)$ ; $a>0$ donc minimum.
\end{exemple}
