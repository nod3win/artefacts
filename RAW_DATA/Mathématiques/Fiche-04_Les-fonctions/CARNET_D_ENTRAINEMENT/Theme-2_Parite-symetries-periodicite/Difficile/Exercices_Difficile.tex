\titrecarnet{Thème 2 — Parité, symétries \& périodicité}{Exercices — niveau Difficile}{Compositions, identités trigonométriques et synthèse.}

\exercice{1}
Soit $f(x)=x^4-2x^2+1$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item Montrer que $f$ est paire.
  \item Que peut-on en déduire pour l'axe de symétrie de sa courbe ?
  \item En remarquant que $f(x)=(x^2-1)^2$, déterminer les racines de $f$.
\end{enumerate}

\exercice{2}
Soit $f(x)=x^2+4x+7$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item Déterminer l'axe de symétrie de la parabole.
  \item Écrire $f$ sous forme canonique $a(x-\alpha)^2+\beta$ et en déduire le sommet.
\end{enumerate}

\exercice{3}
On pose $g(x)=\cos^4 x-\sin^4 x$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item Montrer que $g(x)=\cos(2x)$. \emph{Indication : $\cos^2 x+\sin^2 x=1$.}
  \item En déduire la période de $g$.
\end{enumerate}

\exercice{4}
Soit $f(x)=\sin x+\cos x$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item À l'aide d'une formule d'addition, montrer que $f(x)=\sqrt2\,\sin\!\Big(x+\dfrac{\pi}{4}\Big)$.
  \item En déduire la période de $f$.
  \item $f$ est-elle paire, impaire, ou ni l'une ni l'autre ?
\end{enumerate}

\exercice{5}
\textbf{Mise en situation.} Un signal est modélisé par $f(x)=\lvert\sin x\rvert$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item Montrer que $f$ est paire.
  \item Montrer que sa période est $T=\pi$ (et non $2\pi$).
\end{enumerate}
