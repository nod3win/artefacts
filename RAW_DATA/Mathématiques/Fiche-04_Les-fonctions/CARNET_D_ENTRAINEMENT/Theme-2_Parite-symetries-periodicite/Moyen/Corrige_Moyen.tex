\titrecarnet{Thème 2 — Parité, symétries \& périodicité}{Corrigé — niveau Moyen}{Détail des calculs.}

\exercice{1}
$f(x)=x^2-6x+5$, donc $a=1$, $b=-6$.
\begin{enumerate}[label=\alph*),leftmargin=1.8em,topsep=2pt]
  \item Axe de symétrie : $x=-\dfrac{b}{2a}=-\dfrac{-6}{2}=\boxed{3}$.
  \item $f(3-x)=(3-x)^2-6(3-x)+5=9-6x+x^2-18+6x+5=x^2-4$.\\
        $f(3+x)=(3+x)^2-6(3+x)+5=9+6x+x^2-18-6x+5=x^2-4$.\\
        Les deux sont égaux : $f(3-x)=f(3+x)$, l'axe $x=3$ est bien axe de symétrie. \checkmark
\end{enumerate}

\exercice{2}
\begin{itemize}[leftmargin=1.6em,topsep=2pt]
  \item $\cos$ : \textbf{paire} car $\cos(-x)=\cos x$ ; \textbf{période} $T=2\pi$ car $\cos(x+2\pi)=\cos x$.
  \item $\tan$ : \textbf{impaire} car $\tan(-x)=\dfrac{\sin(-x)}{\cos(-x)}=\dfrac{-\sin x}{\cos x}=-\tan x$ ; \textbf{période} $T=\pi$ (et non $2\pi$) car $\tan(x+\pi)=\tan x$.
\end{itemize}

\exercice{3}
On teste $f(0-x)+f(0+x)=2b$ avec $a=0$, $b=2$ :
\[ f(-x)+f(x)=\big((-x)^3+2\big)+\big(x^3+2\big)=-x^3+2+x^3+2=4=2\times 2. \]
La relation $f(a-x)+f(a+x)=2b$ est vérifiée avec $(a;b)=(0;2)$ : $(0;2)$ est un \textbf{centre de symétrie}. \checkmark

\exercice{4}
$f(x+\pi)=\sin\big(2(x+\pi)\big)=\sin(2x+2\pi)=\sin(2x)=f(x)$ pour tout $x$.
Donc $f$ est périodique et $\pi$ est une période. C'est la \emph{plus petite} : $\boxed{T=\pi}$.

\exercice{5}
$f(-x)=\dfrac{1}{-x}=-\dfrac1x=-f(x)$ : $f$ est \textbf{impaire}. Une fonction impaire admet l'\textbf{origine} $(0;0)$ pour centre de symétrie ; le graphe de $\frac1x$ (hyperbole) est donc symétrique par rapport à $O$.
