\exo{Valeurs négatives}
\begin{itemize}[leftmargin=1.4em,itemsep=3pt]
\item[\textbf{(a)}] \rep $\cos x=-\dfrac12$. Référence : $\cos\dfrac{\pi}{3}=\dfrac12$, et le signe
$-$ place l'angle en quadrants II/III : $-\dfrac12=\cos\dfrac{2\pi}{3}$. Donc
$\cos x=\cos\dfrac{2\pi}{3}$ et
\[
x=\frac{2\pi}{3}+2k\pi \ \text{ou}\ x=-\frac{2\pi}{3}+2k\pi,\ k\in\Z
\ \Rightarrow\
\boxed{\,S=\left\{\pm\tfrac{2\pi}{3}+2k\pi \ \middle|\ k\in\Z\right\}.}
\]
\item[\textbf{(b)}] \rep $\sin x=-\dfrac{\sqrt3}{2}$. Référence : $\sin\dfrac{\pi}{3}=\dfrac{\sqrt3}{2}$ ;
le signe $-$ donne $-\dfrac{\sqrt3}{2}=\sin\!\left(-\dfrac{\pi}{3}\right)$. Donc
$\sin x=\sin\!\left(-\dfrac{\pi}{3}\right)$ et
\[
x=-\frac{\pi}{3}+2k\pi \ \text{ou}\ x=\pi-\left(-\frac{\pi}{3}\right)+2k\pi=\frac{4\pi}{3}+2k\pi,\ k\in\Z.
\]
\[
\boxed{\,S=\left\{-\tfrac{\pi}{3}+2k\pi\ ;\ \tfrac{4\pi}{3}+2k\pi \ \middle|\ k\in\Z\right\}.}
\]
\end{itemize}

\exo{Isoler puis sélectionner (cosinus)}
\rep \textbf{(1) Isoler :} $2\cos x+1=0\Leftrightarrow \cos x=-\dfrac12$.
\textbf{(2) Reconnaître :} $-\dfrac12=\cos\dfrac{2\pi}{3}$.
\textbf{(3) Familles :} $x=\dfrac{2\pi}{3}+2k\pi$ ou $x=-\dfrac{2\pi}{3}+2k\pi$.
\textbf{(4) Sélection :} le 2\textsuperscript{e} quadrant, c'est $x\in\left]\dfrac{\pi}{2}\,;\,\pi\right[$.
La valeur $\dfrac{2\pi}{3}\ (=120\degr)$ y est ; $-\dfrac{2\pi}{3}\equiv\dfrac{4\pi}{3}\ (=240\degr)$ est
au quadrant III. On retient
\[
\boxed{\,x=\frac{2\pi}{3}.}
\]

\exo{Isoler puis sélectionner (sinus)}
\rep \textbf{(1) Isoler :} $2\sin x-\sqrt3=0\Leftrightarrow \sin x=\dfrac{\sqrt3}{2}$.
\textbf{(2) Reconnaître :} $\dfrac{\sqrt3}{2}=\sin\dfrac{\pi}{3}$.
\textbf{(3) Familles :} $x=\dfrac{\pi}{3}+2k\pi$ ou $x=\pi-\dfrac{\pi}{3}+2k\pi=\dfrac{2\pi}{3}+2k\pi$.
\textbf{(4) Sélection :} au 2\textsuperscript{e} quadrant $\left]\dfrac{\pi}{2}\,;\,\pi\right[$, seule
$\dfrac{2\pi}{3}\ (=120\degr)$ convient ($\dfrac{\pi}{3}=60\degr$ est au quadrant I). On retient
\[
\boxed{\,x=\frac{2\pi}{3}.}
\]

\exo{Sélection sur un intervalle (tangente)}
\rep $\tg x=-1$. Référence $\tg\dfrac{\pi}{4}=1$, donc $-1=\tg\!\left(-\dfrac{\pi}{4}\right)$ :
famille (période $\pi$)
\[
x=-\frac{\pi}{4}+k\pi,\qquad k\in\Z.
\]
On teste dans $[0\,;\,2\pi[$ : $k=1\Rightarrow -\dfrac{\pi}{4}+\pi=\dfrac{3\pi}{4}$ ;
$k=2\Rightarrow -\dfrac{\pi}{4}+2\pi=\dfrac{7\pi}{4}$ (et $k=0$ donne $-\frac{\pi}{4}<0$,
$k=3$ dépasse $2\pi$). Donc
\[
\boxed{\,S=\left\{\frac{3\pi}{4}\,;\,\frac{7\pi}{4}\right\}.}
\]

\exo{Sélection sur un intervalle (sinus)}
\rep $\sin x=-\dfrac12$. Référence $\sin\dfrac{\pi}{6}=\dfrac12$, donc
$-\dfrac12=\sin\!\left(-\dfrac{\pi}{6}\right)$ : familles
\[
x=-\frac{\pi}{6}+2k\pi \ \text{ou}\ x=\pi-\left(-\frac{\pi}{6}\right)+2k\pi=\frac{7\pi}{6}+2k\pi.
\]
Dans $[0\,;\,2\pi[$ : la première donne $-\dfrac{\pi}{6}+2\pi=\dfrac{11\pi}{6}$ ; la seconde donne
$\dfrac{7\pi}{6}$. Donc
\[
\boxed{\,S=\left\{\frac{7\pi}{6}\,;\,\frac{11\pi}{6}\right\}.}
\]
