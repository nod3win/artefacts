\begin{consigne}
\textbf{Objectif :} traiter l'\textbf{angle double} (écrire les familles \emph{en $2x$} puis diviser
par $2$ — sans perdre de solution), ramener une \textbf{équation linéaire} $a\sin x+b\cos x=0$ à une
tangente, et appliquer tout cela à une \textbf{situation périodique}. Les sous-questions
s'enchaînent. \textbf{Sans calculatrice}, valeurs exactes, conclure par $S$.
\end{consigne}

\exo{Angle double — le sinus}
On veut résoudre $2\sin(2x)=1$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2em,itemsep=2pt]
\item Isole $\sin(2x)$, puis écris les \textbf{deux familles} de solutions \emph{pour l'angle $2x$}.
\item Divise par $2$ pour obtenir les familles en $x$. Que devient le « $+2k\pi$ » ?
\item Donne toutes les solutions dans $[0\,;\,2\pi[$.
\end{enumerate}

\exo{Angle double — la tangente}
On veut résoudre $\tg(2x)=1$ dans $[0\,;\,2\pi[$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2em,itemsep=2pt]
\item Écris la famille de solutions \emph{pour $2x$}, puis divise par $2$ pour obtenir $x$.
\item Combien attends-tu de solutions dans $[0\,;\,2\pi[$~? Donne-les toutes.
\end{enumerate}

\exo{Équation linéaire ramenée à une tangente}
On considère $\sin x-\sqrt3\,\cos x=0$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2em,itemsep=2pt]
\item Explique pourquoi $\cos x=0$ ne peut pas être solution.
\item En divisant par $\cos x$, ramène l'équation à une équation en $\tg x$, puis résous dans $\mathbb{R}$.
\item Donne les solutions appartenant à $[0\,;\,2\pi[$.
\end{enumerate}

\exo{Mise en situation périodique — deux roues}
La hauteur (comptée à partir de l'axe, unité arbitraire) d'un point d'une roue en fonction de sa
position angulaire $x\in[0\,;\,2\pi[$ est modélisée par $h(x)=\sin x$.
\begin{enumerate}[label=\textbf{(\alph*)},leftmargin=2em,itemsep=2pt]
\item Pour quelles positions $x$ la hauteur vaut-elle $\dfrac12$~?
\item Pour quelles positions $x$ la hauteur vaut-elle $-\dfrac{\sqrt2}{2}$~?
\item Une seconde roue tourne \emph{deux fois plus vite} : sa hauteur est $g(x)=\sin(2x)$, toujours
pour $x\in[0\,;\,2\pi[$. Pour quelles positions $x$ vaut-elle $\dfrac{\sqrt2}{2}$~?
(Attention à l'angle double.)
\end{enumerate}
