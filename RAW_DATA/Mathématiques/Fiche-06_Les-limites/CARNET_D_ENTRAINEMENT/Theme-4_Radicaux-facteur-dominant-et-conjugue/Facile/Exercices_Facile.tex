{\small\itshape But : manipuler $\sqrt{x^2}=|x|$, reconnaître un conjugué et l'utiliser sur des cas
simples.}
\vspace{6pt}

\begin{exercice}[$\sqrt{x^2}=|x|$]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Calculer $\sqrt{x^2}$ pour $x=5$, puis pour $x=-5$.
  \item Simplifier $\sqrt{(x-3)^2}$ en fonction du signe de $x-3$.
  \item Simplifier $\sqrt{9x^2}$ pour $x>0$.
\end{enumerate}
\end{exercice}

\begin{exercice}[racines simples à l'infini]
Donner la limite (toutes tendent vers $+\infty$, mais vérifier).
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item $\displaystyle\lim_{x\to+\infty}\sqrt{x^2+3}$
  \item $\displaystyle\lim_{x\to+\infty}\sqrt{x^4+x}$
  \item $\displaystyle\lim_{x\to+\infty}\bigl(\sqrt{x}+2\bigr)$
\end{enumerate}
\end{exercice}

\begin{exercice}[écrire le conjugué]
Donner l'expression conjuguée de chacune (on change le signe central).
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item $\sqrt{x}-3$ \qquad b) $\sqrt{x+1}-\sqrt{x}$ \qquad c) $2-\sqrt{x}$
\end{enumerate}
\end{exercice}

\begin{exercice}[produit par le conjugué]
Développer et simplifier chaque produit (utiliser $(\sqrt A-\sqrt B)(\sqrt A+\sqrt B)=A-B$).
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item $(\sqrt{x}-2)(\sqrt{x}+2)$
  \item $\bigl(\sqrt{x+1}-1\bigr)\bigl(\sqrt{x+1}+1\bigr)$
\end{enumerate}
\end{exercice}

\begin{exercice}[une première indétermination $\frac00$]
On veut $\displaystyle\lim_{x\to 4}\frac{\sqrt{x}-2}{x-4}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Vérifier que le remplacement direct donne la forme $\frac00$.
  \item Multiplier haut et bas par le conjugué $\sqrt{x}+2$ et simplifier.
  \item Conclure la valeur de la limite.
\end{enumerate}
\end{exercice}
