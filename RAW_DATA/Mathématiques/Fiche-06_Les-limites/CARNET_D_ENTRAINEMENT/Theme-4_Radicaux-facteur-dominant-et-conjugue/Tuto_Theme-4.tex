\begin{tcolorbox}[boxbase,colback=primary!12,colframe=primary,
   title={\Large Thème 4 — Radicaux : facteur dominant \& binôme conjugué}]
\textbf{Mécanique travaillée :} calculer une limite contenant des \emph{racines carrées}. Deux gestes :
(1) \textbf{mettre le terme dominant en facteur} sous la racine, en se rappelant $\sqrt{x^2}=|x|$ ;
(2) multiplier par l'\textbf{expression conjuguée} pour lever une indétermination $\infty-\infty$ ou
$\frac00$ due à une différence de racines.
\end{tcolorbox}

\section{1. Sortir le terme dominant d'une racine : $\sqrt{x^2}=|x|$}

\begin{idee}[la valeur absolue est incontournable]
\[ \boxed{\ \sqrt{x^2}=|x|\ }\quad\text{avec}\quad |x|=\begin{cases} x & \text{si }x\geq 0\ (\text{côté }+\infty),\\ -x & \text{si }x<0\ (\text{côté }-\infty).\end{cases} \]
\end{idee}

\begin{methode}[radical à l'infini]
\begin{enumerate}[leftmargin=1.9em,topsep=2pt,itemsep=1pt]
  \item Sous la racine, mettre en facteur la plus haute puissance (souvent $x^2$).
  \item Sortir : $\sqrt{x^2\cdot(\dots)}=|x|\sqrt{\dots}$, puis remplacer $|x|$ par $x$ (en $+\infty$) ou $-x$ (en $-\infty$).
  \item La parenthèse sous la racine tend vers $1$ : conclure.
\end{enumerate}
\end{methode}

\begin{exemple}[le rôle de $|x|$]
$\sqrt{x^2-2x+5}=|x|\sqrt{1-\tfrac2x+\tfrac{5}{x^2}}\ \xrightarrow[x\to\pm\infty]{}\ |x|$.
\[ \lim_{x\to+\infty}\sqrt{x^2-2x+5}=\lim_{x\to+\infty} x=+\infty,\qquad
   \lim_{x\to-\infty}\sqrt{x^2-2x+5}=\lim_{x\to-\infty}(-x)=+\infty. \]
\end{exemple}

\section{2. Le binôme conjugué}

\begin{idee}[faire disparaître les racines]
\[ \boxed{\ \bigl(\sqrt{A}-\sqrt{B}\bigr)\bigl(\sqrt{A}+\sqrt{B}\bigr)=A-B\ } \]
On multiplie \emph{haut et bas} par le conjugué de la partie qui pose problème : les racines
disparaissent, l'indétermination se lève.
\end{idee}

\begin{methode}[lever $\infty-\infty$ ou $\frac00$ avec des racines]
\begin{enumerate}[leftmargin=1.9em,topsep=2pt,itemsep=1pt]
  \item Repérer la différence de racines responsable de l'indétermination.
  \item Multiplier numérateur et dénominateur par son conjugué (on change le signe central).
  \item Développer avec $A-B$ : les racines s'en vont. Simplifier, puis calculer la limite.
\end{enumerate}
\end{methode}

\begin{exemple}[$\infty-\infty$ : $\displaystyle\lim_{x\to+\infty}\bigl(\sqrt{x+2}-\sqrt{x-2}\bigr)$]
\[ \sqrt{x+2}-\sqrt{x-2}
   =\frac{(x+2)-(x-2)}{\sqrt{x+2}+\sqrt{x-2}}=\frac{4}{\sqrt{x+2}+\sqrt{x-2}}\xrightarrow[x\to+\infty]{}\frac{4}{+\infty}=0. \]
\end{exemple}

\begin{exemple}[$\frac00$ : $\displaystyle\lim_{x\to 1}\frac{\sqrt{x+1}-\sqrt2}{x-1}$]
\[ \frac{\sqrt{x+1}-\sqrt2}{x-1}
  =\frac{(x+1)-2}{(x-1)\bigl(\sqrt{x+1}+\sqrt2\bigr)}
  =\frac{x-1}{(x-1)\bigl(\sqrt{x+1}+\sqrt2\bigr)}
  =\frac{1}{\sqrt{x+1}+\sqrt2}\xrightarrow[x\to1]{}\frac{1}{2\sqrt2}=\frac{\sqrt2}{4}. \]
Le facteur $(x-1)$ se simplifie : l'indétermination est levée.
\end{exemple}

\begin{attention}[ne pas « distribuer » la racine]
$\sqrt{a+b}\neq\sqrt{a}+\sqrt{b}$ et $\sqrt{x^2}\neq x$ (c'est $|x|$ !). Le côté $-\infty$ change le
signe : oublier le $|x|$ y donne un signe faux.
\end{attention}

\begin{astuce}[différence $\sqrt{\text{polynôme}}-\text{polynôme}$]
Pour $\sqrt{x^2+bx+c}-x$ (forme $\infty-\infty$ en $+\infty$), le conjugué est $\sqrt{x^2+bx+c}+x$ ;
le numérateur devient $(x^2+bx+c)-x^2=bx+c$, et l'on conclut par le terme dominant. Résultat utile :
$\displaystyle\lim_{x\to+\infty}\bigl(\sqrt{x^2+bx+c}-x\bigr)=\frac{b}{2}$.
\end{astuce}
