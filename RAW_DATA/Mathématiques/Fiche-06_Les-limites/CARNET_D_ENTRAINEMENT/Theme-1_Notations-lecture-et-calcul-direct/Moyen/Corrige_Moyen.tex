\begin{correction}[calcul direct avec fonctions usuelles]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item $\displaystyle\lim_{x\to 0}(e^{x}+3\cos x)=e^{0}+3\cos 0=1+3=\boxed{4}$.
  \item $\displaystyle\lim_{x\to 1}\frac{\ln x+2}{x+1}=\frac{\ln 1+2}{1+1}=\frac{0+2}{2}=\boxed{1}$.
  \item $\displaystyle\lim_{x\to \frac{\pi}{2}}(2\sin x-x)=2\sin\tfrac{\pi}{2}-\tfrac{\pi}{2}=2-\tfrac{\pi}{2}=\boxed{2-\tfrac{\pi}{2}}\approx 0{,}43$.
  \item $\displaystyle\lim_{x\to 9}\frac{\sqrt{x}-1}{x-5}=\frac{\sqrt{9}-1}{9-5}=\frac{3-1}{4}=\frac{2}{4}=\boxed{\tfrac12}$
        (le dénominateur vaut $4\neq 0$).
\end{enumerate}
\end{correction}

\begin{correction}[raccord de deux morceaux]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item À gauche $x\leq 1$ : $f(x)=x^2+1$, donc $\displaystyle\lim_{x\to 1^{-}} f(x)=1^2+1=2$.\\
        À droite $x>1$ : $f(x)=4x-2$, donc $\displaystyle\lim_{x\to 1^{+}} f(x)=4\cdot 1-2=2$.
  \item Les deux limites latérales sont \emph{égales} à $2$ : $f$ possède la limite $\displaystyle\lim_{x\to 1} f(x)=2$.
  \item $f(1)$ se calcule avec le morceau $x\leq 1$ : $f(1)=1^2+1=2$. Comme
        $\displaystyle\lim_{x\to 1} f(x)=2=f(1)$, \textbf{$f$ est continue en $1$}.
\end{enumerate}
\end{correction}

\begin{correction}[ajuster un paramètre pour la continuité]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item À gauche $x<3$ : $f(x)=2x+k$, donc $\displaystyle\lim_{x\to 3^{-}} f(x)=6+k$.\\
        À droite $x\geq 3$ : $f(x)=x^2$, donc $\displaystyle\lim_{x\to 3^{+}} f(x)=9$ ; de plus $f(3)=9$.
  \item $f$ est continue en $3$ ssi limite à gauche $=$ limite à droite $=f(3)$, soit
        $6+k=9$, donc $\boxed{k=3}$.
\end{enumerate}
\end{correction}

\begin{correction}[lecture graphique fine]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item La branche de gauche arrive au niveau $2$ : $\displaystyle\lim_{x\to 1^{-}} f(x)=2$.
        La branche de droite part du niveau $0{,}8$ : $\displaystyle\lim_{x\to 1^{+}} f(x)=0{,}8$.
  \item Le rond plein est placé en $(1\,;0{,}8)$, donc $f(1)=0{,}8$.
  \item Les deux limites latérales diffèrent ($2\neq 0{,}8$) : $f$ n'a pas de limite en $1$, donc
        \textbf{n'est pas continue} en $1$. En revanche, à droite $\displaystyle\lim_{x\to 1^{+}} f(x)=0{,}8=f(1)$ :
        \textbf{$f$ est continue à droite} en $1$ (mais pas à gauche).
\end{enumerate}
\end{correction}

\begin{correction}[vrai / faux justifié]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item \textbf{Faux.} L'égalité des limites latérales garantit seulement que la \emph{limite existe} ; il faut
        \emph{en plus} que cette limite vaille $f(x_0)$ (et que $x_0\in D_f$) pour la continuité.
  \item \textbf{Vrai.} Par exemple $f(x)=\dfrac{x^2-1}{x-1}$ n'est pas définie en $1$, mais tend vers $2$ en $1$.
        La limite décrit une tendance, pas une valeur au point.
  \item \textbf{Vrai.} C'est exactement la définition de la continuité en $x_0$ :
        $\displaystyle\lim_{x\to x_0} f(x)=f(x_0)$.
\end{enumerate}
\end{correction}
