\begin{exercice}[une fonction à deux paramètres]
Soit $a$ et $b$ deux réels et
\[ f(x)=\begin{cases} ax+1 & \text{si } x<2,\\[2pt] 3 & \text{si } x=2,\\[2pt] x^2+b & \text{si } x>2.\end{cases} \]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Exprimer $\displaystyle\lim_{x\to 2^{-}} f(x)$ et $\displaystyle\lim_{x\to 2^{+}} f(x)$ en fonction de $a$ et $b$.
  \item Écrire la condition sur $a$ et $b$ pour que $f$ \emph{possède une limite} en $2$.
  \item Écrire la (les) condition(s) supplémentaire(s) pour que $f$ soit \emph{continue} en $2$.
  \item En déduire toutes les paires $(a,b)$ rendant $f$ continue en $2$, puis donner un exemple numérique.
\end{enumerate}
\end{exercice}

\begin{exercice}[lecture graphique complète]
On donne ci-dessous la courbe d'une fonction $f$ (ronds pleins et ouverts, asymptotes en pointillés).
\begin{center}
\begin{tikzpicture}[scale=0.95]
\begin{axis}[width=12cm,height=6.2cm,axis lines=middle,xlabel={$x$},ylabel={$y$},
   xmin=-5.6,xmax=5.8,ymin=-3.4,ymax=3.8,xtick={-4,-2,2,4},ytick={-2,-1,1,2,3},
   tick label style={font=\scriptsize}]
  % branche pour x<2 : 1 + 2/(x-2) -> asympt verticale x=2 (vers -inf a gauche), horiz y=1
  \addplot[courbe,domain=-5.4:1.4,samples=140]{1+2/(x-2)};
  % branche pour x>2 : part d'un rond ouvert en (2,3) et decroit vers y=1
  \addplot[courbe,domain=2.25:5.6,samples=120]{1+0.9/(x-2)};
  \draw[asy,primary](axis cs:-5.6,1)--(axis cs:5.8,1);
  \draw[asy,primary](axis cs:2,-3.4)--(axis cs:2,3.8);
  \node[ptouvert] at (axis cs:2,3){};   % limite droite = 3 (ouvert)
  \node[pt,fill=curvecol] at (axis cs:2,0){}; % f(2)=0 (plein, isole)
  \node[primary,font=\scriptsize,anchor=south west] at (axis cs:-5.4,1.05){$y=1$};
  \node[primary,font=\scriptsize] at (axis cs:2.9,-2.7){$x=2$};
\end{axis}
\end{tikzpicture}
\end{center}
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Donner $\displaystyle\lim_{x\to +\infty} f(x)$ et $\displaystyle\lim_{x\to -\infty} f(x)$, et nommer l'asymptote correspondante.
  \item Donner $\displaystyle\lim_{x\to 2^{-}} f(x)$ ; que peut-on en conclure sur l'asymptote verticale ?
  \item Donner $\displaystyle\lim_{x\to 2^{+}} f(x)$ et la valeur $f(2)$ (rond plein).
  \item $f$ possède-t-elle une limite en $2$ ? est-elle continue en $2$ ? à droite ? Justifier chaque réponse.
\end{enumerate}
\end{exercice}

\begin{exercice}[mise en situation : un tarif par paliers]
Une clinique facture une consultation selon la durée $t$ (en minutes). Le montant $M(t)$ (en euros) est :
\[ M(t)=\begin{cases} 30 & \text{si } 0< t\leq 20,\\[2pt] 30+2\,(t-20) & \text{si } t>20.\end{cases} \]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Calculer $\displaystyle\lim_{t\to 20^{-}} M(t)$ et $\displaystyle\lim_{t\to 20^{+}} M(t)$.
  \item La fonction montant est-elle continue en $t=20$ ? Interpréter concrètement (le tarif « saute »-t-il ?).
  \item Même question au point $t=20$ si l'on remplace le second morceau par $35+2\,(t-20)$ :
        la facturation présenterait-elle un saut ? de combien d'euros ?
\end{enumerate}
\end{exercice}
