\begin{exercice}[deux asymptotes verticales]
Soit $f(x)=\dfrac{x+5}{x^2-x-6}$.
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Factoriser $x^2-x-6$ et donner le domaine $D_f$.
  \item Dresser le tableau de signes du d├®nominateur sur $\R$.
  \item ├ëtudier les limites lat├®rales en $x=3$ : $\displaystyle\lim_{x\to 3^{-}} f$ et $\displaystyle\lim_{x\to 3^{+}} f$.
  \item ├ëtudier les limites lat├®rales en $x=-2$.
  \item Conclure : combien d'asymptotes verticales ? Donner leurs ├®quations.
\end{enumerate}
\end{exercice}

\begin{exercice}[contraste : multiplicit├® paire ou impaire]

\[ f(x)=\frac{3}{x-1}\qquad\text{et}\qquad g(x)=\frac{3}{(x-1)^2}. \]
\begin{enumerate}[label=\alph*),leftmargin=2em,itemsep=3pt]
  \item Calculer $\displaystyle\lim_{x\to 1^{-}} f$ et $\displaystyle\lim_{x\to 1^{+}} f$. $f$ a-t-elle une limite en $1$ ?
  \item Calculer $\displaystyle\lim_{x\to 1^{-}} g$ et $\displaystyle\lim_{x\to 1^{+}} g$. $g$ a-t-elle une limite en $1$ ?
  \item Expliquer, ├á l'aide du signe du d├®nominateur, pourquoi $g$ a une limite globale mais pas $f$.
  \item Associer chaque fonction ├á l'une des deux allures ci-dessous pr├¿s de l'asymptote $x=1$.

\begin{exercice}[mise en situation : dilution qui s'emballe]

  \item Calculer $C(0)$, $C(3)$ et $C(4{,}9)$. Que constate-t-on quand $t$ approche $5$ ?
  \item Calculer $\displaystyle\lim_{t\to 5^{-}} C(t)$ ├á l'aide de la r├¿gle $\frac{a}{0^{\pm}}$.
  \item Interpr├®ter : que traduit cette asymptote verticale $t=5$ pour le mod├¿le ? Est-il r├®aliste
        au-del├á de $t=5$ ?
\end{enumerate}
\end{exercice}
